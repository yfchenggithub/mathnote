\begin{summarybox}
\textbf{\textcolor{CSummary}{一句话核心}}

\textbf{椭圆长轴端点切线与任意点切线交点到端点的距离乘积恒为 $b^2$，所围四边形面积最小值为 $2ab$。}

\medskip
\textbf{\textcolor{CSummary}{使用条件}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{条件 1（椭圆标准方程）：} 椭圆方程为 $\frac{x^2}{a^2}+\frac{y^2}{b^2}=1$，长轴在 $x$ 轴上。
    \item \textbf{条件 2（端点切线垂直）：} 过端点 $A_1(-a,0)$、$A_2(a,0)$ 的切线为 $x=-a$ 和 $x=a$。
\end{itemize}

\medskip
\textbf{\textcolor{CSummary}{关键提醒}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{易错点（切线公式）：} 确保正确写出椭圆在点 $(x_0,y_0)$ 处的切线方程。
    \item \textbf{检查项（乘积定值）：} 计算后验证 $|P_1A_1|\cdot|P_2A_2|$ 是否等于 $b^2$，以检查推导正确性。
\end{itemize}
\end{summarybox}
