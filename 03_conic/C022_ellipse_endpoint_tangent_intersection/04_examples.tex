\begin{examplebox}
\textbf{\textcolor{CExample}{例 1（基础）}}

\textbf{\textcolor{CExample}{题目：}} 已知椭圆 $C:\frac{x^2}{4}+\frac{y^2}{3}=1$，$A_1(-2,0)$，$A_2(2,0)$，$P$ 为椭圆上一点，过 $P$ 的切线交直线 $x=-2$ 于 $P_1$，交 $x=2$ 于 $P_2$，求 $|P_1A_1|\cdot|P_2A_2|$ 的值。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（识别结论）：}] 由结论，无论 $P$ 在椭圆上何处，$|P_1A_1|\cdot|P_2A_2|=b^2$。
    \par\textbf{理由：} 直接应用已知二级结论。

    \item[\textbf{第二步（确定参数）：}] 椭圆中 $a^2=4$，$b^2=3$，故 $b^2=3$。
    \par\textbf{理由：} 从椭圆方程读取 $a$ 和 $b$。

    \item[\textbf{第三步（写出答案）：}] 所以 $|P_1A_1|\cdot|P_2A_2|=3$。
    \par\textbf{理由：} 代入结论。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{对于椭圆 $\frac{x^2}{a^2}+\frac{y^2}{b^2}=1$，在给定条件下，$|P_1A_1|\cdot|P_2A_2|=b^2$ 恒成立。}

\medskip
\textbf{\textcolor{CExample}{例 2（稍变形）}}

\textbf{\textcolor{CExample}{题目：}} 椭圆 $C:\frac{x^2}{9}+\frac{y^2}{4}=1$，$A_1(-3,0)$，$A_2(3,0)$，$P$ 为椭圆上动点，过 $P$ 的切线交 $x=-3$ 于 $P_1$，交 $x=3$ 于 $P_2$，求四边形 $A_1P_1P_2A_2$ 面积的最小值。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（面积表达式）：}] 由结论，四边形面积 $S \ge 2ab$，且最小值为 $2ab$。
    \par\textbf{理由：} 直接应用面积最小值结论。

    \item[\textbf{第二步（计算参数）：}] 椭圆中 $a=3$，$b=2$，故 $2ab = 2\times 3\times 2 = 12$。
    \par\textbf{理由：} 从椭圆方程读取 $a$ 和 $b$。

    \item[\textbf{第三步（确定取等条件）：}] 当 $P$ 在短轴端点 $(0,\pm 2)$ 时，面积取得最小值 $12$。
    \par\textbf{理由：} 最小值在 $P(0,\pm b)$ 时取得。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{四边形 $A_1P_1P_2A_2$ 面积的最小值为 $2ab$，当且仅当 $P$ 在椭圆短轴端点时取等。}
\end{examplebox}
