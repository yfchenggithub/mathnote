\begin{trapbox}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CTrap}{易错点一（焦点位置误判）}}] 误以为焦点在 $y$ 轴上，或混淆 $c$ 的计算公式。
    \item[\textbf{\textcolor{CTrap}{正确理解（焦点在长轴）：}}] 对于椭圆 $\frac{x^2}{a^2}+\frac{y^2}{b^2}=1$（$a>b>0$），焦点在 $x$ 轴上，坐标为 $(\pm c,0)$，其中 $c=\sqrt{a^2-b^2}$。
    \item[\textbf{\textcolor{CTrap}{错因分析（标准形式混淆）：}}] 未注意椭圆方程中 $a$ 对应分母较大的轴，导致焦点位置错误。
\end{description}

\medskip
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CTrap}{易错点二（距离公式漏项）}}] 计算 $d$ 时，未将切线方程化为一般式 $Ax+By+C=0$，或忽略分母平方和的开方。
    \item[\textbf{\textcolor{CTrap}{正确理解（点到直线距离）：}}] 直线 $\frac{x_0 x}{a^2}+\frac{y_0 y}{b^2}=1$ 的一般式为 $\frac{x_0}{a^2}x + \frac{y_0}{b^2}y - 1 = 0$，距离公式为 $d = \frac{|C|}{\sqrt{A^2+B^2}}$。
    \item[\textbf{\textcolor{CTrap}{错因分析（公式记忆偏差）：}}] 对点到直线距离公式的条件理解不清，或直接代入错误系数。
\end{description}

\medskip
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CTrap}{易错点三（焦半径公式错用）}}] 误用 $r_1 = a + ex_0$ 和 $r_2 = a - ex_0$ 时，忽略 $x_0$ 的符号或 $e$ 的计算。
    \item[\textbf{\textcolor{CTrap}{正确理解（焦半径与坐标）：}}] 焦半径公式 $r_1 = a + ex_0$，$r_2 = a - ex_0$ 适用于焦点在 $x$ 轴且 $x_0$ 为点 $P$ 的横坐标，其中 $e=\frac{c}{a}$。
    \item[\textbf{\textcolor{CTrap}{错因分析（符号与范围混淆）：}}] 未注意 $x_0$ 可正可负，或误记 $e$ 为 $\sqrt{a^2+b^2}$ 等。
\end{description}
\end{trapbox}
