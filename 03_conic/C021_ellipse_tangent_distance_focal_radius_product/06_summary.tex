\begin{summarybox}
\textbf{\textcolor{CSummary}{一句话核心}}

\textbf{椭圆上任意点处的切线到原点距离，与该点到两焦点距离的几何平均的乘积，恒等于椭圆半轴长乘积。}

\medskip
\textbf{\textcolor{CSummary}{使用条件}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{条件 1（标准椭圆）：} 椭圆方程为 $\frac{x^2}{a^2}+\frac{y^2}{b^2}=1$，且 $a>b>0$。
    \item \textbf{条件 2（焦点在轴）：} 两焦点在 $x$ 轴上，坐标为 $(\pm c,0)$，$c=\sqrt{a^2-b^2}$。
\end{itemize}

\medskip
\textbf{\textcolor{CSummary}{关键提醒}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{易错点（焦点位置）：} 务必确认焦点在长轴（$x$ 轴），避免误用在 $y$ 轴椭圆。
    \item \textbf{检查项（公式代入）：} 计算 $d$ 和 $r_1,r_2$ 后，代入 $\sqrt{r_1 r_2}\cdot d$ 检查是否等于 $ab$。
\end{itemize}
\end{summarybox}
