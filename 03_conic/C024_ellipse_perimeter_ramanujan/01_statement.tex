\begin{statementbox}
	\textbf{\textcolor{CStatement}{条件}}

	\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
		\item[\textbf{\textcolor{CStatement}{条件1（椭圆标准型）：}}]
		      椭圆方程为 $\frac{x^2}{a^2}+\frac{y^2}{b^2}=1$，且 $a\ge b>0$。
	\end{description}

	\textbf{\textcolor{CStatement}{结论}}

	\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
		\item[\textbf{\textcolor{CStatement}{结论一（Ramanujan近似）：}}]
		      \textbf{\textcolor{CStatement}{直观描述：}} 用长短半轴计算椭圆周长的高精度近似值。
		      \[
			      C\approx \pi(a+b)\left(1+\frac{3h}{10+\sqrt{4-3h}}\right),
			      \qquad
			      h=\left(\frac{a-b}{a+b}\right)^2.
		      \]

		\item[\textbf{\textcolor{CStatement}{推广结论（圆特例）：}}]
		      \textbf{\textcolor{CStatement}{直观描述：}} 当 $a=b$ 时公式退化为精确圆周长 $2\pi a$。
		      \[
			      C=2\pi a.
		      \]
	\end{description}
\end{statementbox}
