\begin{trapbox}
	\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
		\item[\textbf{\textcolor{CTrap}{易错点一（计算h错误）}}]
		      计算 $h=\left(\frac{a-b}{a+b}\right)^2$ 时，忘记平方，或分子分母顺序颠倒。

		\item[\textbf{\textcolor{CTrap}{正确理解（正确步骤）：}}]
		      严格按公式计算：先求 $a-b$ 与 $a+b$ 的比值，再平方得到 $h$。

		\item[\textbf{\textcolor{CTrap}{错因分析（运算粗心）：}}]
		      对平方运算不敏感，或急于求成跳步导致漏算。
	\end{description}

	\medskip

	\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
		\item[\textbf{\textcolor{CTrap}{易错点二（误当精确公式）}}]
		      将该近似公式当作精确等式，在需要精确值的题目中直接使用，导致答案不被认可。

		\item[\textbf{\textcolor{CTrap}{正确理解（近似性质）：}}]
		      该公式是近似公式，并非精确值；应用中需注意题目是否允许近似。

		\item[\textbf{\textcolor{CTrap}{错因分析（概念模糊）：}}]
		      对椭圆周长无初等表达式缺乏认识，混淆近似与精确。
	\end{description}

	\medskip

	\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
		\item[\textbf{\textcolor{CTrap}{易错点三（特例处理不当）}}]
		      当 $a=b$ 时，不知道 $h=0$，仍按一般步骤计算，或认为公式不适用。

		\item[\textbf{\textcolor{CTrap}{正确理解（退化情形）：}}]
		      当 $a=b$ 时，$h=0$，公式右端 $\pi(a+b)$ 即为 $2\pi a$，与圆周长相同，是精确值。

		\item[\textbf{\textcolor{CTrap}{错因分析（灵活不足）：}}]
		      未注意公式退化为圆的情况，缺乏对公式结构的深入理解。
	\end{description}
\end{trapbox}
