\begin{examplebox}
	\textbf{\textcolor{CExample}{例 1（基础）}}

	\textbf{\textcolor{CExample}{题目：}} 已知椭圆方程为 $\frac{x^2}{25}+\frac{y^2}{9}=1$，求该椭圆周长的近似值（结果保留三位小数）。

	\textbf{\textcolor{CExample}{解题步骤：}}

	\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
		\item[\textbf{第一步（识别参数）：}]
		      由标准方程得 $a=5,b=3$。

		      \par\textbf{理由：} 椭圆方程 $\frac{x^2}{a^2}+\frac{y^2}{b^2}=1$，分母为平方项，故 $a^2=25,b^2=9$。

		\item[\textbf{第二步（计算h）：}]
		      计算 $h=\left(\frac{a-b}{a+b}\right)^2=\left(\frac{5-3}{5+3}\right)^2=\left(\frac{2}{8}\right)^2=\frac{1}{16}=0.0625$。

		      \par\textbf{理由：} $h$ 是公式中反映椭圆扁平程度的中间量。

		\item[\textbf{第三步（代入求值）：}]
		      代入近似公式：
		      \[
			      \begin{aligned}
				      C & \approx \pi(a+b)\left(1+\frac{3h}{10+\sqrt{4-3h}}\right)                         \\
				        & = \pi\times8\times\left(1+\frac{3\times0.0625}{10+\sqrt{4-3\times0.0625}}\right) \\
				        & = 8\pi\left(1+\frac{0.1875}{10+\sqrt{3.8125}}\right)                             \\
				        & \approx 8\pi\left(1+\frac{0.1875}{10+1.9526}\right)                              \\
				        & = 8\pi\left(1+\frac{0.1875}{11.9526}\right)                                      \\
				        & \approx 8\pi\times1.01568 \approx 25.529.
			      \end{aligned}
		      \]

		      \par\textbf{理由：} 按公式逐次计算 $h$、根号、分数，最后乘以 $\pi(a+b)$ 得到近似周长。
	\end{description}

	\textbf{\textcolor{CExample}{关键结论：}} \textbf{椭圆周长近似为 $25.529$（单位与半轴相同）。}

	\textbf{\textcolor{CExample}{例 2（稍变形）}}

	\textbf{\textcolor{CExample}{题目：}} 已知椭圆长半轴 $a=5$，离心率 $e=\frac{3}{5}$，求该椭圆周长的近似值（结果保留三位小数）。

	\textbf{\textcolor{CExample}{解题步骤：}}

	\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
		\item[\textbf{第一步（求短半轴）：}]
		      由 $e=\frac{c}{a}$，得 $c=3$，从而 $b=\sqrt{a^2-c^2}=\sqrt{25-9}=4$。

		      \par\textbf{理由：} 椭圆参数关系 $c^2=a^2-b^2$。

		\item[\textbf{第二步（计算h）：}]
		      $h=\left(\frac{a-b}{a+b}\right)^2=\left(\frac{5-4}{5+4}\right)^2=\left(\frac{1}{9}\right)^2=\frac{1}{81}\approx0.0123457$。

		      \par\textbf{理由：} h为形状参数。

		\item[\textbf{第三步（代入公式）：}]
		      \[
			      \begin{aligned}
				      C & \approx \pi(a+b)\left(1+\frac{3h}{10+\sqrt{4-3h}}\right)                                           \\
				        & = \pi\times9\times\left(1+\frac{3\times0.0123457}{10+\sqrt{4-3\times0.0123457}}\right)             \\
				        & = 9\pi\left(1+\frac{0.0370371}{10+\sqrt{4-0.0370371}}\right)                                       \\
				        & = 9\pi\left(1+\frac{0.0370371}{10+\sqrt{3.9629629}}\right)                                         \\
				        & \approx 9\pi\left(1+\frac{0.0370371}{10+1.990718}\right)                                           \\
				        & = 9\pi\left(1+\frac{0.0370371}{11.990718}\right)                                                   \\
				        & \approx 9\pi(1+0.003088) = 9\pi\times1.003088 \approx 9\times3.14159\times1.003088 \approx 28.362.
			      \end{aligned}
		      \]

		      \par\textbf{理由：} 代入数值逐步计算。
	\end{description}

	\textbf{\textcolor{CExample}{关键结论：}} \textbf{该椭圆周长近似为 $28.362$。}

	\textbf{\textcolor{CExample}{例 3（综合应用）}}

	\textbf{\textcolor{CExample}{题目：}} 已知椭圆 $E_1:\frac{x^2}{100}+\frac{y^2}{36}=1$，椭圆 $E_2:\frac{x^2}{81}+\frac{y^2}{49}=1$，试比较它们的周长大小。

	\textbf{\textcolor{CExample}{解题步骤：}}

	\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
		\item[\textbf{第一步（计算E1周长）：}]
		      对 $E_1$，$a=10,b=6$，$h=\left(\frac{10-6}{10+6}\right)^2=\left(\frac{4}{16}\right)^2=0.0625$。代入公式：
		      \[
			      \begin{aligned}
				      C_1 & \approx \pi(10+6)\left(1+\frac{3\times0.0625}{10+\sqrt{4-3\times0.0625}}\right) \\
				          & = 16\pi\left(1+\frac{0.1875}{10+\sqrt{3.8125}}\right)                           \\
				          & \approx 16\pi\left(1+\frac{0.1875}{10+1.9526}\right)                            \\
				          & = 16\pi\left(1+\frac{0.1875}{11.9526}\right)                                    \\
				          & \approx 16\pi\times1.01568 \approx 51.053.
			      \end{aligned}
		      \]

		      \par\textbf{理由：} 直接套用公式。

		\item[\textbf{第二步（计算E2周长）：}]
		      对 $E_2$，$a=9,b=7$，$h=\left(\frac{9-7}{9+7}\right)^2=\left(\frac{2}{16}\right)^2=\frac{1}{64}=0.015625$。代入公式：
		      \[
			      \begin{aligned}
				      C_2 & \approx \pi(9+7)\left(1+\frac{3\times0.015625}{10+\sqrt{4-3\times0.015625}}\right) \\
				          & = 16\pi\left(1+\frac{0.046875}{10+\sqrt{4-0.046875}}\right)                        \\
				          & = 16\pi\left(1+\frac{0.046875}{10+\sqrt{3.953125}}\right)                          \\
				          & \approx 16\pi\left(1+\frac{0.046875}{10+1.9883}\right)                             \\
				          & = 16\pi\left(1+\frac{0.046875}{11.9883}\right)                                     \\
				          & \approx 16\pi(1+0.003910) \approx 16\pi\times1.00391 \approx 50.463.
			      \end{aligned}
		      \]

		      \par\textbf{理由：} 同公式计算。

		\item[\textbf{第三步（比较大小）：}]
		      由于 $C_1\approx51.053 > C_2\approx50.463$，故 $E_1$ 的周长大于 $E_2$ 的周长。

		      \par\textbf{理由：} 直接比较数值大小。
	\end{description}

	\textbf{\textcolor{CExample}{关键结论：}} \textbf{椭圆 $E_1$ 周长更大}（尽管两者 $a+b$ 相等，但 $E_1$ 更扁平导致修正因子更大）。
