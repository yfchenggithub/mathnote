\begin{trapbox}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CTrap}{易错点一（忽略条件）}}] 忘记椭圆方程中 $a>b>0$ 的条件，误用 $a,b$ 符号或大小关系。
    \item[\textbf{\textcolor{CTrap}{正确理解（标准方程）：}}] 公式仅适用于椭圆标准方程 $\frac{x^2}{a^2}+\frac{y^2}{b^2}=1$ 且 $a>b>0$，否则 $a^2,b^2$ 意义不同。
    \item[\textbf{\textcolor{CTrap}{错因分析（前提混淆）：}}] 未注意结论前提，直接套用于非标准椭圆或 $a,b$ 顺序颠倒的情况。
\end{description}

\medskip
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CTrap}{易错点二（公式记错）}}] 记忆 $m^2 = a^2 k^2 + b^2$ 时，误写成减号或混淆 $a^2,b^2$ 位置。
    \item[\textbf{\textcolor{CTrap}{正确理解（相加结构）：}}] 右边是 $a^2 k^2$ 与 $b^2$ 相加，顺序固定，且为加号。
    \item[\textbf{\textcolor{CTrap}{错因分析（符号混淆）：}}] 与其他圆锥曲线切线公式（如双曲线）混淆，或受判别式展开中间步骤干扰。
\end{description}

\medskip
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CTrap}{易错点三（斜率特殊）}}] 忽略直线斜率不存在（即 $x=c$ 型）的情况，此时公式不直接适用。
    \item[\textbf{\textcolor{CTrap}{正确理解（分类讨论）：}}] 当直线垂直时，应单独考虑切线方程 $x=\pm a$，其对应 $k$ 无穷大，但公式需极限处理。
    \item[\textbf{\textcolor{CTrap}{错因分析（漏解）：}}] 只记斜截式公式，未涵盖所有切线情形，导致漏解。
\end{description}
\end{trapbox}
