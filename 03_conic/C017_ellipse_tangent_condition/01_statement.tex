\textbf{\textcolor{CStatement}{条件}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{条件 1（标准方程）：}}] 椭圆方程为 $\dfrac{x^2}{a^2}+\dfrac{y^2}{b^2}=1$，其中 $a>b>0$；直线方程为 $y=kx+m$。
\end{description}

\textbf{\textcolor{CStatement}{结论}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{\textcolor{CStatement}{结论一（相切条件）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 直线与椭圆相切时，截距平方满足固定关系。
    \[
    m^2 = a^2 k^2 + b^2
    \]

    \item[\textbf{\textcolor{CStatement}{推广结论（切线方程）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 切线方程可直接写出截距表达式。
    \[
    y = kx \pm \sqrt{a^2 k^2 + b^2}
    \]
\end{description}

\end{statementbox}
