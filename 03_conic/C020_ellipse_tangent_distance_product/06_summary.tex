\begin{summarybox}
\textbf{\textcolor{CSummary}{一句话核心}}

\textbf{椭圆上任意点处的切线到原点的距离与该点到两焦点距离的几何均值的乘积，恒等于椭圆半长轴与半短轴的乘积。}

\medskip
\textbf{\textcolor{CSummary}{使用条件}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{条件 1（椭圆标准方程）：} 椭圆方程为 $\dfrac{x^2}{a^2}+\dfrac{y^2}{b^2}=1$，且 $a>b>0$，焦点在 $x$ 轴上。
    \item \textbf{条件 2（点在椭圆上）：} 点 $P(x_0,y_0)$ 必须满足椭圆方程，即 $\dfrac{x_0^2}{a^2}+\dfrac{y_0^2}{b^2}=1$。
    \item \textbf{条件 3（切线定义）：} 直线 $l$ 是椭圆在点 $P$ 处的切线，需用正确公式表达。
\end{itemize}

\medskip
\textbf{\textcolor{CSummary}{关键提醒}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{易错点（参数混淆）：} 混淆 $a,b,c$ 关系，尤其在求焦点坐标或 $r_1, r_2$ 时，务必用 $c^2=a^2-b^2$。
    \item \textbf{检查项（公式验证）：} 应用结论前，先检查椭圆方程是否标准、点是否在椭圆上，避免误用。
\end{itemize}
\end{summarybox}
