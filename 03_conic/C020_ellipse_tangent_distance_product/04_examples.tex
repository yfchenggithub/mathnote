\begin{examplebox}
\textbf{\textcolor{CExample}{例 1（基础）}}

\textbf{\textcolor{CExample}{题目：}} 已知椭圆 $\dfrac{x^2}{16}+\dfrac{y^2}{9}=1$，点 $P(2,\frac{3\sqrt{3}}{2})$ 在椭圆上。求椭圆在点 $P$ 处的切线到原点的距离 $d$，并验证 $\sqrt{r_1 r_2} \cdot d = 12$，其中 $r_1, r_2$ 为 $P$ 到两焦点的距离。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（求椭圆参数）：}] 由椭圆方程得 $a^2=16$, $b^2=9$，故 $a=4$, $b=3$，焦距 $c=\sqrt{a^2-b^2}=\sqrt{7}$。焦点为 $F_1(-\sqrt{7},0)$, $F_2(\sqrt{7},0)$。
    \par\textbf{理由：} 标准椭圆参数定义。

    \item[\textbf{第二步（计算 $r_1 r_2$）：}] 点 $P(2,\frac{3\sqrt{3}}{2})$，计算 $r_1 = \sqrt{(2+\sqrt{7})^2 + \left(\frac{3\sqrt{3}}{2}\right)^2}$, $r_2 = \sqrt{(2-\sqrt{7})^2 + \left(\frac{3\sqrt{3}}{2}\right)^2}$。先算平方：
    \[
    \begin{aligned}
    r_1^2 &= (2+\sqrt{7})^2 + \frac{27}{4} = 4+4\sqrt{7}+7+6.75 = 17.75+4\sqrt{7}, \\
    r_2^2 &= (2-\sqrt{7})^2 + \frac{27}{4} = 4-4\sqrt{7}+7+6.75 = 17.75-4\sqrt{7}.
    \end{aligned}
    \]
    故 $r_1^2 r_2^2 = (17.75+4\sqrt{7})(17.75-4\sqrt{7}) = (17.75)^2 - (4\sqrt{7})^2 = 315.0625 - 112 = 203.0625$。
    所以 $r_1 r_2 = \sqrt{203.0625} = 14.25$（精确值可保留为 $\frac{57}{4}$）。
    \par\textbf{理由：} 直接代入距离公式，利用平方差简化计算。

    \item[\textbf{第三步（利用结论求 $d$）：}] 由结论 $\sqrt{r_1 r_2} \cdot d = ab = 4\times3=12$，得
    \[
    d = \frac{12}{\sqrt{r_1 r_2}} = \frac{12}{\sqrt{14.25}} = \frac{12}{\frac{\sqrt{57}}{2}} = \frac{24}{\sqrt{57}} = \frac{24\sqrt{57}}{57}.
    \]
    \par\textbf{理由：} 直接套用结论，避免计算切线方程和距离公式。

    \item[\textbf{第四步（验证）：}] 计算 $\sqrt{r_1 r_2} = \sqrt{14.25} = \frac{\sqrt{57}}{2}$，则 $\sqrt{r_1 r_2} \cdot d = \frac{\sqrt{57}}{2} \times \frac{24\sqrt{57}}{57} = \frac{24\times57}{2\times57} = 12$，符合 $ab=12$。
    \par\textbf{理由：} 验证结论自洽，巩固理解。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{对于给定椭圆和点 $P$，可直接用 $\sqrt{r_1 r_2} \cdot d = ab$ 求 $d$ 或验证定值，节省计算量。}

\textbf{\textcolor{CExample}{例 2（稍变形）}}

\textbf{\textcolor{CExample}{题目：}} 椭圆 $\dfrac{x^2}{25}+\dfrac{y^2}{16}=1$，点 $P$ 在椭圆上，且椭圆在 $P$ 处的切线到原点的距离 $d=2$。求 $\sqrt{r_1 r_2}$ 的值，其中 $r_1, r_2$ 为 $P$ 到焦点的距离。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（确定参数）：}] 椭圆中 $a^2=25$, $b^2=16$，故 $a=5$, $b=4$，$ab=20$。
    \par\textbf{理由：} 提取基本量。

    \item[\textbf{第二步（套用结论）：}] 由 $\sqrt{r_1 r_2} \cdot d = ab$，代入 $d=2$, $ab=20$，得
    \[
    \sqrt{r_1 r_2} \times 2 = 20 \quad \Rightarrow \quad \sqrt{r_1 r_2} = 10.
    \]
    \par\textbf{理由：} 结论直接变形求解。

    \item[\textbf{第三步（解释意义）：}] 因此 $\sqrt{r_1 r_2}=10$，即点 $P$ 到两焦点距离的几何均值为 $10$。结合椭圆定义 $r_1+r_2=2a=10$，可进一步求 $r_1, r_2$，但非本题要求。
    \par\textbf{理由：} 结论给出了焦点距离几何均值与切线距离的关系，无需具体点 $P$ 坐标。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{当已知切线距离 $d$ 时，可利用结论反推 $\sqrt{r_1 r_2}$，体现了结论的双向应用价值。}
\end{examplebox}
