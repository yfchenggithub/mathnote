\begin{summarybox}
\textbf{\textcolor{CSummary}{一句话核心}}

\textbf{椭圆围成区域的面积等于 $\pi$ 乘以两条半轴长度之积：$S=\pi ab$。}

\medskip
\textbf{\textcolor{CSummary}{使用条件}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{条件 1（标准椭圆形式）：} 椭圆边界为 $\frac{x^2}{a^2}+\frac{y^2}{b^2}=1$，且 $a>0,\ b>0$。
    \item \textbf{条件 2（半轴含义清楚）：} $a,b$ 表示两条半轴长度；可以是 $x,y$ 方向半轴，也可以按教材记作长半轴、短半轴。
\end{itemize}

\medskip
\textbf{\textcolor{CSummary}{关键提醒}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{易错点（区分分母与半轴）：} 标准方程中的分母是半轴的平方，代入前需开方。
    \item \textbf{检查项（方程标准化）：} 使用前确认椭圆方程已化为右边等于 $1$ 的标准形式，否则会误读半轴。
    \item \textbf{综合题提醒（焦距）：} 椭圆焦距为 $2c$，不要把题目给出的焦距直接当作 $c$。
\end{itemize}
\end{summarybox}
