\begin{examplebox}
\textbf{\textcolor{CExample}{例 1（基础）}}

\textbf{\textcolor{CExample}{题目：}} 已知椭圆方程为 $\frac{x^2}{4}+\frac{y^2}{9}=1$，求该椭圆的面积。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（识别半轴）：}] 由方程得 $x$ 方向半轴为 $\sqrt{4}=2$，$y$ 方向半轴为 $\sqrt{9}=3$。
    \[
    \text{两条半轴长分别为 }2,\ 3.
    \]
    \par\textbf{理由：} 标准方程中分母是半轴长的平方，读半轴前必须开方。

    \item[\textbf{第二步（套用公式）：}] 椭圆面积公式为 $S=\pi\times\text{半轴}_1\times\text{半轴}_2$。
    \[
    S = \pi \times 2 \times 3.
    \]
    \par\textbf{理由：} 面积只与两条半轴长度的乘积有关。

    \item[\textbf{第三步（计算面积）：}] 计算乘积。
    \[
    S = 6\pi.
    \]
    \par\textbf{理由：} $2\times3=6$，再乘以 $\pi$。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{椭圆面积为 $6\pi$。}

\textbf{\textcolor{CExample}{例 2（稍变形）}}

\textbf{\textcolor{CExample}{题目：}} 已知椭圆的面积为 $20\pi$，且其一条半轴长为 $5$，求另一条半轴长。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（设未知数）：}] 设另一条半轴长为 $t$，已知一条半轴长为 $5$，面积 $S=20\pi$。
    \[
    S=20\pi,\quad \text{一条半轴}=5.
    \]
    \par\textbf{理由：} 面积公式只需要两条半轴长度，不必先判断哪条是长半轴。

    \item[\textbf{第二步（代入公式）：}] 由 $S=\pi\times 5\times t$ 得
    \[
    20\pi = \pi\cdot 5\cdot t.
    \]
    \par\textbf{理由：} 将已知面积和已知半轴代入公式。

    \item[\textbf{第三步（解方程）：}] 两边同除以 $\pi$ 得 $20=5t$，解得 $t=4$。
    \[
    t = 4.
    \]
    \par\textbf{理由：} 等式两边除以 $\pi$（$\pi\neq0$），再除以 $5$。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{另一条半轴长为 $4$。}

\medskip
\textbf{\textcolor{CExample}{例 3（综合应用）}}

\textbf{\textcolor{CExample}{题目：}} 已知椭圆的短半轴长为 $3$，焦距为 $8$，求该椭圆的面积。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（求半焦距）：}] 焦距 $2c=8$，所以半焦距 $c=4$。
    \[
    c = 4.
    \]
    \par\textbf{理由：} 椭圆的焦距是两个焦点之间的距离，等于 $2c$。

    \item[\textbf{第二步（求长半轴）：}] 按圆锥曲线常规记号，设长半轴为 $a$，短半轴为 $b$。已知 $b=3,\ c=4$，由 $a^2=b^2+c^2$ 得
    \[
    a^2 = 3^2 + 4^2 = 25,\quad a=5.
    \]
    \par\textbf{理由：} 对椭圆有 $a^2=b^2+c^2$，其中 $a$ 是长半轴，$b$ 是短半轴。

    \item[\textbf{第三步（计算面积）：}] 面积公式 $S=\pi ab$，代入 $a=5,
    b=3$。
    \[
    S = \pi \times 5 \times 3 = 15\pi.
    \]
    \par\textbf{理由：} $5\times3=15$，再乘以 $\pi$。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{椭圆面积为 $15\pi$。}
\end{examplebox}
