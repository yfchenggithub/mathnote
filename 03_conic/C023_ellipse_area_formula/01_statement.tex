\begin{statementbox}
\textbf{\textcolor{CStatement}{条件}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{条件1（标准椭圆方程）：}}] 椭圆边界为
    $\frac{x^2}{a^2}+\frac{y^2}{b^2}=1$，且 $a>0,\ b>0$。其中 $a,b$ 分别表示$x$方向、$y$方向的半轴长，不预设 $a>b$。
\end{description}

\textbf{\textcolor{CStatement}{结论}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{\textcolor{CStatement}{结论一（椭圆面积公式）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 椭圆围成区域的面积等于 $\pi$ 乘以两条半轴长度之积。
    \[
    S = \pi ab.
    \]

    \item[\textbf{\textcolor{CStatement}{推广结论（圆为特例）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 当 $a=b$ 时，椭圆变为半径为 $a$ 的圆，面积公式退化为圆面积。
    \[
    S = \pi a^2.
    \]
\end{description}

\textbf{\textcolor{CStatement}{适用范围：}} 该公式适用于中心在原点、轴与坐标轴平行的标准椭圆。对于平移或旋转后的椭圆，面积仍等于 $\pi$ 乘以两条半轴长度，但不能直接从原方程的系数读出 $a,b$。
\end{statementbox}
