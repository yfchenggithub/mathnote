\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{条件 1（椭圆方程）：}}] 椭圆 $C$ 的方程为 $\frac{x^2}{a^2}+\frac{y^2}{b^2}=1\ (a>b>0)$，点 $P(x_0,y_0)$ 在椭圆 $C$ 上。
\end{description}

\textbf{\textcolor{CStatement}{结论}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{\textcolor{CStatement}{结论一（斜率乘积）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 原点与椭圆上点连线的斜率，乘以该点处切线的斜率，结果为离心率平方减一。
    \[
    k_{OP} \cdot k_l = e^{2} - 1
    \]

    \item[\textbf{\textcolor{CStatement}{推广结论（圆特例）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 当椭圆退化为圆时，两斜率乘积为 $-1$，即切线垂直于半径。
    \[
    \text{若 } e=0 \text{(即圆)} k_{OP} \cdot k_l = -1.
    \]
\end{description}
