\begin{summarybox}
\textbf{\textcolor{CSummary}{一句话核心}}
\textbf{椭圆上任意一点（非坐标轴点）处，径向斜率与切线斜率之积等于离心率平方减一，这是一个只与椭圆形状有关的定值。}

\medskip
\textbf{\textcolor{CSummary}{使用条件}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{条件 1（标准椭圆）：} 椭圆方程为 $\frac{x^2}{a^2}+\frac{y^2}{b^2}=1$，且 $a>b>0$。
    \item \textbf{条件 2（点 P 在椭圆上）：} 点 $P(x_0,y_0)$ 满足椭圆方程，且通常要求 $x_0 \neq 0, y_0 \neq 0$ 以避免斜率不存在。
\end{itemize}

\medskip
\textbf{\textcolor{CSummary}{关键提醒}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{易错点（符号与记忆）：} 牢记公式 $k_{OP} \cdot k_l = e^2-1$（负值），切勿与双曲线结论混淆。
    \item \textbf{检查项（离心率计算）：} 使用前务必正确计算 $e^2 = 1 - \frac{b^2}{a^2}$，确保 $a, b$ 对应准确。
\end{itemize}
\end{summarybox}
