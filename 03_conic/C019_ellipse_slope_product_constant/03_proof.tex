\begin{proofbox}
\textbf{\textcolor{CProof}{思路提示}}
目标是证明 $k_{OP} \cdot k_l = e^2 - 1$。核心步骤是：先分别用点 $P$ 的坐标表示 $k_{OP}$ 和 $k_l$，然后计算它们的乘积，并利用点 $P$ 在椭圆上满足的方程以及离心率定义 $e^2 = 1 - \frac{b^2}{a^2}$ 进行化简。

\medskip
\textbf{\textcolor{CProof}{正式推导}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.55em]
    \item[\textbf{步骤一（表示斜率）：}] 由定义，直线 $OP$ 的斜率为 $k_{OP} = \dfrac{y_0}{x_0}$（其中 $x_0 \neq 0$）。
    \par\textbf{理由：} 两点 $O(0,0)$ 与 $P(x_0,y_0)$ 确定直线的斜率公式。

    \item[\textbf{步骤二（求切线斜率）：}] 椭圆在点 $P$ 处的切线 $l$ 的斜率 $k_l$，可通过隐函数求导得到。
    \[
    \begin{aligned}
        \frac{x^2}{a^2}+\frac{y^2}{b^2} &= 1 \\
        \frac{d}{dx}\left(\frac{x^2}{a^2}+\frac{y^2}{b^2}\right) &= \frac{d}{dx}(1) \\
        \frac{2x}{a^2} + \frac{2y}{b^2} \cdot \frac{dy}{dx} &= 0
    \end{aligned}
    \]
    \par\textbf{理由：} 对方程两边关于 $x$ 求导，注意 $y$ 是 $x$ 的函数。
    \par 解出 $\dfrac{dy}{dx}$，并在点 $P(x_0,y_0)$ 处取值：
    \[
        k_l = \left.\frac{dy}{dx}\right|_{(x_0,y_0)} = -\frac{b^2 x_0}{a^2 y_0} \quad (y_0 \neq 0)。
    \]

    \item[\textbf{步骤三（计算乘积）：}] 计算 $k_{OP} \cdot k_l$。
    \[
        k_{OP} \cdot k_l = \frac{y_0}{x_0} \cdot \left(-\frac{b^2 x_0}{a^2 y_0}\right) = -\frac{b^2}{a^2}。
    \]
    \par\textbf{理由：} 将步骤一和步骤二的表达式代入，并约去 $x_0, y_0$（假设 $x_0 \neq 0$ 且 $y_0 \neq 0$）。

    \item[\textbf{步骤四（引入离心率）：}] 利用离心率 $e$ 的定义进行转化。
    \[
        e^2 = \frac{c^2}{a^2} = \frac{a^2 - b^2}{a^2} = 1 - \frac{b^2}{a^2}。
    \]
    \par\textbf{理由：} 这是椭圆离心率的标准定义，其中 $c^2 = a^2 - b^2$。
    \par 因此，$ -\frac{b^2}{a^2} = e^2 - 1$。

    \item[\textbf{步骤五（合并结果）：}] 将步骤三与步骤四结合，得到最终关系。
    \[
        k_{OP} \cdot k_l = -\frac{b^2}{a^2} = e^2 - 1。
    \]
    \par\textbf{理由：} 直接代入化简。

    \item[\textbf{步骤六（补充特殊情况）：}] 当 $x_0=0$ 或 $y_0=0$ 时，需单独验证。
    \par 若 $x_0=0$，则 $P$ 在 $y$ 轴上，$k_{OP}$ 不存在（或视为无穷大），此时切线水平，$k_l=0$。乘积 $k_{OP} \cdot k_l$ 可理解为 $0$（或极限情形），而 $e^2-1 \neq 0$，看似矛盾。
    \par \textbf{理由：} 实际上，当 $x_0=0$ 时，点 $P$ 坐标为 $(0, \pm b)$，切线方程为 $y = \pm b$，斜率为 $0$。而 $OP$ 斜率不存在，公式 $k_{OP} \cdot k_l = e^2-1$ 应理解为在极限意义下成立（当 $P$ 趋近于 $y$ 轴时，$k_{OP} \to \infty$，$k_l \to 0$，乘积的极限为 $e^2-1$）。在严格应用中，我们通常避开这种退化点。类似地，$y_0=0$ 时，$P$ 在 $x$ 轴上，$k_{OP}=0$，切线垂直，斜率不存在，结论在极限意义下也成立。
\end{description}

\medskip
\textbf{\textcolor{CProof}{结论回扣}}
\textbf{综上所述，对于椭圆 $\frac{x^2}{a^2}+\frac{y^2}{b^2}=1$ 上非坐标轴上的点 $P$，恒有 $k_{OP} \cdot k_l = e^2 - 1$。该关系将几何中的两个斜率与椭圆的离心率紧密联系在一起。}
\end{proofbox}
