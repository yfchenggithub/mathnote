\begin{examplebox}
\textbf{\textcolor{CExample}{例 1（基础）}}

\textbf{\textcolor{CExample}{题目：}} 已知椭圆 $C: \frac{x^2}{16}+\frac{y^2}{9}=1$，点 $P(2, \frac{3\sqrt{3}}{2})$ 在 $C$ 上。求椭圆在点 $P$ 处的切线 $l$ 的斜率 $k_l$。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（求离心率）：}] 由椭圆方程得 $a^2=16, b^2=9$，故 $c^2 = a^2 - b^2 = 7$，离心率 $e = \frac{\sqrt{7}}{4}$。
    \par\textbf{理由：} 先确定椭圆的基本参数，为使用结论做准备。

    \item[\textbf{第二步（求 $k_{OP}$）：}] 计算直线 $OP$ 的斜率。
    \[
        k_{OP} = \frac{y_0}{x_0} = \frac{3\sqrt{3}/2}{2} = \frac{3\sqrt{3}}{4}.
    \]
    \par\textbf{理由：} 直接利用斜率公式。

    \item[\textbf{第三步（用结论求 $k_l$）：}] 由结论 $k_{OP} \cdot k_l = e^2 - 1$ 得
    \[
        k_l = \frac{e^2 - 1}{k_{OP}} = \frac{\frac{7}{16} - 1}{\frac{3\sqrt{3}}{4}} = \frac{-\frac{9}{16}}{\frac{3\sqrt{3}}{4}} = -\frac{9}{16} \cdot \frac{4}{3\sqrt{3}} = -\frac{3}{4\sqrt{3}} = -\frac{\sqrt{3}}{4}.
    \]
    \par\textbf{理由：} 代入已知量计算。注意 $e^2 = \frac{7}{16}$，$e^2-1 = -\frac{9}{16}$。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{利用 $k_{OP} \cdot k_l = e^2-1$，可快速求得切线斜率 $k_l = -\frac{\sqrt{3}}{4}$，避免了隐函数求导的步骤。}

\textbf{\textcolor{CExample}{例 2（稍变形）}}

\textbf{\textcolor{CExample}{题目：}} 已知椭圆 $C: \frac{x^2}{a^2}+\frac{y^2}{b^2}=1 (a>b>0)$ 的离心率为 $\frac{1}{2}$。过椭圆上一点 $P$ 作切线，若该切线的斜率为 $-\frac{1}{2}$，求直线 $OP$ 的斜率 $k_{OP}$。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（列出关系）：}] 由已知，$e=\frac{1}{2}$，$k_l = -\frac{1}{2}$。根据结论有
    \[
        k_{OP} \cdot k_l = e^2 - 1.
    \]
    \par\textbf{理由：} 直接套用结论公式。

    \item[\textbf{第二步（计算 $e^2-1$）：}] 计算常数项。
    \[
        e^2 - 1 = \left(\frac{1}{2}\right)^2 - 1 = \frac{1}{4} - 1 = -\frac{3}{4}.
    \]
    \par\textbf{理由：} 先确定等号右边的值。

    \item[\textbf{第三步（解出 $k_{OP}$）：}] 将 $k_l = -\frac{1}{2}$ 和 $e^2-1 = -\frac{3}{4}$ 代入关系式：
    \[
        k_{OP} \cdot \left(-\frac{1}{2}\right) = -\frac{3}{4}.
    \]
    \par 解得
    \[
        k_{OP} = \frac{-\frac{3}{4}}{-\frac{1}{2}} = \frac{3}{2}.
    \]
    \par\textbf{理由：} 简单代数运算。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{直线 $OP$ 的斜率为 $\frac{3}{2}$。本题展示了已知离心率和切线斜率，反求径向斜率的逆向应用。}
\end{examplebox}
