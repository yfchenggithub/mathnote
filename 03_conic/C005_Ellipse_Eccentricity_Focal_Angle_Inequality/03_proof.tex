\begin{proofbox}
\textbf{利用焦半径公式推导：}
\begin{enumerate}
    \item 在 $\triangle F_1PF_2$ 中，设 $|PF_1|=r_1, |PF_2|=r_2$。由定义知 $r_1+r_2=2a$，底边 $|F_1F_2|=2c$。
    \item 由余弦定理：$\cos\theta = \frac{r_1^2 + r_2^2 - (2c)^2}{2r_1r_2} = \frac{(r_1+r_2)^2 - 2r_1r_2 - 4c^2}{2r_1r_2} = \frac{4a^2 - 4c^2 - 2r_1r_2}{2r_1r_2} = \frac{2b^2}{r_1r_2} - 1$。
    \item 由半角公式：$\sin^2 \frac{\theta}{2} = \frac{1-\cos\theta}{2} = 1 - \frac{b^2}{r_1r_2}$。
    \item 由均值不等式：$r_1r_2 \le (\frac{r_1+r_2}{2})^2 = a^2$，等号在 $r_1=r_2=a$（即短轴端点）时成立。
    \item 故 $\sin^2 \frac{\theta}{2} \le 1 - \frac{b^2}{a^2} = \frac{c^2}{a^2} = e^2$。
    \item 开方得 $\sin \frac{\theta}{2} \le e$。
\end{enumerate}
\end{proofbox}
