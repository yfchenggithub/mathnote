\begin{examplebox}
\textbf{\textcolor{CExample}{例 1（基础）}}

\textbf{\textcolor{CExample}{题目：}} 已知椭圆 $\dfrac{x^2}{9} + \dfrac{y^2}{4} = 1$，点 $P(3,0)$ 在椭圆上，求椭圆在点 $P$ 处的切线方程。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（验证点位置）：}] 验证点 $P(3,0)$ 满足椭圆方程。
    \[
    \frac{3^2}{9} + \frac{0^2}{4} = \frac{9}{9} + 0 = 1
    \]
    \par\textbf{理由：} 确保点 $P$ 在椭圆上，才能使用切线公式。

    \item[\textbf{第二步（代入公式）：}] 将 $a^2=9$, $b^2=4$, $x_0=3$, $y_0=0$ 代入切线公式 $\dfrac{x_0 x}{a^2} + \dfrac{y_0 y}{b^2} = 1$。
    \[
    \frac{3x}{9} + \frac{0 \cdot y}{4} = 1
    \]
    \par\textbf{理由：} 直接应用切线方程结论。

    \item[\textbf{第三步（化简方程）：}] 化简得切线方程。
    \[
    \frac{x}{3} = 1 \quad \Rightarrow \quad x = 3
    \]
    \par\textbf{理由：} 简化方程，得到最终形式。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{切线方程为 $x = 3$，是一条垂直直线。}

\textbf{\textcolor{CExample}{例 2（稍变形）}}

\textbf{\textcolor{CExample}{题目：}} 已知椭圆 $\dfrac{x^2}{4} + \dfrac{y^2}{3} = 1$，点 $P(1, \dfrac{3}{2})$ 在椭圆上。求椭圆在点 $P$ 处的切线方程，并求该切线在 $x$ 轴和 $y$ 轴上的截距，以及截距围成的三角形面积。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（求切线方程）：}] 将 $a^2=4$, $b^2=3$, $x_0=1$, $y_0=\dfrac{3}{2}$ 代入切线公式。
    \[
    \frac{1 \cdot x}{4} + \frac{\frac{3}{2} \cdot y}{3} = 1
    \]
    \par\textbf{理由：} 直接应用切线方程结论。

    \item[\textbf{第二步（化简切线）：}] 化简方程。
    \[
    \frac{x}{4} + \frac{y}{2} = 1
    \]
    \par\textbf{理由：} $\dfrac{\frac{3}{2} y}{3} = \dfrac{3y}{6} = \dfrac{y}{2}$，所以方程为 $\dfrac{x}{4} + \dfrac{y}{2} = 1$。

    \item[\textbf{第三步（求截距）：}] 在切线方程中，令 $y=0$ 得 $x$ 截距，令 $x=0$ 得 $y$ 截距。
    \[
    \text{令 } y=0: \frac{x}{4} = 1 \Rightarrow x = 4 \quad \text{（$x$ 截距）}
    \]
    \[
    \text{令 } x=0: \frac{y}{2} = 1 \Rightarrow y = 2 \quad \text{（$y$ 截距）}
    \]
    \par\textbf{理由：} 截距是直线与坐标轴交点的坐标。

    \item[\textbf{第四步（求面积）：}] 截距围成的直角三角形面积 $S = \frac{1}{2} \times |x\text{截距}| \times |y\text{截距}|$。
    \[
    S = \frac{1}{2} \times 4 \times 2 = 4
    \]
    \par\textbf{理由：} 三角形面积公式。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{切线方程为 $\dfrac{x}{4} + \dfrac{y}{2} = 1$，$x$ 截距为 $4$，$y$ 截距为 $2$，三角形面积为 $4$。}
\end{examplebox}
