\textbf{\textcolor{CStatement}{条件}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{条件 1（椭圆上点）：}}] 点 $P(x_0, y_0)$ 在椭圆 $\dfrac{x^2}{a^2} + \dfrac{y^2}{b^2} = 1$ 上。
\end{description}

\textbf{\textcolor{CStatement}{结论}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{\textcolor{CStatement}{结论一（标准形式）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 椭圆在点 $P$ 处的切线方程。
    \[
    \frac{x_0 x}{a^2} + \frac{y_0 y}{b^2} = 1.
    \]

    \item[\textbf{\textcolor{CStatement}{推广结论（斜率表达）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 切线斜率的坐标表达式。
    \[
    k = -\frac{b^2 x_0}{a^2 y_0} \quad (y_0 \neq 0).
    \]
\end{description}
\end{statementbox}
