\begin{summarybox}
\textbf{\textcolor{CSummary}{一句话核心}}

\textbf{椭圆上任意一点的切线方程，只需将椭圆标准方程中的 $x^2$ 替换为 $x_0 x$，$y^2$ 替换为 $y_0 y$ 即可得到。}

\medskip
\textbf{\textcolor{CSummary}{使用条件}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{条件 1（点在椭圆上）：} 点 $P(x_0, y_0)$ 必须满足椭圆方程 $\dfrac{x_0^2}{a^2} + \dfrac{y_0^2}{b^2} = 1$。
    \item \textbf{条件 2（标准方程）：} 椭圆方程必须为标准形式 $\dfrac{x^2}{a^2} + \dfrac{y^2}{b^2} = 1$（$a > b > 0$）。
\end{itemize}

\medskip
\textbf{\textcolor{CSummary}{关键提醒}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{易错点（点不在椭圆）：} 使用公式前务必验证点是否在椭圆上，否则结果错误。
    \item \textbf{检查项（验证点坐标）：} 解题时先计算 $\dfrac{x_0^2}{a^2} + \dfrac{y_0^2}{b^2}$ 是否等于 $1$，确保公式适用。
\end{itemize}
\end{summarybox}
