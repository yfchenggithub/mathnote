\begin{proofbox}
\textbf{利用判别式法推导}:
\begin{enumerate}
    \item 联立直线 $y = kx + m$ 与椭圆 $b^2x^2 + a^2y^2 = a^2b^2$.
    \item 代入得: $b^2x^2 + a^2(kx + m)^2 = a^2b^2$.
    \item 展开整理为关于 $x$ 的一元二次方程: $(b^2 + a^2k^2)x^2 + 2a^2kmx + a^2(m^2 - b^2) = 0$.
    \item 直线与椭圆相切 $\iff \Delta = 0$.
    \item $\Delta = (2a^2km)^2 - 4(b^2 + a^2k^2) \cdot a^2(m^2 - b^2) = 0$.
    \item 化简得: $4a^4k^2m^2 - 4a^2(b^2m^2 - b^4 + a^2k^2m^2 - a^2k^2b^2) = 0$.
    \item 进一步消项得: $b^2m^2 = b^4 + a^2k^2b^2$.
    \item 同时除以 $b^2$, 最终得证: $m^2 = a^2k^2 + b^2$.
\end{enumerate}
\end{proofbox}
