\begin{examplebox}
\textbf{例题1（量词否定）}：写出命题"所有实数的平方都大于等于0"的否定形式。
\textit{解析}：原命题为 $\forall x\in\mathbb{R},\;x^2\ge 0$。否定为 $\exists x\in\mathbb{R},\;x^2<0$。注意原命题为真，否定为假。
\end{examplebox}

\begin{examplebox}
\textbf{例题2（联结词否定）}：写出命题"他既聪明又勤奋"的否定。
\textit{解析}：原命题为 $p\land q$（$p$：聪明，$q$：勤奋）。否定为 $\lnot p\lor\lnot q$，即"他不聪明或者不勤奋"。
\end{examplebox}

\begin{examplebox}
\textbf{例题3（数量词否定）}：命题"这个班至少有10个女生"的否定是什么？
\textit{解析}：原命题"至少有10个"即人数 $\ge 10$。否定为"至多有9个女生"，即女生人数 $\le 9$。
\end{examplebox}

\begin{examplebox}
\textbf{例题4（综合练习）}：已知命题 $p$：所有 $x\in(0,+\infty)$ 都有 $x+\frac{1}{x}\ge 2$。写出 $\lnot p$，并判断真假。
\textit{解析}：$\lnot p$：存在 $x\in(0,+\infty)$ 使得 $x+\frac{1}{x}<2$。由于原命题为真（基本不等式），故 $\lnot p$ 为假。
\end{examplebox}
