\begin{examplebox}
\textbf{例题1：} 设全集 $U = \{1,2,3,4,5,6\}$，$A = \{1,2,3\}$，$B = \{2,4,6\}$。验证德摩根公式。

\textit{解析：} 计算 $A \cap B = \{2\}$，$C_U(A \cap B) = \{1,3,4,5,6\}$。计算 $C_U A = \{4,5,6\}$，$C_U B = \{1,3,5\}$，则 $C_U A \cup C_U B = \{1,3,4,5,6\}$，相等。

计算 $A \cup B = \{1,2,3,4,6\}$，$C_U(A \cup B) = \{5\}$。$C_U A \cap C_U B = \{4,5,6\} \cap \{1,3,5\} = \{5\}$，相等。

\end{examplebox}

\begin{examplebox}
\textbf{例题2：} 证明 $A \setminus (B \cup C) = (A \setminus B) \cap (A \setminus C)$。

\textit{解析：} 由集合差定义 $A \setminus B = A \cap C_U B$，则
\[
A \setminus (B \cup C) = A \cap C_U(B \cup C) = A \cap (C_U B \cap C_U C) = (A \cap C_U B) \cap (A \cap C_U C) = (A \setminus B) \cap (A \setminus C).
\]
这里用到了德摩根公式 $C_U(B \cup C) = C_U B \cap C_U C$。
\end{examplebox}
