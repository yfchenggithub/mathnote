\begin{proofbox}
\textbf{证明} $C_U(A \cap B) = C_U A \cup C_U B$：
\begin{enumerate}
    \item 设 $x \in C_U(A \cap B)$，则 $x \notin A \cap B$，即 $x$ 不同时属于 $A$ 和 $B$。因此 $x \notin A$ 或 $x \notin B$，即 $x \in C_U A$ 或 $x \in C_U B$，故 $x \in C_U A \cup C_U B$。
    \item 反之，设 $x \in C_U A \cup C_U B$，则 $x \in C_U A$ 或 $x \in C_U B$，即 $x \notin A$ 或 $x \notin B$。于是 $x$ 不可能同时属于 $A$ 和 $B$，即 $x \notin A \cap B$，所以 $x \in C_U(A \cap B)$。
    \item 由集合相等的定义，$C_U(A \cap B) \subseteq C_U A \cup C_U B$ 且 $C_U A \cup C_U B \subseteq C_U(A \cap B)$，故等式成立。
\end{enumerate}
\end{proofbox}

\begin{proofbox}
\textbf{证明} $C_U(A \cup B) = C_U A \cap C_U B$（类似可证）：
\begin{enumerate}
    \item $x \in C_U(A \cup B) \iff x \notin A \cup B \iff x \notin A$ 且 $x \notin B \iff x \in C_U A$ 且 $x \in C_U B \iff x \in C_U A \cap C_U B$。
    \item 因此两边互为充要条件，集合相等。
\end{enumerate}
\end{proofbox}
