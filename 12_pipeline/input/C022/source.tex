\noindent
\textbf{性质3}\quad
如图1.1-1，经过椭圆
\[
\frac{x^2}{a^2}+\frac{y^2}{b^2}=1 \quad (a>b>0)
\]
长轴的两个端点 $A_1$ 和 $A_2$ 的切线，与过椭圆上任意一点 $P$ 的切线分别相交于点
$P_1$ 和 $P_2$，则有：

\begin{enumerate}
    \item $\left|P_1A_1\right|\left|P_2A_2\right|=b^2$；
    \item 四边形 $A_1P_1P_2A_2$ 面积的最小值是 $2ab$。
\end{enumerate}

\begin{figure}[h]
\centering
\begin{tikzpicture}[scale=1.0, >=Stealth]

    % 参数
    \def\SemiA{3.0}
    \def\SemiB{2.0}
    \def\T{40}

    % 椭圆上一点 P
    \pgfmathsetmacro{\Px}{\SemiA*cos(\T)}
    \pgfmathsetmacro{\Py}{\SemiB*sin(\T)}

    % 过 P 的切线与 x = -SemiA, x = SemiA 的交点纵坐标
    \pgfmathsetmacro{\Poney}{(\SemiB*\SemiB/\Py)*(1+\Px/\SemiA)}
    \pgfmathsetmacro{\Ptwoy}{(\SemiB*\SemiB/\Py)*(1-\Px/\SemiA)}

    % 坐标轴
    \draw[->] (-4.2,0) -- (4.2,0) node[right] {$x$};
    \draw[->] (0,-3.0) -- (0,4.0) node[above] {$y$};

    % 椭圆
    \draw[thick] (0,0) ellipse [x radius=\SemiA cm, y radius=\SemiB cm];

    % A1, A2
    \coordinate (A1) at (-\SemiA,0);
    \coordinate (A2) at (\SemiA,0);
    \fill (A1) circle (1.2pt);
    \fill (A2) circle (1.2pt);
    \node[below left] at (A1) {$A_1$};
    \node[below right] at (A2) {$A_2$};

    % O
    \fill (0,0) circle (1.2pt);
    \node[below left] at (0,0) {$O$};

    % 过 A1, A2 的切线（竖直）
    \draw[thick] (-\SemiA,0) -- (-\SemiA,\Poney+0.2);
    \draw[thick] (\SemiA,0) -- (\SemiA,\Ptwoy+0.2);

    % P, P1, P2
    \coordinate (P) at (\Px,\Py);
    \coordinate (P1) at (-\SemiA,\Poney);
    \coordinate (P2) at (\SemiA,\Ptwoy);

    % 过 P 的切线
    \draw[thick] (P1) -- (P2);

    % 点与标注
    \fill (P) circle (1.2pt);
    \fill (P1) circle (1.2pt);
    \fill (P2) circle (1.2pt);

    \node[above right] at (P) {$P$};
    \node[above left] at (P1) {$P_1$};
    \node[above right] at (P2) {$P_2$};

\end{tikzpicture}

\end{figure}
