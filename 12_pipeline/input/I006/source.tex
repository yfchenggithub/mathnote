\section*{1.6 权方和不等式}
设 \( a_i \in \mathbb{R} \), \( b_i > 0 \)：
\begin{enumerate}
    \item 二维形式：\( \frac{a_1^2}{b_1} + \frac{a_2^2}{b_2} \geq \frac{(a_1+a_2)^2}{b_1+b_2} \)；
    \item 一般形式：\( \sum_{i=1}^n \frac{a_i^2}{b_i} \geq \frac{(\sum_{i=1}^n a_i)^2}{\sum_{i=1}^n b_i} \)；
    \item 推广形式：\( \sum_{i=1}^n \frac{a_i^m}{b_i^{m-1}} \geq \frac{(\sum a_i)^m}{(\sum b_i)^{m-1}} \)（\( m \geq 1 \)）；
    \item 等号成立条件：当且仅当 \( \frac{a_1}{b_1} = \frac{a_2}{b_2} = \dots = \frac{a_n}{b_n} \)。
\end{enumerate}
