\begin{proofbox}
\textbf{利用导数法推导}:
\begin{enumerate}
    \item 对椭圆方程 $\frac{x^2}{a^2} + \frac{y^2}{b^2} = 1$ 两边关于 $x$ 求导.
    \item $\frac{2x}{a^2} + \frac{2y}{b^2} \cdot y' = 0$, 解得 $y' = -\frac{b^2x}{a^2y}$.
    \item 代入切点 $P(x_0, y_0)$, 得到切线斜率 $k = -\frac{b^2x_0}{a^2y_0}$.
    \item 根据点斜式方程: $y - y_0 = -\frac{b^2x_0}{a^2y_0}(x - x_0)$.
    \item 展开整理得: $b^2x_0x + a^2y_0y = b^2x_0^2 + a^2y_0^2$.
    \item 同时除以 $a^2b^2$, 得 $\frac{x_0x}{a^2} + \frac{y_0y}{b^2} = \frac{x_0^2}{a^2} + \frac{y_0^2}{b^2}$.
    \item 因为 $P$ 在椭圆上, 右端项等于 $1$, 故 $\frac{x_0x}{a^2} + \frac{y_0y}{b^2} = 1$ 得证.
\end{enumerate}
\end{proofbox}
