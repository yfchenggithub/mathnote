\begin{proofbox}
\textbf{1. 内心 $I$ 坐标公式的详细推导（核心）}

\textit{核心思路：利用角平分线定理及内心分点公式，配合双曲线焦半径公式。}

\begin{enumerate}
    \item \textbf{第一步：确定边长} (设 $P$ 在右支 $x_0 \ge a$):
    \begin{itemize}
        \item 底边 $p = |F_1F_2| = 2c$;
        \item 侧边 $f_1 = |PF_2| = ex_0 - a$;
        \item 侧边 $f_2 = |PF_1| = ex_0 + a$.
        \item 周长 $L = 2c + (ex_0 - a) + (ex_0 + a) = 2c + 2ex_0$.
    \end{itemize}
    \item \textbf{第二步：代入内心坐标公式}
    \[ x_I = \frac{p \cdot x_0 + f_1 \cdot (-c) + f_2 \cdot c}{L} \]
    \item \textbf{第三步：分子化简}
    \[ \text{分子} = 2cx_0 - c(ex_0 - a) + c(ex_0 + a) = 2cx_0 - cex_0 + ac + cex_0 + ac = 2cx_0 + 2ac \]
    \item \textbf{第四步：求值}
    \[ x_I = \frac{2c(x_0 + a)}{2(c + ex_0)} = \frac{c(x_0 + a)}{c + \frac{c}{a}x_0} = \frac{c(x_0 + a)}{\frac{c}{a}(a + x_0)} = a \]
    同理可得 $y_I = \frac{2c \cdot y_0 + 0 + 0}{2c + 2ex_0} = \frac{cy_0}{c+ex_0}$。
\end{enumerate}

\tcbline

\textbf{2. 垂心 $H$ 与外心 $O'$ 的推导简述}

\begin{itemize}
    \item \textbf{垂心}: 过 $P$ 的高线为 $x = x_0$。过 $F_1$ 的高线斜率为 $k = -\frac{1}{k_{PF_2}} = -\frac{x_0-c}{y_0}$。方程为 $y = -\frac{x_0-c}{y_0}(x+c)$。联立得 $y_H = \frac{c^2-x_0^2}{y_0}$。
    \item \textbf{外心}: 底边中垂线为 $x=0$。设外心 $(0, y_{O'})$, 利用距离相等 $O'F_2^2 = O'P^2$:
    $c^2 + y_{O'}^2 = x_0^2 + (y_0 - y_{O'})^2 \implies 2y_0y_{O'} = x_0^2 + y_0^2 - c^2$。
\end{itemize}
\end{proofbox}
