\begin{proofbox}
\textbf{参数方程推导步骤：}
\begin{enumerate}
    \item \textbf{设点}：设椭圆上任意点 $P(a\cos\theta, b\sin\theta)$。
    \item \textbf{代入公式}：点 $P$ 到直线 $Ax + By + C = 0$ 的距离为 $d = \frac{|Aa\cos\theta + Bb\sin\theta + C|}{\sqrt{A^2 + B^2}}$。
    \item \textbf{辅助角合一}：令 $Aa\cos\theta + Bb\sin\theta = \sqrt{A^2a^2 + B^2b^2}\sin(\theta + \phi)$。
    \item \textbf{求最值}：当 $\sin(\theta + \phi) = \pm 1$ 时，分子取得极值 $|C \pm \sqrt{A^2a^2 + B^2b^2}|$，由此得距离 $d$ 的最值。
\end{enumerate}
\end{proofbox}
