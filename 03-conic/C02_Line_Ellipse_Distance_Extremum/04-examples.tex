\begin{examplebox}
\textbf{例题 1：} 求椭圆 $\frac{x^2}{4} + y^2 = 1$ 上的点到直线 $x - y + 4 = 0$ 的最短距离。

\textit{解析：} 
由题意 $a=2, b=1, A=1, B=-1, C=4$。
计算项 $\sqrt{A^2a^2 + B^2b^2} = \sqrt{1^2 \cdot 2^2 + (-1)^2 \cdot 1^2} = \sqrt{5}$。
最短距离 $d_{min} = \frac{4 - \sqrt{5}}{\sqrt{1^2 + (-1)^2}} = \frac{4 - \sqrt{5}}{\sqrt{2}} = \frac{4\sqrt{2} - \sqrt{10}}{2}$。

\textbf{例题 2：} 椭圆 $\frac{x^2}{16} + \frac{y^2}{9} = 1$ 上一点 $P$ 到直线 $l: x + y - 10 = 0$ 的最大距离。

\textit{解析：}
$a=4, b=3, A=1, B=1, C=-10$。
$\sqrt{A^2a^2 + B^2b^2} = \sqrt{16 + 9} = 5$。
$d_{max} = \frac{|-10| + 5}{\sqrt{1^2 + 1^2}} = \frac{15}{\sqrt{2}} = \frac{15\sqrt{2}}{2}$。
\end{examplebox}
