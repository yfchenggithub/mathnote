\begin{proofbox}
\textbf{1. 乘积定值的证明}:
\begin{enumerate}
    \item 设 $P(x_0, y_0)$, 则过 $P$ 的切线为 $l: \frac{x_0x}{a^2} + \frac{y_0y}{b^2} = 1$.
    \item $A_1$ 处切线方程为 $x = -a$, 代入 $l$ 得 $P_1$ 纵坐标: $\frac{-ax_0}{a^2} + \frac{y_0y_1}{b^2} = 1 \implies y_1 = \frac{b^2(a+x_0)}{ay_0}$.
    \item $A_2$ 处切线方程为 $x = a$, 代入 $l$ 得 $P_2$ 纵坐标: $\frac{ax_0}{a^2} + \frac{y_0y_2}{b^2} = 1 \implies y_2 = \frac{b^2(a-x_0)}{ay_0}$.
    \item 则 $|P_1A_1| \cdot |P_2A_2| = |y_1 \cdot y_2| = \frac{b^4(a^2-x_0^2)}{a^2y_0^2}$.
    \item 由点 $P$ 在椭圆上知 $y_0^2 = \frac{b^2(a^2-x_0^2)}{a^2}$, 代入上式:
    $|y_1 y_2| = \frac{b^4(a^2-x_0^2)}{a^2 \cdot \frac{b^2(a^2-x_0^2)}{a^2}} = b^2$. 证毕.
\end{enumerate}

\textbf{2. 面积最小值的证明}:
\begin{enumerate}
    \item 四边形 $A_1P_1P_2A_2$ 为直角梯形, 面积 $S = \frac{1}{2}(|y_1| + |y_2|) \cdot 2a = a(|y_1| + |y_2|)$.
    \item 由算术-几何平均值不等式: $|y_1| + |y_2| \ge 2\sqrt{|y_1 y_2|}$.
    \item 代入定值 $|y_1 y_2| = b^2$, 得 $S \ge a \cdot 2\sqrt{b^2} = 2ab$.
    \item 当且仅当 $|y_1| = |y_2| = b$ 时取等号, 此时 $P$ 为短轴顶点 $(0, \pm b)$.
\end{enumerate}
\end{proofbox}
