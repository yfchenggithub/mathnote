\begin{proofbox}
\textbf{1. 焦点距离积 $d_1 d_2 = b^2$ 的详细代数证明}:
\begin{enumerate}
    \item 设椭圆方程为 $\frac{x^2}{a^2} + \frac{y^2}{b^2} = 1$, 切线 $l$ 的斜率为 $k$. 则切线方程可设为 $y = kx + m$.
    \item 根据直线与椭圆相切的判别式 $\Delta = 0$, 得到切线背景下的定值关系: $m^2 = a^2k^2 + b^2$.
    \item 左右焦点分别为 $F_1(-c, 0), F_2(c, 0)$. 利用点到直线距离公式 $d = \frac{|kx - y + m|}{\sqrt{k^2+1}}$:
    \begin{itemize}
        \item $d_1 = \frac{|-kc - 0 + m|}{\sqrt{k^2+1}} = \frac{|m - kc|}{\sqrt{k^2+1}}$;
        \item $d_2 = \frac{|kc - 0 + m|}{\sqrt{k^2+1}} = \frac{|m + kc|}{\sqrt{k^2+1}}$.
    \end{itemize}
    \item 计算乘积 $d_1 \cdot d_2 = \frac{|(m - kc)(m + kc)|}{k^2+1} = \frac{|m^2 - k^2c^2|}{k^2+1}$.
    \item 代入 $m^2 = a^2k^2 + b^2$ 和 $c^2 = a^2 - b^2$:
    $d_1 d_2 = \frac{a^2k^2 + b^2 - k^2(a^2 - b^2)}{k^2+1} = \frac{a^2k^2 + b^2 - a^2k^2 + b^2k^2}{k^2+1} = \frac{b^2(1 + k^2)}{k^2+1} = b^2$. 证毕.
\end{enumerate}

\textbf{2. 垂足 $H_1, H_2$ 在主圆 $x^2 + y^2 = a^2$ 上的证明}:
\begin{enumerate}
    \item 以 $F_2(c, 0)$ 为例, 过 $F_2$ 且垂直于切线 $y = kx + m$ 的直线 (垂线) 方程为: $y = -\frac{1}{k}(x - c)$, 即 $x + ky = c$.
    \item 切线方程整理为: $kx - y = -m$.
    \item 将两方程分别平方后相加: $(x + ky)^2 + (kx - y)^2 = c^2 + (-m)^2$.
    \item 左边展开: $x^2 + 2kxy + k^2y^2 + k^2x^2 - 2kxy + y^2 = (1 + k^2)(x^2 + y^2)$.
    \item 右边代入 $m^2$: $c^2 + a^2k^2 + b^2 = (a^2 - b^2) + a^2k^2 + b^2 = a^2(1 + k^2)$.
    \item 约去 $(1 + k^2)$ 得 $x^2 + y^2 = a^2$. 垂足 $H_1, H_2$ 均在圆上. 证毕.
\end{enumerate}

\textbf{3. 光学性质 (切线平分 $\angle F_1PF_2$ 外角) 的几何证明}:
\begin{enumerate}
    \item 延长 $F_1P$ 到 $F_1'$, 使得 $PF_1' = PF_1$. 由椭圆定义 $PF_1 + PF_2 = 2a$, 故 $F_2F_1' = 2a$.
    \item 根据垂足性质, $F_1$ 关于切线 $l$ 的对称点 $Q$ 满足 $F_2, P, Q$ 三点共线 (这是反射定律的几何表达).
    \item 由于 $H_1$ 是 $F_1$ 在切线上的射影且 $H_1$ 在主圆上, 利用中位线定理可证 $P$ 处法线即为 $\angle F_1PF_2$ 的角平分线.
    \item 因此, 切线 $l$ 作为法线的垂线, 必为该角的内角平分线的垂线, 即外角平分线. 证毕.
\end{enumerate}
\end{proofbox}
