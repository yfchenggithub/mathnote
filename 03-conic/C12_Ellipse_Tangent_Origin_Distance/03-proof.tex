\begin{proofbox}
\textbf{利用坐标解析法证明}:
\begin{enumerate}
    \item 设 $P(x_0, y_0)$, 切线 $l: \frac{x_0x}{a^2} + \frac{y_0y}{b^2} = 1$. 原点到 $l$ 的距离 $d = \frac{1}{\sqrt{\frac{x_0^2}{a^4} + \frac{y_0^2}{b^4}}}$.
    \item 整理得 $d^2 = \frac{a^4b^4}{b^4x_0^2 + a^4y_0^2}$. 
    \item 焦点坐标 $F(\pm c, 0)$, 焦半径公式为 $r_1 = a + ex_0, r_2 = a - ex_0$.
    \item 乘积 $r_1r_2 = a^2 - e^2x_0^2 = a^2 - \frac{c^2}{a^2}x_0^2 = \frac{a^4 - (a^2-b^2)x_0^2}{a^2}$.
    \item 利用 $y_0^2 = b^2(1 - \frac{x_0^2}{a^2})$, 代入 $d^2$ 分母:
    $b^4x_0^2 + a^4 \cdot b^2(1 - \frac{x_0^2}{a^2}) = b^4x_0^2 + a^2b^2(a^2 - x_0^2) = b^2(b^2x_0^2 + a^4 - a^2x_0^2)$.
    \item 进一步化简分母为 $b^2[a^4 - (a^2-b^2)x_0^2]$.
    \item 观察发现 $d^2 = \frac{a^4b^4}{b^2(a^2 \cdot r_1r_2)} = \frac{a^2b^2}{r_1r_2}$.
    \item 两边开方得 $d = \frac{ab}{\sqrt{r_1r_2}}$, 即 $\sqrt{r_1r_2} \cdot d = ab$. 证毕.
\end{enumerate}
\end{proofbox}
