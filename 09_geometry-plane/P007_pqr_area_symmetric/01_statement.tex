\begin{statementbox}

\textbf{\textcolor{CStatement}{条件}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{条件1（三边平方组）：}}] 设三角形三边长分别为 $x,y,z$. 定义三个数 $P,Q,R$ 满足
    \[
    \begin{cases}
    P+Q = x^2,\\
    Q+R = y^2,\\
    R+P = z^2.
    \end{cases}
    \]
\end{description}

\textbf{\textcolor{CStatement}{结论}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{\textcolor{CStatement}{结论一（面积对称式）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 三角形面积可用三边平方的交叉乘积和表示.
    \[
    2S = \sqrt{PQ + QR + RP}.
    \]

    \item[\textbf{\textcolor{CStatement}{常见变形（中间量显式）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 将 $P,Q,R$ 用边长平方直接表示.
    \[
    P = \frac{x^2+z^2-y^2}{2},\quad
    Q = \frac{x^2+y^2-z^2}{2},\quad
    R = \frac{y^2+z^2-x^2}{2}.
    \]
\end{description}

% 非不等式类结论，补充适用范围
\textbf{\textcolor{CStatement}{适用范围}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{适用场景（根式边长）：}}] 当三边为 $\sqrt{a},\sqrt{b},\sqrt{c}$ 时，直接取 $x^2=a,y^2=b,z^2=c$，避免计算半周长时的二次根式嵌套.
\end{description}

\end{statementbox}
