\begin{proofbox}

\textbf{\textcolor{CProof}{思路提示}}

由方程组解出 $P,Q,R$ 用 $x,y,z$ 表示的表达式，再计算 $PQ+QR+RP$，化简后与海伦公式的平方形式 $16S^2$ 对比，证明二者相等.

\medskip
\textbf{\textcolor{CProof}{正式推导}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.55em]
    \item[\textbf{步骤一（解出辅助量）：}] 将方程组三式相加得 $2(P+Q+R)=x^2+y^2+z^2$，故
    \[
    P+Q+R = \frac{x^2+y^2+z^2}{2}.
    \]
    分别减去每式得
    \[
    P = \frac{x^2+z^2-y^2}{2},\quad
    Q = \frac{x^2+y^2-z^2}{2},\quad
    R = \frac{y^2+z^2-x^2}{2}.
    \]
    \par\textbf{理由：} 这是线性方程组的基本解法，利用整体消元.

    \item[\textbf{步骤二（计算乘积和）：}] 计算 $4(PQ+QR+RP)$ 以消去分母.
    \[
    \begin{aligned}
    4PQ &= (x^2+z^2-y^2)(x^2+y^2-z^2),\\
    4QR &= (x^2+y^2-z^2)(y^2+z^2-x^2),\\
    4RP &= (y^2+z^2-x^2)(x^2+z^2-y^2).
    \end{aligned}
    \]
    三式相加得
    \[
    4(PQ+QR+RP) = \sum_{\text{cyc}} (x^2+z^2-y^2)(x^2+y^2-z^2).
    \]
    \par\textbf{理由：} 按对称性列出三个乘积，统一形式.

    \item[\textbf{步骤三（化简代数和）：}] 展开上式，分组整理.
    \[
    \begin{aligned}
    4(PQ+QR+RP)
    &= (x^2+z^2-y^2)(x^2+y^2-z^2) \\
    &\quad + (x^2+y^2-z^2)(y^2+z^2-x^2) \\
    &\quad + (y^2+z^2-x^2)(x^2+z^2-y^2) \\
    &= (x^2+y^2+z^2)^2 - 2(x^4+y^4+z^4).
    \end{aligned}
    \]
    (直接展开合并即得)
    \par\textbf{理由：} 通过观察对称性，合并后得到简洁表达式.

    \item[\textbf{步骤四（对比海伦公式）：}] 已知海伦公式 $16S^2 = 2(x^2y^2+y^2z^2+z^2x^2) - (x^4+y^4+z^4)$. 而刚得到
    \[
    4(PQ+QR+RP) = (x^2+y^2+z^2)^2 - 2(x^4+y^4+z^4).
    \]
    展开右边：
    \[
    (x^2+y^2+z^2)^2 = x^4+y^4+z^4 + 2(x^2y^2+y^2z^2+z^2x^2).
    \]
    代入得
    \[
    \begin{aligned}
    4(PQ+QR+RP)
    &= (x^4+y^4+z^4) + 2(x^2y^2+y^2z^2+z^2x^2) \\
    &\quad - 2(x^4+y^4+z^4) \\
    &= 2(x^2y^2+y^2z^2+z^2x^2) - (x^4+y^4+z^4).
    \end{aligned}
    \]
    这正是 $16S^2$，故 $PQ+QR+RP = 4S^2$，即 $2S = \sqrt{PQ+QR+RP}$.
    \par\textbf{理由：} 等量代换，直接比较得到面积表达式.
\end{description}

\medskip
\textbf{\textcolor{CProof}{结论回扣}}

\textbf{因此，由三边平方构造的 $P,Q,R$ 满足 $2S = \sqrt{PQ+QR+RP}$，与海伦公式完全一致.}

\end{proofbox}
