\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CTrap}{易错点一（辅助量混淆）}}] 误把 $P,Q,R$ 当成边长或半周长，不清楚方程组中每一项对应哪个平方.
    \item[\textbf{\textcolor{CTrap}{正确理解（有序对应）：}}] 严格按照 $P+Q=x^2,\;Q+R=y^2,\;R+P=z^2$ 的顺序，不可互换.
    \item[\textbf{\textcolor{CTrap}{错因分析（记忆不清）：}}] 方程组的形式容易记反，建议记住“循环和”结构：$P$ 缺 $y^2$，$Q$ 缺 $z^2$，$R$ 缺 $x^2$ 或通过解表达式记忆.
\end{description}

\medskip
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CTrap}{易错点二（公式因子遗漏）}}] 直接写 $S = \sqrt{PQ+QR+RP}$，忘记公式中是 $2S$.
    \item[\textbf{\textcolor{CTrap}{正确理解（强调因子2）：}}] 公式为 $2S = \sqrt{PQ+QR+RP}$，因此 $S = \dfrac{\sqrt{PQ+QR+RP}}{2}$.
    \item[\textbf{\textcolor{CTrap}{错因分析（粗心复制）：}}] 将海伦公式中 $S = \sqrt{s(s-a)(s-b)(s-c)}$ 的系数混淆，未注意本公式多了一个因子2.
\end{description}

\medskip
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CTrap}{易错点三（条件误判）}}] 认为该公式只适用于三边平方和为整数或边长特殊的三角形.
    \item[\textbf{\textcolor{CTrap}{正确理解（普遍适用）：}}] 对任意正边长 $x,y,z$ 满足三角形不等式，公式均成立.
    \item[\textbf{\textcolor{CTrap}{错因分析（思维定势）：}}] 因例题中多选用整数或简单根式，误以为有条件限制；实际上推导与海伦公式等价，无额外限制.
\end{description}
