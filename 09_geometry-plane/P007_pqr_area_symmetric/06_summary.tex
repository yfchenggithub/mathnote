\begin{summarybox}

\textbf{\textcolor{CSummary}{一句话核心}}

\textbf{三角形面积 $S = \dfrac12\sqrt{PQ+QR+RP}$，其中 $P+Q=x^2,\;Q+R=y^2,\;R+P=z^2$，$x,y,z$ 为三边长.}

\medskip
\textbf{\textcolor{CSummary}{使用条件}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{条件1（三边已知）：} 需要知道三角形的三边长度 $x,y,z$，不需角度.
    \item \textbf{条件2（解方程组）：} 必须准确解出 $P,Q,R$，可使用对称表达式 $P = (x^2+z^2-y^2)/2$ 等.
\end{itemize}

\medskip
\textbf{\textcolor{CSummary}{关键提醒}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{易错点（公式因子）：} 牢记 $2S = \sqrt{\cdots}$，不要遗漏因子2.
    \item \textbf{检查项（计算顺序）：} 核对 $P+Q$ 是否等于 $x^2$，确保方程组正确.
\end{itemize}

\end{summarybox}
