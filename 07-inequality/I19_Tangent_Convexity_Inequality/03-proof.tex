\begin{proofbox}
\textbf{二阶导数证明法：}
\begin{enumerate}
    \item 设 $f(x) = \tan x$ ，$x \in (0, \frac{\pi}{2})$ ；
    \item 一阶导数：$f'(x) = \sec^2 x = \frac{1}{\cos^2 x}$ ；
    \item 二阶导数：$f''(x) = 2\sec x \cdot (\sec x \tan x) = 2\sec^2 x \tan x$ ；
    \item 在 $(0, \frac{\pi}{2})$ 上，$\sec^2 x > 0$ 且 $\tan x > 0$，故 $f''(x) > 0$ ；
    \item 由琴生不等式，下凸函数满足 $\frac{f(x)+f(y)}{2} \geqslant f(\frac{x+y}{2})$ ；
    \item 代入得 $\frac{\tan x + \tan y}{2} \geqslant \tan(\frac{x+y}{2})$。
\end{enumerate}
\end{proofbox}
