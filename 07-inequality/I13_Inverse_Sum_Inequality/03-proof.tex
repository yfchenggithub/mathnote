\begin{proofbox}
\textbf{方法一：配方法（代数法）}
\begin{enumerate}
    \item 考虑 $x + \frac{1}{x} - 2$ ；
    \item 变形为 $(\sqrt{x})^2 - 2 \cdot \sqrt{x} \cdot \frac{1}{\sqrt{x}} + (\frac{1}{\sqrt{x}})^2$ ；
    \item 利用完全平方公式得：$(\sqrt{x} - \frac{1}{\sqrt{x}})^2$ ；
    \item 因为实数的平方恒 $\geqslant 0$，故 $x + \frac{1}{x} - 2 \geqslant 0$ ，即 $x + \frac{1}{x} \geqslant 2$。
\end{enumerate}

\textbf{方法二：判别式法}
设 $x + \frac{1}{x} = k$，整理得 $x^2 - kx + 1 = 0$。
因为 $x$ 是实数，所以 $\Delta = k^2 - 4 \geqslant 0$，解得 $k \geqslant 2$ 或 $k \leqslant -2$。
\end{proofbox}
