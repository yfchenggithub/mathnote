\begin{proofbox}
\textbf{导数法证明：}
\begin{enumerate}
    \item 构造辅助函数 $f(x) = x - 1 - \ln x$，其定义域为 $(0, +\infty)$ ；
    \item 对 $f(x)$ 求导：$f'(x) = 1 - \frac{1}{x} = \frac{x-1}{x}$ ；
    \item 分析导数符号：
    \begin{itemize}
        \item 当 $0 < x < 1$ 时，$f'(x) < 0$，$f(x)$ 单调递减；
        \item 当 $x > 1$ 时，$f'(x) > 0$，$f(x)$ 单调递增；
    \end{itemize}
    \item 故 $f(x)$ 在 $x = 1$ 处取得极小值，也是最小值 ；
    \item $f(x)_{min} = f(1) = 1 - 1 - \ln 1 = 0$ ；
    \item 因此 $f(x) \geqslant 0$，即 $x - 1 \geqslant \ln x$ 恒成立。
\end{enumerate}
\end{proofbox}
