\begin{proofbox}
\textbf{二阶导数判定法：}
\begin{enumerate}
    \item 构造函数 $f(t) = t^n$ ($t > 0, n > 1$) ；
    \item 求导得 $f'(t) = nt^{n-1}$，二阶导得 $f''(t) = n(n-1)t^{n-2}$ ；
    \item 因为 $n > 1$ 且 $t > 0$，所以 $f''(t) > 0$，函数为下凸函数 ；
    \item 根据琴生不等式 $\frac{f(x)+f(y)}{2} \geqslant f\left(\frac{x+y}{2}\right)$ ；
    \item 代入 $f(t) = t^n$ 即证得：$\frac{x^n + y^n}{2} \geqslant \left(\frac{x+y}{2}\right)^n$。
\end{enumerate}
\end{proofbox}
