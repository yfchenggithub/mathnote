\begin{proofbox}
\textbf{方法一：和差化积法}
\begin{enumerate}
    \item 考虑差值：$\sin\left(\frac{x+y}{2}\right) - \frac{\sin x + \sin y}{2}$ ；
    \item 利用和差化积公式：$\sin x + \sin y = 2\sin\frac{x+y}{2}\cos\frac{x-y}{2}$ ；
    \item 原式 $= \sin\frac{x+y}{2} - \sin\frac{x+y}{2}\cos\frac{x-y}{2} = \sin\frac{x+y}{2} \left(1 - \cos\frac{x-y}{2}\right)$ ；
    \item 因为 $x, y \in (0, \pi)$，故 $\frac{x+y}{2} \in (0, \pi)$，$\sin\frac{x+y}{2} > 0$ ；
    \item 又因为 $\cos\frac{x-y}{2} \leqslant 1$，故 $1 - \cos\frac{x-y}{2} \geqslant 0$ ；
    \item 乘积 $\geqslant 0$，结论得证。
\end{enumerate}
\end{proofbox}
