\begin{proofbox}
\textbf{利用因式分解公式证明：}
\begin{enumerate}
    \item 回顾恒等式：$a^3+b^3+c^3-3abc = (a+b+c)(a^2+b^2+c^2-ab-bc-ca)$ ；
    \item 令 $a = \sqrt[3]{x}, b = \sqrt[3]{y}, c = \sqrt[3]{z}$ ；
    \item 则 $x+y+z - 3\sqrt[3]{xyz} = (\sqrt[3]{x}+\sqrt[3]{y}+\sqrt[3]{z}) \cdot \frac{1}{2}[(\sqrt[3]{x}-\sqrt[3]{y})^2 + (\sqrt[3]{y}-\sqrt[3]{z})^2 + (\sqrt[3]{z}-\sqrt[3]{x})^2]$ ；
    \item 由于 $x,y,z \geqslant 0$，右侧各项均为非负，故 $x+y+z - 3\sqrt[3]{xyz} \geqslant 0$ ；
    \item 整理得：$\frac{x+y+z}{3} \geqslant \sqrt[3]{xyz}$。
\end{enumerate}
\end{proofbox}
