\begin{proofbox}
\textbf{因式分解法证明：}
\begin{enumerate}
    \item 利用恒等式：$a^3+b^3+c^3-3abc = (a+b+c)(a^2+b^2+c^2-ab-bc-ca)$ ；
    \item 对后半部分配方：$a^2+b^2+c^2-ab-bc-ca = \frac{1}{2}[(a-b)^2 + (b-c)^2 + (c-a)^2]$ ；
    \item 故：$a^3+b^3+c^3-3abc = \frac{1}{2}(a+b+c)[(a-b)^2 + (b-c)^2 + (c-a)^2]$ ；
    \item 因为 $a,b,c \geqslant 0$，所以 $a+b+c \geqslant 0$，且平方项之和恒 $\geqslant 0$；
    \item 所以 $a^3+b^3+c^3-3abc \geqslant 0$，即 $a^3+b^3+c^3 \geqslant 3abc$。
\end{enumerate}
\end{proofbox}
