\begin{proofbox}
\textbf{作差法证明：}
\begin{enumerate}
    \item 考虑 $\frac{a+m}{b+m} - \frac{a}{b}$；
    \item 通分得：$\frac{b(a+m) - a(b+m)}{b(b+m)}$；
    \item 展开分子：$\frac{ab + bm - ab - am}{b(b+m)} = \frac{m(b-a)}{b(b+m)}$；
    \item 因为 $a, b, m > 0$ 且 $b > a$，所以 $m(b-a) > 0$ 且分母 $b(b+m) > 0$；
    \item 故 $\frac{a+m}{b+m} - \frac{a}{b} > 0$，即 $\frac{a}{b} < \frac{a+m}{b+m}$ 成立。
\end{enumerate}
\end{proofbox}
