\begin{proofbox}
\textbf{代数证明（利用均值不等式）：}
\begin{enumerate}
    \item 观察左式：$\frac{a^x + a^y}{2}$，右式：$a^{\frac{x+y}{2}} = \sqrt{a^{x+y}} = \sqrt{a^x \cdot a^y}$ ；
    \item 令 $X = a^x$，$Y = a^y$。由于 $a > 0$，故 $X, Y > 0$ ；
    \item 根据算术-几何平均值不等式（AM-GM）：$\frac{X+Y}{2} \geqslant \sqrt{XY}$ ；
    \item 将 $X, Y$ 还原：$\frac{a^x + a^y}{2} \geqslant \sqrt{a^x \cdot a^y} = a^{\frac{x+y}{2}}$ ；
    \item 证毕。
\end{enumerate}
\end{proofbox}
