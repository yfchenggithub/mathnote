\begin{summarybox}
\textbf{【结论小结 · 对数二阶放缩】}
\vspace{0.8em}
\hrule
\vspace{0.8em}

\begin{description}
    \item[核心不等式组] $x - \frac{x^2}{2} < \ln(1+x) < x \quad (x > 0)$
    \item[高阶修正] $\ln(1+x) < x - \frac{x^2}{2} + \frac{x^3}{3}$
    \item[实战建议] 记住"左边减二次，右边不减"。如果遇到 $e^x$ 与 $\ln x$ 的复合，注意两者的对偶放缩关系。
\end{description}
\end{summarybox}
