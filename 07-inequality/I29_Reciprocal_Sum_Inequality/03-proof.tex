\begin{proofbox}
\textbf{方法一：完全平方法（最本质证明）}
\begin{enumerate}
    \item 考虑差值：$\frac{a}{b} + \frac{b}{a} - 2 = \frac{a^2 + b^2 - 2ab}{ab} = \frac{(a-b)^2}{ab}$
    \item 因为 $(a-b)^2 \geqslant 0$ 且已知 $ab > 0$
    \item 故 $\frac{(a-b)^2}{ab} \geqslant 0$，即 $\frac{a}{b} + \frac{b}{a} \geqslant 2$
\end{enumerate}

\textbf{方法二：基本不等式法 (AM-GM)}
\begin{enumerate}
    \item 因为 $ab > 0$，所以 $\frac{a}{b} > 0$ 且 $\frac{b}{a} > 0$
    \item 直接应用均值不等式：$\frac{a}{b} + \frac{b}{a} \geqslant 2\sqrt{\frac{a}{b} \cdot \frac{b}{a}} = 2\sqrt{1} = 2$
\end{enumerate}
\end{proofbox}
