\begin{proofbox}
\textbf{二阶导数判定法：}
\begin{enumerate}
    \item 构造辅助函数 $f(t) = t\ln t$，其定义域为 $(0, +\infty)$ ；
    \item 一阶求导：$f'(t) = \ln t + t \cdot \frac{1}{t} = \ln t + 1$ ；
    \item 二阶求导：$f''(t) = \frac{1}{t}$ ；
    \item 因为 $t > 0$，所以 $f''(t) > 0$ 恒成立，表明 $f(t)$ 在其定义域内为严格下凸函数 ；
    \item 由琴生不等式 $\frac{f(x)+f(y)}{2} \geqslant f\left(\frac{x+y}{2}\right)$ 直接代入：
    \item $\frac{x\ln x + y\ln y}{2} \geqslant \frac{x+y}{2}\ln\left(\frac{x+y}{2}\right)$ ；
    \item 证毕。
\end{enumerate}
\end{proofbox}
