\begin{proofbox}
\textbf{方法一：导数法（最通用证明）}
\begin{enumerate}
    \item 构造辅助函数 $f(x) = \tan x - x$，定义域 $x \in [0, \frac{\pi}{2})$
    \item 对 $f(x)$ 求导：$f'(x) = (\tan x)' - (x)' = \frac{1}{\cos^2 x} - 1$
    \item 通分整理得：$f'(x) = \frac{1 - \cos^2 x}{\cos^2 x} = \frac{\sin^2 x}{\cos^2 x} = \tan^2 x$
    \item 在 $x \in [0, \frac{\pi}{2})$ 时，$\tan^2 x \geqslant 0$ 恒成立，且仅在 $x=0$ 处导数为 $0$
    \item 因此 $f(x)$ 在该区间上单调递增
    \item 结合初值 $f(0) = \tan 0 - 0 = 0$，得出 $f(x) \geqslant f(0) = 0$
    \item 即 $\tan x \geqslant x$，证毕。
\end{enumerate}
\end{proofbox}
