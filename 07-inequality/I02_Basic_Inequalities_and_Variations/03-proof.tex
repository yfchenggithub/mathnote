\begin{proofbox}
\textbf{代数证明（作差法）：}
\begin{enumerate}
    \item 考虑 $\frac{a+b}{2} - \sqrt{ab}$；
    \item 通分得 $\frac{a+b-2\sqrt{ab}}{2} = \frac{(\sqrt{a}-\sqrt{b})^2}{2}$；
    \item 因为 $(\sqrt{a}-\sqrt{b})^2 \geqslant 0$，所以 $\frac{a+b}{2} \geqslant \sqrt{ab}$ 恒成立；
    \item 当且仅当 $\sqrt{a}-\sqrt{b}=0$ 即 $a=b$ 时，等号成立。
\end{enumerate}
\end{proofbox}
