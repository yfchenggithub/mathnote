\begin{proofbox}
\textbf{导数法证明：}
\begin{enumerate}
    \item 构造辅助函数 $f(x) = e^x - x - 1$ ，定义域为 $\mathbb{R}$ ；
    \item 求导：$f'(x) = e^x - 1$ ；
    \item 分析导数符号：
    \begin{itemize}
        \item 当 $x < 0$ 时，$e^x < 1$，则 $f'(x) < 0$，$f(x)$ 单调递减；
        \item 当 $x > 0$ 时，$e^x > 1$，则 $f'(x) > 0$，$f(x)$ 单调递增；
    \end{itemize}
    \item 故 $f(x)$ 在 $x = 0$ 处取得最小值 ；
    \item $f(x)_{min} = f(0) = e^0 - 0 - 1 = 0$ ；
    \item 因此 $f(x) \geqslant 0$ 恒成立，即 $e^x \geqslant x + 1$。
\end{enumerate}
\end{proofbox}
