\begin{proofbox}
\textbf{方法一：和差化积法}
\begin{enumerate}
    \item 考虑差值：$\cos\left(\frac{x+y}{2}\right) - \frac{\cos x + \cos y}{2}$ ；
    \item 利用和差化积公式：$\cos x + \cos y = 2\cos\frac{x+y}{2}\cos\frac{x-y}{2}$ ；
    \item 提取公因式：$\cos\frac{x+y}{2} \left(1 - \cos\frac{x-y}{2}\right)$ ；
    \item 因为 $x, y \in (0, \frac{\pi}{2})$，故 $\frac{x+y}{2} \in (0, \frac{\pi}{2})$，$\cos\frac{x+y}{2} > 0$ ；
    \item 又因为 $\cos\frac{x-y}{2} \leqslant 1$，故差值 $\geqslant 0$ ；
    \item 结论得证。
\end{enumerate}
\end{proofbox}
