\begin{proofbox}
\textbf{方法一：通分比较法}
\begin{enumerate}
    \item 左边通分：$\frac{1}{x} + \frac{1}{y} = \frac{x+y}{xy}$ ；
    \item 考虑差值：$\frac{x+y}{xy} - \frac{4}{x+y} = \frac{(x+y)^2 - 4xy}{xy(x+y)}$ ；
    \item 分子展开：$x^2 + 2xy + y^2 - 4xy = x^2 - 2xy + y^2 = (x-y)^2$ ；
    \item 因为 $(x-y)^2 \geqslant 0$ 且 $x, y > 0$，故差值 $\geqslant 0$。
\end{enumerate}

\textbf{方法二：柯西不等式法}
根据 $\left(\frac{1}{x} + \frac{1}{y}\right)(x + y) \geqslant (1 \cdot 1 + 1 \cdot 1)^2 = 4$，
两边除以 $(x+y)$ 即得 $\frac{1}{x} + \frac{1}{y} \geqslant \frac{4}{x+y}$。
\end{proofbox}
