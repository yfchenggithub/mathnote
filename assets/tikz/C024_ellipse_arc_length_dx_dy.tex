\begin{tikzpicture}[scale=1.05]

	%========================
	% 参数
	%========================
	\def\ea{4.8}
	\def\eb{2.3}
	\def\angP{70}
	\def\angQ{35}

	%========================
	% 计算点坐标
	%========================
	\pgfmathsetmacro{\Px}{\ea*cos(\angP)}
	\pgfmathsetmacro{\Py}{\eb*sin(\angP)}

	\pgfmathsetmacro{\Qx}{\ea*cos(\angQ)}
	\pgfmathsetmacro{\Qy}{\eb*sin(\angQ)}

	% R 点：从 P 竖直向下，从 Q 水平向左，相交于 R
	\pgfmathsetmacro{\Rx}{\Px}
	\pgfmathsetmacro{\Ry}{\Qy}

	%========================
	% 坐标轴
	%========================
	\draw[->, line width=0.8pt] (-5.3,0) -- (6.3,0);
	\draw[->, line width=0.8pt] (0,-3.0) -- (0,3.9);

	\node at (6.45,-0.12) {$x$};
	\node at (-0.12,4.05) {$y$};
	\node at (-0.28,-0.25) {$O$};

	%========================
	% 椭圆
	%========================
	\draw[blue, line width=0.9pt] (0,0) ellipse ({\ea} and {\eb});

	%========================
	% 椭圆弧 dl
	%========================
	\draw[
		red,
		line width=0.9pt,
		smooth,
		samples=80,
		domain=\angQ:\angP
	]
	plot ({\ea*cos(\x)}, {\eb*sin(\x)});

	%========================
	% P Q 两点之间的直线弦
	%========================
	\draw[red, line width=0.9pt] (\Px,\Py) -- (\Qx,\Qy);

	%========================
	% dy 与 dx
	% dy：P 到 R
	% dx：Q 水平向左到 R
	%========================
	\draw[red, dashed, line width=0.8pt] (\Px,\Py) -- (\Rx,\Ry);
	\draw[red, dashed, line width=0.8pt] (\Qx,\Qy) -- (\Rx,\Ry);

	%========================
	% R 点直角标记
	%========================
	\def\rs{0.16}
	\draw[red, line width=0.8pt]
	(\Rx,\Ry+\rs) -- (\Rx+\rs,\Ry+\rs) -- (\Rx+\rs,\Ry);

	%========================
	% 点 P, Q
	%========================
	\fill (\Px,\Py) circle (1.8pt);
	\fill (\Qx,\Qy) circle (1.8pt);

	%========================
	% 标签
	%========================
	\node at ({\Px-0.15},{\Py+0.35}) {$P(x,y)$};

	\node at ({\Qx+1.55},{\Qy+0.10}) {$Q(x+dx,y+dy)$};

	\node at ({\Rx-0.18},{\Ry-0.28}) {$R$};

	\node[red] at ({\Px-0.35},{(\Py+\Ry)/2}) {$dy$};

	\node[red] at ({(\Rx+\Qx)/2},{\Ry-0.27}) {$dx$};

	\node[red] at (2.95,2.05) {$dl$};

	%========================
	% 公式
	%========================
	\node at (4.55,3.15) {$\left(dl\right)^2=\left(dx\right)^2+\left(dy\right)^2$};

\end{tikzpicture}
