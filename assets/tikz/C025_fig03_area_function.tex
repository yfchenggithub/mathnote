\documentclass[tikz,border=6pt]{standalone}
\usepackage[UTF8]{ctex}
\usepackage{amsmath}
\usetikzlibrary{arrows.meta,calc,patterns}
\definecolor{mainblue}{RGB}{36,92,130}
\definecolor{softblue}{RGB}{226,241,250}
\definecolor{accent}{RGB}{202,87,48}
\definecolor{greenish}{RGB}{46,125,95}
\begin{document}
\begin{tikzpicture}[>=Stealth,line cap=round,line join=round,font=\small]
  \def\a{2.55}
  \def\b{1.18}

  % left panel: radius from ellipse
  \begin{scope}[shift={(0,0)}]
    \node[font=\bfseries\small,mainblue] at (0,2.1) {从椭圆读出截面半径};
    \draw[-{Stealth[length=2mm]},thick] (-3.0,0)--(3.15,0) node[right] {$x$};
    \draw[-{Stealth[length=2mm]},thick] (0,-1.55)--(0,1.65) node[above] {$y$};
    \fill[softblue] (0,0) ellipse [x radius=\a,y radius=\b];
    \draw[mainblue,thick] (0,0) ellipse [x radius=\a,y radius=\b];
    \draw[mainblue,thick,domain=-\a:\a,samples=100] plot (\x,{\b*sqrt(1-(\x*\x)/(\a*\a))});
    \pgfmathsetmacro{\tx}{1.25}
    \pgfmathsetmacro{\ty}{\b*sqrt(1-(\tx*\tx)/(\a*\a))}
    \draw[dashed] (\tx,0)--(\tx,\ty)--(0,\ty);
    \fill[accent] (\tx,\ty) circle (1.7pt);
    \draw[accent,thick,->] (\tx,0)--(\tx,\ty) node[midway,right] {$y$};
    \node[below] at (\tx,0) {$x$};
    \node[below] at (0,-1.8) {$\displaystyle y^2=b^2\left(1-\frac{x^2}{a^2}\right)$};
  \end{scope}

  % arrow
  \draw[-{Stealth[length=3mm]},very thick,accent] (3.45,0)--(4.65,0);
  \node[accent,align=center] at (4.05,.48) {平方\\乘 $\pi$};

  % right panel: area function
  \begin{scope}[shift={(7.3,0)}]
    \node[font=\bfseries\small,mainblue] at (0,2.1) {面积函数：开口向下的抛物线};
    \draw[-{Stealth[length=2mm]},thick] (-3.0,0)--(3.15,0) node[right] {$x$};
    \draw[-{Stealth[length=2mm]},thick] (0,-.15)--(0,1.70);\node[right] at (0,1.62) {$S(x)$};
    \draw[greenish,thick,domain=-2.55:2.55,samples=100] plot (\x,{1.45*(1-(\x*\x)/(2.55*2.55))});
    \fill[greenish!18,domain=-2.55:2.55,samples=100] (-2.55,0) -- plot (\x,{1.45*(1-(\x*\x)/(2.55*2.55))}) -- (2.55,0) -- cycle;
    \draw[greenish,thick,domain=-2.55:2.55,samples=100] plot (\x,{1.45*(1-(\x*\x)/(2.55*2.55))});
    \node[below] at (-2.55,0) {$-a$};
    \node[below] at (2.55,0) {$a$};
    \node[left] at (0,1.45) {$\pi b^2$};
    \draw[dashed] (-2.55,1.45)--(0,1.45);
    \node[align=center,greenish] at (0,-1.03) {$\displaystyle S(x)=\pi b^2\left(1-\frac{x^2}{a^2}\right)$\\[2pt]
    $\displaystyle V=\int_{-a}^{a}S(x)\,dx$};
  \end{scope}
\end{tikzpicture}
