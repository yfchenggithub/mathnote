\textbf{\textcolor{CStatement}{条件}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{条件 1（任意角）：}}] 对任意角$\alpha,\beta$，等式恒成立，无需额外限制.
\end{description}

\textbf{\textcolor{CStatement}{结论}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{\textcolor{CStatement}{结论一（正弦平方差）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 正弦平方差化为两角差与和的正弦之积.
    \[
    \sin^2\alpha - \sin^2\beta = \sin(\alpha-\beta)\,\sin(\alpha+\beta).
    \]

    \item[\textbf{\textcolor{CStatement}{推广结论（余弦平方差）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 余弦平方差与正弦乘积仅差一个负号.
    \[
    \cos^2\alpha - \cos^2\beta = -\sin(\alpha-\beta)\,\sin(\alpha+\beta).
    \]
\end{description}

\end{statementbox}
