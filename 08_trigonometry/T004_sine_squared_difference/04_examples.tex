\begin{examplebox}
\textbf{\textcolor{CExample}{例 1（基础）}}

\textbf{\textcolor{CExample}{题目：}} 计算 $\sin^2 75^\circ - \sin^2 15^\circ$ 的值.

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（确定差和角）：}] 令$\alpha = 75^\circ$, $\beta = 15^\circ$, 则 $\alpha-\beta = 60^\circ$, $\alpha+\beta = 90^\circ$.
    \par\textbf{理由：}
    识别公式所需两角差与两角和.
    \item[\textbf{第二步（代入公式）：}] 
    \[
    \sin^2 75^\circ - \sin^2 15^\circ = \sin 60^\circ \cdot \sin 90^\circ.
    \]
    \par\textbf{理由：}
    直接使用正弦平方差公式.
    \item[\textbf{第三步（计算乘积）：}]
    \[
    \sin 60^\circ \cdot \sin 90^\circ = \frac{\sqrt{3}}{2} \times 1 = \frac{\sqrt{3}}{2}.
    \]
    \par\textbf{理由：}
    特殊角三角函数值代入.
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{原式的值为 $\frac{\sqrt{3}}{2}$，直接代公式避免复杂展开.}

\textbf{\textcolor{CExample}{例 2（稍变形）}}

\textbf{\textcolor{CExample}{题目：}} 证明恒等式 $\sin^2\left(\frac{\pi}{4}+\theta\right) - \sin^2\left(\frac{\pi}{4}-\theta\right) = \sin 2\theta$.

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（设定两角）：}] 记 $\alpha = \frac{\pi}{4}+\theta$, $\beta = \frac{\pi}{4}-\theta$, 则 $\alpha-\beta = 2\theta$, $\alpha+\beta = \frac{\pi}{2}$.
    \par\textbf{理由：}
    将问题转化为公式中的标准形式.
    \item[\textbf{第二步（套用公式）：}] 
    \[
    \sin^2\alpha - \sin^2\beta = \sin(\alpha-\beta)\sin(\alpha+\beta) = \sin 2\theta \cdot \sin\frac{\pi}{2}.
    \]
    \par\textbf{理由：}
    正弦平方差公式直接转化.
    \item[\textbf{第三步（化简得证）：}]
    \[
    \sin\frac{\pi}{2} = 1 \quad \Rightarrow \quad \sin 2\theta \cdot 1 = \sin 2\theta.
    \]
    \par\textbf{理由：}
    约简即得右边，恒等式成立.
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{形如 $\sin^2(A\pm\theta)$ 的差可巧妙套用公式，化为单角正弦.}
\end{examplebox}
