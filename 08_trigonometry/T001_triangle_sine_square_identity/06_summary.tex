\begin{summarybox}
\textbf{\textcolor{CSummary}{一句话核心}}

\textbf{三角形内角的正弦平方和恒等于 $2$ 加两倍余弦乘积，轮换对称且无条件成立。}

\medskip
\textbf{\textcolor{CSummary}{使用条件}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{条件 1（三角形内角）：} $A,B,C$ 为 $\triangle ABC$ 的内角，满足 $A+B+C=\pi$。
    \item \textbf{条件 2（恒成立）：} 无其他限制，任意形状三角形均可直接套用。
\end{itemize}

\medskip
\textbf{\textcolor{CSummary}{关键提醒}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{易错点（余弦符号）：} $\cos A\cos B\cos C$ 可正可负，锐角为正、直角为零、钝角为负，勿忽略符号。
    \item \textbf{检查项（内角和验证）：} 使用前务必确认 $A+B+C=\pi$，非三角形角数组严禁套用。
\end{itemize}
\end{summarybox}
