\begin{examplebox}
\textbf{\textcolor{CExample}{例 1（基础）}}

\textbf{\textcolor{CExample}{题目：}} 在 $\triangle ABC$ 中，若 $\cos A\cos B\cos C = \frac{1}{8}$，求 $\sin^2 A+\sin^2 B+\sin^2 C$ 的值。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（代入公式）：}] 直接应用恒等式 $\sin^2 A+\sin^2 B+\sin^2 C = 2+2\cos A\cos B\cos C$。
    \par\textbf{理由：} 题目条件正好给出余弦乘积，可直接套用结论。
    \item[\textbf{第二步（计算求值）：}] 代入数值计算：
    \[ 2+2\times\frac{1}{8} = 2+\frac{1}{4} = \frac{9}{4}. \]
    \par\textbf{理由：} 简单运算得到最终结果。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{所求值为 $\frac{9}{4}$，此时三角形恰为等边三角形。}

\medskip
\textbf{\textcolor{CExample}{例 2（稍变形）}}

\textbf{\textcolor{CExample}{题目：}} 在 $\triangle ABC$ 中，若 $\sin^2 A+\sin^2 B+\sin^2 C = \frac{9}{4}$，判断三角形的形状。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（求余弦积）：}] 由恒等式得
    \[ 2+2\cos A\cos B\cos C = \frac{9}{4} \quad\Rightarrow\quad \cos A\cos B\cos C = \frac{1}{8}. \]
    \par\textbf{理由：} 移项化简即可解出余弦乘积。
    \item[\textbf{第二步（引用不等式）：}] 在任意三角形中，恒有 $\cos A\cos B\cos C \le \frac{1}{8}$，等号成立当且仅当 $A=B=C=60^\circ$。
    \par\textbf{理由：} 该不等式可从均值不等式或 Jensen 不等式推出，为常用结论。
    \item[\textbf{第三步（等号条件）：}] 由第二步知已取得最大值，故必有 $A=B=C$。
    \par\textbf{理由：} 等号成立对应角度全部相等。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{该三角形为等边三角形。}
\end{examplebox}
