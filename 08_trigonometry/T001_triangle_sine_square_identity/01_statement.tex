\textbf{\textcolor{CStatement}{条件}}

\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{条件 1（三角形内角）：}}] 在 $\triangle ABC$ 中，$A,B,C$ 为三个内角，满足 $A+B+C=\pi$。
\end{description}

\textbf{\textcolor{CStatement}{结论}}

\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{\textcolor{CStatement}{结论一（恒等关系）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 三个内角的正弦平方和恒等于 $2$ 加上两倍余弦乘积。
    \[
    \sin^2 A+\sin^2 B+\sin^2 C = 2+2\cos A\cos B\cos C.
    \]

    \item[\textbf{\textcolor{CStatement}{推广结论（直角情形）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 若三角形有一个直角，等式退化为勾股定理的正弦形式。
    \[
    \text{若 }C=90^\circ,\text{ 则 }\sin^2 A+\sin^2 B+1=2,\quad \sin^2 A+\sin^2 B=1.
    \]
\end{description}

\textbf{\textcolor{CStatement}{等号/取等条件：}} 恒成立，无额外限制条件。
\end{statementbox}
