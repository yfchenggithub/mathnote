\begin{summarybox}
\textbf{\textcolor{CSummary}{一句话核心}}
\textbf{三角形内角余弦平方和等于 $1$ 减去二倍三余弦乘积。}

\medskip
\textbf{\textcolor{CSummary}{使用条件}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{条件 1（内角和为π）：} $A,B,C$ 是 $\triangle ABC$ 的内角，$A+B+C=\pi$。
    \item \textbf{条件 2（余弦可取值）：} $\cos A,\cos B,\cos C$ 均为实数，无其他定义域限制。
\end{itemize}

\medskip
\textbf{\textcolor{CSummary}{关键提醒}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{易错点（条件检查）：} 使用前务必确认三个角满足内角和为 $\pi$。
    \item \textbf{检查项（特值验证）：} 可用 $A=B=C=60^\circ$ 快速检验符号与系数：左 $\frac34$，右 $1-2\cdot\frac18=\frac34$。
\end{itemize}
\end{summarybox}
