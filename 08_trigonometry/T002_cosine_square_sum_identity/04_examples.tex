\begin{examplebox}
\textbf{\textcolor{CExample}{例 1（基础）}}

\textbf{\textcolor{CExample}{题目：}} 在 $\triangle ABC$ 中，已知 $\cos A\cos B\cos C = \frac{1}{8}$，求 $\cos^2 A+\cos^2 B+\cos^2 C$ 的值。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（识别恒等式）：}] 由三角形内角余弦平方和恒等式
    \[
    \cos^2 A+\cos^2 B+\cos^2 C = 1-2\cos A\cos B\cos C.
    \]
    \par\textbf{理由：}该恒等式对任意三角形内角均成立。

    \item[\textbf{第二步（代入乘积值）：}] 将已知 $\cos A\cos B\cos C = \frac{1}{8}$ 代入右边：
    \[
    1 - 2\times\frac{1}{8} = 1 - \frac{1}{4}.
    \]
    \par\textbf{理由：}直接代值计算。

    \item[\textbf{第三步（计算结果）：}] 得出
    \[
    \cos^2 A+\cos^2 B+\cos^2 C = \frac{3}{4}.
    \]
    \par\textbf{理由：}进行简单分数运算。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{利用恒等式可以绕过具体角度的计算，直接由乘积得出平方和。}

\textbf{\textcolor{CExample}{例 2（稍变形）}}

\textbf{\textcolor{CExample}{题目：}} 在 $\triangle ABC$ 中，$A=2B$，且 $\cos B = \frac{3}{4}$，求 $\cos^2 A+\cos^2 B+\cos^2 C$ 的值。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（转化角关系）：}] 由 $A=2B$ 及 $A+B+C=\pi$ 得 $C = \pi - 3B$。
    \par\textbf{理由：}利用内角和与已知倍数关系。

    \item[\textbf{第二步（计算各余弦）：}] 
    \[
    \cos A = \cos 2B = 2\cos^2 B - 1 = 2\cdot\left(\frac{3}{4}\right)^2 - 1 = \frac{1}{8},
    \]
    \[
    \cos C = \cos(\pi-3B) = -\cos 3B = -\bigl(4\cos^3 B - 3\cos B\bigr).
    \]
    代入 $\cos B = \frac{3}{4}$：
    \[
    4\cos^3 B = 4\cdot\frac{27}{64} = \frac{27}{16},\quad 3\cos B = \frac{9}{4} = \frac{36}{16},
    \]
    \[
    \cos 3B = \frac{27}{16} - \frac{36}{16} = -\frac{9}{16},\quad \cos C = \frac{9}{16}.
    \]
    \par\textbf{理由：}运用倍角公式与三倍角公式，注意诱导公式符号。

    \item[\textbf{第三步（代入恒等式）：}] 
    \[
    \cos^2 A+\cos^2 B+\cos^2 C = 1 - 2\cos A\cos B\cos C.
    \]
    将数值代入：
    \[
    = 1 - 2\cdot\frac{1}{8}\cdot\frac{3}{4}\cdot\frac{9}{16}.
    \]
    \par\textbf{理由：}直接使用恒等式计算。

    \item[\textbf{第四步（化简得值）：}] 
    \[
    2\cdot\frac{1}{8}\cdot\frac{3}{4}\cdot\frac{9}{16} = \frac{27}{256},
    \]
    \[
    \cos^2 A+\cos^2 B+\cos^2 C = 1 - \frac{27}{256} = \frac{229}{256}.
    \]
    \par\textbf{理由：}进行逐步分式化简得出最终结果。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{即使角度关系复杂，恒等式仍能将问题转化为代数运算，体现三角与代数的结合。}
\end{examplebox}
