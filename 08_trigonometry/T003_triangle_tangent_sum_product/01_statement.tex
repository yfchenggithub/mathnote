\textbf{\textcolor{CStatement}{条件}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{条件 1（非直角三角形）：}}] 三角形 $\triangle ABC$ 不是直角三角形，即 $A,B,C \neq \dfrac{\pi}{2}$，且 $A+B+C=\pi$。
\end{description}

\textbf{\textcolor{CStatement}{结论}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{\textcolor{CStatement}{结论一（正切积和等式）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 三个内角的正切之和恒等于它们的乘积。
    \[
    \tan A + \tan B + \tan C = \tan A \cdot \tan B \cdot \tan C.
    \]

    \item[\textbf{\textcolor{CStatement}{推广结论（负和钝角）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 若三正切之和为负数，则三角形必有一个钝角。
    \[
    \tan A + \tan B + \tan C < 0 \quad\Longrightarrow\quad \triangle ABC\ \text{为钝角三角形}.
    \]
\end{description}

\end{statementbox}
