\begin{summarybox}
\textbf{\textcolor{CSummary}{一句话核心}}

\textbf{在非直角三角形中，$\tan A+\tan B+\tan C = \tan A\tan B\tan C$ 恒成立；若三者之和为负，则该三角形为钝角三角形。}

\medskip
\textbf{\textcolor{CSummary}{使用条件}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{条件 1（非直角三角形）：} 三角形无直角，$A,B,C \neq \pi/2$，保证正切存在且分母 $1-\tan A\tan B$ 非零。
    \item \textbf{条件 2（三角形内角和）：} $A+B+C=\pi$，这是导出恒等式的根源，不可忽略。
\end{itemize}

\medskip
\textbf{\textcolor{CSummary}{关键提醒}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{易错点（直角排除盲区）：} 切勿在直角三角形或未判定形状前直接引用，必须先确认无直角。
    \item \textbf{检查项（符号判形单向）：} “和负 $\Rightarrow$ 钝角”不可逆，判形后勿反向推符号，需结合具体数值或乘积符号。
\end{itemize}
\end{summarybox}
