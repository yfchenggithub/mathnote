\begin{proofbox}
利用向量数量积的运算律：
\begin{enumerate}
    \item 展开和的平方：
    \[ |\vec{a} + \vec{b}|^2 = (\vec{a} + \vec{b})^2 = |\vec{a}|^2 + 2\vec{a} \cdot \vec{b} + |\vec{b}|^2 \quad \text{--- \textcircled{1}} \]
    \item 展开差的平方：
    \[ |\vec{a} - \vec{b}|^2 = (\vec{a} - \vec{b})^2 = |\vec{a}|^2 - 2\vec{a} \cdot \vec{b} + |\vec{b}|^2 \quad \text{--- \textcircled{2}} \]
    \item 式\textcircled{1} 减去 式\textcircled{2} 得：
    \[ |\vec{a} + \vec{b}|^2 - |\vec{a} - \vec{b}|^2 = 4\vec{a} \cdot \vec{b} \]
    \item 两边同时除以 4，得证：
    \[ \vec{a} \cdot \vec{b} = \frac{1}{4} \left[ |\vec{a} + \vec{b}|^2 - |\vec{a} - \vec{b}|^2 \right] \]
\end{enumerate}
\end{proofbox}
