\begin{proofbox}
\textbf{九点圆心向量表示与半径证明}：
\begin{enumerate}
    \item \textbf{定义圆心}：已知外心 $O$ 与垂心 $H$。令 $E$ 为 $OH$ 中点，则 $\vec{OE} = \frac{1}{2}\vec{OH}$。
    \item \textbf{计算到中点的距离}：设 $M$ 为 $BC$ 中点。
    \[ \vec{EM} = \vec{OM} - \vec{OE} = \vec{OM} - \frac{1}{2}(\vec{OA} + \vec{OB} + \vec{OC}) \]
    利用 $\vec{OB} + \vec{OC} = 2\vec{OM}$，代入得：
    \[ \vec{EM} = \vec{OM} - \frac{1}{2}\vec{OA} - \vec{OM} = -\frac{1}{2}\vec{OA} \]
    \item \textbf{求模长}：$|\vec{EM}| = |-\frac{1}{2}\vec{OA}| = \frac{1}{2}R$。
    \item \textbf{结论}：同理可证 $E$ 到其他中点、垂足及欧拉点的距离均为 $\frac{1}{2}R$。故这九点共圆，且半径为外接圆的一半。
\end{enumerate}
\end{proofbox}

\begin{proofbox}
\textbf{九点圆心位置与性质的完整证明}：

\begin{enumerate}
    \item \textbf{圆心位置的设定}：
    设 $\triangle ABC$ 的外心为 $O$，垂心为 $H$。定义点 $E$ 为线段 $OH$ 的中点，即：
    \[ \vec{OE} = \frac{1}{2}\vec{OH} = \frac{1}{2}(\vec{OA} + \vec{OB} + \vec{OC}) \]

    \item \textbf{证明 $E$ 到边中点的距离为 $\frac{R}{2}$}：
    设 $BC$ 边中点为 $M$，则 $\vec{OM} = \frac{1}{2}(\vec{OB} + \vec{OC})$。
    考察向量 $\vec{EM}$：
    \[ \vec{EM} = \vec{OM} - \vec{OE} = \frac{1}{2}(\vec{OB} + \vec{OC}) - \frac{1}{2}(\vec{OA} + \vec{OB} + \vec{OC}) \]
    \[ \vec{EM} = -\frac{1}{2}\vec{OA} \]
    由于 $|\vec{OA}| = R$（外接圆半径），故 $|\vec{EM}| = \frac{1}{2}R$。

    \item \textbf{证明 $E$ 到欧拉点的距离为 $\frac{R}{2}$}：
    设 $AH$ 的中点为 $X$（欧拉点），则 $\vec{OX} = \frac{1}{2}(\vec{OA} + \vec{OH})$。
    考察向量 $\vec{EX}$：
    \[ \vec{EX} = \vec{OX} - \vec{OE} = \frac{1}{2}(\vec{OA} + \vec{OH}) - \frac{1}{2}\vec{OH} \]
    \[ \vec{EX} = \frac{1}{2}\vec{OA} \]
    同理得 $|\vec{EX}| = \frac{1}{2}R$。

    \item \textbf{证明 $E$ 到垂足的距离}：
    设 $BC$ 边上的垂足为 $D$。在直角三角形 $XDM$ 中，$XM$ 是九点圆的一条直径（长度为 $R$），根据直角三角形斜边中线定理，$E$ 作为 $XM$ 的中点，到直角顶点 $D$ 的距离必然满足 $ED = \frac{1}{2}XM = \frac{1}{2}R$。

    \item \textbf{结论}：
    综上所述，点 $E$ 到九个特征点的距离均为 $\frac{1}{2}R$。因此，$E$ 必为九点圆圆心，且其位置处于外心 $O$ 与垂心 $H$ 的中点。
\end{enumerate}
\end{proofbox}
