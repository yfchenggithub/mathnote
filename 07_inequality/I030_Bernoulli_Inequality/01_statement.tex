\textbf{\textcolor{CStatement}{条件}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{条件 1（非负区间）：}}] $x \in \bigl[0,\frac{\pi}{2}\bigr)$
    \item[\textbf{\textcolor{CStatement}{条件 2（非正区间）：}}] $x \in \bigl(-\frac{\pi}{2},0\bigr]$
\end{description}

\textbf{\textcolor{CStatement}{结论}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{\textcolor{CStatement}{结论一（正切下界）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 在非负区间内，正切函数值不小于自变量。
    \[
    \tan x \geq x
    \]

    \item[\textbf{\textcolor{CStatement}{结论二（正切上界）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 在非正区间内，正切函数值不大于自变量。
    \[
    \tan x \leq x
    \]

    \item[\textbf{\textcolor{CStatement}{推广结论（正弦正切夹逼）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 在正开区间内，正弦小于自变量，自变量小于正切。
    \[
    \sin x < x < \tan x
    \]
\end{description}

\textbf{\textcolor{CStatement}{等号/取等条件：}} 等号成立当且仅当 $x=0$（适用于结论一和结论二）。

\end{statementbox}
