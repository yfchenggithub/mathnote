\begin{examplebox}
\textbf{\textcolor{CExample}{例 1（基础）}}

\textbf{\textcolor{CExample}{题目：}} 判断 $\tan 0.5$ 与 $0.5$ 的大小关系，并说明理由。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（确认区间）：}] 检查 $0.5$ 是否属于 $[0, \frac{\pi}{2})$。由于 $\frac{\pi}{2} \approx 1.57 > 0.5$，故 $0.5 \in [0, \frac{\pi}{2})$。
    \par\textbf{理由：}
    定理条件要求 $x$ 在非负区间内才能应用 $\tan x \geq x$。

    \item[\textbf{第二步（应用定理）：}] 根据定理，当 $x \in [0, \frac{\pi}{2})$ 时，有 $\tan x \geq x$。取 $x=0.5$，则
    \[
    \tan 0.5 \geq 0.5
    \]
    \par\textbf{理由：}
    直接代入结论一。

    \item[\textbf{第三步（严格性判断）：}] 由于 $0.5 > 0$，等号仅当 $x=0$ 时成立，故 $\tan 0.5 > 0.5$。
    \par\textbf{理由：}
    定理中等号条件为 $x=0$，此处 $x=0.5 \neq 0$，因此不等式严格成立。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{$\tan 0.5 > 0.5$。}

\textbf{\textcolor{CExample}{例 2（稍变形）}}

\textbf{\textcolor{CExample}{题目：}} 设 $x \in (0, \frac{\pi}{2})$，证明 $\frac{\sin x}{x} < 1 < \frac{\tan x}{x}$。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（分离不等式）：}] 要证 $\frac{\sin x}{x} < 1 < \frac{\tan x}{x}$，即证 $\sin x < x$ 且 $x < \tan x$。
    \par\textbf{理由：}
    将分式不等式乘以正数 $x$（因 $x>0$），不等号方向不变。

    \item[\textbf{第二步（应用定理）：}] 根据定理结论三，当 $x \in (0, \frac{\pi}{2})$ 时，有 $\sin x < x < \tan x$。
    \[
    \sin x < x \quad \text{且} \quad x < \tan x
    \]
    \par\textbf{理由：}
    直接引用推广结论中的不等式链。

    \item[\textbf{第三步（转化回分式）：}] 由 $\sin x < x$，两边同除以正数 $x$，得 $\frac{\sin x}{x} < 1$；由 $x < \tan x$，两边同除以正数 $x$，得 $1 < \frac{\tan x}{x}$。
    \[
    \frac{\sin x}{x} < 1 \quad \text{且} \quad 1 < \frac{\tan x}{x}
    \]
    \par\textbf{理由：}
    除以正数不改变不等号方向。

    \item[\textbf{第四步（合并结论）：}] 综上，有 $\frac{\sin x}{x} < 1 < \frac{\tan x}{x}$。
    \par\textbf{理由：}
    将两个不等式结合即可。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{当 $x \in (0, \frac{\pi}{2})$ 时，$\frac{\sin x}{x} < 1 < \frac{\tan x}{x}$ 成立。}
\end{examplebox}
