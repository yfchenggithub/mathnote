\begin{summarybox}
\textbf{\textcolor{CSummary}{一句话核心}}

\textbf{在 $x$ 接近零时，$\tan x$ 略大于 $x$，而 $\sin x$ 略小于 $x$，形成 $\sin x < x < \tan x$ 的不等式链。}

\medskip
\textbf{\textcolor{CSummary}{使用条件}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{条件 1（区间非负）：} $x \in [0, \frac{\pi}{2})$，此时适用 $\tan x \geq x$ 且 $\sin x < x$。
    \item \textbf{条件 2（区间非正）：} $x \in (-\frac{\pi}{2}, 0]$，此时适用 $\tan x \leq x$。
\end{itemize}

\medskip
\textbf{\textcolor{CSummary}{关键提醒}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{易错点（符号方向）：} 负区间时不等号方向与正区间相反，务必注意。
    \item \textbf{检查项（区间验证）：} 使用前先检查 $x$ 是否在定理指定的区间内，避免错误应用。
\end{itemize}
\end{summarybox}
