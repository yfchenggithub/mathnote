\begin{examplebox}
\textbf{\textcolor{CExample}{例 1（基础）}}

\textbf{\textcolor{CExample}{题目：}} 设 $a,b>0$，证明 $\dfrac{a^{2}}{b}+\dfrac{b^{2}}{a}\geqslant a+b$。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（识别结构）：}] 观察不等式形式，与定理中 $n=2$ 的特例一致。
    \par\textbf{理由：}
    定理推广结论给出当 $n=2$ 时，有 $\frac{a^{2}}{b}+\frac{b^{2}}{a}\geqslant a+b$。

    \item[\textbf{第二步（直接应用）：}] 由定理，直接得出结论。
    \[
    \frac{a^{2}}{b}+\frac{b^{2}}{a}\geqslant a+b
    \]
    \par\textbf{理由：}
    定理条件满足（$a,b>0$，$n=2\in\mathbb{N}^{*}$），故不等式成立。

    \item[\textbf{第三步（取等条件）：}] 指出等号成立条件。
    \par\textbf{理由：}
    根据定理，等号当且仅当 $a=b$ 时成立。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{对于任意正实数 $a,b$，有 $\frac{a^{2}}{b}+\frac{b^{2}}{a}\geqslant a+b$，当且仅当 $a=b$ 时取等。}

\medskip
\textbf{\textcolor{CExample}{例 2（稍变形）}}

\textbf{\textcolor{CExample}{题目：}} 已知 $a,b>0$，且 $a+b=5$，求 $\dfrac{a^{3}}{b^{2}}+\dfrac{b^{3}}{a^{2}}$ 的最小值。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（应用定理）：}] 使用定理中 $n=3$ 的情形。
    \[
    \frac{a^{3}}{b^{2}}+\frac{b^{3}}{a^{2}}\geqslant a+b
    \]
    \par\textbf{理由：}
    由定理，当 $n=3$ 时，不等式 $\frac{a^{n}}{b^{n-1}}+\frac{b^{n}}{a^{n-1}}\geqslant a+b$ 成立。

    \item[\textbf{第二步（代入条件）：}] 将 $a+b=5$ 代入不等式右边。
    \[
    \frac{a^{3}}{b^{2}}+\frac{b^{3}}{a^{2}}\geqslant 5
    \]
    \par\textbf{理由：}
    已知 $a+b=5$，所以右边为 $5$。

    \item[\textbf{第三步（确定最值）：}] 得出最小值并检查取等条件。
    \par\textbf{理由：}
    等号当 $a=b$ 时成立，结合 $a+b=5$，得 $a=b=2.5$，此时 $\frac{a^{3}}{b^{2}}+\frac{b^{3}}{a^{2}} = 5$，所以最小值为 $5$。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{在 $a,b>0$ 且 $a+b=5$ 的条件下，$\frac{a^{3}}{b^{2}}+\frac{b^{3}}{a^{2}}$ 的最小值为 $5$，当 $a=b=2.5$ 时取得。}
\end{examplebox}
