\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CTrap}{易错点一（忽视正数条件）}}] 错误地认为 $a,b$ 可以为任意实数，忽略 $a,b>0$ 的前提。
    \item[\textbf{\textcolor{CTrap}{正确理解（严格正数）：}}] 定理仅对正实数 $a,b$ 成立，因为分母和指数运算需要正数保证。
    \item[\textbf{\textcolor{CTrap}{错因分析（条件遗漏）：}}] 在应用时未检查变量是否为正，导致不等式可能不成立或证明错误。
\end{description}

\medskip
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CTrap}{易错点二（指数混淆）}}] 将指数 $n$ 和 $n-1$ 记错，错误写成 $\frac{a^n}{b^n}$ 等形式。
    \item[\textbf{\textcolor{CTrap}{正确理解（指数关系）：}}] 分子指数为 $n$，分母指数为 $n-1$，保持齐次性；推广时需对应正确。
    \item[\textbf{\textcolor{CTrap}{错因分析（记忆偏差）：}}] 对定理结构不熟悉，混淆分子和分母的指数，导致应用错误。
\end{description}

\medskip
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CTrap}{易错点三（取等忽略）}}] 在求解最值问题时，只得到不等式，未验证等号是否能取得。
    \item[\textbf{\textcolor{CTrap}{正确理解（等号验证）：}}] 应用不等式求最值时，必须检查等号成立条件是否满足题目约束。
    \item[\textbf{\textcolor{CTrap}{错因分析（完整性缺失）：}}] 忽略取等条件，可能使得求得的值不是实际最值，或解不完整。
\end{description}
\end{trapbox}
