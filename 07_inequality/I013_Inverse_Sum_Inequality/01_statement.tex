\begin{statementbox}
\textbf{\textcolor{CStatement}{条件}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{条件 1（变量范围）：}}] $x$ 为非零实数。
\end{description}

\textbf{\textcolor{CStatement}{结论}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{\textcolor{CStatement}{结论一（正数情形）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 正数与其倒数之和至少为2。
    \[
    x + \frac{1}{x} \geq 2, \quad x > 0
    \]

    \item[\textbf{\textcolor{CStatement}{结论二（负数情形）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 负数与其倒数之和至多为-2。
    \[
    x + \frac{1}{x} \leq -2, \quad x < 0
    \]

    \item[\textbf{\textcolor{CStatement}{推广结论（一般形式）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 正系数线性与倒数组合的最小值。
    \[
    a x + \frac{b}{x} \geq 2\sqrt{ab}, \quad a, b, x > 0
    \]
\end{description}

\textbf{\textcolor{CStatement}{等号/取等条件：}} 对于结论一，当 $x=1$ 时等号成立；对于结论二，当 $x=-1$ 时等号成立；对于推广结论，当 $x=\sqrt{b/a}$ 时等号成立。
\end{statementbox}
