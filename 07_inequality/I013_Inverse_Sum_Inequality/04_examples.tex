\begin{examplebox}
\textbf{\textcolor{CExample}{例 1（基础）}}

\textbf{\textcolor{CExample}{题目：}} 已知 $x > 0$，求 $f(x) = x + \frac{1}{x}$ 的最小值。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（识别形式）：}] 函数为 $x + 1/x$，符合对勾不等式正数情形。
    \par\textbf{理由：} 直接应用结论一。

    \item[\textbf{第二步（应用不等式）：}] 根据不等式，$x + 1/x \geq 2$。
    \[
    x + \frac{1}{x} \geq 2
    \]
    \par\textbf{理由：} 当 $x > 0$ 时，结论一成立。

    \item[\textbf{第三步（取等条件）：}] 等号当且仅当 $x = 1$ 时成立。
    \par\textbf{理由：} 结论一的取等条件。

    \item[\textbf{第四步（得最小值）：}] 因此，$f(x)$ 的最小值为 $2$。
    \par\textbf{理由：} 不等式给出了下界，且可达到。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{当 $x > 0$ 时，$x + 1/x$ 的最小值为 $2$，在 $x=1$ 时取得。}

\textbf{\textcolor{CExample}{例 2（稍变形）}}

\textbf{\textcolor{CExample}{题目：}} 已知 $x > 0$，求 $g(x) = 2x + \frac{8}{x}$ 的最小值。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（匹配推广形式）：}] 将 $g(x)$ 写成 $a x + b/x$ 形式，其中 $a=2$, $b=8$。
    \par\textbf{理由：} 推广结论要求 $a, b, x > 0$，此处满足。

    \item[\textbf{第二步（应用推广不等式）：}] 根据推广形式，$2x + 8/x \geq 2\sqrt{2 \cdot 8}$。
    \[
    2x + \frac{8}{x} \geq 2\sqrt{16} = 8
    \]
    \par\textbf{理由：} 计算 $2\sqrt{ab} = 2\sqrt{16} = 8$。

    \item[\textbf{第三步（取等条件）：}] 等号当且仅当 $x = \sqrt{b/a} = \sqrt{8/2} = 2$ 时成立。
    \par\textbf{理由：} 推广结论的取等条件。

    \item[\textbf{第四步（得最小值）：}] 因此，$g(x)$ 的最小值为 $8$。
    \par\textbf{理由：} 不等式给出了下界，且可达到。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{当 $x > 0$ 时，$2x + 8/x$ 的最小值为 $8$，在 $x=2$ 时取得。}
\end{examplebox}
