\begin{examplebox}
\textbf{\textcolor{CExample}{例 1（基础）}}

\textbf{\textcolor{CExample}{题目：}} 已知 $x\geq0$，求证：$\ln(1+x)\geq x-\frac{x^2}{2}$。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（选用更强下界）：}] 由结论二（标准下界）知，$\ln(1+x)\geq x-\frac{x^2}{2(1+x)}$。
    \par\textbf{理由：}
    直接应用已证明的不等式。

    \item[\textbf{第二步（比较两个下界）：}] 需证 $x-\frac{x^2}{2(1+x)}\geq x-\frac{x^2}{2}$。
    \par\textbf{理由：}
    若能证明此式，则结合第一步可得 $\ln(1+x)\geq x-\frac{x^2}{2}$。

    \item[\textbf{第三步（简化比较）：}] 比较两式，即证 $-\frac{x^2}{2(1+x)}\geq -\frac{x^2}{2}$。
    \[
    \text{由于 } x\geq0,\quad \frac{1}{2(1+x)}\leq \frac{1}{2} \Rightarrow -\frac{x^2}{2(1+x)}\geq -\frac{x^2}{2}.
    \]
    \par\textbf{理由：}
    分母 $1+x\geq1$，故倒数减小，乘以 $-x^2$ 后不等号反转。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{因此，$\ln(1+x)\geq x-\frac{x^2}{2(1+x)}\geq x-\frac{x^2}{2}$，原不等式得证。}

\medskip
\textbf{\textcolor{CExample}{例 2（稍变形）}}

\textbf{\textcolor{CExample}{题目：}} 已知 $x>0$，求证：$\frac{2}{x+2}<\frac{\ln(1+x)}{x}<\frac{x+2}{2(x+1)}$。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（分离对数）：}] 由结论一（稳健双边）知 $\frac{2x}{x+2}\leq\ln(1+x)\leq\frac{x(x+2)}{2(x+1)}$。
    \par\textbf{理由：}
    直接应用双边估计。

    \item[\textbf{第二步（同除正数 $x$）：}] 由于 $x>0$，不等式两边同时除以 $x$，得
    \[
    \frac{2}{x+2}\leq\frac{\ln(1+x)}{x}\leq\frac{x+2}{2(x+1)}.
    \]
    \par\textbf{理由：}
    除以正数不等号方向不变。

    \item[\textbf{第三步（严格不等号）：}] 等号成立条件为 $x=0$，而题目中 $x>0$，故等号不成立。
    \par\textbf{理由：}
    取等条件不满足，因此不等式严格成立。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{所以，当 $x>0$ 时，有严格不等式 $\frac{2}{x+2}<\frac{\ln(1+x)}{x}<\frac{x+2}{2(x+1)}$ 成立。}

\end{examplebox}
