\begin{proofbox}
\textbf{\textcolor{CProof}{思路提示}}
考虑构造函数 $f(x) = \sin x - \frac{2}{\pi}x$，通过研究其在区间 $[0,\pi/2]$ 上的单调性来证明 $f(x) \geq 0$。这是处理函数不等式的通用方法。

\medskip
\textbf{\textcolor{CProof}{正式推导}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.55em]
    \item[\textbf{步骤一（构造函数）：}] 设辅助函数 $f(x) = \sin x - \frac{2}{\pi}x$，其中 $x \in [0, \frac{\pi}{2}]$。
    \[
    f(x) = \sin x - \frac{2}{\pi}x
    \]
    \par\textbf{理由：} 要证 $\sin x \geq \frac{2}{\pi}x$，等价于证明 $f(x) = \sin x - \frac{2}{\pi}x \geq 0$。

    \item[\textbf{步骤二（求导判单调）：}] 对 $f(x)$ 求导，并分析导函数符号。
    \[
    f'(x) = \cos x - \frac{2}{\pi}
    \]
    \par\textbf{理由：} 导数 $f'(x)$ 的正负决定了 $f(x)$ 的单调性。由于 $\cos x$ 在 $[0, \pi/2]$ 上单调递减，$f'(x)$ 也单调递减。

    \item[\textbf{步骤三（寻找驻点）：}] 令 $f'(x_0)=0$，解得唯一驻点 $x_0 = \arccos\frac{2}{\pi}$，且 $0 < x_0 < \frac{\pi}{2}$。
    \[
    \begin{aligned}
        f'(x) &= 0 \\
        \cos x_0 &= \frac{2}{\pi} \\
        x_0 &= \arccos\frac{2}{\pi}
    \end{aligned}
    \]
    \par\textbf{理由：} 解方程找到导数为零的点，这是函数单调性可能发生改变的位置。

    \item[\textbf{步骤四（分析单调区间）：}] 判断 $f(x)$ 在 $[0, x_0]$ 和 $[x_0, \pi/2]$ 上的单调性。
    \par\textbf{理由：} 当 $x \in [0, x_0)$ 时，$\cos x > \cos x_0 = \frac{2}{\pi}$，故 $f'(x) > 0$，$f(x)$ 单调递增。
    \par 当 $x \in (x_0, \pi/2]$ 时，$\cos x < \cos x_0 = \frac{2}{\pi}$，故 $f'(x) < 0$，$f(x)$ 单调递减。

    \item[\textbf{步骤五（比较端点值）：}] 计算 $f(x)$ 在区间端点 $x=0$ 和 $x=\pi/2$ 处的值。
    \[
    \begin{aligned}
        f(0) &= \sin 0 - \frac{2}{\pi} \cdot 0 = 0 \\
        f\left(\frac{\pi}{2}\right) &= \sin\frac{\pi}{2} - \frac{2}{\pi} \cdot \frac{\pi}{2} = 1 - 1 = 0
    \end{aligned}
    \]
    \par\textbf{理由：} $f(x)$ 在区间两端点的函数值均为 $0$。

    \item[\textbf{步骤六（得出结论）：}] 综合以上单调性，$f(x)$ 在 $[0, \pi/2]$ 上先增后减，且在两端点取最小值 $0$。
    \[
    \begin{aligned}
        &\text{故对任意 } x \in [0, \tfrac{\pi}{2}],\ f(x) \geq 0. \\
        &\text{即 } \sin x - \frac{2}{\pi}x \geq 0.
    \end{aligned}
    \]
    \par\textbf{理由：} 函数的最小值在端点取得，因此在整个区间上 $f(x) \geq 0$。等号成立当且仅当 $x=0$ 或 $x=\pi/2$。
\end{description}

\medskip
\textbf{\textcolor{CProof}{结论回扣}}
\textbf{因此，原命题 $\sin x \geq \frac{2}{\pi}x \ (x \in [0,\frac{\pi}{2}])$ 得证，且等号成立的条件已验证。}
\end{proofbox}
