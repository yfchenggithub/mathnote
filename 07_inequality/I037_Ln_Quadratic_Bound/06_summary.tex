\begin{summarybox}
\textbf{\textcolor{CSummary}{一句话核心}}
\textbf{在 $[0,\frac{\pi}{2}]$ 上，$\sin x$ 图像总在连接 $(0,0)$ 和 $(\frac{\pi}{2},1)$ 的直线之上。}

\medskip
\textbf{\textcolor{CSummary}{使用条件}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{条件 1（角范围）：} 自变量 $x$（或对应角）必须满足 $0 \leq x \leq \frac{\pi}{2}$。
    \item \textbf{条件 2（结构匹配）：} 待处理的不等式或比较式需能化为 $\sin(\text{角}) \geq k \times (\text{同一个角})$ 的形式，且 $k=\frac{2}{\pi}$。
\end{itemize}

\medskip
\textbf{\textcolor{CSummary}{关键提醒}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{易错点（区间遗忘）：} 使用前务必先检查角的范围，这是结论的生命线。
    \item \textbf{检查项（等号成立）：} 等号只在区间两端点成立，内部为严格大于。证明题中若需严格不等式，必须说明角不在端点。
\end{itemize}
\end{summarybox}
