\begin{statementbox}

\textbf{\textcolor{CStatement}{条件}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{条件 1（定义域范围）：}}] 自变量 $x$ 的取值范围为 $\big[0, \frac{\pi}{2}\big]$。
\end{description}

\textbf{\textcolor{CStatement}{结论}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{\textcolor{CStatement}{结论一（线性下界）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} $\sin x$ 在该区间上不小于一条连接端点的直线。
    \[
    \sin x \geq \frac{2}{\pi}x, \quad x \in \big[0,\tfrac{\pi}{2}\big]
    \]

    \item[\textbf{\textcolor{CStatement}{推广结论（上下界夹挤）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 结合 $\sin x \leq x$，可得到 $\sin x$ 的上下界估计。
    \[
    \frac{2}{\pi}x \leq \sin x \leq x, \quad x \in \big[0,\tfrac{\pi}{2}\big]
    \]
\end{description}

\textbf{\textcolor{CStatement}{等号/取等条件：}} 当 $x=0$ 或 $x=\frac{\pi}{2}$ 时，不等式 $\sin x \geq \frac{2}{\pi}x$ 取等号。

\end{statementbox}
