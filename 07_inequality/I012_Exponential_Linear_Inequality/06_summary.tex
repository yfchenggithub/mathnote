\begin{summarybox}
\textbf{\textcolor{CSummary}{一句话核心}}
\textbf{指数函数 $e^x$ 在其切线和二阶展开下方，提供了一系列多项式下界不等式。}

\medskip
\textbf{\textcolor{CSummary}{使用条件}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{条件 1（任意实数）：} 不等式 $e^x \geq x+1$ 和 $e^x \geq ex$ 对任意实数 $x$ 成立。
    \item \textbf{条件 2（非负实数）：} 不等式 $e^x \geq 1+x+\frac{x^2}{2}$ 仅当 $x \geq 0$ 时成立。
\end{itemize}

\medskip
\textbf{\textcolor{CSummary}{关键提醒}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{易错点（取等条件）：} 每个不等式都有特定取等点，使用时需核对。
    \item \textbf{检查项（范围验证）：} 应用前务必检查变量是否满足不等式的前提条件。
\end{itemize}
\end{summarybox}
