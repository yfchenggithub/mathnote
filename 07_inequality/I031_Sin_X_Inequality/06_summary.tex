\begin{summarybox}
\textbf{\textcolor{CSummary}{一句话核心}}
\textbf{余弦函数有一个全局的二次多项式下界：$\cos x \geq 1 - \frac{x^2}{2}$，仅在原点处取等。}

\medskip
\textbf{\textcolor{CSummary}{使用条件}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{条件 1（实数自变量）：} 自变量 $x$ 可以是\textbf{任意实数}，无范围限制。
    \item \textbf{条件 2（下界放缩）：} 当需要寻找 $\cos x$ 的一个\textbf{下界}，且该下界是简单的二次函数时。
\end{itemize}

\medskip
\textbf{\textcolor{CSummary}{关键提醒}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{易错点（方向与取等）：} 牢记是“$\ge$”而不是“$\le$”；牢记取等仅当 $x=0$。
    \item \textbf{检查项（放缩有效性）：} 使用此不等式放缩后，需检查得到的下界在当前问题中是否还有效（非平凡），尤其当 $|x|$ 较大时。
\end{itemize}
\end{summarybox}
