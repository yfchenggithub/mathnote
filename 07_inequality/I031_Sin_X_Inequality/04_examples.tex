\begin{examplebox}
\textbf{\textcolor{CExample}{例 1（基础）}}

\textbf{\textcolor{CExample}{题目：}} 利用不等式 $\cos x \geq 1 - \frac{x^2}{2}$，证明：当 $x \in \mathbb{R}$ 时，$\cos 2x \geq 1 - 2x^2$。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（变量替换）：}] 在原不等式中，将 $x$ 替换为 $2x$。
    \par\textbf{理由：}
    结论中角是 $2x$，因此直接使用 $2x$ 代入原不等式中的自变量位置。

    \item[\textbf{第二步（直接代入）：}] 代入得 $\cos(2x) \geq 1 - \frac{(2x)^2}{2}$。
    \[
    \cos(2x) \geq 1 - \frac{4x^2}{2}
    \]
    \par\textbf{理由：}
    严格遵循原不等式形式，注意替换后分母的 $2$ 保持不变。

    \item[\textbf{第三步（化简）：}] 计算右边的表达式。
    \[
    1 - \frac{4x^2}{2} = 1 - 2x^2.
    \]
    \par\textbf{理由：}
    进行简单的算术化简，得到最终形式。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{因此，我们直接得到 $\cos 2x \geq 1 - 2x^2$，等号成立当且仅当 $2x=0$，即 $x=0$。}

\medskip
\textbf{\textcolor{CExample}{例 2（稍变形）}}

\textbf{\textcolor{CExample}{题目：}} 已知 $x \in \mathbb{R}$，求证：$\frac{\cos x - 1}{x^2} \geq -\frac{1}{2}$。（约定当 $x=0$ 时，极限值为 $-\frac{1}{2}$）

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（变形目标）：}] 要证 $\frac{\cos x - 1}{x^2} \geq -\frac{1}{2}$，即证 $\cos x - 1 \geq -\frac{x^2}{2}$（当 $x^2 > 0$ 时）。
    \par\textbf{理由：}
    将分式不等式乘以正数 $x^2$ 化为整式不等式。注意 $x=0$ 时需单独处理。

    \item[\textbf{第二步（应用结论）：}] 由已知结论 $\cos x \geq 1 - \frac{x^2}{2}$，将 $1 - \frac{x^2}{2}$ 移到左边。
    \[
    \cos x - 1 \geq -\frac{x^2}{2}
    \]
    \par\textbf{理由：}
    这正是原不等式移项后的结果。

    \item[\textbf{第三步（分情况说明）：}] 当 $x \neq 0$ 时，两边同除以 $x^2$（正数），即得 $\frac{\cos x - 1}{x^2} \geq -\frac{1}{2}$。当 $x=0$ 时，结论中的极限值 $-\frac{1}{2}$ 恰好是 $\cos x$ 在 $x=0$ 处的二阶泰勒系数。
    \par\textbf{理由：}
    此步骤完成了从整式到分式的转换，并解释了 $x=0$ 时的连续性。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{因此，对于任意实数 $x$（$x=0$ 时按极限理解），均有 $\frac{\cos x - 1}{x^2} \geq -\frac{1}{2}$ 成立。}
\end{examplebox}
