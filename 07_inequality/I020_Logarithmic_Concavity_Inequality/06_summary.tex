\begin{summarybox}
\textbf{\textcolor{CSummary}{一句话核心}}

\textbf{对数函数是凹函数，因此对数的算术平均不超过算术平均的对数。}

\medskip
\textbf{\textcolor{CSummary}{使用条件}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{条件 1（正实数）：} 所有变量必须为正实数，确保对数有定义。
    \item \textbf{条件 2（凹函数适用）：} 结论依赖于 $\ln t$ 的凹性，推广时需确认函数性质。
\end{itemize}

\medskip
\textbf{\textcolor{CSummary}{关键提醒}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{易错点（方向记忆）：} 牢记不等式方向为 $\frac{\ln x + \ln y}{2} \leq \ln\!\left( \frac{x+y}{2} \right)$，避免反号。
    \item \textbf{检查项（取等条件）：} 应用后务必检查等号是否可能成立，通常仅在变量相等时取等。
\end{itemize}
\end{summarybox}
