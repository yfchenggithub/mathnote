\begin{statementbox}

\textbf{\textcolor{CStatement}{条件}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{条件 1（正实数）：}}] 设 $x, y \in (0, +\infty)$。
\end{description}

\textbf{\textcolor{CStatement}{结论}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{\textcolor{CStatement}{结论一（对数平均）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 算术平均的对数不小于对数的算术平均。
    \[
    \frac{\ln x + \ln y}{2} \leq \ln\!\left( \frac{x+y}{2} \right).
    \]

    \item[\textbf{\textcolor{CStatement}{推广结论（Jensen推广）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 对任意有限个正实数，对数平均不等式成立。
    \[
    \frac{1}{n} \sum_{i=1}^{n} \ln x_i \leq \ln\!\left( \frac{1}{n} \sum_{i=1}^{n} x_i \right).
    \]
\end{description}

\textbf{\textcolor{CStatement}{等号/取等条件：}} 当且仅当 $x = y$（主结论）或 $x_1 = x_2 = \cdots = x_n$（推广结论）时取等。

\end{statementbox}
