\begin{proofbox}
\textbf{\textcolor{CProof}{思路提示}}

通过构造柯西-施瓦茨不等式的形式，将权方和不等式转化为已知不等式。

\medskip
\textbf{\textcolor{CProof}{正式推导}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.55em]
    \item[\textbf{步骤一（构造平方和）：}] 定义 $u_i = \dfrac{a_i}{\sqrt{b_i}}$ 和 $v_i = \sqrt{b_i}$。
    \[
        u_i = \frac{a_i}{\sqrt{b_i}}, \quad v_i = \sqrt{b_i} \quad (i=1,2,\dots,n)
    \]
    \par\textbf{理由：}
    这样构造使得 $u_i^2 = \frac{a_i^2}{b_i}$，且 $v_i^2 = b_i$，便于应用柯西-施瓦茨。

    \item[\textbf{步骤二（应用柯西）：}] 对 $u_i$ 和 $v_i$ 应用柯西-施瓦茨不等式。
    \[
        \left( \sum_{i=1}^{n} u_i^2 \right) \left( \sum_{i=1}^{n} v_i^2 \right) \geq \left( \sum_{i=1}^{n} u_i v_i \right)^2
    \]
    \par\textbf{理由：}
    柯西-施瓦茨不等式对于实数序列成立。

    \item[\textbf{步骤三（代入计算）：}] 将 $u_i$ 和 $v_i$ 的表达式代入。
    \[
        \begin{aligned}
            \sum_{i=1}^{n} u_i^2 &= \sum_{i=1}^{n} \frac{a_i^2}{b_i}, \\
            \sum_{i=1}^{n} v_i^2 &= \sum_{i=1}^{n} b_i, \\
            \sum_{i=1}^{n} u_i v_i &= \sum_{i=1}^{n} \left( \frac{a_i}{\sqrt{b_i}} \cdot \sqrt{b_i} \right) = \sum_{i=1}^{n} a_i.
        \end{aligned}
    \]
    \par\textbf{理由：}
    直接计算各项和。

    \item[\textbf{步骤四（得到不等式）：}] 代入后，柯西-施瓦茨不等式变为：
    \[
        \left( \sum_{i=1}^{n} \frac{a_i^2}{b_i} \right) \left( \sum_{i=1}^{n} b_i \right) \geq \left( \sum_{i=1}^{n} a_i \right)^2
    \]
    \par\textbf{理由：}
    将步骤三的结果代入步骤二的不等式。

    \item[\textbf{步骤五（整理形式）：}] 由于 $\sum b_i > 0$，两边除以 $\sum b_i$ 得到权方和不等式。
    \[
        \sum_{i=1}^{n} \frac{a_i^2}{b_i} \geq \frac{\left( \sum_{i=1}^{n} a_i \right)^2}{\sum_{i=1}^{n} b_i}
    \]
    \par\textbf{理由：}
    不等号方向不变，因为除以正数。

    \item[\textbf{步骤六（取等条件）：}] 等号成立当且仅当柯西-施瓦茨等号成立，即 $u_i$ 与 $v_i$ 成比例。
    \[
        \frac{u_1}{v_1} = \frac{u_2}{v_2} = \dots = \frac{u_n}{v_n}
    \]
    代入 $u_i = \frac{a_i}{\sqrt{b_i}}$, $v_i = \sqrt{b_i}$，得 $\frac{a_1}{b_1} = \frac{a_2}{b_2} = \dots = \frac{a_n}{b_n}$。
    \par\textbf{理由：}
    柯西-施瓦茨等号条件为序列成比例。
\end{description}

\medskip
\textbf{\textcolor{CProof}{结论回扣}}

\textbf{通过柯西-施瓦茨不等式，我们证明了权方和不等式的一般形式，并明确了取等条件。推广形式可通过类似方法或数学归纳法证明。}
\end{proofbox}
