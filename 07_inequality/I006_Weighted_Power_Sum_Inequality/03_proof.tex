\begin{proofbox}
\textbf{思路提示}

利用赫尔德（H\"older）不等式的特例，或通过构造并应用均值不等式（如加权幂平均不等式）进行证明。这里采用构造法和加权幂平均不等式（或广义柯西不等式）的思路，通过配凑系数来证明。

\medskip
\textbf{正式推导}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.55em]
    \item[\textbf{步骤一（构造目标比）：}] 
    令 $\lambda = \frac{a_i}{b_i}$，则 $a_i = \lambda_i b_i$。我们需要证明：
    \[
    \sum_{i=1}^{n} \frac{(\lambda_i b_i)^m}{b_i^{m-1}} \geq \frac{\bigl(\sum_{i=1}^{n} \lambda_i b_i\bigr)^m}{\bigl(\sum_{i=1}^{n} b_i\bigr)^{m-1}}
    \]
    \par\textbf{理由：}
    将 $a_i$ 用比值和分母表示，是探究取等条件 $\frac{a_i}{b_i}$ 为常数的自然思路。

    \item[\textbf{步骤二（化简与重组）：}] 
    化简左边为 $\sum_{i=1}^{n} \lambda_i^m b_i$。将不等式两边同乘以 $\bigl(\sum_{i=1}^{n} b_i\bigr)^{m-1}$，目标转化为证明：
    \[
    \bigl(\sum_{i=1}^{n} b_i\bigr)^{m-1} \bigl(\sum_{i=1}^{n} \lambda_i^m b_i\bigr) \geq \bigl(\sum_{i=1}^{n} \lambda_i b_i\bigr)^m
    \]
    \par\textbf{理由：}
    消除分母，将分式不等式转化为多项式（或加权和）的不等式，便于应用已知不等式。

    \item[\textbf{步骤三（应用加权幂平均）：}] 
    考虑以 $b_i$ 为权重的加权平均。由加权幂平均不等式（或赫尔德不等式），对 $m \geq 1$ 有：
    \[
    \bigl(\frac{\sum_{i=1}^{n} b_i \cdot \lambda_i^m}{\sum_{i=1}^{n} b_i}\bigr)^{\frac{1}{m}} \geq \frac{\sum_{i=1}^{n} b_i \cdot \lambda_i}{\sum_{i=1}^{n} b_i}
    \]
    \par\textbf{理由：}
    函数 $f(x)=x^m$ 在 $x>0, m\ge1$ 时是下凸的，加权幂平均不等式成立。左边是 $\lambda_i$ 的 $m$ 次幂平均，右边是 $\lambda_i$ 的 $1$ 次（算术）平均。

    \item[\textbf{步骤四（整理得结论）：}] 
    将步骤三的不等式两边同时取 $m$ 次幂，并乘以 $(\sum b_i)^m$：
    \[
    \begin{aligned}
    \bigl(\sum_{i=1}^{n} b_i\bigr)^{m} \cdot \frac{\sum_{i=1}^{n} b_i \lambda_i^m}{\sum_{i=1}^{n} b_i} &\geq \bigl(\sum_{i=1}^{n} b_i \lambda_i\bigr)^m \\
    \bigl(\sum_{i=1}^{n} b_i\bigr)^{m-1} \bigl(\sum_{i=1}^{n} b_i \lambda_i^m\bigr) &\geq \bigl(\sum_{i=1}^{n} b_i \lambda_i\bigr)^m
    \end{aligned}
    \]
    \par\textbf{理由：}
    这正是步骤二需要证明的不等式。代回 $\lambda_i = a_i / b_i$ 即得原不等式。

    \item[\textbf{步骤五（验证取等）：}] 
    加权幂平均不等式取等条件为所有 $\lambda_i$ 相等，即 $\frac{a_1}{b_1} = \frac{a_2}{b_2} = \dots = \frac{a_n}{b_n}$。
    \par\textbf{理由：}
    等号传递性保证了原不等式在同一条件下取等。
\end{description}

\medskip
\textbf{结论回扣}

\textbf{证明的关键在于将 $a_i$ 表示为比值与 $b_i$ 的乘积，然后利用关于 $b_i$ 加权的幂平均不等式，最终整理得到所需形式。} 当 $m=2$ 时，该证明过程即退化为柯西不等式的标准证明之一。
\end{proofbox}
