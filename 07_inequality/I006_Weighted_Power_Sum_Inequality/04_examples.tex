\begin{examplebox}
\textbf{例 1（基础）}

\textbf{题目：} 设 $x, y, z > 0$，求证：$\frac{x^2}{y+z} + \frac{y^2}{z+x} + \frac{z^2}{x+y} \geq \frac{x+y+z}{2}$。

\textbf{解题步骤：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（识别形式）：}] 观察不等式左边，是三个分式的和，分子是平方项，分母是一次项。这提示我们可能应用 $m=2$ 时的权方和不等式。
    \par\textbf{理由：}
    权方和不等式（平方特例）的结构是 $\sum \frac{a_i^2}{b_i}$。我们需要将题目中的分母与某个 $b_i$ 对应起来。

    \item[\textbf{第二步（构造变量）：}] 令 $a_1=x, a_2=y, a_3=z$。为了匹配分母 $y+z, z+x, x+y$，我们设 $b_1 = y+z, b_2 = z+x, b_3 = x+y$。
    \par\textbf{理由：}
    直接应用结论二：$\sum \frac{a_i^2}{b_i} \geq \frac{(\sum a_i)^2}{\sum b_i}$。其中 $\sum a_i = x+y+z$，$\sum b_i = (y+z)+(z+x)+(x+y)=2(x+y+z)$。

    \item[\textbf{第三步（代入计算）：}] 代入权方和不等式：
    \[
    \frac{x^2}{y+z} + \frac{y^2}{z+x} + \frac{z^2}{x+y} \geq \frac{(x+y+z)^2}{(y+z)+(z+x)+(x+y)} = \frac{(x+y+z)^2}{2(x+y+z)}
    \]
    \par\textbf{理由：}
    这是不等式结论的直接代入。

    \item[\textbf{第四步（化简得证）：}] 化简右端：$\frac{(x+y+z)^2}{2(x+y+z)} = \frac{x+y+z}{2}$。
    \par\textbf{理由：}
    由于 $x,y,z>0$，$x+y+z>0$，可以约分。因此原不等式成立。
\end{description}

\textbf{关键结论：} \textbf{通过构造 $a_i=x,y,z$ 和 $b_i$ 为对应的两数之和，直接应用权方和不等式（$m=2$）可一步到位完成证明。}

\textbf{例 2（稍变形）}

\textbf{题目：} 已知 $a, b, c > 0$，且 $a+b+c=1$，求 $\frac{a^3}{b} + \frac{b^3}{c} + \frac{c^3}{a}$ 的最小值。

\textbf{解题步骤：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（匹配指数）：}] 所求表达式为 $\frac{a^3}{b} + \frac{b^3}{c} + \frac{c^3}{a}$。分子是三次，分母是一次。这对应权方和不等式中 $m=3$ 的情形：$\sum \frac{a_i^3}{b_i^2}$。
    \par\textbf{理由：}
    我们目前的表达式是 $\sum \frac{a_i^3}{b_i^1}$，与标准形式 $\sum \frac{a_i^m}{b_i^{m-1}}$ 在分母指数上差 $1$。

    \item[\textbf{第二步（构造标准形）：}] 将原式视为 $\sum \frac{a_i^3}{b_i} = \sum \frac{a_i^3}{(b_i^{1/2})^2}$。令 $x_i = a_i$, $y_i = b_i^{1/2}$，则表达式变为 $\sum \frac{x_i^3}{y_i^2}$，这符合 $m=3$ 的标准形式（$b_i^{m-1}=y_i^2$）。
    \par\textbf{理由：}
    为了应用不等式，我们需要分母是 $b_i^{m-1}=b_i^{2}$。因此需要将 $b_i$ 写成某个量的平方。

    \item[\textbf{第三步（应用不等式）：}] 应用 $m=3$ 的权方和不等式：
    \[
    \frac{a^3}{b} + \frac{b^3}{c} + \frac{c^3}{a} = \sum \frac{x_i^3}{y_i^2} \geq \frac{(\sum x_i)^3}{(\sum y_i)^2} = \frac{(a+b+c)^3}{(\sqrt{b}+\sqrt{c}+\sqrt{a})^2}
    \]
    \par\textbf{理由：}
    这里 $\sum x_i = a+b+c=1$，$\sum y_i = \sqrt{a}+\sqrt{b}+\sqrt{c}$。

    \item[\textbf{第四步（进一步放缩）：}] 由已知 $a+b+c=1$，及 $\sqrt{a}+\sqrt{b}+\sqrt{c} \leq \sqrt{3(a+b+c)} = \sqrt{3}$（柯西不等式或均值不等式）。因此 $(\sum y_i)^2 \leq 3$。
    \par\textbf{理由：}
    为了得到整个表达式的最小值下界，需要放大分母 $(\sum y_i)^2$ 的上界，从而得到整个分式的一个下界。

    \item[\textbf{第五步（得到最小值）：}] 结合第三、四步：
    \[
    \frac{a^3}{b} + \frac{b^3}{c} + \frac{c^3}{a} \geq \frac{1^3}{(\sqrt{a}+\sqrt{b}+\sqrt{c})^2} \geq \frac{1}{3}
    \]
    当 $a=b=c=\frac{1}{3}$ 时，等号同时成立（满足比值相等 $\frac{a}{\sqrt{b}} = \frac{b}{\sqrt{c}} = \frac{c}{\sqrt{a}}$ 且 $\sqrt{a}+\sqrt{b}+\sqrt{c}$ 取最大值 $\sqrt{3}$）。
    \par\textbf{理由：}
    故最小值为 $\frac{1}{3}$。
\end{description}

\textbf{关键结论：} \textbf{当表达式与标准形式不完全匹配时，需通过变量代换（如 $y_i=\sqrt{b_i}$）构造出 $b_i^{m-1}$ 的形式。有时需要结合其他不等式（如柯西）进行二次放缩才能定出最值。}
\end{examplebox}
