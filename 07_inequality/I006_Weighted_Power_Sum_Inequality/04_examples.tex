\begin{examplebox}
\textbf{\textcolor{CExample}{例 1（基础）}}

\textbf{\textcolor{CExample}{题目：}} 已知 $a, b, c > 0$，且 $a+b+c=1$，求 $\displaystyle \frac{a^2}{b} + \frac{b^2}{c} + \frac{c^2}{a}$ 的最小值。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（识别结构）：}] 表达式为 $\frac{a^2}{b} + \frac{b^2}{c} + \frac{c^2}{a}$，符合权方和不等式的一般形式。
    \par\textbf{理由：}
    分子平方，分母一次，且分母为正。

    \item[\textbf{第二步（应用不等式）：}] 由权方和不等式，有
    \[
        \frac{a^2}{b} + \frac{b^2}{c} + \frac{c^2}{a} \geq \frac{(a+b+c)^2}{b+c+a}
    \]
    \par\textbf{理由：}
    直接应用，其中 $a_1=a, a_2=b, a_3=c$，$b_1=b, b_2=c, b_3=a$。

    \item[\textbf{第三步（代入计算）：}] 已知 $a+b+c=1$，且分母和 $b+c+a=1$，所以
    \[
        \frac{a^2}{b} + \frac{b^2}{c} + \frac{c^2}{a} \geq \frac{1^2}{1} = 1
    \]
    \par\textbf{理由：}
    将条件代入不等式右边。

    \item[\textbf{第四步（取等检查）：}] 等号成立当且仅当 $\frac{a}{b} = \frac{b}{c} = \frac{c}{a}$，结合 $a+b+c=1$，可得 $a=b=c=\frac{1}{3}$。
    \par\textbf{理由：}
    验证取等条件，确保最小值可达。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{最小值为 $1$，当且仅当 $a=b=c=\frac{1}{3}$ 时取到。}

\textbf{\textcolor{CExample}{例 2（稍变形）}}

\textbf{\textcolor{CExample}{题目：}} 设 $x, y, z > 0$，证明 $\displaystyle \frac{x^2}{y+z} + \frac{y^2}{z+x} + \frac{z^2}{x+y} \geq \frac{x+y+z}{2}$。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（分母变形）：}] 注意到分母 $y+z, z+x, x+y$ 不是单个变量，但可视为权方和中的 $b_i$。
    \par\textbf{理由：}
    权方和不等式要求 $b_i>0$，这里分母满足条件。

    \item[\textbf{第二步（应用权方和）：}] 由权方和不等式，
    \[
        \frac{x^2}{y+z} + \frac{y^2}{z+x} + \frac{z^2}{x+y} \geq \frac{(x+y+z)^2}{(y+z)+(z+x)+(x+y)}
    \]
    \par\textbf{理由：}
    将 $x,y,z$ 视为 $a_i$，$y+z, z+x, x+y$ 视为 $b_i$。

    \item[\textbf{第三步（计算分母和）：}] 分母和为 $(y+z)+(z+x)+(x+y) = 2(x+y+z)$。
    \[
        \frac{(x+y+z)^2}{2(x+y+z)} = \frac{x+y+z}{2}
    \]
    \par\textbf{理由：}
    简化表达式。

    \item[\textbf{第四步（得到不等式）：}] 因此，
    \[
        \frac{x^2}{y+z} + \frac{y^2}{z+x} + \frac{z^2}{x+y} \geq \frac{x+y+z}{2}
    \]
    \par\textbf{理由：}
    结合步骤二和步骤三。

    \item[\textbf{第五步（取等条件）：}] 等号成立当且仅当 $\frac{x}{y+z} = \frac{y}{z+x} = \frac{z}{x+y}$。在正数条件下，这等价于 $x=y=z$。
    \par\textbf{理由：}
    权方和等号条件，代入具体变量。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{不等式成立，等号当 $x=y=z$ 时取得。}
\end{examplebox}
