\begin{summarybox}
\textbf{一句话核心}

\textbf{对满足指数差为1的加权分式和，其最小值可用整体和的幂来简洁表示。}

\medskip
\textbf{使用条件}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{条件 1（分母为正）：} 所有权重 $b_i > 0$，这是应用的基础。
    \item \textbf{条件 2（指数匹配）：} 表达式形如 $\sum \frac{a_i^m}{b_i^{m-1}}$，其中 $m \ge 1$。
\end{itemize}

\medskip
\textbf{关键提醒}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{易错点（形式套用）：} 套用前务必检查分子分母指数是否严格满足“$m$”与“$m-1$”的关系。
    \item \textbf{检查项（取等条件）：} 最值问题必须检查并说明当且仅当所有 $\frac{a_i}{b_i}$ 相等时取得等号。
\end{itemize}
\end{summarybox}
