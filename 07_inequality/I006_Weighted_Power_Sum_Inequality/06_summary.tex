\begin{summarybox}
\textbf{\textcolor{CSummary}{一句话核心}}

\textbf{权方和不等式将分式平方和与总和的平方联系起来，是求最值和证明不等式的有效方法。}

\medskip
\textbf{\textcolor{CSummary}{使用条件}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{条件 1（实数变量）：} $a_i$ 为任意实数。
    \item \textbf{条件 2（正分母）：} $b_i > 0$ 必须严格成立。
    \item \textbf{条件 3（指数范围）：} $m \geq 1$，通常 $m$ 为整数或实数。
\end{itemize}

\medskip
\textbf{\textcolor{CSummary}{关键提醒}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{易错点（分母正负）：} 务必检查 $b_i > 0$，否则不等式不适用。
    \item \textbf{检查项（取等条件）：} 应用后必须验证 $\frac{a_1}{b_1} = \frac{a_2}{b_2} = \dots = \frac{a_n}{b_n}$ 是否可达。
\end{itemize}
\end{summarybox}
