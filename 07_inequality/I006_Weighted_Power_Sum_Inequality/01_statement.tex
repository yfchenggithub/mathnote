\begin{statementbox}

\textbf{\textcolor{CStatement}{条件}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{条件 1（实数变量）：}}] $a_i \in \mathbb{R}$（$i=1,2,\dots,n$）为任意实数。
    \item[\textbf{\textcolor{CStatement}{条件 2（正分母）：}}] $b_i > 0$（$i=1,2,\dots,n$）为正实数。
    \item[\textbf{\textcolor{CStatement}{条件 3（指数范围）：}}] 实数 $m \geq 1$。
\end{description}

\textbf{\textcolor{CStatement}{结论}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{\textcolor{CStatement}{结论一（二维形式）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 两个分式平方和大于等于和的平方除以分母和。
    \[
        \frac{a_1^2}{b_1} + \frac{a_2^2}{b_2} \geq \frac{(a_1 + a_2)^2}{b_1 + b_2}
    \]

    \item[\textbf{\textcolor{CStatement}{结论二（一般形式）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} n 个分式平方和大于等于总和的平方除以分母总和。
    \[
        \sum_{i=1}^{n} \frac{a_i^2}{b_i} \geq \frac{\left(\sum_{i=1}^{n} a_i\right)^2}{\sum_{i=1}^{n} b_i}
    \]

    \item[\textbf{\textcolor{CStatement}{推广结论（m次形式）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 将平方推广到 m 次方，分母指数为 m-1。
    \[
        \sum_{i=1}^{n} \frac{a_i^{m}}{b_i^{\,m-1}} \geq \frac{\left(\sum_{i=1}^{n} a_i\right)^{m}}{\left(\sum_{i=1}^{n} b_i\right)^{m-1}}
    \]
\end{description}

\textbf{\textcolor{CStatement}{等号/取等条件：}} 当且仅当 $\displaystyle\frac{a_1}{b_1} = \frac{a_2}{b_2} = \dots = \frac{a_n}{b_n}$ 时等号成立。

\end{statementbox}
