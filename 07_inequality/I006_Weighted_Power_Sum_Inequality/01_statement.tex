\begin{statementbox}

\textbf{条件：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{条件 1（实数变量）：}] 变量 $a_i$ 为任意实数。
    \item[\textbf{条件 2（正权分母）：}] 分母 $b_i$ 均为正数。
    \item[\textbf{条件 3（指数要求）：}] 指数 $m$ 满足 $m \geq 1$。
\end{description}

\textbf{结论：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{结论一（一般权方和）：}]
    \textbf{直观描述：} 加权幂和的商不小于总和的对应幂的商。
    \[
    \sum_{i=1}^{n} \frac{a_i^m}{b_i^{m-1}} \geq \frac{\bigl(\sum_{i=1}^{n} a_i\bigr)^m}{\bigl(\sum_{i=1}^{n} b_i\bigr)^{m-1}}
    \]
    \item[\textbf{结论二（平方特例）：}]
    \textbf{直观描述：} $m=2$ 时形式简化，类似加权柯西不等式。
    \[
    \sum_{i=1}^{n} \frac{a_i^2}{b_i} \geq \frac{\bigl(\sum_{i=1}^{n} a_i\bigr)^2}{\sum_{i=1}^{n} b_i}
    \]
\end{description}

\textbf{等号/取等条件：} 等号成立当且仅当所有比值 $\frac{a_1}{b_1} = \frac{a_2}{b_2} = \dots = \frac{a_n}{b_n}$ 相等。

\end{statementbox}
