\begin{examplebox}
\textbf{\textcolor{CExample}{例 1（基础）}}

\textbf{\textcolor{CExample}{题目：}} 设 $x \neq 0$，求证 $\left| \dfrac{\sin x}{x} \right| \le 1$。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（应用原结论）：}] 由已知定理 $|\sin x| \le |x|$。
    \[
    |\sin x| \le |x|.
    \]
    \par\textbf{理由：}
    直接使用主结论。

    \item[\textbf{第二步（除以正数）：}] 因为 $x \neq 0$，所以 $|x| > 0$，不等式两边同除以 $|x|$。
    \[
    \frac{|\sin x|}{|x|} \le 1.
    \]
    \par\textbf{理由：}
    不等式两边同除以正数，不等号方向不变。

    \item[\textbf{第三步（化简形式）：}] 注意到 $\frac{|\sin x|}{|x|} = \left| \frac{\sin x}{x} \right|$，故得证。
    \[
    \left| \frac{\sin x}{x} \right| \le 1.
    \]
    \par\textbf{理由：}
    绝对值商等于商的绝对值。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{当 $x \neq 0$ 时，$\left| \frac{\sin x}{x} \right| \le 1$。}

\medskip
\textbf{\textcolor{CExample}{例 2（稍变形）}}

\textbf{\textcolor{CExample}{题目：}} 设 $x, y \in \mathbb{R}$，求证 $|\sin x - \sin y| \le |x - y|$。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（和差化积）：}] 利用三角恒等式。
    \[
    \sin x - \sin y = 2 \cos \frac{x+y}{2} \sin \frac{x-y}{2}.
    \]
    \par\textbf{理由：}
    将差化为乘积形式，便于处理绝对值。

    \item[\textbf{第二步（取绝对值）：}] 取绝对值并利用 $|\cos \theta| \le 1$。
    \[
    |\sin x - \sin y| = \left| 2 \cos \frac{x+y}{2} \sin \frac{x-y}{2} \right| \le 2 \left| \sin \frac{x-y}{2} \right|.
    \]
    \par\textbf{理由：}
    $|\cos \theta| \le 1$，且 $|ab| = |a||b|$。

    \item[\textbf{第三步（应用结论）：}] 由 $|\sin t| \le |t|$，令 $t = \frac{x-y}{2}$。
    \[
    \left| \sin \frac{x-y}{2} \right| \le \left| \frac{x-y}{2} \right|.
    \]
    \par\textbf{理由：}
    直接使用主结论。

    \item[\textbf{第四步（综合得证）：}] 代入上步不等式。
    \[
    |\sin x - \sin y| \le 2 \cdot \left| \frac{x-y}{2} \right| = |x-y|.
    \]
    \par\textbf{理由：}
    简单代数运算。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{对于任意实数 $x, y$，有 $|\sin x - \sin y| \le |x - y|$。}
\end{examplebox}
