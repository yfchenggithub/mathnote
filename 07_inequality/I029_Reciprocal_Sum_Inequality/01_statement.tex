\begin{statementbox}
\textbf{\textcolor{CStatement}{条件}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{条件 1（实数范围）：}}] 设 $x \in \mathbb{R}$。
\end{description}

\textbf{\textcolor{CStatement}{结论}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{\textcolor{CStatement}{结论一（绝对值不等式）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 正弦绝对值不超过自变量绝对值。
    \[
    |\sin x| \leq |x|
    \]

    \item[\textbf{\textcolor{CStatement}{推广结论（平方形式）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 正弦平方不超过自变量平方。
    \[
    \sin^2 x \leq x^2
    \]
\end{description}

\textbf{\textcolor{CStatement}{等号/取等条件：}} 当且仅当 $x = 0$ 时等号成立。
\end{statementbox}
