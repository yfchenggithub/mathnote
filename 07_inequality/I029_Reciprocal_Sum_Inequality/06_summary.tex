\begin{summarybox}
\textbf{\textcolor{CSummary}{一句话核心}}

\textbf{对于任意实数 $x$，正弦绝对值 $|\sin x|$ 总是不超过 $|x|$，且等号只在原点成立。}

\medskip
\textbf{\textcolor{CSummary}{使用条件}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{条件 1（实数变量）：} 变量 $x$ 必须是实数，不能是复数。
    \item \textbf{条件 2（任意取值）：} $x$ 可取任意实数值，无需限制范围。
\end{itemize}

\medskip
\textbf{\textcolor{CSummary}{关键提醒}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{易错点（绝对值遗漏）：} 务必保留绝对值符号，避免写成 $\sin x \le x$。
    \item \textbf{检查项（等号验证）：} 使用后检查等号是否仅在 $x=0$ 时成立，防止误用。
\end{itemize}
\end{summarybox}
