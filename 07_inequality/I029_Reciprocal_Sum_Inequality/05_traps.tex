\begin{trapbox}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CTrap}{易错点一（忽略绝对值）：}}] 将不等式错误记为 $\sin x \le x$ 而不加绝对值。
    \item[\textbf{\textcolor{CTrap}{正确理解（绝对值不可省）：}}] 必须写成 $|\sin x| \le |x|$，否则当 $x<0$ 或 $\sin x<0$ 时不成立。
    \item[\textbf{\textcolor{CTrap}{错因分析（符号混淆）：}}] 忽略了正弦函数和自变量的符号变化，导致不等式方向错误。
\end{description}

\medskip
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CTrap}{易错点二（等号条件扩大）：}}] 认为等号在 $x=2k\pi$（$k\in\mathbb{Z}$）时成立。
    \item[\textbf{\textcolor{CTrap}{正确理解（等号唯一）：}}] 等号仅当 $x=0$ 时成立，其他位置严格小于。
    \item[\textbf{\textcolor{CTrap}{错因分析（导数零点误解）：}}] 误将导数 $f'(x)=0$ 的点当作函数取等点，但 $f(x)=x-\sin x$ 在 $x=2k\pi$ 处 $f(x)>0$。
\end{description}

\medskip
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CTrap}{易错点三（复数误用）：}}] 在复数 $z$ 中错误使用 $|\sin z| \le |z|$。
    \item[\textbf{\textcolor{CTrap}{正确理解（实数范围）：}}] 结论仅对实数 $x$ 成立，复数情形不保证。
    \item[\textbf{\textcolor{CTrap}{错因分析（条件忽视）：}}] 未注意定理条件 $x \in \mathbb{R}$，盲目推广到复数。
\end{description}
\end{trapbox}
