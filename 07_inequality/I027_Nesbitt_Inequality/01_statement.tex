\textbf{\textcolor{CStatement}{条件}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{条件 1（非零实数）：}}] $a, b$ 为非零实数。
\end{description}

\textbf{\textcolor{CStatement}{结论}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{\textcolor{CStatement}{结论一（同号倒数下界）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 若两数同号，则它们的互为倒数之和不小于 2。
    \[
    \text{若 } ab > 0,\quad \frac{a}{b} + \frac{b}{a} \geq 2.
    \]

    \item[\textbf{\textcolor{CStatement}{结论二（异号倒数上界）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 若两数异号，则它们的互为倒数之和不大于 $-2$。
    \[
    \text{若 } ab < 0,\quad \frac{a}{b} + \frac{b}{a} \leq -2.
    \]

    \item[\textbf{\textcolor{CStatement}{推广结论（正数倒数和）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 正数与其倒数的和不小于 2。
    \[
    \text{若 } x > 0,\quad x + \frac{1}{x} \geq 2.
    \]
\end{description}

\textbf{\textcolor{CStatement}{等号/取等条件：}} 当且仅当 $a = b$（或 $x = 1$）时等号成立。

\end{statementbox}
