\begin{examplebox}
\textbf{\textcolor{CExample}{例 1（基础）}}

\textbf{\textcolor{CExample}{题目：}} 已知 $x > 0$，求 $x + \frac{1}{x}$ 的最小值，并说明取最小值时 $x$ 的值。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（确认条件）：}] 由题，$x > 0$，满足 $ab > 0$ 的条件（这里 $a = x$，$b = 1$，且 $x \cdot 1 > 0$）。
    \par\textbf{理由：} 结论要求 $a,b$ 同号。
    \item[\textbf{第二步（应用结论）：}] 根据互为倒数和不等式，当 $ab > 0$ 时，$\frac{a}{b} + \frac{b}{a} \geq 2$。此处 $a = x$，$b = 1$，所以
    \[
    \frac{x}{1} + \frac{1}{x} = x + \frac{1}{x} \geq 2.
    \]
    \par\textbf{理由：} 直接代入结论。
    \item[\textbf{第三步（取等条件）：}] 等号成立当且仅当 $a = b$，即 $x = 1$。
    \par\textbf{理由：} 结论的等号条件。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{所以 $x + \frac{1}{x}$ 的最小值为 $2$，当且仅当 $x = 1$ 时取得。}

\textbf{\textcolor{CExample}{例 2（稍变形）}}

\textbf{\textcolor{CExample}{题目：}} 设 $a, b$ 为非零实数，且 $ab < 0$，求 $\frac{a}{b} + \frac{b}{a}$ 的取值范围。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（确认条件）：}] 由题，$ab < 0$，满足互为倒数和不等式中 $ab < 0$ 的情形。
    \par\textbf{理由：} 结论中第二种情况的条件。
    \item[\textbf{第二步（应用结论）：}] 根据结论，当 $ab < 0$ 时，有
    \[
    \frac{a}{b} + \frac{b}{a} \leq -2.
    \]
    \par\textbf{理由：} 直接代入结论。
    \item[\textbf{第三步（取等分析）：}] 结论中等号成立的条件为 $a = b$。但在 $ab < 0$ 的条件下，$a = b$ 意味着 $a^2 < 0$，矛盾，故等号无法取得。
    \par\textbf{理由：} 等号条件与已知条件矛盾。
    \item[\textbf{第四步（确定范围）：}] 由于等号取不到，所以实际上 $\frac{a}{b} + \frac{b}{a} < -2$。又因为 $\frac{a}{b} + \frac{b}{a}$ 可以取任意小的值（例如令 $a$ 固定，$b \to 0^+$ 或 $b \to 0^-$），所以取值范围为 $(-\infty, -2)$。
    \par\textbf{理由：} 结合不等式的严格性和函数的变化趋势。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{所以 $\frac{a}{b} + \frac{b}{a}$ 的取值范围是 $(-\infty, -2)$，即它恒小于 $-2$，且可以任意小。}
\end{examplebox}
