\begin{summarybox}
\textbf{一句话核心}

\textbf{三个非负实数的立方和不小于它们乘积的三倍，等号仅在三数相等时成立。}

\medskip
\textbf{使用条件}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{条件 1：} $a, b, c$ 必须是非负实数，否则不等式可能不成立。
    \item \textbf{条件 2：} 不等式适用于对称或轮换对称表达式，便于直接比较。
\end{itemize}

\medskip
\textbf{关键提醒}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{易错点：} 忽视非负条件或误认为取等条件在部分相等时成立。
    \item \textbf{检查项：} 应用前先验证变量非负，并确认取等条件 $a=b=c$ 是否在问题中有意义。
\end{itemize}
\end{summarybox}
