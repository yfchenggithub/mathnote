\begin{summarybox}
\textbf{一句话核心}

\textbf{非负实数的立方和，不小于它们乘积的三倍，取等当三数全等。}

\medskip
\textbf{使用条件}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{条件 1：} 变量 $a, b, c$ 必须全部是非负实数。
    \item \textbf{条件 2：} 适用于比较或放缩形如 $a^3+b^3+c^3$ 与 $abc$ 的表达式。
\end{itemize}

\medskip
\textbf{关键提醒}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{易错点：} 务必注意变量范围，负数情形下结论不成立。
    \item \textbf{检查项：} 应用前先确认是否满足 $a, b, c \geq 0$；取等时严格检查是否 $a=b=c$。
\end{itemize}
\end{summarybox}
