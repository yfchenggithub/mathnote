\begin{summarybox}
\textbf{一句话核心}

\textbf{三个非负实数的立方和不小于三倍它们的乘积。}

\medskip
\textbf{使用条件}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{条件 1（变量非负）：} $a, b, c \geq 0$，必须满足非负性。
    \item \textbf{条件 2（实数范围）：} $a, b, c$ 为实数，但非负是关键。
\end{itemize}

\medskip
\textbf{关键提醒}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{易错点（检查非负）：} 应用前务必检查所有变量是否非负。
    \item \textbf{检查项（取等条件）：} 等号仅当 $a=b=c$ 时成立，确保对称性。
\end{itemize}
\end{summarybox}
