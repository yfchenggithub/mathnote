\begin{summarybox}
\textbf{\textcolor{CSummary}{一句话核心}}

\textbf{对于非负实数 $a, b, c$，有 $a^3+b^3+c^3 \geq 3abc$，等号当且仅当 $a=b=c$。}

\medskip
\textbf{\textcolor{CSummary}{使用条件}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{条件 1（非负实数）：} $a, b, c \geq 0$。
    \item \textbf{条件 2（对称结构）：} 表达式为对称立方和与乘积。
\end{itemize}

\medskip
\textbf{\textcolor{CSummary}{关键提醒}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{易错点（符号忽视）：} 使用前务必检查变量是否非负。
    \item \textbf{检查项（取等验证）：} 最值问题中验证等号能否达到。
\end{itemize}
\end{summarybox}
