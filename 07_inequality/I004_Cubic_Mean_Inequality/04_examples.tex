\begin{examplebox}
\textbf{\textcolor{CExample}{例 1（基础）}}

\textbf{\textcolor{CExample}{题目：}} 取 $a=1, b=2, c=3$，验证不等式 $a^3+b^3+c^3 \geq 3abc$ 成立。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（代入数值）：}] 将具体值代入左边和右边。
    \[
    a^3+b^3+c^3 = 1^3 + 2^3 + 3^3, \quad 3abc = 3 \times 1 \times 2 \times 3
    \]
    \par\textbf{理由：}
    直接计算各立方项和乘积。

    \item[\textbf{第二步（计算数值）：}] 分别计算数值。
    \[
    1^3 + 2^3 + 3^3 = 1 + 8 + 27 = 36, \quad 3 \times 1 \times 2 \times 3 = 18
    \]
    \par\textbf{理由：}
    执行算术运算。

    \item[\textbf{第三步（比较大小）：}] 比较两个结果。
    \[
    36 \geq 18
    \]
    \par\textbf{理由：}
    由于 $36$ 大于 $18$，不等式成立。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{对于这些具体值，不等式成立，符合一般结论。}

\textbf{\textcolor{CExample}{例 2（稍变形）}}

\textbf{\textcolor{CExample}{题目：}} 已知 $a, b, c > 0$ 且 $abc=1$，证明 $a^3+b^3+c^3 \geq 3$。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（应用条件）：}] 由 $abc=1$，得 $3abc = 3$。
    \[
    3abc = 3 \times 1 = 3
    \]
    \par\textbf{理由：}
    将乘积条件转化为常数。

    \item[\textbf{第二步（使用不等式）：}] 根据原不等式，有 $a^3+b^3+c^3 \geq 3abc$。
    \[
    a^3+b^3+c^3 \geq 3abc
    \]
    \par\textbf{理由：}
    直接应用三项立方和不等式。

    \item[\textbf{第三步（代入结论）：}] 将 $3abc=3$ 代入不等式。
    \[
    a^3+b^3+c^3 \geq 3
    \]
    \par\textbf{理由：}
    结合前两步即得所求。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{在 $abc=1$ 的条件下，$a^3+b^3+c^3$ 的最小值为 $3$，当且仅当 $a=b=c=1$ 时取到。}
\end{examplebox}
