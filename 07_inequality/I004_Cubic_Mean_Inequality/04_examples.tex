\begin{examplebox}
\textbf{例 1（基础）}

\textbf{题目：} 已知 $a, b, c$ 是非负实数，且 $abc=1$，证明 $a^3+b^3+c^3 \geq 3$。

\textbf{解题步骤：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步：}] 应用三项立方和不等式 $a^3+b^3+c^3 \geq 3abc$。（理由：直接使用已知结论）
    \item[\textbf{第二步：}] 代入条件 $abc=1$，得到 $a^3+b^3+c^3 \geq 3 \times 1 = 3$。（理由：将已知值代入不等式右侧）
    \item[\textbf{第三步：}] 检查取等条件：当 $a=b=c$ 且 $abc=1$ 时，即 $a=b=c=1$，等号成立。（理由：验证结论的严谨性）
\end{description}

\textbf{关键结论：} \textbf{在 $abc=1$ 的条件下，$a^3+b^3+c^3$ 的最小值为 $3$，当 $a=b=c=1$ 时取到。}

\medskip
\textbf{例 2（稍变形）}

\textbf{题目：} 设 $x, y, z$ 是非负实数，证明 $x^6+y^6+z^6 \geq 3x^2y^2z^2$。

\textbf{解题步骤：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步：}] 令 $a=x^2, b=y^2, c=z^2$，则 $a, b, c$ 为非负实数。（理由：变量替换以匹配原不等式形式）
    \item[\textbf{第二步：}] 对 $a, b, c$ 应用原不等式：$a^3+b^3+c^3 \geq 3abc$。（理由：直接套用）
    \item[\textbf{第三步：}] 将 $a, b, c$ 代回，即 $(x^2)^3+(y^2)^3+(z^2)^3 \geq 3(x^2)(y^2)(z^2)$，化简得 $x^6+y^6+z^6 \geq 3x^2y^2z^2$。（理由：还原变量并简化）
\end{description}

\textbf{关键结论：} \textbf{通过平方形式的变量替换，原不等式可推广为 $x^6+y^6+z^6 \geq 3x^2y^2z^2$，取等条件为 $x=y=z$。}
\end{examplebox}
