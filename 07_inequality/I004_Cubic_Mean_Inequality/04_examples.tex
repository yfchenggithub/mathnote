\begin{examplebox}
\textbf{例 1（基础）}

\textbf{题目：} 已知 $x, y, z \geq 0$ 且 $xyz=1$，求证 $x^3 + y^3 + z^3 \geq 3$。

\textbf{解题步骤：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（应用不等式）：}] 由三项立方和不等式，对非负 $x, y, z$ 有
    \[
    x^3 + y^3 + z^3 \geq 3xyz.
    \]
    \par\textbf{理由：}
    直接套用结论，其中 $a=x, b=y, c=z$。

    \item[\textbf{第二步（代入条件）：}] 已知 $xyz=1$，代入上式右边。
    \[
    x^3 + y^3 + z^3 \geq 3 \times 1 = 3.
    \]
    \par\textbf{理由：}
    将条件 $xyz=1$ 代入不等式 $x^3+y^3+z^3 \geq 3xyz$。

    \item[\textbf{第三步（得出结论）：}] 因此原不等式成立。
    \[
    x^3 + y^3 + z^3 \geq 3.
    \]
    \par\textbf{理由：}
    结合以上两步，得证。
\end{description}

\textbf{关键结论：} \textbf{在 $xyz=1$ 的条件下，立方和 $x^3+y^3+z^3$ 的最小值为 $3$，当且仅当 $x=y=z=1$ 时取到。}

\medskip
\textbf{例 2（稍变形）}

\textbf{题目：} 设 $a, b, c \geq 0$，证明 $\dfrac{a^3 + b^3 + c^3}{3} \geq abc$。

\textbf{解题步骤：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（写出原式）：}] 从三项立方和不等式出发。
    \[
    a^3 + b^3 + c^3 \geq 3abc.
    \]
    \par\textbf{理由：}
    这是已知结论。

    \item[\textbf{第二步（变形不等式）：}] 将不等式两边同时除以 $3$。
    \[
    \frac{a^3 + b^3 + c^3}{3} \geq abc.
    \]
    \par\textbf{理由：}
    除以正数 $3$ 不改变不等号方向。

    \item[\textbf{第三步（确认等号）：}] 等号成立当且仅当 $a=b=c$。
    \par\textbf{理由：}
    原不等式等号条件为 $a=b=c$，除以 $3$ 不影响等号条件。
\end{description}

\textbf{关键结论：} \textbf{立方和的算术平均不小于乘积 $abc$，等号在 $a=b=c$ 时成立。}
\end{examplebox}
