\begin{examplebox}
\textbf{例 1（基础）}

\textbf{题目：} 已知 $x, y, z \geq 0$，求证：$x^3 + y^3 + z^3 \geq 3xyz$。

\textbf{解题步骤：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步：}] 识别题目结构与已知结论。
    \par\textbf{理由：}
    题中 $x, y, z$ 均为非负实数，结论形式与 $a^3+b^3+c^3 \geq 3abc$ 完全一致，故可直接应用。

    \item[\textbf{第二步：}] 直接写出结论。
    \[
    \text{由三项立方和不等式，得 } x^3 + y^3 + z^3 \geq 3xyz。
    \]
    \par\textbf{理由：}
    令 $a=x, b=y, c=z$，直接代入结论即可。

    \item[\textbf{第三步：}] 说明取等条件。
    \[
    \text{等号成立当且仅当 } x = y = z。
    \]
    \par\textbf{理由：}
    这是原结论的一部分，必须明确指出。
\end{description}

\textbf{关键结论：} \textbf{对于非负实数，立方和不等式可直接作为放缩依据。}

\medskip
\textbf{例 2（稍变形）}

\textbf{题目：} 设 $a, b, c > 0$，且 $abc = 1$。证明：$a^3 + b^3 + c^3 \geq 3$。

\textbf{解题步骤：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步：}] 利用条件 $abc=1$ 将目标与已知结论联系起来。
    \par\textbf{理由：}
    已知结论右边是 $3abc$，而 $abc=1$，所以 $3abc = 3$。

    \item[\textbf{第二步：}] 应用三项立方和不等式。
    \[
    a^3 + b^3 + c^3 \geq 3abc
    \]
    \par\textbf{理由：}
    由于 $a, b, c > 0$，满足不等式条件。

    \item[\textbf{第三步：}] 代入 $abc=1$，得到最终结论。
    \[
    a^3 + b^3 + c^3 \geq 3 \times 1 = 3
    \]
    \par\textbf{理由：}
    将已知条件代入不等式右端，完成证明。等号成立当且仅当 $a=b=c=1$。
\end{description}

\textbf{关键结论：} \textbf{当已知乘积为常数时，此不等式可用于证明立方和的最小值。}
\end{examplebox}
