\begin{trapbox}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CTrap}{易错点一（忽略非负条件）}}] 认为不等式对所有实数 $a, b, c$ 都成立。
    \item[\textbf{\textcolor{CTrap}{正确理解（变量范围）：}}] 不等式仅对非负实数成立，负数时可能反向。
    \item[\textbf{\textcolor{CTrap}{错因分析（符号影响）：}}] 当 $a, b, c$ 为负时，因式分解中 $a+b+c$ 可能为负，导致不等式不成立。
\end{description}

\medskip
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CTrap}{易错点二（取等误解）}}] 认为只要有两个变量相等即可取等号。
    \item[\textbf{\textcolor{CTrap}{正确理解（全等条件）：}}] 等号成立当且仅当 $a=b=c$，三者全相等。
    \item[\textbf{\textcolor{CTrap}{错因分析（对称性误解）：}}] 由于不等式对称，但取等要求平方和为零，必须所有差为零。
\end{description}

\medskip
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CTrap}{易错点三（错误推广）}}] 将不等式推广到 $a^4+b^4+c^4 \geq 3abc$ 或其他幂次。
    \item[\textbf{\textcolor{CTrap}{正确理解（幂次特定）：}}] 该不等式只针对三次方成立，高次需另证。
    \item[\textbf{\textcolor{CTrap}{错因分析（形式混淆）：}}] 不同幂次的不等式结构不同，不能直接类比。
\end{description}
\end{trapbox}
