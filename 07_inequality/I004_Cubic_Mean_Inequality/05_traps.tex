\begin{trapbox}
\textbf{易错点一}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{错误想法：}] 认为 $a, b, c$ 可以是任意实数。
    \item[\textbf{正确理解：}] 结论成立的前提是 $a, b, c \geq 0$，即必须是非负实数。
    \item[\textbf{错因分析：}] 证明过程中依赖于 $a+b+c \geq 0$ 以及配方恒等式。若 $a, b, c$ 中有负数，$a+b+c$ 可能为负，而平方和仍非负，乘积符号无法确定，不等式不一定成立。
\end{description}

\medskip
\textbf{易错点二}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{错误想法：}] 认为等号成立的条件是 $a, b, c$ 中任意两个相等即可。
    \item[\textbf{正确理解：}] 等号成立当且仅当 $a = b = c$，即三个数必须全相等。
    \item[\textbf{错因分析：}] 从证明的配方形式 $\frac{1}{2}[(a-b)^2+(b-c)^2+(c-a)^2]=0$ 可知，必须每一项的平方都为零，故 $a=b, b=c, c=a$，缺一不可。
\end{description}

\medskip
\textbf{易错点三}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{错误想法：}] 将此不等式与算术-几何平均不等式 $\frac{a+b+c}{3} \geq \sqrt[3]{abc}$ 混淆，认为两者可以直接互推。
    \item[\textbf{正确理解：}] 两者是不同的不等式。虽然都涉及非负实数和取等条件 $a=b=c$，但形式不同，不能通过简单代数变换直接相互推导。
    \item[\textbf{错因分析：}] 算术-几何平均不等式两边立方后得到 $(a+b+c)^3 \geq 27abc$，这与 $a^3+b^3+c^3 \geq 3abc$ 结构不同，前者包含交叉项，后者没有。它们是独立但有关联的结论。
\end{description}
\end{trapbox}
