\begin{statementbox}

\textbf{条件：} $a, b, c$ 均为非负实数，即 $a \geq 0,\, b \geq 0,\, c \geq 0$。

\textbf{结论：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{结论 1：}]
    \textbf{直观描述：} 三数立方和不小于它们乘积的三倍。
    \[
    a^3 + b^3 + c^3 \geq 3abc
    \]
\end{description}

\textbf{等号/取等条件：} 当且仅当 $a = b = c$ 时，等号成立。

\end{statementbox}
