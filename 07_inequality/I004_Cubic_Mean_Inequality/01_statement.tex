\textbf{\textcolor{CStatement}{条件}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{条件 1（变量范围）：}}] $a, b, c$ 为非负实数。
\end{description}

\textbf{\textcolor{CStatement}{结论}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{\textcolor{CStatement}{结论一（立方和不等式）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 三项立方和不小于三倍乘积。
    \[
    a^3 + b^3 + c^3 \geq 3abc
    \]

    \item[\textbf{\textcolor{CStatement}{推广结论（因式分解）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 不等式等价于平方和非负。
    \[
    a^3 + b^3 + c^3 - 3abc = \frac{1}{2}(a+b+c)\bigl[(a-b)^2+(b-c)^2+(c-a)^2\bigr] \geq 0
    \]
\end{description}

\textbf{\textcolor{CStatement}{等号/取等条件：}} 当且仅当 $a = b = c$。
