\begin{proofbox}
\textbf{\textcolor{CProof}{思路提示}}

利用柯西不等式将原式与平方和联系起来，再通过基本不等式放缩得到下界。

\medskip
\textbf{\textcolor{CProof}{正式推导}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.55em]
    \item[\textbf{步骤一（应用柯西）：}] 由柯西不等式，
    \[
    \left( \frac{a}{b+c} + \frac{b}{c+a} + \frac{c}{a+b} \right) \big( a(b+c) + b(c+a) + c(a+b) \big) \ge (a+b+c)^2.
    \]
    \par\textbf{理由：}
    取柯西不等式形式 $\left( \sum \frac{x_i^2}{y_i} \right) \left( \sum y_i \right) \ge \left( \sum x_i \right)^2$，其中 $x_1=a, y_1=b+c$ 等。

    \item[\textbf{步骤二（计算分母和）：}] 计算 $a(b+c)+b(c+a)+c(a+b)$：
    \[
    a(b+c)+b(c+a)+c(a+b) = ab+ac+bc+ba+ca+cb = 2(ab+bc+ca).
    \]
    \par\textbf{理由：}
    展开并合并同类项。

    \item[\textbf{步骤三（代入整理）：}] 令 $T = \frac{a}{b+c} + \frac{b}{c+a} + \frac{c}{a+b}$，则
    \[
    T \cdot 2(ab+bc+ca) \ge (a+b+c)^2.
    \]
    \par\textbf{理由：}
    将步骤二结果代入步骤一。

    \item[\textbf{步骤四（分离 T）：}] 由上式，
    \[
    T \ge \frac{(a+b+c)^2}{2(ab+bc+ca)}.
    \]
    \par\textbf{理由：}
    两边除以正数 $2(ab+bc+ca)$。

    \item[\textbf{步骤五（证明下界）：}] 需要证明 $\frac{(a+b+c)^2}{2(ab+bc+ca)} \ge \frac{3}{2}$，即
    \[
    (a+b+c)^2 \ge 3(ab+bc+ca).
    \]
    展开得 $a^2+b^2+c^2+2(ab+bc+ca) \ge 3(ab+bc+ca)$，所以
    \[
    a^2+b^2+c^2 \ge ab+bc+ca.
    \]
    \par\textbf{理由：}
    由基本不等式，$a^2+b^2 \ge 2ab$，$b^2+c^2 \ge 2bc$，$c^2+a^2 \ge 2ca$，求和得 $2(a^2+b^2+c^2) \ge 2(ab+bc+ca)$，即 $a^2+b^2+c^2 \ge ab+bc+ca$。

    \item[\textbf{步骤六（得出结论）：}] 因此，
    \[
    T \ge \frac{3}{2}.
    \]
    等号成立当且仅当 $a=b=c$。
    \par\textbf{理由：}
    步骤五中等号成立当 $a=b=c$，且步骤一柯西不等式等号成立条件相同。
\end{description}

\medskip
\textbf{\textcolor{CProof}{结论回扣}}

\textbf{从而证明了 Nesbitt 不等式：对于任意正实数 $a,b,c$，有 $\frac{a}{b+c} + \frac{b}{c+a} + \frac{c}{a+b} \ge \frac{3}{2}$，等号当且仅当 $a=b=c$。}
\end{proofbox}
