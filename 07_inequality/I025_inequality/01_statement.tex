\textbf{\textcolor{CStatement}{条件}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{条件 1（正实数）：}}] 变量 $a,b,c$ 均为正实数，即 $a>0,\ b>0,\ c>0$。
\end{description}

\textbf{\textcolor{CStatement}{结论}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{\textcolor{CStatement}{结论一（核心形式）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 三个对称分数之和的最小值为 $\frac{3}{2}$。
    \[
    \frac{a}{b+c} + \frac{b}{c+a} + \frac{c}{a+b} \ge \frac{3}{2}.
    \]

    \item[\textbf{\textcolor{CStatement}{推广结论（和表示形式）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 用总和 $s=a+b+c$ 改写分母的等价形式。
    \[
    \frac{a}{s-a} + \frac{b}{s-b} + \frac{c}{s-c} \ge \frac{3}{2}.
    \]
\end{description}

\textbf{\textcolor{CStatement}{等号/取等条件：}} 当且仅当 $a = b = c$ 时，上述不等式等号成立。

\end{statementbox}
