\begin{examplebox}
\textbf{\textcolor{CExample}{例 1（基础）}}

\textbf{\textcolor{CExample}{题目：}} 设 $a,b,c$ 为正实数，证明 $\frac{a}{b+c} + \frac{b}{c+a} + \frac{c}{a+b} \ge \frac{3}{2}$。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（确认条件）：}] 由题设，$a>0,\ b>0,\ c>0$，满足 Nesbitt 不等式的条件。
    \par\textbf{理由：}
    Nesbitt 不等式要求变量为正实数。

    \item[\textbf{第二步（应用不等式）：}] 直接应用 Nesbitt 不等式的核心形式：
    \[
    \frac{a}{b+c} + \frac{b}{c+a} + \frac{c}{a+b} \ge \frac{3}{2}.
    \]
    \par\textbf{理由：}
    这是已知结论，无需额外推导。

    \item[\textbf{第三步（检查取等）：}] 等号成立当且仅当 $a=b=c$。
    \par\textbf{理由：}
    根据不等式等号条件。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{对于任意正实数 $a,b,c$，总有 $\frac{a}{b+c} + \frac{b}{c+a} + \frac{c}{a+b} \ge \frac{3}{2}$，等号当 $a=b=c$。}

\medskip
\textbf{\textcolor{CExample}{例 2（稍变形）}}

\textbf{\textcolor{CExample}{题目：}} 已知 $x,y,z$ 为正实数，且 $x+y+z=3$，求 $\frac{x}{3-x} + \frac{y}{3-y} + \frac{z}{3-z}$ 的最小值。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（设和变量）：}] 令 $s = x+y+z = 3$，则 $3-x = s-x = y+z$，类似地 $3-y = z+x$，$3-z = x+y$。
    \par\textbf{理由：}
    引入总和以匹配推广形式。

    \item[\textbf{第二步（改写表达式）：}] 原式化为
    \[
    \frac{x}{3-x} + \frac{y}{3-y} + \frac{z}{3-z} = \frac{x}{s-x} + \frac{y}{s-y} + \frac{z}{s-z}.
    \]
    \par\textbf{理由：}
    代入 $s=3$。

    \item[\textbf{第三步（应用推广形式）：}] 由 Nesbitt 不等式的推广形式，有
    \[
    \frac{x}{s-x} + \frac{y}{s-y} + \frac{z}{s-z} \ge \frac{3}{2}.
    \]
    \par\textbf{理由：}
    推广形式要求 $s=x+y+z$，且变量为正，此处满足。

    \item[\textbf{第四步（得出最小值）：}] 因此，原式的最小值为 $\frac{3}{2}$，当且仅当 $x=y=z=1$ 时取得。
    \par\textbf{理由：}
    等号成立条件为 $x=y=z$，结合 $x+y+z=3$ 得 $x=y=z=1$。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{在 $x+y+z=3$ 的条件下，$\frac{x}{3-x} + \frac{y}{3-y} + \frac{z}{3-z}$ 的最小值为 $\frac{3}{2}$，当 $x=y=z=1$ 时取到。}
\end{examplebox}
