\textbf{\textcolor{CStatement}{条件}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{条件 1（正实数）：}}] $a, b \in \mathbb{R}^{+}$（$a$和$b$均为正实数）。
\end{description}

\textbf{\textcolor{CStatement}{结论}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{\textcolor{CStatement}{结论一（四均值不等式链）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}}两个正实数的四种平均数存在确定的大小序关系。
    \[
    \begin{aligned}
    \frac{2}{\dfrac{1}{a}+\dfrac{1}{b}} &\leq \sqrt{ab} \\
    &\leq \frac{a+b}{2} \\
    &\leq \sqrt{\frac{a^{2}+b^{2}}{2}}.
    \end{aligned}
    \]

    \item[\textbf{\textcolor{CStatement}{推广结论（n 元不等式链）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}}对于 $n$ 个正实数，四种平均数之间亦有类似序关系。
    \[
    \begin{aligned}
    \frac{n}{\frac{1}{x_1}+\cdots+\frac{1}{x_n}} &\leq \sqrt[n]{x_1 x_2 \cdots x_n} \\
    &\leq \frac{x_1+x_2+\cdots+x_n}{n} \\
    &\leq \sqrt{\frac{x_1^2+x_2^2+\cdots+x_n^2}{n}}.
    \end{aligned}
    \]
\end{description}

\textbf{\textcolor{CStatement}{等号/取等条件：}}所有不等式中等号成立当且仅当所有变量相等（二元时即 $a = b$）。

\end{statementbox}
