    \item[\textbf{条件 1（正实数）：}] $a, b$ 为正实数。

\textbf{结论：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{结论一（均值不等式链）：}]
    \textbf{直观描述：} 调和平均最小，几何平均次之，算术平均再次，平方平均最大。
    \[
    H(a,b) \leq G(a,b) \leq A(a,b) \leq Q(a,b)
    \]
\end{description}

\textbf{等号/取等条件：} 当且仅当 $a=b$ 时所有等号同时成立。
