\begin{proofbox}
\textbf{思路提示}

从基本不等式出发，先证 $G \leq A$，再通过倒数关系证 $H \leq G$，最后用平方差证 $A \leq Q$，全程注意取等条件。

\medskip
\textbf{正式推导}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.55em]
    \item[\textbf{步骤一（证 $G \leq A$）：}] 
    \[
    (\sqrt{a} - \sqrt{b})^2 \geq 0 \;\Rightarrow\; a - 2\sqrt{ab} + b \geq 0 \;\Rightarrow\; \sqrt{ab} \leq \frac{a+b}{2}.
    \]
    \par\textbf{理由：} 平方非负，展开后移项即得 $G(a,b) \leq A(a,b)$。

    \item[\textbf{步骤二（证 $H \leq G$）：}] 
    将 $G \leq A$ 应用于 $\frac{1}{a}, \frac{1}{b}$，得 $\sqrt{\frac{1}{a}\cdot\frac{1}{b}} \leq \frac{\frac{1}{a}+\frac{1}{b}}{2}$，即 $\frac{1}{\sqrt{ab}} \leq \frac{1}{2}\bigl(\frac{1}{a}+\frac{1}{b}\bigr)$。
    \[
    \text{取倒数得 } \sqrt{ab} \geq \frac{2}{\frac{1}{a}+\frac{1}{b}} \;\Rightarrow\; G(a,b) \geq H(a,b).
    \]
    \par\textbf{理由：} 倒数运算反转不等号方向，故 $H \leq G$。

    \item[\textbf{步骤三（证 $A \leq Q$）：}] 
    由 $(a-b)^2 \geq 0$ 得 $a^2+b^2 \geq 2ab$，两边加 $a^2+b^2$ 得 $2(a^2+b^2) \geq (a+b)^2$，即 $\frac{a^2+b^2}{2} \geq \bigl(\frac{a+b}{2}\bigr)^2$。
    \[
    \text{开方得 } \sqrt{\frac{a^2+b^2}{2}} \geq \frac{a+b}{2} \;\Rightarrow\; Q(a,b) \geq A(a,b).
    \]
    \par\textbf{理由：} 平方差非负推导出平方和与和的关系，开方保持不等号。

    \item[\textbf{步骤四（取等条件）：}] 
    步骤一等号成立当且仅当 $\sqrt{a}=\sqrt{b}$ 即 $a=b$；步骤二等号成立当且仅当 $\frac{1}{a}=\frac{1}{b}$ 即 $a=b$；步骤三等号成立当且仅当 $a=b$。因此整个链等号成立当且仅当 $a=b$。
    \par\textbf{理由：} 各步等号条件一致，传递后整体取等条件相同。
\end{description}

\medskip
\textbf{结论回扣}

\textbf{综上，严格证明了对于任意正实数 $a,b$，有 $H(a,b) \leq G(a,b) \leq A(a,b) \leq Q(a,b)$，且等号成立当且仅当 $a=b$。}
\end{proofbox}
