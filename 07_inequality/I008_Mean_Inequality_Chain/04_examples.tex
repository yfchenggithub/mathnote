\begin{examplebox}
\textbf{例 1（基础）}

\textbf{题目：} 设 $a, b$ 为正实数，求证 $\dfrac{2ab}{a+b} \leq \dfrac{a+b}{2}$。

\textbf{解题步骤：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（识别平均数）：}] 左边 $\dfrac{2ab}{a+b} = H(a,b)$，右边 $\dfrac{a+b}{2} = A(a,b)$。
    \par\textbf{理由：} 调和平均数 $H(a,b) = \dfrac{2}{\frac{1}{a}+\frac{1}{b}} = \dfrac{2ab}{a+b}$，算术平均数 $A(a,b) = \frac{a+b}{2}$。

    \item[\textbf{第二步（应用不等式）：}] 由均值不等式链，$H(a,b) \leq A(a,b)$。
    \par\textbf{理由：} 因为 $H \leq G \leq A$，所以 $H \leq A$。

    \item[\textbf{第三步（得出结论）：}] 因此，$\dfrac{2ab}{a+b} \leq \dfrac{a+b}{2}$。
    \par\textbf{理由：} 直接代入即得不等式。
\end{description}

\textbf{关键结论：} \textbf{对于正实数 $a,b$，调和平均不超过算术平均，即 $\frac{2ab}{a+b} \leq \frac{a+b}{2}$，等号当 $a=b$ 时成立。}

\textbf{例 2（稍变形）}

\textbf{题目：} 设 $a, b$ 为正实数，且 $a+b=2$。求 $G(a,b)$ 的最大值，并证明 $Q(a,b) \geq 1$。

\textbf{解题步骤：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（求 $G$ 最大值）：}] 由 $a+b=2$ 得 $A(a,b) = \frac{a+b}{2} = 1$。根据 $G(a,b) \leq A(a,b)$，有 $G(a,b) \leq 1$。
    \par\textbf{理由：} 算术平均数为定值，几何平均数不超过算术平均数。

    \item[\textbf{第二步（取等条件）：}] 等号成立当且仅当 $a=b$，此时 $a=b=1$，$G(1,1)=1$。所以 $G(a,b)$ 的最大值为 $1$。
    \par\textbf{理由：} 当 $a=b$ 时 $G=A$，且 $a+b=2$ 推出 $a=b=1$。

    \item[\textbf{第三步（证 $Q \geq 1$）：}] 由 $A(a,b)=1$ 和 $A(a,b) \leq Q(a,b)$，得 $Q(a,b) \geq 1$。
    \par\textbf{理由：} 算术平均数不超过平方平均数，且此处算术平均数为 $1$。
\end{description}

\textbf{关键结论：} \textbf{当 $a+b=2$ 时，$G(a,b)$ 的最大值为 $1$，且 $Q(a,b) \geq 1$，等号成立当 $a=b=1$。}
\end{examplebox}
