\begin{summarybox}
\textbf{一句话核心}

\textbf{对于正实数 $a,b$，调和、几何、算术、平方平均数满足 $H \leq G \leq A \leq Q$，等号成立当且仅当 $a=b$。}

\medskip
\textbf{使用条件}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{条件 1（正实数）：} $a, b$ 必须为正实数。
    \item \textbf{条件 2（取等一致）：} 等号成立当且仅当 $a=b$。
\end{itemize}

\medskip
\textbf{关键提醒}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{易错点（忽略正数）：} 使用前务必检查 $a,b>0$，否则不等式可能不成立。
    \item \textbf{检查项（验证取等）：} 解题后验证等号是否可达，确保最值或不等式紧性。
\end{itemize}
\end{summarybox}
