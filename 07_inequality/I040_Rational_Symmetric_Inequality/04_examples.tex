\begin{examplebox}
\textbf{\textcolor{CExample}{例 1（基础）}}

\textbf{\textcolor{CExample}{题目：}} 设 $a, b, c > 0$，且 $a+b+c=1$，求 $\frac{a^2}{b+c} + \frac{b^2}{c+a} + \frac{c^2}{a+b}$ 的最小值。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（应用不等式）：}] 由三元权方和不等式，
    \[
    \frac{a^2}{b+c} + \frac{b^2}{c+a} + \frac{c^2}{a+b} \ge \frac{a+b+c}{2}.
    \]
    \par\textbf{理由：}
    直接应用定理。

    \item[\textbf{第二步（代入条件）：}] 已知 $a+b+c=1$，所以
    \[
    \frac{a^2}{b+c} + \frac{b^2}{c+a} + \frac{c^2}{a+b} \ge \frac{1}{2}.
    \]
    \par\textbf{理由：}
    将条件代入不等式。

    \item[\textbf{第三步（取等检查）：}] 当且仅当 $a=b=c$ 时等号成立，结合 $a+b+c=1$，得 $a=b=c=\frac{1}{3}$。
    \par\textbf{理由：}
    等号成立条件为 $a=b=c$，代入条件得具体值。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{最小值为 $\frac{1}{2}$，当 $a=b=c=\frac{1}{3}$ 时取得。}

\textbf{\textcolor{CExample}{例 2（稍变形）}}

\textbf{\textcolor{CExample}{题目：}} 设 $a, b, c > 0$，且 $abc=1$，证明 $\frac{a^2}{b+c} + \frac{b^2}{c+a} + \frac{c^2}{a+b} \ge \frac{3}{2}$。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（应用权方和）：}] 由三元权方和不等式，
    \[
    \frac{a^2}{b+c} + \frac{b^2}{c+a} + \frac{c^2}{a+b} \ge \frac{a+b+c}{2}.
    \]
    \par\textbf{理由：}
    直接应用定理。

    \item[\textbf{第二步（均值不等式）：}] 由均值不等式，$a+b+c \ge 3\sqrt[3]{abc} = 3$。
    \[
    \frac{a+b+c}{2} \ge \frac{3}{2}.
    \]
    \par\textbf{理由：}
    已知 $abc=1$，所以 $\sqrt[3]{abc}=1$，从而 $a+b+c \ge 3$。

    \item[\textbf{第三步（结合得证）：}] 因此，
    \[
    \frac{a^2}{b+c} + \frac{b^2}{c+a} + \frac{c^2}{a+b} \ge \frac{a+b+c}{2} \ge \frac{3}{2}.
    \]
    \par\textbf{理由：}
    由第一步和第二步，传递不等式。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{不等式成立，当且仅当 $a=b=c=1$ 时等号成立。}
\end{examplebox}
