\textbf{\textcolor{CStatement}{条件}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{条件 1（正实数）：}}] $a, b, c$ 为正实数。
\end{description}

\textbf{\textcolor{CStatement}{结论}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{\textcolor{CStatement}{结论一（三元权方和）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 平方项除以两数和的下界为半和。
    \[
    \frac{a^{2}}{b+c} + \frac{b^{2}}{c+a} + \frac{c^{2}}{a+b} \geq \frac{a+b+c}{2}
    \]

    \item[\textbf{\textcolor{CStatement}{推广结论（n元形式）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 各项平方除以其余项和的下界为总和除以 $n-1$。
    \[
    \sum_{i=1}^{n} \frac{a_i^{2}}{\sum_{j\neq i} a_j} \geq \frac{\sum_{i=1}^{n} a_i}{n-1}, \quad a_i > 0
    \]
\end{description}

\textbf{\textcolor{CStatement}{等号/取等条件：}} 当且仅当 $a = b = c$ 时等号成立（对于推广，当且仅当所有 $a_i$ 相等）。

\end{statementbox}
