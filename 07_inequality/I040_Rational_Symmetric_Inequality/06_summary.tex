\begin{summarybox}
\textbf{\textcolor{CSummary}{一句话核心}}

\textbf{对于正实数 $a, b, c$，平方项除以两数和的和小下界为半和。}

\medskip
\textbf{\textcolor{CSummary}{使用条件}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{条件 1（正实数）：} $a, b, c > 0$。
    \item \textbf{条件 2（分母形式）：} 分母必须为 $b+c$, $c+a$, $a+b$ 的形式。
\end{itemize}

\medskip
\textbf{\textcolor{CSummary}{关键提醒}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{易错点（取等条件）：} 等号成立当且仅当 $a=b=c$。
    \item \textbf{检查项（分母匹配）：} 应用前检查分母是否匹配定理要求。
\end{itemize}
\end{summarybox}
