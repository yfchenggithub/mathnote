\begin{proofbox}
\textbf{思路提示}

从 $N < f(x) < M$ 出发，通过移项、配方、取倒数等基本代数操作，可系统地导出四种等价形式。关键点是保持等价变形，注意不等号方向。

\medskip
\textbf{正式推导}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.55em]
    \item[\textbf{步骤一：}] 推导乘积形式。
    \[
    N < f(x) < M \iff f(x) - N > 0 \text{ 且 } M - f(x) > 0
    \]
    \textbf{理由：} 将连不等式拆为两个独立不等式。
    \[
    \iff \bigl(f(x) - N\bigr)\bigl(M - f(x)\bigr) > 0
    \]
    \textbf{理由：} 两正数乘积为正。注意 $M - f(x) = -\bigl(f(x) - M\bigr)$，所以
    \[
    \iff \bigl[f(x) - M\bigr]\bigl[f(x) - N\bigr] < 0
    \]
    \textbf{理由：} 提取负号，不等号反向。

    \item[\textbf{步骤二：}] 推导绝对值形式。
    \[
    N < f(x) < M \iff f(x) - \frac{M+N}{2} \text{ 的绝对值小于 } \frac{M-N}{2}
    \]
    \textbf{理由：} 区间中点与半径的几何表示。具体地：
    \[
    \left|\, f(x) - \frac{M+N}{2} \,\right| < \frac{M-N}{2}
    \]
    \textbf{理由：} 直接由 $N < f(x) < M$ 通过移项和绝对值定义可得。

    \item[\textbf{步骤三：}] 推导分式形式。
    \[
    N < f(x) < M \iff f(x)-N > 0 \text{ 且 } M-f(x) > 0
    \]
    \textbf{理由：} 同步骤一。
    \[
    \iff \frac{f(x)-N}{M-f(x)} > 0
    \]
    \textbf{理由：} 两正数之比为正。

    \item[\textbf{步骤四：}] 推导倒数形式。
    \[
    N < f(x) < M \iff f(x)-N > 0 \text{ 且 } M-N > 0
    \]
    \textbf{理由：} $M>N$ 所以 $M-N>0$。由于 $f(x)-N < M-N$，且均为正，取倒数得：
    \[
    \frac{1}{f(x)-N} > \frac{1}{M-N}
    \]
    \textbf{理由：} 正数范围内，倒数函数单调递减，所以较大的原数对应较小的倒数。
\end{description}

\medskip
\textbf{结论回扣}

\textbf{以上推导表明，连不等式 $N < f(x) < M$ 与所列四种形式完全等价，可根据解题需要灵活选用。}
\end{proofbox}
