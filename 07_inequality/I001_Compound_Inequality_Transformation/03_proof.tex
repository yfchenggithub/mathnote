\begin{proofbox}
\textbf{思路提示}

从 $N < f(x) < M$ 出发，通过代数变形（如移项、乘除、绝对值）得到其他形式。注意变形过程中的符号保持。

\medskip
\textbf{正式推导}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.55em]
    \item[\textbf{步骤一（乘积形式推导）：}] 
    由 $N < f(x) < M$，可得 $f(x)-N > 0$ 且 $f(x)-M < 0$。
    \[
    (f(x)-M)(f(x)-N) < 0
    \]
    \par\textbf{理由：}
    正数与负数相乘得负数。

    \item[\textbf{步骤二（绝对值形式推导）：}]
    将 $N < f(x) < M$ 改写为关于中点的形式。
    \[
    \begin{aligned}
    N &< f(x) < M \\
    \Rightarrow f(x) - \frac{M+N}{2} &> N - \frac{M+N}{2} = -\frac{M-N}{2}, \\
    f(x) - \frac{M+N}{2} &< M - \frac{M+N}{2} = \frac{M-N}{2}.
    \end{aligned}
    \]
    因此，
    \[
    \left| f(x) - \frac{M+N}{2} \right| < \frac{M-N}{2}.
    \]
    \par\textbf{理由：}
    将不等式平移至中点，然后取绝对值。

    \item[\textbf{步骤三（比值形式推导）：}]
    由 $N < f(x) < M$，有 $f(x)-N > 0$ 且 $M-f(x) > 0$。
    \[
    \frac{f(x)-N}{M-f(x)} > 0
    \]
    \par\textbf{理由：}
    正数除以正数得正数。

    \item[\textbf{步骤四（倒数形式推导）：}]
    由 $N < f(x) < M$，且 $f(x) \neq N, M$（自动满足），则 $f(x)-N > 0$ 且 $M-N > 0$。
    由于 $f(x) < M$，有 $f(x)-N < M-N$。取倒数并注意正数倒数递减：
    \[
    \frac{1}{f(x)-N} > \frac{1}{M-N}
    \]
    \par\textbf{理由：}
    对于正数，若 $a < b$，则 $\frac{1}{a} > \frac{1}{b}$。

    \item[\textbf{步骤五（等价循环）：}]
    类似地，可以从其他形式推回(1)，从而所有陈述等价。
\end{description}

\medskip
\textbf{结论回扣}

\textbf{因此，五个陈述彼此等价，都表示 $f(x)$ 在开区间 $(N, M)$ 内。}
\end{proofbox}
