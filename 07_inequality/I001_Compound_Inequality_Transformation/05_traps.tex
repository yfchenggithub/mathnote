\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{易错点一（忽略严格不等）}] 误认为 $N \le f(x) \le M$ 时结论也成立。
    \item[\textbf{正确理解（严格区间）：}] 结论只针对严格不等式 $N < f(x) < M$，等号成立时需单独考虑。
    \item[\textbf{错因分析（等号混淆）：}] 例如，若 $f(x)=N$，则乘积形式为 $0$，不满足 $<0$；绝对值形式等号可能成立。
\end{description}

\medskip
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{易错点二（分母为零）}] 在使用比值或倒数形式时，忽略 $f(x) \neq N$ 和 $f(x) \neq M$ 的条件。
    \item[\textbf{正确理解（自动满足）：}] 在 $N < f(x) < M$ 下，自动有 $f(x) \neq N,M$，但单独使用形式五时须先检查。
    \item[\textbf{错因分析（未验条件）：}] 若直接应用形式五而不验证分母非零，可能导致无意义表达式或错误结论。
\end{description}

\medskip
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{易错点三（符号误判）}] 在乘积形式中，误写为 $(f(x)-M)(f(x)-N) > 0$ 或错误因式分解。
    \item[\textbf{正确理解（乘积负号）：}] 只有当 $f(x)$ 介于 $M$ 和 $N$ 之间时，乘积才为负；若 $f(x)$ 在外，乘积为正。
    \item[\textbf{错因分析（区间颠倒）：}] 没有注意 $M > N$，导致区间方向错误，或记错乘积符号。
\end{description}
\end{trapbox}
