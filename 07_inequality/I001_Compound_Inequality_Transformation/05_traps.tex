\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=1.0em, topsep=0.5em]
    \item[\textbf{易错点 1}]
    \begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
        \item[\textbf{错误想法：}] 认为乘积形式 $(f(x)-M)(f(x)-N) < 0$ 无需条件 $M > N$。
        \item[\textbf{正确理解：}] 当 $M > N$ 时，乘积形式才等价于 $N < f(x) < M$；若 $M < N$，符号反向。
        \item[\textbf{错因分析：}] 忽略了原不等式中 $M$ 和 $N$ 的大小关系，导致等价性失效。
    \end{description}
    \item[\textbf{易错点 2}]
    \begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
        \item[\textbf{错误想法：}] 在倒数形式中直接写 $\frac{1}{f(x)-N} > \frac{1}{M-N}$ 而不验证 $f(x)-N > 0$。
        \item[\textbf{正确理解：}] 倒数形式成立的前提是 $f(x)-N > 0$ 且 $M-N > 0$，否则倒数不等式方向可能改变。
        \item[\textbf{错因分析：}] 倒数函数的单调性仅在正数区间一致递减；若 $f(x)-N$ 为负，结论不成立。
    \end{description}
    \item[\textbf{易错点 3}]
    \begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
        \item[\textbf{错误想法：}] 分式形式 $\frac{f(x)-N}{M-f(x)} > 0$ 中，忽略分母 $M-f(x) \neq 0$ 的条件。
        \item[\textbf{正确理解：}] 分式形式要求 $M-f(x) \neq 0$，即 $f(x) \neq M$，这与原不等式 $f(x) < M$ 一致，但需在变形中保持。
        \item[\textbf{错因分析：}] 未考虑分式分母非零，可能导致无意义表达式或漏掉边界情况。
    \end{description}
\end{description}
\end{trapbox}
