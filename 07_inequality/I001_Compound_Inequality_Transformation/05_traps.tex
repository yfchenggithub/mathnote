\begin{trapbox}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CTrap}{易错点一（忽略条件）：}}] 使用倒数形式 $\frac{1}{f(x)-N} > \frac{1}{M-N}$ 时，未检查 $f(x)-N > 0$ 是否成立。
    \item[\textbf{\textcolor{CTrap}{正确理解（条件先行）：}}] 倒数形式仅在 $f(x)-N > 0$ 的条件下与原始不等式等价；否则需另行讨论。
    \item[\textbf{\textcolor{CTrap}{错因分析（单调性误用）：}}] 函数 $\frac{1}{t}$ 在 $t>0$ 时递减，但在 $t<0$ 时性质不同，直接应用可能导致不等号方向错误。
\end{description}

\medskip
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CTrap}{易错点二（大小关系）：}}] 在乘积形式 $(f(x)-M)(f(x)-N) < 0$ 中，认为 $f(x)$ 一定在 $M$ 和 $N$ 之间，但未明确 $M$ 与 $N$ 的大小。
    \item[\textbf{\textcolor{CTrap}{正确理解（顺序重要）：}}] 当 $M > N$ 时，乘积为负等价于 $N < f(x) < M$；若 $M < N$，则等价于 $M < f(x) < N$。
    \item[\textbf{\textcolor{CTrap}{错因分析（默认顺序）：}}] 惯性思维认为 $M$ 和 $N$ 已按大小排列，但题目可能未明确，需先比较 $M$ 和 $N$。
\end{description}

\medskip
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CTrap}{易错点三（半长计算）：}}] 在绝对值形式中，将半长误写为 $\frac{M+N}{2}$ 或 $\left| \frac{M-N}{2} \right|$ 而未简化。
    \item[\textbf{\textcolor{CTrap}{正确理解（正半长）：}}] 由于 $M > N$，半长 $\frac{M-N}{2} > 0$，无需加绝对值；中点 $\frac{M+N}{2}$ 为实数。
    \item[\textbf{\textcolor{CTrap}{错因分析（符号混淆）：}}] 对绝对值概念和区间几何意义理解不透，导致公式记忆错误。
\end{description}
\end{trapbox}
