\begin{examplebox}
\textbf{例 1（基础）}

\textbf{题目：} 设 $f(x)=2x+1$，且 $-1 < f(x) < 3$，试用四种等价形式表示该约束。

\textbf{解题步骤：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步：}] 确定 $N$ 与 $M$。
    \par\textbf{理由：} 由题知 $N=-1$，$M=3$，且 $M>N$ 成立。
    \item[\textbf{第二步：}] 写出乘积形式。
    \[
    [f(x)-3][f(x)+1] < 0.
    \]
    \par\textbf{理由：} 直接代入结论 1，注意 $f(x)-N = f(x)+1$。
    \item[\textbf{第三步：}] 写出绝对值形式。
    \[
    \left| f(x)-\frac{3+(-1)}{2} \right| < \frac{3-(-1)}{2} \quad\Rightarrow\quad \left| f(x)-1 \right| < 2.
    \]
    \par\textbf{理由：} 计算中点 $\frac{M+N}{2}=1$，半长度 $\frac{M-N}{2}=2$。
    \item[\textbf{第四步：}] 写出分式形式与倒数形式。
    \[
    \frac{f(x)+1}{3-f(x)} > 0,\qquad \frac{1}{f(x)+1} > \frac{1}{4}.
    \]
    \par\textbf{理由：} 分别代入结论 3 与结论 4，注意 $M-N=4$。
\end{description}

\textbf{关键结论：} \textbf{四种形式均可描述 $-1 < 2x+1 < 3$，但代数表达不同，可根据后续计算需求选择。}

\textbf{例 2（稍变形）}

\textbf{题目：} 已知函数 $g(x)=x^2$，且 $0 < g(x) < 4$，求 $x$ 的取值范围，并用乘积形式验证。

\textbf{解题步骤：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步：}] 代入 $g(x)=x^2$ 得到原始不等式。
    \[
    0 < x^2 < 4.
    \]
    \par\textbf{理由：} 直接代入函数表达式。
    \item[\textbf{第二步：}] 解原始不等式。
    \[
    x^2 > 0 \ \text{且}\ x^2 < 4 \quad\Rightarrow\quad x \neq 0 \ \text{且}\ -2 < x < 2.
    \]
    \par\textbf{理由：} 平方非负，且开平方得 $|x|<2$，综合得 $x \in (-2,0) \cup (0,2)$。
    \item[\textbf{第三步：}] 用乘积形式验证。
    \[
    [x^2-4][x^2-0] < 0 \quad\Rightarrow\quad x^2(x^2-4) < 0.
    \]
    \par\textbf{理由：} 代入结论 1 得 $[g(x)-4][g(x)-0]<0$。分析 $x^2(x^2-4)<0$：因 $x^2 \geq 0$，故只需 $x^2>0$ 且 $x^2-4<0$，与第二步结果一致。
\end{description}

\textbf{关键结论：} \textbf{乘积形式 $x^2(x^2-4)<0$ 能快速导出 $x^2>0$ 且 $x^2<4$，验证了原始解集。}
\end{examplebox}
