\begin{examplebox}
\textbf{例 1（基础）}

\textbf{题目：} 设函数 $g(x)$ 满足 $2 < g(x) < 5$，写出其乘积形式和绝对值形式。

\textbf{解题步骤：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步：}] 识别 $N=2$, $M=5$。
    \item[\textbf{第二步：}] 乘积形式：$\bigl[g(x) - 5\bigr]\bigl[g(x) - 2\bigr] < 0$。（理由：直接套用结论1）
    \item[\textbf{第三步：}] 绝对值形式：中点 $\frac{2+5}{2}=3.5$，半长 $\frac{5-2}{2}=1.5$，所以 $\left| g(x) - 3.5 \right| < 1.5$。（理由：套用结论2）
\end{description}

\textbf{关键结论：} \textbf{对于 $2 < g(x) < 5$，等价于 $(g(x)-5)(g(x)-2) < 0$ 或 $|g(x)-3.5| < 1.5$。}

\textbf{例 2（稍变形）}

\textbf{题目：} 已知 $1 < h(x) + 3 < 4$，求 $h(x)$ 的范围，并用分式形式表示。

\textbf{解题步骤：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步：}] 令 $f(x) = h(x) + 3$，则 $1 < f(x) < 4$，其中 $N=1$, $M=4$。
    \item[\textbf{第二步：}] $h(x)$ 的范围：从 $1 < h(x)+3 < 4$ 得 $-2 < h(x) < 1$。（理由：不等式同时减3）
    \item[\textbf{第三步：}] 分式形式：$\frac{f(x)-1}{4-f(x)} > 0$，即 $\frac{h(x)+3-1}{4-(h(x)+3)} > 0$，化简为 $\frac{h(x)+2}{1-h(x)} > 0$。（理由：套用结论3）
\end{description}

\textbf{关键结论：} \textbf{不等式 $1 < h(x)+3 < 4$ 等价于 $-2 < h(x) < 1$，分式形式为 $\frac{h(x)+2}{1-h(x)} > 0$。}
\end{examplebox}
