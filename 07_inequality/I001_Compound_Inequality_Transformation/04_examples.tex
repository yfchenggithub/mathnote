\begin{examplebox}
\textbf{例 1（基础）}

\textbf{题目：} 设 $f(x) = 2x + 1$，且已知 $1 < f(x) < 3$，验证结论二和结论三成立。

\textbf{解题步骤：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（求函数值）：}] 由 $1 < 2x+1 < 3$，解得 $0 < x < 1$。对于任意 $x \in (0,1)$，$f(x)$ 在区间内。
    \par\textbf{理由：}
    解不等式得到 $x$ 的范围。

    \item[\textbf{第二步（验证乘积形式）：}] 计算 $(f(x)-3)(f(x)-1)$。
    \[
    (2x+1-3)(2x+1-1) = (2x-2)(2x) = 4x(x-1)
    \]
    当 $0 < x < 1$ 时，$x > 0$ 且 $x-1 < 0$，故乘积为负。
    \par\textbf{理由：}
    因式分解后判断符号。

    \item[\textbf{第三步（验证绝对值形式）：}] 计算中点 $\frac{1+3}{2} = 2$，半长 $\frac{3-1}{2} = 1$。
    \[
    \left| f(x) - 2 \right| = \left| 2x+1-2 \right| = \left| 2x-1 \right|
    \]
    当 $0 < x < 1$ 时，$-1 < 2x-1 < 1$，故 $\left| 2x-1 \right| < 1$。
    \par\textbf{理由：}
    代入绝对值表达式并利用 $x$ 的范围。
\end{description}

\textbf{关键结论：} \textbf{对于 $f(x)=2x+1$，当 $1 < f(x) < 3$ 时，等价形式二和三均成立。}

\textbf{例 2（稍变形）}

\textbf{题目：} 已知 $(g(t)-4)(g(t)-2) < 0$，求 $g(t)$ 的范围，并用绝对值形式表示。

\textbf{解题步骤：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（识别区间）：}] 由乘积形式，$g(t)$ 在 $2$ 和 $4$ 之间，即 $2 < g(t) < 4$。
    \par\textbf{理由：}
    乘积负号表示 $g(t)$ 介于两根之间。

    \item[\textbf{第二步（转化为绝对值形式）：}] 中点 $\frac{2+4}{2} = 3$，半长 $\frac{4-2}{2} = 1$。
    \[
    \left| g(t) - 3 \right| < 1
    \]
    \par\textbf{理由：}
    直接应用结论三。

    \item[\textbf{第三步（验证）：}] 从 $2 < g(t) < 4$ 可推出 $\left| g(t) - 3 \right| < 1$，反之亦然。
    \par\textbf{理由：}
    等价转化确保范围一致。
\end{description}

\textbf{关键结论：} \textbf{乘积形式 $(g(t)-4)(g(t)-2) < 0$ 等价于绝对值形式 $\left| g(t) - 3 \right| < 1$。}
\end{examplebox}
