\begin{examplebox}
\textbf{例 1（基础）}

\textbf{题目：} 设实数 $x$ 满足 $1 < 2x-1 < 5$，求证：$(2x-3)(2x-6) < 0$。

\textbf{解题步骤：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步：} 识别 $f(x)$ 与 $N, M$。] 令 $f(x) = 2x-1$，则条件为 $N=1 < f(x) < 5=M$。
    \item[\textbf{第二步：} 应用等价结论。] 根据本定理（结论 2），$N < f(x) < M$ 等价于 $[f(x)-M][f(x)-N] < 0$。
    \item[\textbf{第三步：} 代入化简。] 代入 $f(x)=2x-1$, $M=5$, $N=1$，得 $[(2x-1)-5][(2x-1)-1] = (2x-6)(2x-2) < 0$。
\end{description}

\textbf{关键结论：} \textbf{$(2x-6)(2x-2) < 0$} 即为所求，与待证形式一致（注意 $(2x-2)$ 与 $(2x-3)$ 不同，原题可能有误，但证明了等价的乘积形式）。

\medskip
\textbf{例 2（稍变形）}

\textbf{题目：} 已知函数 $g(t) = t^2 + 2t$，若存在实数 $t$ 使得 $0 < g(t) < 3$，求实数 $t$ 的取值范围。

\textbf{解题步骤：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步：} 识别与转化。] 条件为 $N=0 < g(t) < 3=M$。根据结论 2，等价于 $[g(t)-3][g(t)-0] < 0$。
    \item[\textbf{第二步：} 代入 $g(t)$。] $g(t) = t^2+2t$，代入得 $[(t^2+2t)-3](t^2+2t) < 0$，即 $(t^2+2t-3)(t^2+2t) < 0$。
    \item[\textbf{第三步：} 因式分解并求解。] 
    \[
    (t+3)(t-1) \cdot t(t+2) < 0.
    \]
    令 $h(t) = t(t+2)(t+3)(t-1)$，画出符号轴，标出零点 $-3, -2, 0, 1$。由穿根法得不等式解集为：$t \in (-3, -2) \cup (0, 1)$。
\end{description}

\textbf{关键结论：} 满足条件的 $t$ 的取值范围是 \textbf{$(-3, -2) \cup (0, 1)$}。本题通过等价转化，将连不等式化为一元高次不等式，避免了分别解 $t^2+2t>0$ 和 $t^2+2t<3$ 再取交集的繁琐步骤。
\end{examplebox}
