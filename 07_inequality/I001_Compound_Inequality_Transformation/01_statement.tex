\begin{statementbox}
\textbf{条件：} 设实数 $M, N$ 满足 $M > N$，$f(x)$ 是定义在某集合上的实值函数。

\textbf{结论：} 不等式 $N < f(x) < M$ 等价于以下任意一种形式：
\begin{enumerate}
\item \textbf{乘积形式：}
\[
[f(x)-M][f(x)-N] < 0
\]
\item \textbf{绝对值形式：}
\[
\left| f(x) - \frac{M+N}{2} \right| < \frac{M-N}{2}
\]
\item \textbf{分式形式：}
\[
\frac{f(x)-N}{M-f(x)} > 0
\]
\item \textbf{倒数形式：}
\[
\frac{1}{f(x)-N} > \frac{1}{M-N}
\]
\end{enumerate}
\end{statementbox}
