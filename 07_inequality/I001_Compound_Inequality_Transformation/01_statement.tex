\begin{statementbox}

\textbf{条件：} 设 $f(x)$ 为定义在某区间上的函数，$M, N \in \mathbb{R}$ 且 $M > N$。

\textbf{结论：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{结论 1：}] (乘积形式)
    \[
    \bigl[f(x) - M\bigr]\bigl[f(x) - N\bigr] < 0
    \]
    \item[\textbf{结论 2：}] (绝对值形式)
    \[
    \left|\, f(x) - \frac{M+N}{2} \,\right| < \frac{M-N}{2}
    \]
    \item[\textbf{结论 3：}] (分式形式)
    \[
    \frac{f(x)-N}{M-f(x)} > 0
    \]
    \item[\textbf{结论 4：}] (倒数形式)
    \[
    \frac{1}{f(x)-N} > \frac{1}{M-N}
    \]
\end{description}

\textbf{等号/取等条件：} 无等号条件，原不等式为严格不等。

\end{statementbox}
