\begin{statementbox}

\textbf{条件：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{条件 1（实数大小）：}] 设 $f(x)$ 是定义在某集合上的实值函数，$M, N \in \mathbb{R}$ 且 $M > N$。
\end{description}

\textbf{结论：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{结论一（基本形式）：}]
    \textbf{直观描述：} 函数值在 $N$ 与 $M$ 之间。
    \[
    N < f(x) < M
    \]
    \item[\textbf{结论二（乘积形式）：}]
    \textbf{直观描述：} 函数值与两端点的差乘积为负。
    \[
    (f(x)-M)(f(x)-N) < 0
    \]
    \item[\textbf{结论三（绝对值形式）：}]
    \textbf{直观描述：} 函数值偏离中点的距离小于半长。
    \[
    \left| f(x) - \frac{M+N}{2} \right| < \frac{M-N}{2}
    \]
    \item[\textbf{结论四（比值形式）：}]
    \textbf{直观描述：} 函数值到 $N$ 与到 $M$ 的差之比为正。
    \[
    \frac{f(x)-N}{M-f(x)} > 0
    \]
    \item[\textbf{结论五（倒数形式）：}]
    \textbf{直观描述：} 函数值到 $N$ 的倒数大于一个常数。
    \[
    \frac{1}{f(x)-N} > \frac{1}{M-N}
    \]
    其中 $f(x) \neq N$ 且 $f(x) \neq M$，在条件 (1) 下自动满足。
\end{description}

\textbf{等号/取等条件：} 无（各不等式均为严格不等，无等号成立）。

\end{statementbox}
