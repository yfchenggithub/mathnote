\begin{statementbox}

\textbf{条件：} 实数 $M,N$ 满足 $M>N$，且函数（或表达式）$f(x)$ 在定义域内的取值使下列表达式均有意义。

\textbf{结论：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{结论 1：}]
    \[
    N < f(x) < M
    \]
    \item[\textbf{结论 2：}]
    \[
    [f(x)-M][f(x)-N] < 0
    \]
    \item[\textbf{结论 3：}]
    \[
    \left| f(x) - \frac{M+N}{2} \right| < \frac{M-N}{2}
    \]
    \item[\textbf{结论 4：}]
    \[
    \frac{f(x)-N}{M-f(x)} > 0
    \]
    \item[\textbf{结论 5：}]
    \[
    \frac{1}{f(x)-N} > \frac{1}{M-N}
    \]
\end{description}

\end{statementbox}
