\begin{statementbox}

\textbf{条件：} $M, N \in \mathbb{R}$，$M > N$，函数 $f(x)$ 定义在某个实数子集上。

\textbf{结论：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{结论 1：}]
    \textbf{直观描述：} 函数值在区间内时，与两端点的差乘积为负。
    \[
    [f(x) - M][f(x) - N] < 0
    \]
    \item[\textbf{结论 2：}]
    \textbf{直观描述：} 函数值到区间中点的距离小于半长度。
    \[
    \left| f(x) - \frac{M+N}{2} \right| < \frac{M-N}{2}
    \]
    \item[\textbf{结论 3：}]
    \textbf{直观描述：} 函数值离左端点的距离与离右端点的距离之比为正。
    \[
    \frac{f(x)-N}{M-f(x)} > 0
    \]
    \item[\textbf{结论 4：}]
    \textbf{直观描述：} 函数值与左端点的差的倒数大于一个正常数。
    \[
    \frac{1}{f(x)-N} > \frac{1}{M-N}
    \]
\end{description}

\textbf{等号/取等条件：} 原不等式为严格不等式，故等号不成立；若允许等号，则相应形式中不等号变为 $\leq$。

\end{statementbox}
