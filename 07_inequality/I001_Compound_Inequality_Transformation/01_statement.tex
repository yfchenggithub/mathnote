\textbf{\textcolor{CStatement}{条件}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{条件 1（实数大小）：}}] $M, N \in \mathbb{R}$ 且 $M > N$；函数 $f(x)$ 在定义域 $D$ 上取实数值。
\end{description}

\textbf{\textcolor{CStatement}{结论}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{\textcolor{CStatement}{结论一（原始形式）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 函数值介于两个给定实数之间。
    \[
    N < f(x) < M
    \]

    \item[\textbf{\textcolor{CStatement}{结论二（乘积形式）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 函数值减端点后乘积为负。
    \[
    (f(x)-M)(f(x)-N) < 0
    \]

    \item[\textbf{\textcolor{CStatement}{结论三（绝对值形式）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 函数值到区间中点的距离小于半长。
    \[
    \left| f(x) - \frac{M+N}{2} \right| < \frac{M-N}{2}
    \]

    \item[\textbf{\textcolor{CStatement}{结论四（分式形式）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 函数值相对于端点的比值大于零。
    \[
    \frac{f(x)-N}{M-f(x)} > 0
    \]

    \item[\textbf{\textcolor{CStatement}{结论五（倒数形式）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 函数值减下界的倒数大于差值的倒数（需正条件）。
    \[
    \frac{1}{f(x)-N} > \frac{1}{M-N}, \quad \text{其中 } f(x)-N > 0
    \]

    \item[\textbf{\textcolor{CStatement}{推广结论（区间表示）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 函数值属于以中点为心、半长为半径的开区间。
    \[
    f(x) \in \left( \frac{M+N}{2} - \frac{M-N}{2}, \frac{M+N}{2} + \frac{M-N}{2} \right)
    \]
\end{description}

\textbf{\textcolor{CStatement}{等号/取等条件：}} 无等号成立情况，因原不等式为严格不等式。
