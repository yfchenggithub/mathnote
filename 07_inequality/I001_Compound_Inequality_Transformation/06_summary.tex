\begin{summarybox}
\textbf{一句话核心}

\textbf{五种等价形式都描述 $f(x)$ 在开区间 $(N, M)$ 内，便于在不同数学场景中灵活转化。}

\medskip
\textbf{使用条件}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{条件 1（实数大小）：} $M, N \in \mathbb{R}$ 且 $M > N$。
    \item \textbf{条件 2（函数实值）：} $f(x)$ 是实值函数，定义域需使不等式有意义。
\end{itemize}

\medskip
\textbf{关键提醒}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{易错点（严格不等）：} 所有不等式均为严格不等，等号不成立。
    \item \textbf{检查项（分母非零）：} 使用形式四和五时，确保分母不为零，或由形式一自动保证。
\end{itemize}
\end{summarybox}
