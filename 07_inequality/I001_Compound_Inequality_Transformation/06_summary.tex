\begin{summarybox}
\textbf{\textcolor{CSummary}{一句话核心}}

\textbf{连不等式 $N < f(x) < M$ 可等价转化为乘积、绝对值、分式、倒数四种形式，本质是函数值落在开区间内。}

\medskip
\textbf{\textcolor{CSummary}{使用条件}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{条件 1（实数大小）：} $M$ 和 $N$ 为实数，且满足 $M > N$。
    \item \textbf{条件 2（函数实值）：} 函数 $f(x)$ 在相关定义域上取实数值。
\end{itemize}

\medskip
\textbf{\textcolor{CSummary}{关键提醒}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{易错点（倒数条件）：} 使用倒数形式时，务必先确保 $f(x)-N > 0$，否则结论不成立。
    \item \textbf{检查项（形式适用）：} 在证明题中优先考虑乘积或绝对值形式；在求解题中可尝试分式或倒数形式简化计算。
\end{itemize}
\end{summarybox}
