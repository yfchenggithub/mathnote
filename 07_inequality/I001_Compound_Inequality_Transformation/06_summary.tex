\begin{summarybox}
\textbf{一句话核心}

\textbf{连不等式 $N < f(x) < M$ 可等价转化为乘积、绝对值、分式、倒数四种代数形式，为不同题型提供适配工具。}

\medskip
\textbf{使用条件}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{条件 1：} $M > N$，即区间左右端点明确且左小右大。
    \item \textbf{条件 2：} $f(x)$ 为实函数，且在该区间内有定义。
\end{itemize}

\medskip
\textbf{关键提醒}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{易错点：} 不同形式隐含的前提不同（如分母非零、正负性），转化时需检查等价条件。
    \item \textbf{检查项：} 遇到具体问题时，优先观察表达式结构，选择能简化计算或避免讨论的形式。
\end{itemize}
\end{summarybox}
