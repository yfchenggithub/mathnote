\begin{summarybox}
\textbf{一句话核心}

\textbf{连不等式 $N < f(x) < M$ 有乘积、绝对值、分式、倒数四种等价代数形式，每种形式在特定问题中可简化计算或提供新视角。}

\medskip
\textbf{使用条件}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{条件 1：} $M, N \in \mathbb{R}$ 且 $M > N$，确保区间非空。
    \item \textbf{条件 2：} $f(x)$ 在相关区间有定义，且不等式涉及的值在函数值域内。
    \item \textbf{条件 3：} 对于倒数形式，需额外保证 $f(x)-N > 0$ 以避免符号错误。
\end{itemize}

\medskip
\textbf{关键提醒}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{易错点：} 忽略 $M > N$ 的前提，直接套用乘积形式；在倒数形式中未验证正负。
    \item \textbf{检查项：} 变形后是否等价；分式分母是否为零；绝对值形式中中点与半径计算是否正确。
\end{itemize}
\end{summarybox}
