\begin{statementbox}

\textbf{条件：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{条件 1（正实数组）：}] $a, b, m$ 均为正实数。
    \item[\textbf{条件 2（大小关系）：}] $a < b$。
\end{description}

\textbf{结论：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{结论一（加糖变甜）：}]
    \textbf{直观描述：} 加糖后糖水变甜，即分数值增大。
    \[
    \frac{a}{b} < \frac{a+m}{b+m}
    \]
    
    \item[\textbf{结论二（减糖变淡）：}]
    \textbf{直观描述：} 当 $b > m$ 时，减糖后分数值减小。
    \[
    \frac{a}{b} > \frac{a-m}{b-m}
    \]
    
    \item[\textbf{结论三（推广形式）：}]
    \textbf{直观描述：} 对于倒数，加糖后值减小。
    \[
    \frac{b}{a} > \frac{b+m}{a+m}
    \]
\end{description}

\end{statementbox}
