\textbf{\textcolor{CStatement}{条件}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{条件 1（正实数）：}}] $a, b, m \in \mathbb{R}^+$（均为正实数）。
    \item[\textbf{\textcolor{CStatement}{条件 2（大小关系）：}}] $a < b$。
    \item[\textbf{\textcolor{CStatement}{条件 3（减糖前提）：}}] 对于减糖原理，需额外满足 $b > m$。
\end{description}

\textbf{\textcolor{CStatement}{结论}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{\textcolor{CStatement}{结论一（加糖原理）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 分子分母同加正数，分式值增大。
    \[
    \frac{a}{b} < \frac{a+m}{b+m}
    \]

    \item[\textbf{\textcolor{CStatement}{结论二（减糖原理）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 分子分母同减正数，分式值减小。
    \[
    \frac{a}{b} > \frac{a-m}{b-m}
    \]

    \item[\textbf{\textcolor{CStatement}{推广结论（倒数形式）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 原分式倒数大于加糖后倒数。
    \[
    \frac{b}{a} > \frac{b+m}{a+m}
    \]
\end{description}

\end{statementbox}
