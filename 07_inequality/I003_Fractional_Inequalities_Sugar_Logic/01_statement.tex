\begin{statementbox}
\textbf{条件：} 设 $a,b,m$ 为正实数，且 $a < b$。对于减糖情形，要求 $0 < m < a$。

\textbf{结论：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{结论 1：}]
    \textbf{直观描述：} 在糖水中加入更多糖，溶液变甜。
    \[
    \frac{a}{b} < \frac{a+m}{b+m}.
    \]
    \item[\textbf{结论 2：}]
    \textbf{直观描述：} 从糖水中减少糖，溶液变淡。
    \[
    \frac{a}{b} > \frac{a-m}{b-m}.
    \]
\end{description}

\textbf{等号/取等条件：} 无等号成立（不等式严格）。
\end{statementbox}
