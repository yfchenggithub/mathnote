\begin{statementbox}
\textbf{条件：} $a, b, m$ 为正实数，且 $a < b$。对于结论 (2)，额外要求 $m < b$。

\textbf{结论：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{结论 1：}]（加糖变甜）
    \[
    \frac{a}{b} < \frac{a+m}{b+m}
    \]
    \item[\textbf{结论 2：}]（减糖变淡，要求 $m < b$）
    \[
    \frac{a}{b} > \frac{a-m}{b-m}
    \]
    \item[\textbf{结论 3：}]（推广形式）
    \[
    \frac{b}{a} > \frac{b+m}{a+m}
    \]
\end{description}
\end{statementbox}
