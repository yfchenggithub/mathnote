\begin{statementbox}
\textbf{条件：}设 $a, b, m$ 均为正实数，且 $a < b$。

\textbf{结论：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{结论 1：}]
    \[
    \text{若 } m > 0, \text{ 则 } \frac{a}{b} < \frac{a+m}{b+m}.
    \]
    \item[\textbf{结论 2：}]
    \[
    \text{若 } 0 < m < a, \text{ 则 } \frac{a}{b} > \frac{a-m}{b-m}.
    \]
    \item[\textbf{结论 3：}]
    \[
    \text{若 } m > 0, \text{ 则 } \frac{b}{a} > \frac{b+m}{a+m}.
    \]
\end{description}

\end{statementbox}
