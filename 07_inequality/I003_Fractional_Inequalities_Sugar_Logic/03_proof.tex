\begin{proofbox}
\textbf{思路提示}

通过计算 $\frac{a+m}{b+m} - \frac{a}{b}$ 的差，并利用条件判断其符号，从而证明不等式。

\medskip
\textbf{正式推导}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.55em]
    \item[\textbf{步骤一：}] 计算加糖情形的差：
    \[
    \frac{a+m}{b+m} - \frac{a}{b} = \frac{b(a+m) - a(b+m)}{b(b+m)} = \frac{ab + bm - ab - am}{b(b+m)} = \frac{m(b-a)}{b(b+m)}.
    \]
    \par\textbf{理由：}
    通分后化简分子。

    \item[\textbf{步骤二：}] 判断差的符号：
    由于 $a,b,m > 0$ 且 $a < b$，所以 $b-a > 0$，因此 $\frac{m(b-a)}{b(b+m)} > 0$。
    \par\textbf{理由：}
    分子为正，分母为正，故整个分式为正。

    \item[\textbf{步骤三：}] 同理，对于减糖情形，计算 $\frac{a}{b} - \frac{a-m}{b-m}$：
    \[
    \frac{a}{b} - \frac{a-m}{b-m} = \frac{a(b-m) - b(a-m)}{b(b-m)} = \frac{ab - am - ab + bm}{b(b-m)} = \frac{m(b-a)}{b(b-m)}.
    \]
    \par\textbf{理由：}
    通分化简，注意 $0 < m < a$ 确保 $a-m > 0$ 和 $b-m > 0$。

    \item[\textbf{步骤四：}] 判断符号：由 $a < b$ 和 $0 < m < a$，得 $b-a > 0$ 且 $b-m > 0$，所以 $\frac{m(b-a)}{b(b-m)} > 0$。
    \par\textbf{理由：}
    分子分母均为正，故差为正。
\end{description}

\medskip
\textbf{结论回扣}

\textbf{因此，加糖变甜不等式 $\frac{a}{b} < \frac{a+m}{b+m}$ 和减糖变淡不等式 $\frac{a}{b} > \frac{a-m}{b-m}$ 均成立。}
\end{proofbox}
