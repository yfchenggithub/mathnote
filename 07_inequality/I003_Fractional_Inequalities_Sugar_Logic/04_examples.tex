\begin{examplebox}
\textbf{例 1（基础）}

\textbf{题目：} 比较 $\frac{2}{5}$ 与 $\frac{3}{6}$ 的大小。

\textbf{解题步骤：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步：}] 观察两个分数，发现 $\frac{3}{6}$ 可以看作 $\frac{2+1}{5+1}$，即 $\frac{2}{5}$ 的分子分母同时加1。
    \item[\textbf{第二步：}] 由于 $2<5$，且 $1>0$，满足糖水不等式条件1。
    \item[\textbf{第三步：}] 根据结论1，有 $\frac{2}{5} < \frac{2+1}{5+1} = \frac{3}{6}$。
\end{description}

\textbf{关键结论：} \textbf{$\frac{2}{5} < \frac{3}{6}$。}

\textbf{例 2（稍变形）}

\textbf{题目：} 设 $0<a<b$，求证：$\frac{a}{b} < \frac{a+1}{b+1} < \frac{a+2}{b+2}$。

\textbf{解题步骤：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步：}] 首先证明左边不等式：由结论1，因为 $a<b$ 且 $1>0$，所以 $\frac{a}{b} < \frac{a+1}{b+1}$。
    \item[\textbf{第二步：}] 再证明右边不等式：考虑 $\frac{a+1}{b+1}$ 与 $\frac{a+2}{b+2}$。由于 $a<b$，有 $a+1 < b+1$，且 $1>0$，再次应用结论1，得 $\frac{a+1}{b+1} < \frac{(a+1)+1}{(b+1)+1} = \frac{a+2}{b+2}$。
    \item[\textbf{第三步：}] 综合两步，得到 $\frac{a}{b} < \frac{a+1}{b+1} < \frac{a+2}{b+2}$。
\end{description}

\textbf{关键结论：} \textbf{对于 $0<a<b$，有 $\frac{a}{b} < \frac{a+1}{b+1} < \frac{a+2}{b+2}$。}
\end{examplebox}
