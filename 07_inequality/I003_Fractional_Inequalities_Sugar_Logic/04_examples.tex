\begin{examplebox}
\textbf{例 1（基础）}

\textbf{题目：} 比较 $\frac{3}{5}$ 与 $\frac{4}{6}$ 的大小。

\textbf{解题步骤：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步：}] 识别条件：这里 $a=3$, $b=5$, $m=1$，满足 $a<b$ 且 $m>0$。
    \par\textbf{理由：}
    糖水不等式的加糖情形要求 $a,b,m$ 为正且 $a<b$。

    \item[\textbf{第二步：}] 应用加糖不等式：$\frac{3}{5} < \frac{3+1}{5+1} = \frac{4}{6}$。
    \par\textbf{理由：}
    直接代入公式 $\frac{a}{b} < \frac{a+m}{b+m}$。

    \item[\textbf{第三步：}] 简化 $\frac{4}{6} = \frac{2}{3}$，因此 $\frac{3}{5} < \frac{2}{3}$。
    \par\textbf{理由：}
    进一步比较或验证。
\end{description}

\textbf{关键结论：} \textbf{糖水不等式直接给出 $\frac{3}{5} < \frac{4}{6}$，且通过计算可验证。}

\textbf{例 2（稍变形）}

\textbf{题目：} 设 $0 < a < b$，且 $m > 0$，证明 $\frac{a}{b} < \frac{a+2m}{b+2m}$。

\textbf{解题步骤：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步：}] 令 $m' = 2m$，则 $m' > 0$。
    \par\textbf{理由：}
    将 $2m$ 视为整体，满足加糖条件。

    \item[\textbf{第二步：}] 应用加糖不等式：由于 $a < b$ 且 $m' > 0$，有 $\frac{a}{b} < \frac{a+m'}{b+m'} = \frac{a+2m}{b+2m}$。
    \par\textbf{理由：}
    直接代入糖水不等式。

    \item[\textbf{第三步：}] 因此，不等式得证。
    \par\textbf{理由：}
    结论已明确。
\end{description}

\textbf{关键结论：} \textbf{通过变量代换，糖水不等式可推广到分子分母同加一个正数倍的情况。}
\end{examplebox}
