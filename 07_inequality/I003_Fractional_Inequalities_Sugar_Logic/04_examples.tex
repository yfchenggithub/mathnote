\begin{examplebox}
\textbf{例 1（基础）}

\textbf{题目：} 设 $a, b$ 为正实数，且 $a < b$，比较 $\dfrac{a}{b}$ 与 $\dfrac{a+1}{b+1}$ 的大小。

\textbf{解题步骤：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（识别条件）：}] 已知 $a, b > 0$ 且 $a < b$，取 $m = 1 > 0$，满足糖水不等式条件。
    \par\textbf{理由：}
    糖水不等式要求 $a, b, m$ 为正实数且 $a < b$。

    \item[\textbf{第二步（应用结论）：}] 根据加糖变甜结论，有
    \[
    \frac{a}{b} < \frac{a+1}{b+1}.
    \]
    \par\textbf{理由：}
    直接代入 $m=1$。

    \item[\textbf{第三步（写出答案）：}] 因此，$\dfrac{a}{b} < \dfrac{a+1}{b+1}$。
    \par\textbf{理由：}
    由不等式直接得出。
\end{description}

\textbf{关键结论：} \textbf{对于正实数 $a<b$，加糖后分数值增大。}

\medskip
\textbf{例 2（稍变形）}

\textbf{题目：} 已知 $x > y > 0$，证明 $\dfrac{y}{x} < \dfrac{y+2}{x+2}$。

\textbf{解题步骤：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（变量对应）：}] 令 $a = y$, $b = x$, $m = 2$，则 $a, b, m > 0$ 且 $a < b$。
    \par\textbf{理由：}
    将题目转化为糖水不等式标准形式。

    \item[\textbf{第二步（应用不等式）：}] 由糖水不等式结论一，
    \[
    \frac{a}{b} < \frac{a+m}{b+m} \quad \Rightarrow \quad \frac{y}{x} < \frac{y+2}{x+2}.
    \]
    \par\textbf{理由：}
    直接代入。

    \item[\textbf{第三步（完成证明）：}] 故原不等式成立。
    \par\textbf{理由：}
    证毕。
\end{description}

\textbf{关键结论：} \textbf{糖水不等式可灵活应用于变量替换后的比较。}
\end{examplebox}
