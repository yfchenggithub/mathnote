\begin{examplebox}
\textbf{例 1（基础）}

\textbf{题目：} 比较 $\frac{3}{8}$ 与 $\frac{5}{10}$ 的大小。

\textbf{解题步骤：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步：}] 观察两个分数。$\frac{5}{10} = \frac{1}{2}$。将 $\frac{3}{8}$ 与 $\frac{1}{2}$ 比较不方便，但注意到从 $\frac{3}{8}$ 到 $\frac{5}{10}$ 可以视为分子和分母同时加上 2。（理由：$3+2=5$，$8+2=10$）
    \item[\textbf{第二步：}] 检查条件。$a=3, b=8, m=2$，满足 $a<b$ 且均为正数，符合“加糖变甜”的条件。
    \item[\textbf{第三步：}] 应用结论。根据糖水不等式（加糖变甜），有 $\frac{3}{8} < \frac{3+2}{8+2} = \frac{5}{10}$。（理由：结论 1）
\end{description}

\textbf{关键结论：} \textbf{$\frac{3}{8} < \frac{5}{10}$}。

\textbf{例 2（稍变形）}

\textbf{题目：} 已知 $x > y > 0$，比较 $\frac{y}{x}$ 与 $\frac{y+1}{x+1}$ 的大小，并证明 $\frac{x}{y} > \frac{x+1}{y+1}$。

\textbf{解题步骤：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步：}] 第一问直接应用。设 $a=y, b=x, m=1$。由于 $x > y > 0$，满足 $a < b$ 且均为正数。（理由：题目条件）
    \item[\textbf{第二步：}] 由糖水不等式（加糖变甜）得：$\frac{y}{x} < \frac{y+1}{x+1}$。（理由：结论 1）
    \item[\textbf{第三步：}] 第二问利用推广形式。由 $x > y > 0$ 知 $\frac{x}{y} > 1$。令 $a' = y, b' = x, m=1$，则 $a' < b'$。由糖水不等式的推广形式（结论 3）直接得：$\frac{x}{y} > \frac{x+1}{y+1}$。（理由：结论 3，即 $\frac{b'}{a'} > \frac{b'+m}{a'+m}$）
\end{description}

\textbf{关键结论：} \textbf{$\frac{y}{x} < \frac{y+1}{x+1}$} 且 \textbf{$\frac{x}{y} > \frac{x+1}{y+1}$}。
\end{examplebox}
