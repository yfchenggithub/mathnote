\begin{examplebox}
\textbf{\textcolor{CExample}{例 1（基础）}}

\textbf{\textcolor{CExample}{题目：}} 比较 $\frac{3}{8}$ 与 $\frac{5}{10}$ 的大小，不直接计算分母。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（识别结构）：}] 注意到 $\frac{5}{10}$ 可视为 $\frac{3+2}{8+2}$。
    \par\textbf{理由：} 分子分母同时加了 $m=2$，符合加糖原理条件。

    \item[\textbf{第二步（应用加糖）：}] 根据加糖原理，由于 $3<8$ 且 $2>0$，有
    \[
    \frac{3}{8} < \frac{3+2}{8+2} = \frac{5}{10}
    \]
    \par\textbf{理由：} 直接代入糖水不等式结论一。

    \item[\textbf{第三步（得出结论）：}] 因此，$\frac{3}{8} < \frac{5}{10}$。
    \par\textbf{理由：} 无需通分计算，直接由不等式得到结果。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{加糖原理允许快速比较分数，避免繁琐通分。}

\textbf{\textcolor{CExample}{例 2（稍变形）}}

\textbf{\textcolor{CExample}{题目：}} 已知 $0 < x < y$，证明 $\frac{y}{x} > \frac{y+1}{x+1}$。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（倒数转换）：}] 考虑原不等式的倒数形式。
    \par\textbf{理由：} 目标式形如糖水不等式的推广结论，但变量顺序需调整。

    \item[\textbf{第二步（应用加糖）：}] 由 $0 < x < y$，令 $a=x, b=y, m=1$，则 $a<b$ 且 $m>0$，根据加糖原理：
    \[
    \frac{x}{y} < \frac{x+1}{y+1}
    \]
    \par\textbf{理由：} 直接代入结论一，注意这里 $\frac{x}{y}<1$。

    \item[\textbf{第三步（取倒数）：}] 对上述不等式两边取倒数，不等号反向。
    \[
    \frac{y}{x} > \frac{y+1}{x+1}
    \]
    \par\textbf{理由：} 由于 $x, y, x+1, y+1$ 均为正，取倒数保持不等号反向。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{通过倒数转换，将问题化归为标准糖水不等式，简化证明。}
\end{examplebox}
