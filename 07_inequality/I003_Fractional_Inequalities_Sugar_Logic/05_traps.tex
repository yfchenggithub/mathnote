\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{易错点一（忽略正实性）}] 误以为 $a, b, m$ 可为任意实数，忽略正实数条件。
    \item[\textbf{正确理解（正数前提）：}] $a, b, m$ 必须均为正实数，否则不等式可能不成立。
    \item[\textbf{错因分析（符号影响）：}] 若 $m$ 为负数，差值符号可能改变；若 $a,b$ 负，大小关系混乱。
\end{description}

\medskip
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{易错点二（减糖条件）}] 应用减糖结论时，未检查 $b > m$ 导致分母 $b-m$ 可能非正。
    \item[\textbf{正确理解（分母正性）：}] 减糖时需保证 $b > m$，确保分母 $b-m > 0$。
    \item[\textbf{错因分析（未验分母）：}] 忽略 $b-m$ 的正负，直接套用公式可能得到错误比较。
\end{description}

\medskip
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{易错点三（推广误用）}] 误将推广形式 $\frac{b}{a} > \frac{b+m}{a+m}$ 与加糖变甜混同，方向记反。
    \item[\textbf{正确理解（倒数关系）：}] 推广形式是倒数不等式，方向与加糖变甜相反。
    \item[\textbf{错因分析（未辨分子分母）：}] 未注意原始分数 $\frac{b}{a}$ 是 $\frac{a}{b}$ 的倒数，导致方向错误。
\end{description}
\end{trapbox}
