\begin{trapbox}
\textbf{易错点一}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{错误想法：}] 糖水不等式对所有实数 $a,b,m$ 都成立。
    \item[\textbf{正确理解：}] 必须满足 $a,b,m > 0$ 且 $a < b$，对于减糖还需 $0 < m < a$。
    \item[\textbf{错因分析：}] 当变量为负时，分母符号变化可能反转不等式方向，导致错误结论。
\end{description}

\medskip
\textbf{易错点二}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{错误想法：}] 减糖不等式中的 $m$ 可以任意正数。
    \item[\textbf{正确理解：}] 减糖时要求 $0 < m < a$，以保证 $a-m$ 和 $b-m$ 为正。
    \item[\textbf{错因分析：}] 若 $m \ge a$，则 $a-m \le 0$，分数可能无意义或不等式不成立。
\end{description}

\medskip
\textbf{易错点三}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{错误想法：}] 加糖会使分数值减小，减糖会使分数值增大。
    \item[\textbf{正确理解：}] 加糖使分数值增大，减糖使分数值减小。
    \item[\textbf{错因分析：}] 没有抓住 $a<b$ 的条件；当 $a<b$ 时，加糖相当于增加分子分母，但分子增加比例更大，导致值增大。
\end{description}
\end{trapbox}
