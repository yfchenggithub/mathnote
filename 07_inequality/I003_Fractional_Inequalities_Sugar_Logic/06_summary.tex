\begin{summarybox}
\textbf{一句话核心}

\textbf{糖水不等式刻画了正分数在分子分母同加（减）正数时，分数值严格增大（减小）的规律。}

\medskip
\textbf{使用条件}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{条件 1：} $a, b, m$ 均为正实数。
    \item \textbf{条件 2：} $a < b$。
    \item \textbf{条件 3：} 对于减糖情形，还需 $0 < m < a$。
\end{itemize}

\medskip
\textbf{关键提醒}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{易错点：} 忘记变量必须为正，或误用减糖条件。
    \item \textbf{检查项：} 应用不等式前，务必验证 $a,b,m>0$ 和 $a<b$；若减糖，检查 $0<m<a$。
\end{itemize}
\end{summarybox}
