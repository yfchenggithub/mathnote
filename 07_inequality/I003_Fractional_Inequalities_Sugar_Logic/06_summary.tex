\begin{summarybox}
\textbf{一句话核心}

\textbf{糖水不等式描述了正实数分式分子分母同加（减）一个正数时，分式值的变化规律：真分数同加变大，同减变小；假分数则相反。}

\medskip
\textbf{使用条件}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{条件 1：} $a, b, m$ 均为正实数。
    \item \textbf{条件 2：} $a < b$（即原分数为真分数）。
    \item \textbf{条件 3：} 对于减糖（结论2），额外要求 $0 < m < a$。
\end{itemize}

\medskip
\textbf{关键提醒}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{易错点：} 忽视 $m$ 的正负和大小条件，直接套用结论。
    \item \textbf{检查项：} 应用前，先确认是否满足“正实数”和“大小关系”，尤其是减糖时 $m$ 是否小于分子。
\end{itemize}
\end{summarybox}
