\begin{summarybox}
\textbf{一句话核心}

\textbf{真分数（分子 $<$ 分母）加同一正数后变大，更靠近 1；减同一正数后变小，更远离 1。}

\medskip
\textbf{使用条件}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{条件 1：} 所有涉及的数（$a, b, m$）必须为正实数，这是前提。
    \item \textbf{条件 2：} 对于“加糖变甜”和推广形式，要求 $a < b$。
    \item \textbf{条件 3：} 对于“减糖变淡”，除 $a < b$ 外，还必须满足 $m < b$（保证分母为正）。
\end{itemize}

\medskip
\textbf{关键提醒}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{易错点：} 使用前必须先判断原分数是大于 1 还是小于 1，以确定变化方向。
    \item \textbf{检查项：} 比较前，确认所加（减）的量 $m$ 是否满足正数条件，对于减法还需检查分母是否保持为正。
\end{itemize}
\end{summarybox}
