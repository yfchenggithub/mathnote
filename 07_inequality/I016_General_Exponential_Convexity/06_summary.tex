\begin{summarybox}
\textbf{\textcolor{CSummary}{一句话核心}}
\textbf{指数函数 $a^{x}$（$a>0$，$a\neq 1$）图像下凸，因此两端点函数值的算术平均不小于中点函数值。}

\medskip
\textbf{\textcolor{CSummary}{使用条件}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{条件 1（底数正非一）：} $a>0$ 且 $a\neq 1$，这是指数函数的基本定义。
    \item \textbf{条件 2（实数自变量）：} $x,y$ 为任意实数，无其他限制。
\end{itemize}

\medskip
\textbf{\textcolor{CSummary}{关键提醒}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{易错点（等号条件）：} 等号仅在 $x=y$ 时成立，使用时务必验证。
    \item \textbf{检查项（不等式方向）：} 确保不等号方向为 $\geq$，表示算术平均大于等于几何平均（中点函数值）。
\end{itemize}
\end{summarybox}
