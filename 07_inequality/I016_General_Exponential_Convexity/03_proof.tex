\begin{proofbox}
\textbf{\textcolor{CProof}{思路提示}}
利用换元法将指数不等式转化为关于新变量的完全平方非负形式，这是处理指数对称式的常用技巧。

\medskip
\textbf{\textcolor{CProof}{正式推导}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.55em]
    \item[\textbf{步骤一（引入中间量）：}] 令 $t = a^{\frac{x-y}{2}}$。因为 $a>0$，所以 $t>0$。
    \[
    a^{\frac{x+y}{2}} > 0.
    \]
    \par\textbf{理由：}
    将 $a^{x}$ 和 $a^{y}$ 用 $t$ 和 $a^{\frac{x+y}{2}}$ 表示。由指数运算法则：
    \[
    a^{x} = a^{\frac{x+y}{2} + \frac{x-y}{2}} = a^{\frac{x+y}{2}} \cdot a^{\frac{x-y}{2}} = a^{\frac{x+y}{2}} t,
    \]
    \[
    a^{y} = a^{\frac{x+y}{2} - \frac{x-y}{2}} = a^{\frac{x+y}{2}} \cdot a^{-\frac{x-y}{2}} = a^{\frac{x+y}{2}} t^{-1}.
    \]

    \item[\textbf{步骤二（代入不等式）：}] 将上述表达式代入原不等式左边。
    \[
    \frac{a^{x}+a^{y}}{2} = \frac{a^{\frac{x+y}{2}} t + a^{\frac{x+y}{2}} t^{-1}}{2} = a^{\frac{x+y}{2}} \cdot \frac{t + t^{-1}}{2}.
    \]
    \par\textbf{理由：}
    提取公因式 $a^{\frac{x+y}{2}}$。

    \item[\textbf{步骤三（化完全平方）：}] 比较 $\frac{t+t^{-1}}{2}$ 与 $1$ 的大小。
    \[
    \frac{t+t^{-1}}{2} - 1 = \frac{t + t^{-1} - 2}{2} = \frac{t^2 - 2t + 1}{2t} = \frac{(t-1)^2}{2t}.
    \]
    \par\textbf{理由：}
    通分并配方。因为 $t>0$，分母 $2t>0$，分子 $(t-1)^2 \geq 0$，所以 $\frac{(t-1)^2}{2t} \geq 0$。

    \item[\textbf{步骤四（得出结论）：}] 由步骤三得 $\frac{t+t^{-1}}{2} \geq 1$，因此
    \[
    \frac{a^{x}+a^{y}}{2} = a^{\frac{x+y}{2}} \cdot \frac{t+t^{-1}}{2} \geq a^{\frac{x+y}{2}} \cdot 1 = a^{\frac{x+y}{2}}.
    \]
    \par\textbf{理由：}
    两边同乘正数 $a^{\frac{x+y}{2}}$，不等号方向不变。

    \item[\textbf{步骤五（取等条件）：}] 等号成立当且仅当 $\frac{(t-1)^2}{2t}=0$，即 $t=1$。
    \[
    t=1 \;\Longleftrightarrow\; a^{\frac{x-y}{2}}=1 \;\Longleftrightarrow\; \frac{x-y}{2}=0 \;\Longleftrightarrow\; x=y.
    \]
    \par\textbf{理由：}
    因为 $a>0$ 且 $a\neq 1$，所以 $a^{k}=1$ 当且仅当 $k=0$。
\end{description}

\medskip
\textbf{\textcolor{CProof}{结论回扣}}
\textbf{由此我们严格证明了 $\frac{a^{x}+a^{y}}{2} \geq a^{\frac{x+y}{2}}$，且等号成立当且仅当 $x=y$，这正是指数函数严格下凸性的一个直接推论。}
\end{proofbox}
