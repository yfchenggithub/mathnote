\textbf{\textcolor{CStatement}{条件}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{条件 1（底数正且非一）：}}] $a>0$ 且 $a\neq 1$。
\end{description}

\textbf{\textcolor{CStatement}{结论}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{\textcolor{CStatement}{结论一（算术几何平均不等式）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 两指数函数值的算术平均不小于中点的函数值。
    \[
    \frac{a^{x}+a^{y}}{2} \geq a^{\frac{x+y}{2}}.
    \]

    \item[\textbf{\textcolor{CStatement}{推广结论（加权凸性形式）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 加权平均形式，系数和为 1 时仍成立。
    \[
    \lambda a^{x} + (1-\lambda) a^{y} \geq a^{\lambda x + (1-\lambda)y}, \quad \lambda \in (0,1).
    \]
\end{description}

\textbf{\textcolor{CStatement}{等号/取等条件：}} 当且仅当 $x=y$ 时等号成立。

\end{statementbox}
