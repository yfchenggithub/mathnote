\begin{examplebox}
\textbf{\textcolor{CExample}{例 1（基础）}}

\textbf{\textcolor{CExample}{题目：}} 设 $a>0$，$a\neq 1$，比较 $a^{3}+a^{5}$ 与 $2a^{4}$ 的大小，并说明理由。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（识别结构）：}] 观察形式为 $a^{x}+a^{y}$ 与 $2a^{\frac{x+y}{2}}$，其中 $x=3$，$y=5$。
    \par\textbf{理由：}
    直接对应结论中的 $\frac{a^{x}+a^{y}}{2} \geq a^{\frac{x+y}{2}}$，两边乘以 2 得 $a^{x}+a^{y} \geq 2a^{\frac{x+y}{2}}$。

    \item[\textbf{第二步（应用结论）：}] 根据指数函数的凸性，有
    \[
    a^{3}+a^{5} \geq 2a^{\frac{3+5}{2}} = 2a^{4}.
    \]
    \par\textbf{理由：}
    结论中取 $x=3$，$y=5$，算术平均为 $\frac{a^{3}+a^{5}}{2}$，几何平均对应 $a^{\frac{3+5}{2}}=a^{4}$。

    \item[\textbf{第三步（判断等号）：}] 等号成立当且仅当 $3=5$，但 $3\neq 5$，故等号不成立。
    \par\textbf{理由：}
    取等条件为 $x=y$，此处不满足。

    \item[\textbf{第四步（写出结论）：}] 因此 $a^{3}+a^{5} > 2a^{4}$。
    \par\textbf{理由：}
    严格大于。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{$a^{3}+a^{5} > 2a^{4}$，因为指数函数严格下凸且 $3\neq 5$。}

\textbf{\textcolor{CExample}{例 2（稍变形）}}

\textbf{\textcolor{CExample}{题目：}} 已知 $a>1$，求证：对任意实数 $x,y$，有 $a^{2x}+a^{2y} \geq 2a^{x+y}$，并指出等号成立条件。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（变量替换）：}] 令 $u=2x$，$v=2y$，则 $x=\frac{u}{2}$，$y=\frac{v}{2}$。
    \par\textbf{理由：}
    将目标不等式化为标准形式 $a^{u}+a^{v} \geq 2a^{\frac{u+v}{2}}$。

    \item[\textbf{第二步（应用标准形式）：}] 根据结论，对任意实数 $u,v$，有
    \[
    \frac{a^{u}+a^{v}}{2} \geq a^{\frac{u+v}{2}}.
    \]
    两边乘以 2 得
    \[
    a^{u}+a^{v} \geq 2a^{\frac{u+v}{2}}.
    \]
    \par\textbf{理由：}
    直接套用指数函数凸性不等式。

    \item[\textbf{第三步（回代变量）：}] 将 $u=2x$，$v=2y$ 代入：
    \[
    a^{2x}+a^{2y} \geq 2a^{\frac{2x+2y}{2}} = 2a^{x+y}.
    \]
    \par\textbf{理由：}
    计算 $\frac{u+v}{2} = \frac{2x+2y}{2}=x+y$。

    \item[\textbf{第四步（等号条件）：}] 等号成立当且仅当 $u=v$，即 $2x=2y$，亦即 $x=y$。
    \par\textbf{理由：}
    原结论等号成立当且仅当 $u=v$。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{不等式 $a^{2x}+a^{2y} \geq 2a^{x+y}$ 恒成立，等号成立当且仅当 $x=y$。}
\end{examplebox}
