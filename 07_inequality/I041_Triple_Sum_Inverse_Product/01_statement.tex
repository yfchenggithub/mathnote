\textbf{\textcolor{CStatement}{条件}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{条件 1（变量正实）：}}] 变量 $a_1, a_2, \dots, a_n$ 均为正实数，并约定 $a_{n+1} = a_1$。
\end{description}

\textbf{\textcolor{CStatement}{结论}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{\textcolor{CStatement}{结论一（循环和不等式）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 循环分式和不小于变量个数。
    \[
    \sum_{i=1}^{n} \frac{a_i}{a_{i+1}} \ge n.
    \]

    \item[\textbf{\textcolor{CStatement}{推广结论（三元特例）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 三变量循环和至少为 3。
    \[
    \frac{a}{b} + \frac{b}{c} + \frac{c}{a} \ge 3.
    \]
\end{description}

\textbf{\textcolor{CStatement}{等号/取等条件：}} 当且仅当所有变量相等，即 $a_1 = a_2 = \dots = a_n$。三元特例等号成立当且仅当 $a=b=c$。

\end{statementbox}
