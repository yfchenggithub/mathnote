\textbf{\textcolor{CStatement}{条件}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{条件 1（定义域与函数）：}}] 函数 $f(x)=x\ln x$，其中 $x>0$，即定义域为正实数。
\end{description}

\textbf{\textcolor{CStatement}{结论}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{\textcolor{CStatement}{结论一（凸性判定）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 函数 $f(x)$ 在其定义域上是凸函数。
    \[
    f''(x) = \frac{1}{x} > 0 \quad (x>0)
    \]

    \item[\textbf{\textcolor{CStatement}{推广结论（Jensen不等式）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 凸函数的性质使得均值函数值不大于函数值的均值。
    \[
    \frac{1}{n}\sum_{i=1}^{n} x_i\ln x_i \geq \left(\frac{1}{n}\sum_{i=1}^{n} x_i\right)\ln\!\left(\frac{1}{n}\sum_{i=1}^{n} x_i\right)
    \]
\end{description}

\textbf{\textcolor{CStatement}{等号/取等条件：}} 当且仅当所有变量相等，即 $x_1 = x_2 = \cdots = x_n$ 时，不等式等号成立。

\end{statementbox}
