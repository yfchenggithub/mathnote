\begin{statementbox}

\textbf{\textcolor{CStatement}{条件}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{条件 1（实数序列）：}}] 设 $a_1, a_2, \dots, a_n$ 和 $b_1, b_2, \dots, b_n$ 是两个实数序列。
\end{description}

\textbf{\textcolor{CStatement}{结论}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{\textcolor{CStatement}{结论一（一般形式）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 平方和的乘积不小于乘积和的平方。
    \[
    \left( \sum_{i=1}^n a_i^2 \right) \left( \sum_{i=1}^n b_i^2 \right) \ge \left( \sum_{i=1}^n a_i b_i \right)^2.
    \]

    \item[\textbf{\textcolor{CStatement}{推广结论（常见特例）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 二维、三维及向量形式是常用的特例。
    \[
    (a^2+b^2)(c^2+d^2) \ge (ac+bd)^2, \quad (a_1^2+a_2^2+a_3^2)(b_1^2+b_2^2+b_3^2) \ge (a_1b_1+a_2b_2+a_3b_3)^2.
    \]
    向量形式：$|\vec{a}|^2 |\vec{b}|^2 \ge (\vec{a} \cdot \vec{b})^2$。
\end{description}

\textbf{\textcolor{CStatement}{等号/取等条件：}} 当且仅当序列 $(a_i)$ 与 $(b_i)$ 成比例，即存在不全为零的实数 $\lambda, \mu$ 使得 $\lambda a_i + \mu b_i = 0$ 对所有 $i$ 成立。

\end{statementbox}
