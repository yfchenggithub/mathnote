\begin{statementbox}

\textbf{条件：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{条件 1（实数序列）：}] 设 $a_i, b_i \in \mathbb{R}$ ($i=1,2,\dots,n$)。
\end{description}

\textbf{结论：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{结论一（平方和乘积）：}]
    \textbf{直观描述：} 两组实数的平方和的乘积不小于它们对应乘积之和的平方。
    \[
    \left( \sum_{i=1}^n a_i^2 \right) \left( \sum_{i=1}^n b_i^2 \right) \ge \left( \sum_{i=1}^n a_i b_i \right)^2
    \]
\end{description}

\textbf{等号/取等条件：} 当且仅当序列 $(a_i)$ 与 $(b_i)$ 成比例，即存在常数 $\lambda$ 使得 $a_i = \lambda b_i$ 对所有 $i$ 成立（若 $b_i$ 不全为零），或者 $b_i = 0$ 对所有 $i$ 成立（此时 $a_i$ 任意，但等号成立要求 $a_i b_i = 0$）。

\end{statementbox}
