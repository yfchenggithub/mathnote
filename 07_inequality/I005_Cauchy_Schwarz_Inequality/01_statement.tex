\begin{statementbox}

\textbf{条件：} 对于任意实数 $a_i, b_i \in \mathbb{R} \ (i = 1, 2, \dots, n)$。

\textbf{结论：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{结论 1：}] \textbf{直观描述：} 两组实数平方和的乘积不小于它们对应项乘积之和的平方。
    \[
    \left(\sum_{i=1}^{n} a_i^2\right)\left(\sum_{i=1}^{n} b_i^2\right) \geq \left(\sum_{i=1}^{n} a_i b_i\right)^2
    \]
    
    \item[\textbf{结论 2：}] \textbf{直观描述：} 二维情形下，两个完全平方和的乘积公式。
    \[
    (a^2 + b^2)(c^2 + d^2) \geq (ac + bd)^2
    \]
    
    \item[\textbf{结论 3：}] \textbf{直观描述：} 三维情形下，对应三个坐标分量的推广。
    \[
    (a_1^2 + a_2^2 + a_3^2)(b_1^2 + b_2^2 + b_3^2) \geq (a_1 b_1 + a_2 b_2 + a_3 b_3)^2
    \]
    
    \item[\textbf{结论 4：}] \textbf{直观描述：} 向量点积的平方不超过模长平方的乘积。
    \[
    |\vec{a}|^2 \, |\vec{b}|^2 \geq (\vec{a} \cdot \vec{b})^2
    \]
\end{description}

\textbf{等号/取等条件：} 当且仅当存在实数 $\lambda$，使得对每个 $i$ 都有 $a_i = \lambda b_i$ 成立。即两组数成比例。

\end{statementbox}
