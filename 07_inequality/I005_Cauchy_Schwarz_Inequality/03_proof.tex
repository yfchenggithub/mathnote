\begin{proofbox}
\textbf{\textcolor{CProof}{思路提示}}

构造一个关于 $t$ 的二次函数 $f(t)=\sum_{i=1}^n (a_i t + b_i)^2$，由于平方和非负，故判别式 $\Delta \le 0$，由此推出不等式。

\medskip
\textbf{\textcolor{CProof}{正式推导}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.55em]
    \item[\textbf{步骤一（构造函数）：}] 令
    \[
    f(t) = \sum_{i=1}^n (a_i t + b_i)^2 \ge 0 \quad (\text{对所有实数 }t)。
    \]
    \par\textbf{理由：}
    每个 $(a_i t + b_i)^2 \ge 0$，故和也非负。

    \item[\textbf{步骤二（展开整理）：}] 将 $f(t)$ 展开：
    \[
    f(t) = \sum_{i=1}^n (a_i^2 t^2 + 2 a_i b_i t + b_i^2) = \left(\sum_{i=1}^n a_i^2\right) t^2 + 2\left(\sum_{i=1}^n a_i b_i\right) t + \left(\sum_{i=1}^n b_i^2\right).
    \]
    \par\textbf{理由：}
    按完全平方公式展开，并利用求和线性性质。

    \item[\textbf{步骤三（判别式非负）：}] 因为 $f(t)\ge 0$ 对所有 $t$ 成立，所以二次方程 $f(t)=0$ 的判别式 $\Delta \le 0$：
    \[
    \Delta = \left[2\sum_{i=1}^n a_i b_i\right]^2 - 4\left(\sum_{i=1}^n a_i^2\right)\left(\sum_{i=1}^n b_i^2\right) \le 0.
    \]
    \par\textbf{理由：}
    二次函数 $At^2+Bt+C \ge 0$ 恒成立，则 $A>0$ 且 $\Delta \le 0$。这里 $A=\sum a_i^2 \ge 0$，若 $A=0$ 则不等式显然成立，故不妨设 $A>0$。

    \item[\textbf{步骤四（导出不等式）：}] 由 $\Delta \le 0$ 直接得到
    \[
    4\left(\sum_{i=1}^n a_i b_i\right)^2 - 4\left(\sum_{i=1}^n a_i^2\right)\left(\sum_{i=1}^n b_i^2\right) \le 0,
    \]
    即
    \[
    \left(\sum_{i=1}^n a_i b_i\right)^2 \le \left(\sum_{i=1}^n a_i^2\right)\left(\sum_{i=1}^n b_i^2\right).
    \]
    \par\textbf{理由：}
    移项并除以正数4即得。

    \item[\textbf{步骤五（取等条件）：}] 等号成立当且仅当 $f(t)=0$ 有唯一实根，即存在 $t_0$ 使 $a_i t_0 + b_i = 0$ 对所有 $i$ 成立，亦即序列 $(a_i)$ 与 $(b_i)$ 成比例。
    \par\textbf{理由：}
    $\Delta=0$ 时方程有重根 $t_0$，此时 $a_i t_0 + b_i = 0$，即 $a_i$ 与 $b_i$ 成比例。
\end{description}

\medskip
\textbf{\textcolor{CProof}{结论回扣}}

\textbf{因此，柯西不等式得证，且等号成立当且仅当两序列成比例。}
\end{proofbox}
