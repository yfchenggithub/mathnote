\begin{summarybox}
\textbf{\textcolor{CSummary}{一句话核心}}

\textbf{平方和的乘积不小于乘积和的平方。}

\medskip
\textbf{\textcolor{CSummary}{使用条件}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{条件 1（实数序列）：} 涉及两个实数序列，且序列长度相同。
    \item \textbf{条件 2（平方和存在）：} 序列的平方和有限（通常为有限项）。
\end{itemize}

\medskip
\textbf{\textcolor{CSummary}{关键提醒}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{易错点（取等条件）：} 等号成立仅当两序列成比例，检查是否满足比例关系。
    \item \textbf{检查项（结构匹配）：} 应用时确保右边是同一下标对应项乘积之和的平方。
\end{itemize}
\end{summarybox}
