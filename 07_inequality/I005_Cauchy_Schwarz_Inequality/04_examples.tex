\begin{examplebox}
\textbf{\textcolor{CExample}{例 1（基础）}}

\textbf{\textcolor{CExample}{题目：}} 已知实数 $a,b,c,d$，求证：$(a^2+b^2)(c^2+d^2) \ge (ac+bd)^2$。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（识别序列）：}] 将不等式左边视为两个序列的平方和乘积：序列一为 $(a,b)$，序列二为 $(c,d)$。
    \par\textbf{理由：}
    柯西不等式要求两个序列，这里恰好对应。
    \item[\textbf{第二步（直接应用）：}] 由柯西不等式（$n=2$ 情形）：
    \[
    (a^2+b^2)(c^2+d^2) \ge (ac+bd)^2.
    \]
    \par\textbf{理由：}
    直接代入，左边平方和乘积，右边对应项乘积和的平方。
    \item[\textbf{第三步（取等条件）：}] 等号成立当且仅当 $\frac{a}{c} = \frac{b}{d}$（当 $c,d$ 不全为零时），即 $ad=bc$。
    \par\textbf{理由：}
    两序列成比例，即存在 $\lambda$ 使 $(a,b)=\lambda(c,d)$。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{二维形式的柯西不等式直接得证，无需额外计算。}

\medskip
\textbf{\textcolor{CExample}{例 2（稍变形）}}

\textbf{\textcolor{CExample}{题目：}} 已知正实数 $x,y,z$ 满足 $x+y+z=1$，求 $x^2+y^2+z^2$ 的最小值。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（构造序列）：}] 考虑两个序列：$(x,y,z)$ 和 $(1,1,1)$。
    \par\textbf{理由：}
    欲求 $x^2+y^2+z^2$，而条件为 $x+y+z=1$，联想到柯西不等式左边为平方和与另一序列平方和的乘积。
    \item[\textbf{第二步（应用柯西）：}] 由柯西不等式：
    \[
    (x^2+y^2+z^2)(1^2+1^2+1^2) \ge (x\cdot 1 + y\cdot 1 + z\cdot 1)^2 = (x+y+z)^2 = 1^2 = 1.
    \]
    \par\textbf{理由：}
    将不等式写成柯西形式：左边为 $\sum a_i^2 \cdot \sum b_i^2$，右边为 $(\sum a_i b_i)^2$。
    \item[\textbf{第三步（解出最值）：}] 由上式得
    \[
    3(x^2+y^2+z^2) \ge 1 \quad \Rightarrow \quad x^2+y^2+z^2 \ge \frac{1}{3}.
    \]
    \par\textbf{理由：}
    移项即可得到平方和的下界。
    \item[\textbf{第四步（验证取等）：}] 等号成立当且仅当 $\frac{x}{1}=\frac{y}{1}=\frac{z}{1}$，即 $x=y=z$，结合 $x+y+z=1$ 得 $x=y=z=\frac{1}{3}$。
    \par\textbf{理由：}
    序列成比例，且满足条件，故最小值可达。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{当 $x=y=z=\frac{1}{3}$ 时，$x^2+y^2+z^2$ 取得最小值 $\frac{1}{3}$。}
\end{examplebox}
