\begin{statementbox}

\textbf{\textcolor{CStatement}{条件}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{条件 1（自变量范围）：}}] 自变量 $x$ 在开区间 $(0, \frac{\pi}{2})$ 内。
\end{description}

\textbf{\textcolor{CStatement}{结论}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{\textcolor{CStatement}{结论一（和式放缩）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}}正弦与正切之和严格大于两倍自变量。
    \[
    \sin x + \tan x > 2x
    \]

    \item[\textbf{\textcolor{CStatement}{推广结论（均值放缩）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}}正弦与正切的算术平均值严格大于自变量。
    \[
    \frac{\sin x + \tan x}{2} > x
    \]
\end{description}

\textbf{\textcolor{CStatement}{等号/取等条件：}} 当且仅当 $x=0$ 时，上述两个不等式中的等号成立。
\end{statementbox}
