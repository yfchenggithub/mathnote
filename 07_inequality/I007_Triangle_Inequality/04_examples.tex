\begin{examplebox}
\textbf{例 1（基础）}

\textbf{题目：} 设 $|x| \le 2$, $|y| \le 3$，求 $|x+y|$ 的最大值。

\textbf{解题步骤：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（应用不等式）：}] 由绝对值三角不等式，$|x+y| \le |x|+|y|$。
    \par\textbf{理由：}
    直接使用右侧不等式 $|a+b| \le |a|+|b|$。

    \item[\textbf{第二步（代入范围）：}] 已知 $|x| \le 2$, $|y| \le 3$，所以 $|x|+|y| \le 2+3 = 5$。
    \par\textbf{理由：}
    将已知条件代入不等式。

    \item[\textbf{第三步（得最大值）：}] 因此 $|x+y| \le 5$，当 $xy \ge 0$ 且 $|x|=2, |y|=3$ 时取等，故最大值为 $5$。
    \par\textbf{理由：}
    结合不等式和取等条件得出结论。
\end{description}

\textbf{关键结论：} \textbf{使用绝对值三角不等式可以求出含有绝对值的表达式的最值。}

\textbf{例 2（稍变形）}

\textbf{题目：} 证明对于任意实数 $x$，有 $|x-2| + |x+3| \ge 5$。

\textbf{解题步骤：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（变形应用）：}] 将表达式视为两个绝对值的和，应用绝对值三角不等式的推广形式。
    \par\textbf{理由：}
    注意 $|x-2| + |x+3| = |2-x| + |x+3|$，但直接使用不等式链。

    \item[\textbf{第二步（利用下界）：}] 由不等式链，$|a|+|b| \ge |a-b|$，这里取 $a=x-2$, $b=x+3$，则 $|x-2| + |x+3| \ge |(x-2) - (x+3)| = | -5| = 5$。
    \par\textbf{理由：}
    使用左侧不等式或类似形式，实际上从 $|a|+|b| \ge |a-b|$ 可得。

    \item[\textbf{第三步（完成证明）：}] 因此，对于任意实数 $x$，$|x-2| + |x+3| \ge 5$ 恒成立。
    \par\textbf{理由：}
    不等式成立，且等号当 $x-2$ 与 $x+3$ 异号且满足条件时取到，例如 $x$ 在 $[-3,2]$ 之间。
\end{description}

\textbf{关键结论：} \textbf{绝对值三角不等式可用于证明含有多个绝对值的不等式。}
\end{examplebox}
