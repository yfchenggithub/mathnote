\begin{examplebox}
\textbf{\textcolor{CExample}{例 1（基础）}}

\textbf{\textcolor{CExample}{题目：}} 已知 $|x| \leq 2$，$|y| \leq 3$，求 $|x+y|$ 的最大值。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（应用不等式）：}] 由绝对值三角不等式，有 $|x+y| \leq |x| + |y|$。
    \par\textbf{理由：} 直接使用右边不等式。

    \item[\textbf{第二步（代入范围）：}] 因为 $|x| \leq 2$，$|y| \leq 3$，所以 $|x| + |y| \leq 2 + 3 = 5$。
    \[
    |x+y| \leq |x| + |y| \leq 5.
    \]
    \par\textbf{理由：} 利用已知条件进行放缩。

    \item[\textbf{第三步（取等检查）：}] 当 $x$ 与 $y$ 同号且 $|x|=2, |y|=3$ 时，等号成立，例如 $x=2, y=3$。
    \par\textbf{理由：} 右边等号成立条件是 $xy \geq 0$，且绝对值取到上界。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{$|x+y|$ 的最大值为 $5$，当 $x$ 与 $y$ 同号且分别取 $2$ 和 $3$ 时达到。}

\textbf{\textcolor{CExample}{例 2（稍变形）}}

\textbf{\textcolor{CExample}{题目：}} 证明对于任意实数 $a, b$，有 $|a-b| \geq \bigl|\,|a| - |b|\,\bigr|$。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（变量替换）：}] 在绝对值三角不等式中，令 $b$ 替换为 $-b$。
    \[
    \bigl|\,|a| - |{-b}|\,\bigr| \leq |a + (-b)| \leq |a| + |{-b}|.
    \]
    \par\textbf{理由：} $|{-b}| = |b|$，且 $a + (-b) = a - b$。

    \item[\textbf{第二步（简化）：}] 由于 $|{-b}| = |b|$，左边不等式变为 $\bigl|\,|a| - |b|\,\bigr| \leq |a - b|$。
    \[
    \bigl|\,|a| - |b|\,\bigr| \leq |a - b|.
    \]
    \par\textbf{理由：} 直接代入化简。

    \item[\textbf{第三步（得出结论）：}] 因此，$|a-b| \geq \bigl|\,|a| - |b|\,\bigr|$ 成立。
    \par\textbf{理由：} 不等式方向不变。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{对于任意实数 $a, b$，有 $|a-b| \geq \bigl|\,|a| - |b|\,\bigr|$，这是绝对值三角不等式的直接推论。}
\end{examplebox}
