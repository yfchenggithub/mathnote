\begin{summarybox}
\textbf{一句话核心}

\textbf{绝对值三角不等式描述了绝对值与加法的界限关系，是放缩和估计的核心工具。}

\medskip
\textbf{使用条件}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{条件 1（实数变量）：} 涉及变量为实数，如 $a,b\in\mathbb{R}$。
    \item \textbf{条件 2（绝对值运算）：} 表达式包含绝对值与加法或减法运算。
\end{itemize}

\medskip
\textbf{关键提醒}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{易错点（取等条件）：} 使用不等式求最值时，必须验证等号能否成立。
    \item \textbf{检查项（方向正确）：} 确保不等式链方向正确，避免符号颠倒。
\end{itemize}
\end{summarybox}
