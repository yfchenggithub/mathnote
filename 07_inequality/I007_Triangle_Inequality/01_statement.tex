\textbf{\textcolor{CStatement}{条件}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{条件 1（实数变量）：}}] 设 $a, b$ 为任意实数。
\end{description}

\textbf{\textcolor{CStatement}{结论}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{\textcolor{CStatement}{结论一（双边不等式）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 两数和的绝对值被夹在它们绝对值之差的绝对值与和之间。
    \[
    \bigl|\,|a| - |b|\,\bigr| \leq |a + b| \leq |a| + |b|.
    \]

    \item[\textbf{\textcolor{CStatement}{推广结论（多实数形式）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 多个实数和的绝对值不超过各自绝对值之和。
    \[
    |a_1 + a_2 + \dots + a_n| \leq |a_1| + |a_2| + \dots + |a_n|.
    \]
\end{description}

\textbf{\textcolor{CStatement}{等号/取等条件：}} 
左边等号 $\bigl|\,|a| - |b|\,\bigr| = |a + b|$ 当且仅当 $ab \leq 0$；
右边等号 $|a + b| = |a| + |b|$ 当且仅当 $ab \geq 0$。
对于 $n$ 个实数，等号成立当且仅当所有 $a_i$ 同号或至少有一个为零。

\end{statementbox}
