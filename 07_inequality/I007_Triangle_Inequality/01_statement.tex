\begin{statementbox}

\textbf{条件：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{条件 1（实数变量）：}] $a,b\in\mathbb{R}$。
\end{description}

\textbf{结论：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{结论一（基本不等式链）：}]
    \textbf{直观描述：} 绝对值之差与和差绝对值的上下界关系。
    \[
    \bigl||a|-|b|\bigr| \le |a\pm b| \le |a|+|b|.
    \]
    \item[\textbf{结论二（推广形式）：}]
    \textbf{直观描述：} 多个实数和的绝对值不超过各自绝对值之和。
    \[
    |a_1+a_2+\cdots+a_n| \le |a_1|+|a_2|+\cdots+|a_n|, \quad n\in\mathbb{N}^*, a_i\in\mathbb{R}.
    \]
\end{description}

\textbf{等号/取等条件：} 右侧不等式 $|a+b| \le |a|+|b|$ 当且仅当 $ab \ge 0$ 时取等；左侧不等式 $|a+b| \ge |a|-|b|$ 当且仅当 $ab \le 0$ 且 $|a| \ge |b|$ 时取等。

\end{statementbox}
