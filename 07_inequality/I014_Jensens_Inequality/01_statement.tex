\textbf{\textcolor{CStatement}{条件}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{条件 1（函数性质）：}}] 函数 $f(x)$ 在区间 $I$ 上连续且二阶可导。
    \item[\textbf{\textcolor{CStatement}{条件 2（二阶导符号）：}}] $f''(x) > 0$（凸函数）或 $f''(x) < 0$（凹函数）。
\end{description}

\textbf{\textcolor{CStatement}{结论}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{\textcolor{CStatement}{结论一（两点平均）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 函数值在区间中点的值，不超过两端点函数值的平均。
    \[
    f\!\left(\frac{x_1+x_2}{2}\right) \le \frac{f(x_1)+f(x_2)}{2}
    \]

    \item[\textbf{\textcolor{CStatement}{推广结论（加权平均）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 函数在加权平均点处的值，不超过函数值的加权平均。
    \[
    f\!\left(\sum_{i=1}^{n}\lambda_i x_i\right) \le \sum_{i=1}^{n}\lambda_i f(x_i)
    \]
\end{description}

\textbf{\textcolor{CStatement}{等号/取等条件：}} 等号成立当且仅当所有自变量 $x_i$ 相等，或函数 $f$ 为线性函数。

\end{statementbox}
