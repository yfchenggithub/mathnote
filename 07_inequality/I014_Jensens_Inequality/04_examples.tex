\begin{examplebox}
\textbf{\textcolor{CExample}{例 1（基础）}}

\textbf{\textcolor{CExample}{题目：}} 设 $f(x)=x^2$，证明对任意实数 $a, b$，有 $\left(\frac{a+b}{2}\right)^2 \le \frac{a^2+b^2}{2}$。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（判断凸凹性）：}] 计算 $f(x)=x^2$ 的二阶导数：$f''(x)=2 > 0$。
    \par\textbf{理由：} 二阶导恒为正，说明 $f(x)$ 是凸函数，满足琴生不等式（凸函数情形）。
    \item[\textbf{第二步（直接套用结论）：}] 根据琴生不等式（两点形式），对任意 $x_1, x_2 \in \mathbb{R}$，有
    \[
    f\!\left(\frac{x_1+x_2}{2}\right) \le \frac{f(x_1)+f(x_2)}{2}
    \]
    令 $x_1=a, x_2=b$，并将 $f(x)=x^2$ 代入得
    \[
    \left(\frac{a+b}{2}\right)^2 \le \frac{a^2+b^2}{2}
    \]
    \par\textbf{理由：} 直接应用结论，注意函数对应关系。
    \item[\textbf{第三步（讨论取等条件）：}] 等号成立当且仅当 $a=b$ 或 $f$ 为线性函数。此处 $f(x)=x^2$ 不是线性函数，故等号成立当且仅当 $a=b$。
    \par\textbf{理由：} 根据定理，取等条件为自变量相等或函数线性。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{对于凸函数 $f(x)=x^2$，其满足中点平方不超过平方的平均。}

\textbf{\textcolor{CExample}{例 2（稍变形）}}

\textbf{\textcolor{CExample}{题目：}} 已知 $f(x)=\ln x$ 在 $(0,+\infty)$ 上是凹函数，利用琴生不等式证明：对任意正数 $a,b$，有 $\sqrt{ab} \le \frac{a+b}{2}$。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（确认函数性质）：}] 题目已告知 $f(x)=\ln x$ 是凹函数，即 $f''(x)=-\frac{1}{x^2}<0$。
    \par\textbf{理由：} 凹函数满足琴生不等式的反向形式。
    \item[\textbf{第二步（应用凹函数不等式）：}] 对于凹函数，琴生不等式为 $f\!\left(\frac{x_1+x_2}{2}\right) \ge \frac{f(x_1)+f(x_2)}{2}$。
    取 $x_1=a, x_2=b$，代入 $f(x)=\ln x$ 得
    \[
    \ln\!\left(\frac{a+b}{2}\right) \ge \frac{\ln a + \ln b}{2} = \ln\sqrt{ab}
    \]
    \par\textbf{理由：} 直接应用凹函数形式的结论，并利用对数运算性质。
    \item[\textbf{第三步（利用单调性化简）：}] 因为 $\ln x$ 是单调递增函数，由上式可得
    \[
    \frac{a+b}{2} \ge \sqrt{ab}
    \]
    等号成立当且仅当 $a=b$。
    \par\textbf{理由：} 对数不等式两边取指数，即 $e^{\ln L} \ge e^{\ln R}$，由于 $e^x$ 单调递增，不等号方向不变。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{利用凹函数 $\ln x$ 的琴生不等式，可推导出算术平均-几何平均不等式。}
\end{examplebox}
