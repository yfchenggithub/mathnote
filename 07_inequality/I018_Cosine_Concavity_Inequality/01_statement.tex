\textbf{\textcolor{CStatement}{条件}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{条件 1（角度范围）：}}] $x, y \in (0, \frac{\pi}{2})$。
    \item[\textbf{\textcolor{CStatement}{条件 2（三角形内角）：}}] $A, B, C$ 是锐角三角形的三个内角。
\end{description}

\textbf{\textcolor{CStatement}{结论}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{\textcolor{CStatement}{结论一（余弦凹性不等式）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 两角余弦算术平均不超过半角余弦。
    \[
    \frac{\cos x + \cos y}{2} \leq \cos\left( \frac{x+y}{2} \right)
    \]

    \item[\textbf{\textcolor{CStatement}{推广结论（三角形余弦和上界）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 锐角三角形内角余弦和不超过 $\frac{3}{2}$。
    \[
    \cos A + \cos B + \cos C \leq \frac{3}{2}
    \]
\end{description}

\textbf{\textcolor{CStatement}{等号/取等条件：}} 对于结论一，等号成立当且仅当 $x = y$；对于推广结论，等号成立当且仅当 $A = B = C = \frac{\pi}{3}$（即三角形为等边三角形）。

\end{statementbox}
