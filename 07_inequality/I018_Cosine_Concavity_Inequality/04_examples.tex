\begin{examplebox}
\textbf{\textcolor{CExample}{例 1（基础）}}

\textbf{\textcolor{CExample}{题目：}} 设 $x, y \in (0, \frac{\pi}{2})$ 且 $x+y = \frac{\pi}{2}$，证明 $\cos x + \cos y \leq \sqrt{2}$。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（应用凹性不等式）：}] 由结论一，$\frac{\cos x + \cos y}{2} \leq \cos\left( \frac{x+y}{2} \right)$。
    \par\textbf{理由：}
    直接代入已知条件。

    \item[\textbf{第二步（代入半角）：}] 由 $x+y = \frac{\pi}{2}$，得 $\frac{x+y}{2} = \frac{\pi}{4}$，故 $\cos\frac{\pi}{4} = \frac{\sqrt{2}}{2}$。
    \[
    \frac{\cos x + \cos y}{2} \leq \frac{\sqrt{2}}{2}
    \]
    \par\textbf{理由：}
    计算余弦值。

    \item[\textbf{第三步（得结论）：}] 两边乘以 2，得到 $\cos x + \cos y \leq \sqrt{2}$。
    \par\textbf{理由：}
    不等式乘法性质。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{当 $x+y=\frac{\pi}{2}$ 时，$\cos x + \cos y$ 的最大值为 $\sqrt{2}$，等号在 $x=y=\frac{\pi}{4}$ 时成立。}

\medskip
\textbf{\textcolor{CExample}{例 2（稍变形）}}

\textbf{\textcolor{CExample}{题目：}} 在锐角三角形 $ABC$ 中，证明 $\cos A + \cos B + \cos C \leq \frac{3}{2}$，并求等号成立条件。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（确认条件）：}] $A, B, C$ 是锐角三角形的内角，故 $A, B, C \in (0,\frac{\pi}{2})$ 且 $A+B+C=\pi$。
    \par\textbf{理由：}
    锐角三角形性质。

    \item[\textbf{第二步（应用Jensen不等式）：}] 由于余弦函数在 $(0,\frac{\pi}{2})$ 上凹，有：
    \[
    \frac{\cos A + \cos B + \cos C}{3} \leq \cos\left( \frac{A+B+C}{3} \right)
    \]
    \par\textbf{理由：}
    凹函数的 Jensen 不等式。

    \item[\textbf{第三步（计算化简）：}] 代入 $A+B+C=\pi$，$\cos\frac{\pi}{3} = \frac{1}{2}$，两边乘以 3 得：
    \[
    \cos A + \cos B + \cos C \leq \frac{3}{2}
    \]
    \par\textbf{理由：}
    代数运算。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{锐角三角形余弦和的最大值为 $\frac{3}{2}$，等号当且仅当 $A=B=C=\frac{\pi}{3}$（等边三角形）时成立。}
\end{examplebox}
