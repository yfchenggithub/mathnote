\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CTrap}{易错点一（区间忽略）：}}] 误将角度取到区间外，如 $x \geq \frac{\pi}{2}$，仍应用不等式。
    \item[\textbf{\textcolor{CTrap}{正确理解（严格锐角）：}}] 结论仅当 $x, y \in (0, \frac{\pi}{2})$ 时成立。
    \item[\textbf{\textcolor{CTrap}{错因分析（凸性改变）：}}] 余弦函数在更大区间上凸性改变，不等式方向可能反转。
\end{description}

\medskip
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CTrap}{易错点二（方向混淆）：}}] 记错凹函数不等式方向，写成 $\frac{\cos x + \cos y}{2} \geq \cos\left( \frac{x+y}{2} \right)$。
    \item[\textbf{\textcolor{CTrap}{正确理解（弦在下）：}}] 凹函数图像弦在曲线下，因此函数值平均不超过平均值的函数值。
    \item[\textbf{\textcolor{CTrap}{错因分析（定义不清）：}}] 对凹函数定义记忆模糊，与凸函数混淆。
\end{description}

\medskip
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CTrap}{易错点三（三角形非锐角）：}}] 在非锐角三角形中应用 $\cos A + \cos B + \cos C \leq \frac{3}{2}$。
    \item[\textbf{\textcolor{CTrap}{正确理解（仅锐角）：}}] 该不等式仅对锐角三角形成立；直角三角形或钝角三角形可能不成立。
    \item[\textbf{\textcolor{CTrap}{错因分析（条件遗漏）：}}] 忽略“锐角三角形”前提，错误推广。
\end{description}
\end{trapbox}
