\begin{examplebox}
\textbf{\textcolor{CExample}{例 1（基础）}}

\textbf{\textcolor{CExample}{题目：}} 设实数 $x$ 满足 $x > -1$，证明 $\ln(1+x) \le x$。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（构造函数）：}] 令 $f(x) = x - \ln(1+x)$，定义域为 $x > -1$。
    \par\textbf{理由：}
    将不等式转化为函数值非负的问题。

    \item[\textbf{第二步（求导分析）：}] 计算导数 $f'(x) = 1 - \frac{1}{1+x} = \frac{x}{1+x}$。
    \par\textbf{理由：}
    通过导数判断函数单调性。

    \item[\textbf{第三步（单调性判断）：}] 当 $x > 0$ 时 $f'(x) > 0$，$f(x)$ 单调递增；当 $-1 < x < 0$ 时 $f'(x) < 0$，$f(x)$ 单调递减。
    \par\textbf{理由：}
    导数符号决定增减性。

    \item[\textbf{第四步（极值点）：}] 在 $x=0$ 处 $f'(x)=0$，且 $f(0)=0$，因此 $f(x) \ge f(0) = 0$。
    \par\textbf{理由：}
    $x=0$ 是极小值点，也是最小值点。

    \item[\textbf{第五步（得出结论）：}] 由 $f(x) \ge 0$ 得 $x \ge \ln(1+x)$，即 $\ln(1+x) \le x$。
    \par\textbf{理由：}
    函数最小值非负，故不等式成立。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{对于 $x > -1$，总有 $\ln(1+x) \le x$，等号当且仅当 $x=0$ 时成立。}

\textbf{\textcolor{CExample}{例 2（稍变形）}}

\textbf{\textcolor{CExample}{题目：}} 设实数 $t > 0$，证明 $\ln t \ge 1 - \frac{1}{t}$。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（变量替换）：}] 令 $x = t - 1$，则 $t = 1+x$，且由 $t > 0$ 得 $x > -1$。
    \par\textbf{理由：}
    将问题转化为已知不等式链的形式。

    \item[\textbf{第二步（代入原式）：}] 根据原不等式链，有 $\frac{x}{1+x} \le \ln(1+x) \le x$。
    \par\textbf{理由：}
    直接应用结论。

    \item[\textbf{第三步（取下界部分）：}] 取左半部分 $\ln(1+x) \ge \frac{x}{1+x}$。
    \par\textbf{理由：}
    我们需要证明 $\ln t \ge 1 - \frac{1}{t}$，对应左半部分。

    \item[\textbf{第四步（换回变量）：}] 将 $x = t-1$ 代入 $\ln(1+x) \ge \frac{x}{1+x}$，得 $\ln t \ge \frac{t-1}{t} = 1 - \frac{1}{t}$。
    \par\textbf{理由：}
    简化表达式即得所求。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{对于 $t > 0$，总有 $\ln t \ge 1 - \frac{1}{t}$，等号当且仅当 $t=1$（即 $x=0$）时成立。}
\end{examplebox}
