\begin{statementbox}
\textbf{\textcolor{CStatement}{条件}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{条件 1（定义域）：}}] 实数 $x$ 满足 $x > -1$。
\end{description}

\textbf{\textcolor{CStatement}{结论}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{\textcolor{CStatement}{结论一（不等式链）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 对数函数被一次函数和分式函数夹逼。
    \[
    \frac{x}{1+x} \le \ln(1+x) \le x
    \]

    \item[\textbf{\textcolor{CStatement}{推广结论（等价变形）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 换元后对数函数与反比例函数的不等式。
    \[
    \ln t \ge 1 - \frac{1}{t}, \quad t > 0.
    \]
\end{description}

\textbf{\textcolor{CStatement}{等号/取等条件：}} 当且仅当 $x=0$（即 $t=1$）时，所有等号成立。
\end{statementbox}
