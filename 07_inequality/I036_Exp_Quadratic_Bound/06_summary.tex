\begin{summarybox}
\textbf{\textcolor{CSummary}{一句话核心}}
\textbf{对于 $x>-1$，$\ln(1+x)$ 满足 $\frac{x}{1+x} \le \ln(1+x) \le x$，等号仅当 $x=0$ 时成立。}

\medskip
\textbf{\textcolor{CSummary}{使用条件}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{条件 1（定义域要求）：} 变量必须满足 $x > -1$（或 $t > 0$）。
    \item \textbf{条件 2（应用场景）：} 需要在证明不等式、求极限或进行对数估计时使用。
\end{itemize}

\medskip
\textbf{\textcolor{CSummary}{关键提醒}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{易错点（等号唯一）：} 等号仅在 $x=0$ 时成立，其他情况为严格不等式。
    \item \textbf{检查项（放缩方向）：} 应用时务必确认是使用上界还是下界，避免方向错误。
\end{itemize}
\end{summarybox}
