\textbf{\textcolor{CStatement}{条件}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{条件 1（任意实数）：}}] $a, b, c$ 为任意实数。
\end{description}

\textbf{\textcolor{CStatement}{结论}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{\textcolor{CStatement}{结论一（平方和与积和）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 平方之和不小于两两乘积之和。
    \[
    a^{2} + b^{2} + c^{2} \geq ab + bc + ca
    \]

    \item[\textbf{\textcolor{CStatement}{推广结论（等价变形）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 与和的平方之间的不等关系。
    \[
    \begin{aligned}
       3\bigl(a^{2} + b^{2} + c^{2}\bigr) &\geq (a+b+c)^{2}, \\
       (a+b+c)^{2} &\geq 3\,(ab+bc+ca).
    \end{aligned}
    \]
\end{description}

\textbf{\textcolor{CStatement}{等号/取等条件：}} 当且仅当 $a = b = c$ 时等号成立。
