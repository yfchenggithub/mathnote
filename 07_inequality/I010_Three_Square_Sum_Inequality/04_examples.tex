\begin{examplebox}
\textbf{\textcolor{CExample}{例 1（基础）}}

\textbf{\textcolor{CExample}{题目：}} 已知 $a, b, c$ 是实数，求证：$a^2+b^2+c^2 \geq ab+bc+ca$。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（移项加倍）：}] 将不等式两边乘以 2 并移项。
    \[
    2(a^2+b^2+c^2) - 2(ab+bc+ca) \geq 0
    \]
    \par\textbf{理由：}
    乘以 2 后便于配方。

    \item[\textbf{第二步（配方组合）：}] 将左边配成三个完全平方和。
    \[
    \begin{aligned}
        &2a^2+2b^2+2c^2-2ab-2bc-2ca \\
        =& (a^2-2ab+b^2) + (b^2-2bc+c^2) + (c^2-2ca+a^2) \\
        =& (a-b)^2 + (b-c)^2 + (c-a)^2
    \end{aligned}
    \]
    \par\textbf{理由：}
    分组后恰好形成平方差。

    \item[\textbf{第三步（利用非负性）：}] 由平方的非负性得出结论。
    \[
    (a-b)^2 + (b-c)^2 + (c-a)^2 \geq 0
    \]
    \par\textbf{理由：}
    平方和总是非负，故原不等式成立。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{原不等式成立，取等当且仅当 $a=b=c$。}

\medskip
\textbf{\textcolor{CExample}{例 2（稍变形）}}

\textbf{\textcolor{CExample}{题目：}} 设 $a, b, c$ 是实数，且 $a+b+c=1$，求 $ab+bc+ca$ 的最大值。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（应用不等式）：}] 由三元平方和不等式的一个等价形式。
    \[
    (a+b+c)^2 \geq 3(ab+bc+ca)
    \]
    \par\textbf{理由：}
    这是原不等式的第三个等价形式。

    \item[\textbf{第二步（代入条件）：}] 将 $a+b+c=1$ 代入不等式。
    \[
    1^2 \geq 3(ab+bc+ca) \quad \Rightarrow \quad ab+bc+ca \leq \frac{1}{3}
    \]
    \par\textbf{理由：}
    不等号方向不变，除以正数 3。

    \item[\textbf{第三步（取等验证）：}] 检查最大值能否取得。
    \[
    \text{当且仅当 } a=b=c=\frac{1}{3} \text{ 时，等号成立。}
    \]
    \par\textbf{理由：}
    等号成立条件为 $a=b=c$，结合 $a+b+c=1$ 即得。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{$ab+bc+ca$ 的最大值为 $\dfrac{1}{3}$，当 $a=b=c=\dfrac{1}{3}$ 时取得。}
\end{examplebox}
