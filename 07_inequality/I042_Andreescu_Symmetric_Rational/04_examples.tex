\begin{examplebox}
\textbf{\textcolor{CExample}{例 1（基础）}}

\textbf{\textcolor{CExample}{题目：}} 已知 $a, b, c > 0$，证明：$\frac{a^2}{b} + \frac{b^2}{c} + \frac{c^2}{a} \geq a+b+c$。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（识别结构）：}] 观察题目结构与已知结论完全一致。
    \par\textbf{理由：} 将 $(x,y,z)$ 对应为 $(a,b,c)$，即为结论的直接形式。

    \item[\textbf{第二步（应用结论）：}] 直接写出已知的不等式结论。
    \[
        \frac{a^2}{b} + \frac{b^2}{c} + \frac{c^2}{a} \geq a + b + c.
    \]
    \par\textbf{理由：} 根据“三元循环平方分式不等式”，该式恒成立。

    \item[\textbf{第三步（说明取等）：}] 指出等号成立的条件。
    \par\textbf{理由：} 当且仅当 $a = b = c$ 时，上述不等式取等号。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{对于任意正实数 $a,b,c$，不等式 $\frac{a^2}{b} + \frac{b^2}{c} + \frac{c^2}{a} \geq a+b+c$ 成立，等号在 $a=b=c$ 时取得。}

\medskip
\textbf{\textcolor{CExample}{例 2（稍变形）}}

\textbf{\textcolor{CExample}{题目：}} 设 $x, y, z > 0$，且 $xyz=1$。求 $\frac{x^2}{y} + \frac{y^2}{z} + \frac{z^2}{x}$ 的最小值。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（应用主结论）：}] 直接应用已知不等式得到一个下界。
    \[
        \frac{x^2}{y} + \frac{y^2}{z} + \frac{z^2}{x} \geq x + y + z.
    \]
    \par\textbf{理由：} 题目条件 $x,y,z>0$ 满足不等式应用前提。

    \item[\textbf{第二步（利用条件放缩）：}] 在 $xyz=1$ 的条件下，对 $x+y+z$ 进一步使用均值不等式。
    \[
        x + y + z \geq 3 \sqrt[3]{xyz} = 3.
    \]
    \par\textbf{理由：} 三个正数的算术平均数不小于几何平均数，且 $xyz=1$。

    \item[\textbf{第三步（得到最小值）：}] 结合前两步，得出原式的最小值。
    \[
        \frac{x^2}{y} + \frac{y^2}{z} + \frac{z^2}{x} \geq x + y + z \geq 3.
    \]
    \par\textbf{理由：} 不等式具有传递性。因此原式的最小值不小于 $3$。

    \item[\textbf{第四步（验证可达到）：}] 检查等号能否同时成立。
    \par\textbf{理由：} 第一个不等式取等需 $x=y=z$，第二个不等式取等也需 $x=y=z$。结合 $xyz=1$，得 $x=y=z=1$。此时原式值为 $3$。故最小值为 $3$。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{在 $x,y,z>0$ 且 $xyz=1$ 的条件下，$\frac{x^2}{y} + \frac{y^2}{z} + \frac{z^2}{x}$ 的最小值为 $3$，当 $x=y=z=1$ 时取得。}
\end{examplebox}
