\textbf{\textcolor{CStatement}{条件}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{条件 1（正实数）：}}] 变量 $x, y, z$ 都是正实数。
\end{description}

\textbf{\textcolor{CStatement}{结论}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{\textcolor{CStatement}{结论一（三元循环不等式）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 分式的平方和大于等于变量直接和。
    \[
        \frac{x^{2}}{y} + \frac{y^{2}}{z} + \frac{z^{2}}{x} \geq x + y + z.
    \]

    \item[\textbf{\textcolor{CStatement}{推广结论（加权推广形式）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 每个分式项乘以正权重的推广。
    \[
        \frac{x^{2}}{y} + \frac{y^{2}}{z} + \frac{z^{2}}{x} \geq \frac{(x+y+z)^{2}}{x+y+z} = x+y+z.
    \]
\end{description}

\textbf{\textcolor{CStatement}{等号/取等条件：}} 当且仅当 $x = y = z$ 时，不等式取等号。

\end{statementbox}
