\begin{summarybox}
\textbf{\textcolor{CSummary}{一句话核心}}
\textbf{伯努利不等式：用切线 $1+nx$ 逼近 $(1+x)^n$，逼近的方向（上或下）由指数 $n$ 是否在 $[0,1]$ 区间内决定。}

\medskip
\textbf{\textcolor{CSummary}{使用条件}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{条件 1（底数非负）：} 必须满足 $x \geq -1$，即 $1+x \geq 0$。
    \item \textbf{条件 2（指数分界）：} 明确 $n$ 的范围：$n \geq 1$ 或 $n \leq 0$ 时用“$\geq$”；$0 \leq n \leq 1$ 时用“$\leq$”。
\end{itemize}

\medskip
\textbf{\textcolor{CSummary}{关键提醒}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{易错点（方向混淆）：} 牢记“出则大（$\geq$），入则小（$\leq$）”的口诀，对应指数 $n$ 跑出或跑进区间 $[0,1]$。
    \item \textbf{检查项（形式匹配）：} 使用前检查式子是否为 $(1+\text{某式})^{\text{指数}}$ 并与 $1+\text{指数} \times \text{某式}$ 比较，若不是，先做变量代换。
    \item \textbf{等号条件（三点）：} 等号成立只在 $n=0$、$n=1$ 或 $x=0$ 时，做题后顺手验证一下。
\end{itemize}
\end{summarybox}
