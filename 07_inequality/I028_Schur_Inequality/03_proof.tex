\begin{proofbox}
\textbf{\textcolor{CProof}{思路提示}}
固定 $x \geq -1$，将 $(1+x)^n$ 与 $1+nx$ 的差视为关于指数 $n$ 的函数，通过研究该函数的单调性来证明不等式。也可以考虑固定 $n$，将差值视为关于 $x$ 的函数，用导数求最值。这里采用第一种思路，构造函数 $g(n) = (1+x)^n - (1+nx)$。

\medskip
\textbf{\textcolor{CProof}{正式推导}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.55em]
    \item[\textbf{步骤一（构造函数）：}] 令 $x \geq -1$ 为固定实数。定义函数
    \[
    g(n) = (1+x)^n - 1 - nx, \quad n \in \mathbb{R}.
    \]
    \par\textbf{理由：} 我们将要证明的结论转化为判断函数 $g(n)$ 的符号。

    \item[\textbf{步骤二（求导判单调）：}] 对 $g(n)$ 关于 $n$ 求导。注意 $(1+x)^n = e^{n \ln(1+x)}$，其中 $1+x > 0$（因为 $x \geq -1$，且当 $x=-1$ 时 $(1+x)^n=0^n$ 需单独讨论，这里先设 $x>-1$）。
    \[
    g'(n) = \frac{d}{dn} \left[ e^{n \ln(1+x)} \right] - x = e^{n \ln(1+x)} \cdot \ln(1+x) - x = (1+x)^n \ln(1+x) - x.
    \]
    \par\textbf{理由：} 为了研究 $g(n)$ 的单调性，需要分析其导数 $g'(n)$ 的符号。

    \item[\textbf{步骤三（分析导函数）：}] 进一步对 $g'(n)$ 关于 $n$ 求导（即求二阶导）：
    \[
    g''(n) = \frac{d}{dn} \left[ (1+x)^n \ln(1+x) \right] = (1+x)^n \left[ \ln(1+x) \right]^2.
    \]
    由于 $(1+x)^n > 0$（$x>-1$），且 $\left[ \ln(1+x) \right]^2 \geq 0$，故 $g''(n) \geq 0$。
    \par\textbf{理由：} $g''(n) \geq 0$ 说明 $g'(n)$ 关于 $n$ 单调不减。

    \item[\textbf{步骤四（寻找临界点）：}] 令 $g'(n)=0$，解出可能的极值点。注意到 $g'(0) = (1+x)^0 \ln(1+x) - x = \ln(1+x) - x$。由常用不等式 $\ln(1+x) \leq x$（$x>-1$，等号当 $x=0$ 时成立）知 $g'(0) \leq 0$。又 $g'(1) = (1+x) \ln(1+x) - x$。可以验证当 $x \neq 0$ 时，$(1+x) \ln(1+x) > x$（可通过构造函数证明），故 $g'(1) > 0$。由于 $g'(n)$ 单调不减，且 $g'(0) \leq 0$，$g'(1) \geq 0$，故存在 $n_0 \in [0,1]$ 使得 $g'(n_0)=0$。
    \par\textbf{理由：} 结合单调性和端点符号，确定 $g'(n)$ 的符号变化情况。

    \item[\textbf{步骤五（确定原函数最值）：}] $g'(n)$ 的符号决定了 $g(n)$ 的单调性：
    \begin{itemize}[leftmargin=*, itemsep=0.2em, topsep=0.2em]
        \item 当 $n \leq n_0$ 时，$g'(n) \leq 0$，$g(n)$ 单调不增。
        \item 当 $n \geq n_0$ 时，$g'(n) \geq 0$，$g(n)$ 单调不减。
    \end{itemize}
    因此，$g(n)$ 在 $n = n_0$ 处取得最小值。计算 $g(0)=0$，$g(1)=0$。由于最小值点 $n_0$ 在 $[0,1]$ 之间，且端点值均为 $0$，故在 $[0,1]$ 上 $g(n) \leq 0$，即 $(1+x)^n \leq 1+nx$；在 $n \leq 0$ 或 $n \geq 1$ 时，$g(n)$ 从 $0$ 出发先减后增或直接增，但端点值 $g(0)=0$ 或 $g(1)=0$，故 $g(n) \geq 0$，即 $(1+x)^n \geq 1+nx$。
    \par\textbf{理由：} 通过单调性和端点值确定函数 $g(n)$ 在整个定义域上的符号。

    \item[\textbf{步骤六（处理边界情况）：}] 当 $x=-1$ 时，左边 $(1+x)^n = 0^n$。若 $n>0$，则 $0^n=0$，右边 $1+n\cdot(-1)=1-n$。显然 $0 \geq 1-n$ 当且仅当 $n \geq 1$；$0 \leq 1-n$ 当且仅当 $n \leq 1$。这与结论一致（注意此时等号成立条件需单独验证）。当 $n=0$ 时，$(1+x)^0=1$，右边 $1+0\cdot x=1$，等号成立。结论仍成立。
    \par\textbf{理由：} 补全 $x=-1$ 的特殊情形，使证明完整。
\end{description}

\medskip
\textbf{\textcolor{CProof}{结论回扣}}
\textbf{综上，对任意实数 $x \geq -1$ 和 $n \in \mathbb{R}$，伯努利不等式 $(1+x)^n \geq 1+nx$（当 $n \geq 1$ 或 $n \leq 0$ 时）与 $(1+x)^n \leq 1+nx$（当 $0 \leq n \leq 1$ 时）成立，等号当且仅当 $n=0,1$ 或 $x=0$ 时成立。}
\end{proofbox}
