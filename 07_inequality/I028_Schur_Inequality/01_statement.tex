\textbf{\textcolor{CStatement}{条件}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{条件 1（实数范围）：}}] 实数 $x \geq -1$，指数 $n \in \mathbb{R}$。
\end{description}

\textbf{\textcolor{CStatement}{结论}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{\textcolor{CStatement}{结论一（指数分界）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 指数范围决定不等式方向。
    \[
    \begin{aligned}
        &\text{当 } n \geq 1 \text{ 或 } n \leq 0 \text{ 时，} && (1+x)^{n} \geq 1+nx; \\
        &\text{当 } 0 \leq n \leq 1 \text{ 时，} && (1+x)^{n} \leq 1+nx.
    \end{aligned}
    \]

    \item[\textbf{\textcolor{CStatement}{推广结论（指数形式）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 指数函数在零点切线的上方。
    \[
    e^{x} \geq 1+x \quad (\forall x \in \mathbb{R})
    \]
\end{description}

\textbf{\textcolor{CStatement}{等号/取等条件：}} 等号成立当且仅当 $n=0$，$n=1$，或 $x=0$。对于 $e^{x} \geq 1+x$，等号成立当且仅当 $x=0$。

\end{statementbox}
