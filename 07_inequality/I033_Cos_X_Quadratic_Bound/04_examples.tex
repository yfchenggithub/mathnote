\begin{examplebox}
\textbf{\textcolor{CExample}{例 1（基础）}}

\textbf{\textcolor{CExample}{题目：}} 已知 $a > b > 0$，比较 $\sqrt{ab}$、$\dfrac{a-b}{\ln a - \ln b}$、$\dfrac{a+b}{2}$ 三者的大小关系。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（识别结构）：}] 观察三个表达式，分别是几何平均 $G$、对数平均 $L$ 和算术平均 $A$。
    \par\textbf{理由：} 题目形式与对数平均值不等式的结构完全一致。

    \item[\textbf{第二步（直接应用）：}] 由对数平均值不等式，对任意两个不相等的正实数 $a, b$，有 $\sqrt{ab} < \dfrac{a-b}{\ln a - \ln b} < \dfrac{a+b}{2}$。
    \par\textbf{理由：} 直接应用结论，无需额外推导。

    \item[\textbf{第三步（写出结论）：}] 因此，三者的大小关系为 $\sqrt{ab} < \dfrac{a-b}{\ln a - \ln b} < \dfrac{a+b}{2}$。
    \par\textbf{理由：} 由于 $a>b>0$，满足不等式条件，结论成立。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{对于不相等的正数，几何平均 $<$ 对数平均 $<$ 算术平均。}

\textbf{\textcolor{CExample}{例 2（稍变形）}}

\textbf{\textcolor{CExample}{题目：}} 设 $x > 0,\, x \neq 1$，证明：$\ln x > \dfrac{2(x-1)}{x+1}$。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（联想等价形式）：}] 对数平均值不等式有一个等价形式：$\ln\dfrac{x_1}{x_2} > \dfrac{2(x_1-x_2)}{x_1+x_2}$。
    \par\textbf{理由：} 题目要证的不等式与这个等价形式结构相似。

    \item[\textbf{第二步（变量替换）：}] 令 $x_1 = x$，$x_2 = 1$，则 $x_1, x_2 > 0$ 且 $x_1 \neq x_2$。代入等价形式：
    \[
    \ln\frac{x}{1} > \frac{2(x-1)}{x+1}.
    \]
    \par\textbf{理由：} 通过取 $x_2=1$ 将一般形式特殊化为题目所给不等式。

    \item[\textbf{第三步（得出结论）：}] 由于对数平均值不等式成立，其等价形式也成立，因此 $\ln x > \dfrac{2(x-1)}{x+1}$ 得证。
    \par\textbf{理由：} 特殊化是原结论的直接推论。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{不等式 $\ln x > \dfrac{2(x-1)}{x+1}$（$x>0,\, x\neq 1$）是对数平均值不等式 $L < A$ 的一个等价形式。}
\end{examplebox}
