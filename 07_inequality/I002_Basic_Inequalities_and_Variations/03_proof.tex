\begin{proofbox}
\textbf{思路提示}

利用非负数的平方总是非负这一基本事实，通过展开和代数变形推导出各个不等式。

\medskip
\textbf{正式推导}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.55em]
    \item[\textbf{步骤一：}] 证明重要不等式 $a^2+b^2 \ge 2ab$。
    \[
    (a-b)^2 \ge 0 \Rightarrow a^2 - 2ab + b^2 \ge 0 \Rightarrow a^2 + b^2 \ge 2ab
    \]
    \par\textbf{理由：}
    任意实数的平方非负，展开后移项即得。

    \item[\textbf{步骤二：}] 证明基本不等式 $a+b \ge 2\sqrt{ab}$（$a,b>0$）。
    \[
    (\sqrt{a} - \sqrt{b})^2 \ge 0 \Rightarrow a - 2\sqrt{ab} + b \ge 0 \Rightarrow a + b \ge 2\sqrt{ab}
    \]
    \par\textbf{理由：}
    正实数的平方根是实数，应用平方非负性直接导出。

    \item[\textbf{步骤三：}] 推导常用变形 $ab \le \left(\frac{a+b}{2}\right)^2$。
    \[
    \text{由基本不等式 } \sqrt{ab} \le \frac{a+b}{2} \text{，两边平方得 } ab \le \left(\frac{a+b}{2}\right)^2
    \]
    \par\textbf{理由：}
    基本不等式两边平方，由于两边非负，不等号方向不变。

    \item[\textbf{步骤四：}] 推导常用变形 $a^2+b^2 \ge \frac{(a+b)^2}{2}$。
    \[
    a^2+b^2 \ge 2ab \Rightarrow 2(a^2+b^2) \ge a^2+b^2+2ab = (a+b)^2 \Rightarrow a^2+b^2 \ge \frac{(a+b)^2}{2}
    \]
    \par\textbf{理由：}
    将重要不等式乘以 2 后与平方和展开结合，再除以 2。
\end{description}

\medskip
\textbf{结论回扣}

\textbf{所有不等式均源于平方非负性，等号成立当且仅当 $a=b$，验证了结论的一致性。}
\end{proofbox}
