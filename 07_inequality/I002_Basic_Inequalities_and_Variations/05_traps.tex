\begin{trapbox}
\textbf{易错点一}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{错误想法：}] 基本不等式 $a+b \ge 2\sqrt{ab}$ 对任意实数 $a,b$ 都成立。
    \item[\textbf{正确理解：}] 基本不等式要求 $a,b$ 为正实数；重要不等式 $a^2+b^2 \ge 2ab$ 才对任意实数成立。
    \item[\textbf{错因分析：}] 混淆了不同不等式的适用条件，忽略正数限制会导致错误结论。
\end{description}

\medskip
\textbf{易错点二}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{错误想法：}] 应用不等式后得到最值，但等号成立条件在问题中无法满足。
    \item[\textbf{正确理解：}] 所有不等式的等号成立当且仅当 $a=b$，使用时必须验证该条件是否可达。
    \item[\textbf{错因分析：}] 未将理论等号条件与实际问题约束结合，可能导致最值不可达。
\end{description}

\medskip
\textbf{易错点三}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{错误想法：}] 误记常用变形为 $ab \ge \left(\frac{a+b}{2}\right)^2$ 或 $a^2+b^2 \le \frac{(a+b)^2}{2}$。
    \item[\textbf{正确理解：}] 正确变形是 $ab \le \left(\frac{a+b}{2}\right)^2$ 和 $a^2+b^2 \ge \frac{(a+b)^2}{2}$。
    \item[\textbf{错因分析：}] 对不等式方向记忆模糊，或未理解变形推导过程，导致放缩方向错误。
\end{description}
\end{trapbox}
