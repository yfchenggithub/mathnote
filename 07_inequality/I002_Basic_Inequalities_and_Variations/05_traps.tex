\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=1.0em, topsep=0.5em]
    \item[\textbf{易错点1}]
    \begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
        \item[\textbf{错误想法：}] 认为基本不等式 $a+b \geq 2\sqrt{ab}$ 对所有实数都成立。
        \item[\textbf{正确理解：}] 该不等式要求 $a, b$ 均为正实数。若 $a,b$ 为负，根式 $\sqrt{ab}$ 可能无意义；若一正一负，不等式方向可能改变。
        \item[\textbf{错因分析：}] 忽视了算术平方根的定义域和不等式成立的前提条件。必须确保 $a,b$ 非负，才能应用。
    \end{description}
    \item[\textbf{易错点2}]
    \begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
        \item[\textbf{错误想法：}] 利用不等式求最值时，仅得出最值表达式，未检查等号能否成立。
        \item[\textbf{正确理解：}] 不等式给出的界不一定能取到。必须验证等号成立的条件是否在题目允许范围内。
        \item[\textbf{错因分析：}] 混淆了“上界”与“最大值”。只有当等号能取到时，这个界才是真正的最值。
    \end{description}
    \item[\textbf{易错点3}]
    \begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
        \item[\textbf{错误想法：}] 将变形 $ab \leq \left(\frac{a+b}{2}\right)^2$ 中的 $\frac{a+b}{2}$ 误认为是算术平均数，而忽略该不等式本身也是基本不等式的直接推论。
        \item[\textbf{正确理解：}] 该变形是 $a+b \geq 2\sqrt{ab}$ 两边平方并整理得到的，其取等条件依然是 $a=b$。
        \item[\textbf{错因分析：}] 对公式的来源理解不透，机械记忆，导致在复杂变形中无法灵活运用或错误放缩。
    \end{description}
\end{description}
\end{trapbox}
