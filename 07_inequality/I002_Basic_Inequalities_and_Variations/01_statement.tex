\begin{statementbox}

\textbf{条件：} 设 $a, b$ 为正实数（其中结论 2 对任意实数 $a, b$ 也成立）。

\textbf{结论：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{结论 1：}]
    \textbf{直观描述：} 算术平均数不小于几何平均数。
    \[
    a+b \ge 2\sqrt{ab}
    \]

    \item[\textbf{结论 2：}]
    \textbf{直观描述：} 平方和不小于两倍乘积。
    \[
    a^2+b^2 \ge 2ab
    \]

    \item[\textbf{结论 3：}]
    \textbf{直观描述：} 乘积有上界，与和的平方相关。
    \[
    ab \le \left(\frac{a+b}{2}\right)^2
    \]

    \item[\textbf{结论 4：}]
    \textbf{直观描述：} 平方和有下界，与和的平方相关。
    \[
    a^2+b^2 \ge \frac{(a+b)^2}{2}
    \]
\end{description}

\textbf{等号成立条件：} 当且仅当 $a = b$。

\end{statementbox}
