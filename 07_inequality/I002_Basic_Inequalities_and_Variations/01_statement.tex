\begin{statementbox}

\textbf{条件：} $a, b$ 为正实数（结论2对任意实数 $a,b$ 也成立）。

\textbf{结论：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{结论1：}]
    \[
    a + b \geq 2\sqrt{ab}
    \]
    \item[\textbf{结论2：}]
    \[
    a^2 + b^2 \geq 2ab
    \]
    \item[\textbf{结论3：}]
    \[
    ab \leq \left(\frac{a+b}{2}\right)^2
    \]
    \item[\textbf{结论4：}]
    \[
    a^2 + b^2 \geq \frac{(a+b)^2}{2}
    \]
\end{description}

\textbf{等号成立条件：} 以上各不等式等号成立当且仅当 $a = b$。

\end{statementbox}
