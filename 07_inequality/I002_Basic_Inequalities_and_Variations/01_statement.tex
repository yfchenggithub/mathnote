\begin{statementbox}

\textbf{条件：}设 $a, b$ 为实数。结论 1 和 3 要求 $a, b \ge 0$；结论 2 和 4 对任意实数 $a, b$ 成立。

\textbf{结论：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{结论 1：}]（算术-几何平均）
        \[
        a + b \ge 2\sqrt{ab}
        \]
    \item[\textbf{结论 2：}]（平方和不等式）
        \[
        a^2 + b^2 \ge 2ab
        \]
    \item[\textbf{结论 3：}]（积的界）
        \[
        ab \le \left(\frac{a+b}{2}\right)^2
        \]
    \item[\textbf{结论 4：}]（和的平方与平方和关系）
        \[
        a^2 + b^2 \ge \frac{(a+b)^2}{2}
        \]
\end{description}

\textbf{等号成立条件：}以上四个不等式等号成立当且仅当 $a = b$。

\end{statementbox}
