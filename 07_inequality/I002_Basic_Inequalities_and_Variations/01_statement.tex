\textbf{\textcolor{CStatement}{条件}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{条件 1（正实数）：}}] $a, b$ 均为正实数。
    \item[\textbf{\textcolor{CStatement}{条件 2（任意实数）：}}] $a, b$ 为任意实数（仅适用于第4条）。
\end{description}

\textbf{\textcolor{CStatement}{结论}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{\textcolor{CStatement}{结论一（算术-几何平均）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 两正数之和不小于其几何平均数两倍。
    \[
    a + b \geq 2\sqrt{ab}
    \]

    \item[\textbf{\textcolor{CStatement}{结论二（积与和平方）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 两正数积不超过其算术平均平方。
    \[
    ab \leq \left(\frac{a+b}{2}\right)^2
    \]

    \item[\textbf{\textcolor{CStatement}{结论三（平方和与和平方）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 两正数平方和不小于其和平方的一半。
    \[
    a^2 + b^2 \geq \frac{(a+b)^2}{2}
    \]

    \item[\textbf{\textcolor{CStatement}{推广结论（平方和与积）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 任意实数平方和不小于其积两倍。
    \[
    a^2 + b^2 \geq 2ab
    \]
\end{description}

\textbf{\textcolor{CStatement}{等号/取等条件：}} 当且仅当 $a = b$ 时，上述所有不等式等号成立。

\end{statementbox}
