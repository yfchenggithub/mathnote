\begin{statementbox}

\textbf{条件：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{条件 1（正实数）：}] $a,b \in \mathbb{R}^+$（即 $a>0,\,b>0$）。
\end{description}

\textbf{结论：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{结论一（算术-几何平均）：}]
    \textbf{直观描述：} 两数算术平均不小于几何平均。
    \[
    a+b \geq 2\sqrt{ab}
    \]
    
    \item[\textbf{结论二（平方和与积）：}]
    \textbf{直观描述：} 两数平方和不小于两倍乘积。
    \[
    a^2+b^2 \geq 2ab
    \]
    
    \item[\textbf{结论三（乘积上界）：}]
    \textbf{直观描述：} 两数乘积不超过算术平均的平方。
    \[
    ab \leq \left(\frac{a+b}{2}\right)^2
    \]
    
    \item[\textbf{结论四（平方和下界）：}]
    \textbf{直观描述：} 两数平方和不小于和的一半的平方。
    \[
    a^2+b^2 \geq \frac{(a+b)^2}{2}
    \]
\end{description}

\textbf{等号/取等条件：} 以上各不等式等号成立当且仅当 $a=b$。

\end{statementbox}
