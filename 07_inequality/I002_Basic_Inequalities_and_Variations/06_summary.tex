\begin{summarybox}
\textbf{一句话核心}

\textbf{基本不等式及其变形刻画了正实数和、积、平方和之间的不等关系，等号成立当且仅当两数相等。}

\medskip
\textbf{使用条件}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{条件 1：} 应用基本不等式 $a+b \ge 2\sqrt{ab}$ 时，必须保证 $a,b > 0$。
    \item \textbf{条件 2：} 重要不等式 $a^2+b^2 \ge 2ab$ 对任意实数成立，但常用变形基于正数条件。
\end{itemize}

\medskip
\textbf{关键提醒}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{易错点：} 忽略等号成立条件 $a=b$ 或误用不等式方向。
    \item \textbf{检查项：} 使用前验证变量为正，应用后确认等号可达，变形时保持不等号方向正确。
\end{itemize}
\end{summarybox}
