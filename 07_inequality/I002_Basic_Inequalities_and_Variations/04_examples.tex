\begin{examplebox}
\textbf{\textcolor{CExample}{例 1（基础）}}

\textbf{\textcolor{CExample}{题目：}} 已知 $x > 0$，求函数 $y = x + \dfrac{9}{x}$ 的最小值。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（判断正数）：}] 由于 $x > 0$，则 $\dfrac{9}{x} > 0$，满足“一正”。
    \par\textbf{理由：} 基本不等式 $a + b \ge 2\sqrt{ab}$ 要求 $a, b$ 均为正数。

    \item[\textbf{第二步（套用公式）：}] 直接应用算术-几何平均不等式：
    \[
    y = x + \frac{9}{x} \ge 2\sqrt{x \cdot \frac{9}{x}} = 2\sqrt{9} = 6.
    \]
    \par\textbf{理由：} 这里 $a = x$， $b = \frac{9}{x}$，且 $ab = x \cdot \frac{9}{x} = 9$ 为定值，满足“二定”。

    \item[\textbf{第三步（验证取等）：}] 等号成立当 $x = \dfrac{9}{x}$，即 $x^2 = 9$，解得 $x = 3$（舍去负值）。
    \par\textbf{理由：} 检查“三相等”，确保最小值可以取到。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{当 $x = 3$ 时，$y_{\min} = 6$。}

\textbf{\textcolor{CExample}{例 2（稍变形）}}

\textbf{\textcolor{CExample}{题目：}} 已知 $a, b > 0$ 且 $a+b=1$，求证：$\left(a+\dfrac{1}{a}\right)\left(b+\dfrac{1}{b}\right) \ge \dfrac{25}{4}$。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（展开化简）：}] 左边展开：
    \[
    \left(a+\frac{1}{a}\right)\left(b+\frac{1}{b}\right) = ab + \frac{a}{b} + \frac{b}{a} + \frac{1}{ab}.
    \]
    \par\textbf{理由：} 多项式乘法。

    \item[\textbf{第二步（利用条件放缩）：}] 由 $a+b=1$ 和 $a,b>0$，根据 $ab \le \left(\frac{a+b}{2}\right)^2 = \frac{1}{4}$，得 $\frac{1}{ab} \ge 4$。又由 $\frac{a}{b} + \frac{b}{a} \ge 2\sqrt{\frac{a}{b} \cdot \frac{b}{a}} = 2$（应用 $a^2+b^2 \ge 2ab$ 的变形）。
    \par\textbf{理由：} 分别对乘积项 $ab$ 和倒数项 $\frac{a}{b}+\frac{b}{a}$ 应用基本不等式或其变形。

    \item[\textbf{第三步（代入求下界）：}] 将 $ab \le \frac{1}{4}$ 代入 $ab + \frac{1}{ab}$，但注意 $ab + \frac{1}{ab}$ 在 $(0, \frac{1}{4}]$ 上单调递减，故当 $ab=\frac{1}{4}$ 时取得最小值。同时 $\frac{a}{b}+\frac{b}{a} \ge 2$。因此
    \[
    \text{原式} \ge \left(\frac{1}{4} + 4\right) + 2 = \frac{1}{4} + 4 + 2 = \frac{25}{4}.
    \]
    \par\textbf{理由：} 将各部分的极小值组合，得到整体的下界。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{不等式得证，且等号当 $a=b=\frac{1}{2}$ 时成立。}
\end{examplebox}
