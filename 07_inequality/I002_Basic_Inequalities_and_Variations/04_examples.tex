\begin{examplebox}
\textbf{例 1（基础）}

\textbf{题目：} 已知 $x > 0$，求函数 $f(x) = x + \frac{4}{x}$ 的最小值。

\textbf{解题步骤：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步：}] 识别形式，直接应用基本不等式。
    \[
    f(x) = x + \frac{4}{x} \ge 2\sqrt{x \cdot \frac{4}{x}}
    \]
    \par\textbf{理由：}
    基本不等式要求两正数，这里 $x$ 和 $\frac{4}{x}$ 均为正。

    \item[\textbf{第二步：}] 简化右侧表达式。
    \[
    2\sqrt{x \cdot \frac{4}{x}} = 2\sqrt{4} = 4
    \]
    \par\textbf{理由：}
    算术计算，注意 $\sqrt{4}=2$。

    \item[\textbf{第三步：}] 确定等号成立条件。
    \[
    \text{等号成立当 } x = \frac{4}{x} \Rightarrow x^2 = 4 \Rightarrow x = 2 \ (x>0)
    \]
    \par\textbf{理由：}
    基本不等式取等条件是两数相等。
\end{description}

\textbf{关键结论：} \textbf{函数 $f(x)$ 的最小值为 $4$，当 $x=2$ 时取得。}

\textbf{例 2（稍变形）}

\textbf{题目：} 已知 $a, b > 0$ 且 $a + b = 2$，求 $ab$ 的最大值。

\textbf{解题步骤：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步：}] 利用已知条件和常用变形。
    \[
    ab \le \left(\frac{a+b}{2}\right)^2
    \]
    \par\textbf{理由：}
    常用变形提供了乘积的上界。

    \item[\textbf{第二步：}] 代入 $a+b=2$。
    \[
    ab \le \left(\frac{2}{2}\right)^2 = 1^2 = 1
    \]
    \par\textbf{理由：}
    代数代入计算。

    \item[\textbf{第三步：}] 检查等号成立条件。
    \[
    \text{等号成立当 } a = b，结合 a+b=2 \Rightarrow a=b=1
    \]
    \par\textbf{理由：}
    常用变形取等条件与基本不等式一致。
\end{description}

\textbf{关键结论：} \textbf{$ab$ 的最大值为 $1$，当 $a=b=1$ 时取得。}
\end{examplebox}
