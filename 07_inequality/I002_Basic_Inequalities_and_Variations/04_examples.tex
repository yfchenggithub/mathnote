\begin{examplebox}
\textbf{例 1（基础）}

\textbf{题目：} 已知 $x>0$，求函数 $f(x)=x+\dfrac{1}{x}$ 的最小值。

\textbf{解题步骤：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（正数确认）：}] 因为 $x>0$，所以 $x$ 和 $\dfrac{1}{x}$ 都是正实数。
    \par\textbf{理由：} 基本不等式要求两个数均为正数。
    \item[\textbf{第二步（应用不等式）：}] 由基本不等式 $a+b \geq 2\sqrt{ab}$，取 $a=x$，$b=\dfrac{1}{x}$，得
    \[
    x+\frac{1}{x} \geq 2\sqrt{x \cdot \frac{1}{x}} = 2.
    \]
    \par\textbf{理由：} 直接代入基本不等式，并计算几何平均。
    \item[\textbf{第三步（取等条件）：}] 等号成立当且仅当 $x = \dfrac{1}{x}$，即 $x^2=1$，结合 $x>0$ 得 $x=1$。
    \par\textbf{理由：} 检查等号成立条件，确保最小值可以取到。
\end{description}

\textbf{关键结论：} \textbf{函数 $f(x)=x+\frac{1}{x}$ 的最小值为 $2$，当 $x=1$ 时取得。}

\textbf{例 2（稍变形）}

\textbf{题目：} 已知 $a,b>0$，且 $a^2+b^2=2$，求证：$a+b \leq 2$。

\textbf{解题步骤：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（选择变形）：}] 由常用变形 $a^2+b^2 \geq \dfrac{(a+b)^2}{2}$ 可得
    \[
    (a+b)^2 \leq 2(a^2+b^2).
    \]
    \par\textbf{理由：} 将不等式 $a^2+b^2 \geq \frac{(a+b)^2}{2}$ 两边乘以 $2$ 并移项。
    \item[\textbf{第二步（代入条件）：}] 代入已知条件 $a^2+b^2=2$，得
    \[
    (a+b)^2 \leq 2 \times 2 = 4.
    \]
    \par\textbf{理由：} 利用条件进行具体计算。
    \item[\textbf{第三步（开方得证）：}] 由于 $a+b>0$，对不等式两边开方得
    \[
    a+b \leq 2.
    \]
    \par\textbf{理由：} 正数开方保持不等号方向。
\end{description}

\textbf{关键结论：} \textbf{在条件 $a,b>0$ 且 $a^2+b^2=2$ 下，有 $a+b \leq 2$，等号当 $a=b=1$ 时成立。}
\end{examplebox}
