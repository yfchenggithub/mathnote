\begin{examplebox}
\textbf{例 1（基础）}

\textbf{题目：} 已知 $x > 0$，求 $x + \frac{4}{x}$ 的最小值。

\textbf{解题步骤：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步：}] 识别结构。 表达式为 $x + \frac{4}{x}$，是两正数 $a=x$ 和 $b=\frac{4}{x}$ 的和。
    \item[\textbf{第二步：}] 应用结论 1。 由于 $x > 0$， $\frac{4}{x} > 0$，满足 $a,b \ge 0$ 的条件。故有
        \[
        x + \frac{4}{x} \ge 2\sqrt{x \cdot \frac{4}{x}}.
        \]
    \item[\textbf{第三步：}] 计算并确定取等条件。
        \[
        2\sqrt{x \cdot \frac{4}{x}} = 2\sqrt{4} = 4.
        \]
        等号成立当且仅当 $x = \frac{4}{x}$，即 $x^2 = 4$，解得 $x = 2$（因为 $x>0$）。
\end{description}

\textbf{关键结论：} \textbf{当 $x = 2$ 时，$x + \frac{4}{x}$ 取得最小值 $4$。}

\medskip
\textbf{例 2（稍变形）}

\textbf{题目：} 已知 $x, y \in \mathbb{R}$，且 $x^2 + y^2 = 4$，求 $xy$ 的最大值。

\textbf{解题步骤：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步：}] 识别可用关系。 已知 $x^2 + y^2$ 为定值 $4$，结论 2 直接给出了 $x^2+y^2$ 与 $2xy$ 的关系，但目标 $xy$ 带系数 2。结论 4 也可用。
    \item[\textbf{第二步：}] 应用结论 2。 $a^2 + b^2 \ge 2ab$ 中，令 $a=x, b=y$，代入 $x^2 + y^2 = 4$，得
        \[
        4 \ge 2xy.
        \]
    \item[\textbf{第三步：}] 解出 $xy$ 并确定取等条件。
        \[
        2xy \le 4 \quad \Rightarrow \quad xy \le 2.
        \]
        等号成立当且仅当 $x = y$。结合 $x^2+y^2=4$，可解得 $x = y = \pm\sqrt{2}$。
\end{description}

\textbf{关键结论：} \textbf{$xy$ 的最大值为 $2$，当且仅当 $x = y = \sqrt{2}$ 或 $x = y = -\sqrt{2}$ 时取得。}
\end{examplebox}
