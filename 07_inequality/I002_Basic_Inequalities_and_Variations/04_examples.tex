\begin{examplebox}
\textbf{例1（基础）}

\textbf{题目：} 设 $x > 0$，求 $x + \frac{1}{x}$ 的最小值。

\textbf{解题步骤：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步：}] 识别形式。表达式为 $x + \frac{1}{x}$，是两个正数之和，符合 $a+b$ 的形式。
    \item[\textbf{第二步：}] 应用基本不等式 $a+b \geq 2\sqrt{ab}$。这里 $a = x$，$b = \frac{1}{x}$，则
    \[
    x + \frac{1}{x} \geq 2\sqrt{x \cdot \frac{1}{x}} = 2
    \]
    \textbf{理由：} 因为 $x>0$，$\frac{1}{x}>0$，满足不等式条件。
    \item[\textbf{第三步：}] 验证取等条件。等号成立当且仅当 $x = \frac{1}{x}$，即 $x^2=1$，由 $x>0$ 得 $x=1$。因此最小值可以取到。
\end{description}

\textbf{关键结论：} \textbf{当 $x=1$ 时，$x + \frac{1}{x}$ 取得最小值 $2$。}

\medskip
\textbf{例2（稍变形）}

\textbf{题目：} 设 $x, y > 0$，且 $x+y=1$，求 $xy$ 的最大值。

\textbf{解题步骤：}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步：}] 已知和为定值，求积的最大值。考虑变形 $ab \leq \left(\frac{a+b}{2}\right)^2$。
    \item[\textbf{第二步：}] 直接应用。令 $a=x$，$b=y$，则
    \[
    xy \leq \left(\frac{x+y}{2}\right)^2 = \left(\frac{1}{2}\right)^2 = \frac{1}{4}
    \]
    \textbf{理由：} 因为 $x,y>0$，满足不等式条件，且 $x+y=1$ 为定值。
    \item[\textbf{第三步：}] 验证取等条件。等号成立当且仅当 $x=y$，结合 $x+y=1$，得 $x=y=\frac{1}{2}$。因此最大值可以取到。
\end{description}

\textbf{关键结论：} \textbf{当 $x=y=\frac{1}{2}$ 时，$xy$ 取得最大值 $\frac{1}{4}$。}
\end{examplebox}
