\textbf{\textcolor{CStatement}{条件}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{条件 1（变量为正）：}}] 实数 $x>0$。
\end{description}

\textbf{\textcolor{CStatement}{结论}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{\textcolor{CStatement}{结论一（基本切线式）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 对数图像在 $x=1$ 处的切线方程。
    \[
    \ln x \leq x-1
    \]

    \item[\textbf{\textcolor{CStatement}{推广结论（平移变形）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 将自变量替换为 $x+1$ 得到常用形式。
    \[
    \ln(x+1) \leq x, \quad (x>-1)
    \]

    \item[\textbf{\textcolor{CStatement}{推广结论（乘积形式）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 不等式两边同乘 $x$ 后的变形。
    \[
    x\ln x \geq x-1, \quad (x>0)
    \]

    \item[\textbf{\textcolor{CStatement}{推广结论（二次逼近）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 利用更强的不等式逼近。
    \[
    \ln x \leq \frac{x^{2}-1}{2x}, \quad (x>0)
    \]
\end{description}

\textbf{\textcolor{CStatement}{等号/取等条件：}} 仅当 $x=1$ 时，所有不等式取等号。

\end{statementbox}
