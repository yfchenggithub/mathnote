\begin{summarybox}
\textbf{\textcolor{CSummary}{一句话核心}}
\textbf{对数函数 $\ln x$ 在 $x=1$ 处切线为 $y=x-1$，图像恒在切线上方（$x\neq1$），由此衍生一系列放缩式。}

\medskip
\textbf{\textcolor{CSummary}{使用条件}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{条件 1（定义域）：} 主结论要求 $x>0$；变式 $\ln(x+1)\leq x$ 要求 $x>-1$。
    \item \textbf{条件 2（形式匹配）：} 待证或待放缩的不等式含有 $\ln x$ 或 $\ln(1+x)$ 与一次/二次多项式比较。
\end{itemize}

\medskip
\textbf{\textcolor{CSummary}{关键提醒}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{易错点（方向与域）：} 牢记各变式不等号方向及定义域，避免反向使用或域外误用。
    \item \textbf{检查项（等号可达性）：} 放缩后务必验证原题条件下等号能否取得，以判断放缩精度是否足够。
    \item \textbf{选择项（精度权衡）：} 一次切线放缩最简，二次变式更紧，根据题目需求灵活选取。
\end{itemize}
\end{summarybox}
