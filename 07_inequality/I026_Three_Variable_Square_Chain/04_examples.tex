\begin{examplebox}
\textbf{\textcolor{CExample}{例 1（基础）}}

\textbf{\textcolor{CExample}{题目：}} 设 $a, b, c \ge 0$，求证：$a^3+b^3+c^3+3abc \ge a^2b+ab^2+b^2c+bc^2+c^2a+ca^2$。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（移项整理）：}] 将不等式右边移至左边：
    \[
    a^3+b^3+c^3+3abc - (a^2b+ab^2+b^2c+bc^2+c^2a+ca^2) \ge 0.
    \]
    \par\textbf{理由：}
    将不等式化为标准形式，便于应用已知结论。

    \item[\textbf{第二步（识别舒尔不等式）：}] 由舒尔不等式（$r=1$），有
    \[
    a^3+b^3+c^3+3abc \ge a^2b+ab^2+b^2c+bc^2+c^2a+ca^2.
    \]
    \par\textbf{理由：}
    这正是舒尔不等式当 $r=1$ 时的特例，直接应用即可。

    \item[\textbf{第三步（得出结论）：}] 因此原不等式成立。
    \par\textbf{理由：}
    由舒尔不等式直接可得。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{舒尔不等式（$r=1$）的直接应用，无需额外变形。}

\medskip
\textbf{\textcolor{CExample}{例 2（稍变形）}}

\textbf{\textcolor{CExample}{题目：}} 已知 $a, b, c > 0$，且 $abc=1$，求证：$a^3+b^3+c^3+3 \ge a^2(b+c)+b^2(c+a)+c^2(a+b)$。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（利用条件）：}] 由 $abc=1$，得 $3abc = 3$。
    \par\textbf{理由：}
    将已知条件转化为等式，便于代入不等式。

    \item[\textbf{第二步（应用舒尔不等式）：}] 由舒尔不等式（$r=1$），有
    \[
    a^3+b^3+c^3+3abc \ge a^2(b+c)+b^2(c+a)+c^2(a+b).
    \]
    \par\textbf{理由：}
    舒尔不等式对任意非负实数成立，此处 $a,b,c>0$，故可直接应用。

    \item[\textbf{第三步（代入条件）：}] 将 $3abc=3$ 代入上式，得到
    \[
    a^3+b^3+c^3+3 \ge a^2(b+c)+b^2(c+a)+c^2(a+b).
    \]
    \par\textbf{理由：}
    等量代换，得到目标不等式。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{在条件 $abc=1$ 下，舒尔不等式中的 $3abc$ 可替换为 $3$，从而得到强化形式。}
\end{examplebox}
