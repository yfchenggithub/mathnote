\textbf{\textcolor{CStatement}{条件}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{条件 1（非负实数与正指数）：}}] $a, b, c \geq 0$，$r > 0$。
\end{description}

\textbf{\textcolor{CStatement}{结论}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{\textcolor{CStatement}{结论一（舒尔不等式）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 三元对称轮换非负和。
    \[
    a^{r}(a-b)(a-c) + b^{r}(b-a)(b-c) + c^{r}(c-a)(c-b) \geq 0
    \]

    \item[\textbf{\textcolor{CStatement}{推广结论（三次舒尔）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 三次对称不等式常见形式。
    \[
    a^{3}+b^{3}+c^{3}+3abc \geq a^{2}b+ab^{2}+b^{2}c+bc^{2}+c^{2}a+ca^{2}
    \]
\end{description}

\textbf{\textcolor{CStatement}{等号/取等条件：}} 等号成立当且仅当 $a = b = c$ 或其中两个相等且第三个为 $0$。

\end{statementbox}
