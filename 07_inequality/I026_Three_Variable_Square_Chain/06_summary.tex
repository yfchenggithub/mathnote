\begin{summarybox}
\textbf{\textcolor{CSummary}{一句话核心}}
\textbf{三元对称轮换多项式在非负变量下恒非负，是处理三元不等式的重要工具。}

\medskip
\textbf{\textcolor{CSummary}{使用条件}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{条件 1（非负变量）：} $a, b, c \ge 0$。
    \item \textbf{条件 2（正指数）：} $r > 0$。
\end{itemize}

\medskip
\textbf{\textcolor{CSummary}{关键提醒}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{易错点（等号遗漏）：} 等号成立条件包括 $a=b=c$ 或两等一零。
    \item \textbf{检查项（范围验证）：} 使用前务必检查变量非负与指数为正。
\end{itemize}
\end{summarybox}
