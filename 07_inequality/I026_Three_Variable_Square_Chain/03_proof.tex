\begin{proofbox}
\textbf{\textcolor{CProof}{思路提示}}
由于不等式关于 $a, b, c$ 对称，不妨设 $a \ge b \ge c \ge 0$，通过因式分解判断符号。

\medskip
\textbf{\textcolor{CProof}{正式推导}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.55em]
    \item[\textbf{步骤一（对称假设）：}] 由对称性，不妨设 $a \ge b \ge c \ge 0$。
    \[
    \text{此时 } a-b \ge 0, \quad a-c \ge 0, \quad b-c \ge 0, \quad c-a \le 0, \quad c-b \le 0.
    \]
    \par\textbf{理由：}
    不等式完全对称，设定大小顺序不影响一般性，且便于判断各项符号。

    \item[\textbf{步骤二（重新分组）：}] 将左边分成两部分：
    \[
    \begin{aligned}
        & a^{r}(a-b)(a-c) + b^{r}(b-a)(b-c) + c^{r}(c-a)(c-b) \\
        =& (a-b)\bigl[a^{r}(a-c) - b^{r}(b-c)\bigr] + c^{r}(c-a)(c-b).
    \end{aligned}
    \]
    \par\textbf{理由：}
    将含 $a,b$ 的项组合，以便利用 $a \ge b$ 比较。

    \item[\textbf{步骤三（判断符号）：}] 分析每部分的符号。
    \begin{itemize}
        \item 因为 $a \ge b \ge c$，所以 $a-c \ge b-c \ge 0$，且 $a^{r} \ge b^{r} \ge 0$（因为 $r>0$），故 $a^{r}(a-c) \ge b^{r}(b-c)$，从而 $(a-b)[a^{r}(a-c)-b^{r}(b-c)] \ge 0$。
        \item 又 $c-a \le 0$，$c-b \le 0$，所以 $(c-a)(c-b) \ge 0$，而 $c^{r} \ge 0$，故 $c^{r}(c-a)(c-b) \ge 0$。
    \end{itemize}
    \par\textbf{理由：}
    利用不等式的传递性和非负数的乘积性质。

    \item[\textbf{步骤四（得出结论）：}] 两部分均非负，故其和也非负。
    \[
    a^{r}(a-b)(a-c) + b^{r}(b-a)(b-c) + c^{r}(c-a)(c-b) \ge 0.
    \]
    \par\textbf{理由：}
    非负数之和仍为非负数。
\end{description}

\medskip
\textbf{\textcolor{CProof}{结论回扣}}
\textbf{综上，原不等式成立，等号成立当且仅当 $a=b=c$ 或其中两个相等且第三个为 $0$（在大小顺序假设下，即 $a=b$ 且 $c=0$ 或 $a=b=c$）。}
\end{proofbox}
