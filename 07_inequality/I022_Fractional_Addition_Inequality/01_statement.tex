\begin{statementbox}

\textbf{\textcolor{CStatement}{条件}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{条件 1（正数条件）：}}] $x, y > 0$。
\end{description}

\textbf{\textcolor{CStatement}{结论}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{\textcolor{CStatement}{结论一（倒数和不等式）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 两数倒数和不小于和的倒数的四倍。
    \[
    \frac{1}{x} + \frac{1}{y} \geq \frac{4}{x+y}
    \]
    \item[\textbf{\textcolor{CStatement}{结论二（和积不等式）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 和与倒数和之积不小于四。
    \[
    (x+y)\left(\frac{1}{x}+\frac{1}{y}\right) \geq 4
    \]
    \item[\textbf{\textcolor{CStatement}{推广结论（权方和形式）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 带权平方和不等式。
    \[
    \frac{a^2}{x} + \frac{b^2}{y} \geq \frac{(a+b)^2}{x+y}
    \]
\end{description}

\textbf{\textcolor{CStatement}{等号/取等条件：}}
\begin{itemize}[leftmargin=*, itemsep=0.35em, topsep=0.3em]
    \item 对于结论一和结论二，当且仅当 $x = y$。
    \item 对于推广结论，当且仅当 $\dfrac{a}{x} = \dfrac{b}{y}$（即 $ay = bx$）。
\end{itemize}
\end{statementbox}
