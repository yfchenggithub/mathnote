\begin{summarybox}
\textbf{\textcolor{CSummary}{一句话核心}}
\textbf{正数倒数和的“调和”关系：倒数和≥4/(和)，其推广是权方和不等式。}

\medskip
\textbf{\textcolor{CSummary}{使用条件}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{条件 1（正数前提）：} $x>0, y>0$ 是使用所有形式的前提。
    \item \textbf{条件 2（分母非零）：} 权方和形式还需 $x+y>0$（由正数条件自然满足）。
    \item \textbf{条件 3（形式匹配）：} 根据目标式是倒数和、乘积还是带平方项选择对应形式。
\end{itemize}

\medskip
\textbf{\textcolor{CSummary}{关键提醒}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{易错点（取等区分）：} 基本形式取等 $x=y$，权方和形式取等 $a/x = b/y$，切勿混淆。
    \item \textbf{检查项（先验验证）：} 使用前必先确认 $x,y>0$；使用后检查等号条件是否可达。
\end{itemize}
\end{summarybox}
