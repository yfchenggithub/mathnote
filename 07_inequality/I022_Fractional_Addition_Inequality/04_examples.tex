\begin{examplebox}
\textbf{\textcolor{CExample}{例 1（基础）}}

\textbf{\textcolor{CExample}{题目：}} 已知 $x>0, y>0$，且 $x+y=1$，求 $\dfrac{1}{x}+\dfrac{1}{y}$ 的最小值。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（识别形式）：}] 题目条件为 $x+y=1$，目标为 $\frac{1}{x}+\frac{1}{y}$，符合倒数和不等式结构。
    \par\textbf{理由：} 直接应用结论一。
    \item[\textbf{第二步（直接应用）：}] 由倒数和不等式，
    \[
    \frac{1}{x} + \frac{1}{y} \ge \frac{4}{x+y}.
    \]
    \par\textbf{理由：} $x,y>0$ 满足不等式条件。
    \item[\textbf{第三步（代入求值）：}] 将 $x+y=1$ 代入右边，得
    \[
    \frac{1}{x} + \frac{1}{y} \ge \frac{4}{1} = 4.
    \]
    \par\textbf{理由：} 不等式右边化为常数。
    \item[\textbf{第四步（取等检查）：}] 等号成立当且仅当 $x=y$，结合 $x+y=1$ 得 $x=y=\frac{1}{2}$。
    \par\textbf{理由：} 检查等号条件是否可达，以确定最小值确实能取到。
\end{description}
\textbf{\textcolor{CExample}{关键结论：}} \textbf{最小值为 $4$，当且仅当 $x=y=\frac{1}{2}$ 时取得。}

\textbf{\textcolor{CExample}{例 2（稍变形）}}

\textbf{\textcolor{CExample}{题目：}} 设 $a,b>0$，证明：$\dfrac{a^2}{b} + \dfrac{b^2}{a} \ge a+b$。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（观察目标）：}] 目标不等式左边是 $\frac{a^2}{b}+\frac{b^2}{a}$，右边是 $a+b$。
    \par\textbf{理由：} 尝试与权方和形式 $\frac{a^2}{x}+\frac{b^2}{y} \ge \frac{(a+b)^2}{x+y}$ 进行比对。
    \item[\textbf{第二步（匹配系数）：}] 令权方和形式中的 $x=b, y=a$，同时分子 $a,b$ 分别对应 $a,b$。则左边即为 $\frac{a^2}{b}+\frac{b^2}{a}$。
    \par\textbf{理由：} 直接代入 $x=b, y=a$ 进行匹配。
    \item[\textbf{第三步（应用不等式）：}] 代入权方和不等式，得
    \[
    \frac{a^2}{b} + \frac{b^2}{a} \ge \frac{(a+b)^2}{b+a} = \frac{(a+b)^2}{a+b} = a+b.
    \]
    \par\textbf{理由：} 严格应用推广结论，注意分母 $x+y=b+a=a+b$。
    \item[\textbf{第四步（取等条件）：}] 等号成立当且仅当 $\dfrac{a}{b} = \dfrac{b}{a}$，即 $a^2=b^2$，由 $a,b>0$ 得 $a=b$。
    \par\textbf{理由：} 验证等号条件 $\frac{a}{x}=\frac{b}{y}$，此处即 $\frac{a}{b}=\frac{b}{a}$。
\end{description}
\textbf{\textcolor{CExample}{关键结论：}} \textbf{不等式得证，等号成立当且仅当 $a=b$。}
\end{examplebox}
