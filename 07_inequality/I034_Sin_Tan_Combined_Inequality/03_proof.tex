\begin{proofbox}
\textbf{\textcolor{CProof}{思路提示}}
通过构造函数差，利用导数分析单调性，证明在 $x \geq 0$ 时函数差非负。

\medskip
\textbf{\textcolor{CProof}{正式推导}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.55em]
    \item[\textbf{步骤一（构造函数差）：}] 考虑函数 $f(x) = e^x - \left(1 + x + \frac{x^2}{2}\right)$，其中 $x \geq 0$。
    \[
    f(x) = e^x - 1 - x - \frac{x^2}{2}
    \]
    \par\textbf{理由：}
    要证明 $e^x \geq 1 + x + \frac{x^2}{2}$，只需证明 $f(x) \geq 0$。

    \item[\textbf{步骤二（求导分析）：}] 计算导数 $f'(x)$。
    \[
    f'(x) = e^x - 1 - x
    \]
    \par\textbf{理由：}
    导数 $f'(x)$ 反映了 $f(x)$ 的单调性。

    \item[\textbf{步骤三（二阶导数判号）：}] 进一步计算 $f''(x)$。
    \[
    f''(x) = e^x - 1
    \]
    \par\textbf{理由：}
    对于 $x \geq 0$，有 $e^x \geq 1$，所以 $f''(x) \geq 0$，等号当 $x=0$ 成立。因此 $f'(x)$ 在 $x \geq 0$ 上单调递增。

    \item[\textbf{步骤四（一阶导数值）：}] 由于 $f'(0) = e^0 - 1 - 0 = 0$，且 $f'(x)$ 单调递增，所以对于 $x \geq 0$，有 $f'(x) \geq 0$。
    \[
    f'(x) \geq 0 \quad \text{对所有 } x \geq 0
    \]
    \par\textbf{理由：}
    $f'(x)$ 在 $x=0$ 处为0，之后非负，表明 $f(x)$ 单调不减。

    \item[\textbf{步骤五（原函数值）：}] 因此 $f(x)$ 在 $x \geq 0$ 上单调递增，而 $f(0) = e^0 - 1 - 0 - 0 = 0$，所以对于 $x \geq 0$，有 $f(x) \geq 0$。
    \[
    f(x) \geq f(0) = 0
    \]
    \par\textbf{理由：}
    单调递增函数在起点取最小值。

    \item[\textbf{步骤六（推广到一般n）：}] 对于一般正整数 $n$，类似地构造 $g(x) = e^x - \sum_{i=0}^{n} \frac{x^i}{i!}$。泰勒展开的拉格朗日余项 $R_n(x) = \frac{e^{\xi}}{(n+1)!} x^{n+1}$ 对于 $\xi$ 在 $0$ 与 $x$ 之间，当 $x \geq 0$ 时 $R_n(x) \geq 0$，因此 $e^x = \sum_{i=0}^{n} \frac{x^i}{i!} + R_n(x) \geq \sum_{i=0}^{n} \frac{x^i}{i!}$。
    \[
    e^x = \sum_{i=0}^{n} \frac{x^i}{i!} + R_n(x), \quad R_n(x) \geq 0
    \]
    \par\textbf{理由：}
    泰勒展开的拉格朗日余项非负，因为 $e^{\xi} > 0$ 且 $x^{n+1} \geq 0$。
\end{description}

\medskip
\textbf{\textcolor{CProof}{结论回扣}}
\textbf{因此，对任意 $x \geq 0$ 和正整数 $n$，都有 $e^x \geq \sum_{i=0}^{n} \frac{x^i}{i!}$，等号当且仅当 $x=0$ 时成立。}
\end{proofbox}
