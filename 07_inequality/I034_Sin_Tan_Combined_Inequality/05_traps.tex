\begin{trapbox}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CTrap}{易错点一（忽略定义域）}}] 在 $x < 0$ 时仍使用该不等式进行放缩。
    \item[\textbf{\textcolor{CTrap}{正确理解（非负域才成立）：}}] 结论仅在 $x \geq 0$ 时成立，对于 $x < 0$，不等式方向可能改变。
    \item[\textbf{\textcolor{CTrap}{错因分析（条件遗忘）：}}] 学生往往只记住公式形式，而忘记前提条件 $x \geq 0$，导致在负数域误用。
\end{description}

\medskip
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CTrap}{易错点二（取等误解）}}] 认为等号在任意点都可能成立，例如误以为当 $x=1$ 时 $e^x = 1 + x + \frac{x^2}{2}$。
    \item[\textbf{\textcolor{CTrap}{正确理解（唯一取等点）：}}] 等号仅当 $x=0$ 时成立，其他点均为严格大于。
    \item[\textbf{\textcolor{CTrap}{错因分析（函数增长性）：}}] $e^x$ 是严格凸函数，其泰勒多项式在 $x>0$ 时总是低估函数值，只有原点处一阶接触。
\end{description}

\medskip
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CTrap}{易错点三（推广滥用）}}] 将不等式推广到任意 $n$ 时，错误地用于 $x$ 的负值或复杂表达式。
    \item[\textbf{\textcolor{CTrap}{正确理解（非负实数域）：}}] 推广结论 $e^x \geq \sum_{i=0}^{n} \frac{x^i}{i!}$ 同样要求 $x \geq 0$，且 $n$ 为正整数。
    \item[\textbf{\textcolor{CTrap}{错因分析（归纳前提忽略）：}}] 在推广过程中，没有注意归纳基础 $x \geq 0$ 的条件，导致错误结论。
\end{description}
\end{trapbox}
