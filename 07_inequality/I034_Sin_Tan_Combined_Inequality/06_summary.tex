\begin{summarybox}
\textbf{\textcolor{CSummary}{一句话核心}}
\textbf{指数函数 $e^x$ 在非负实数上总是不低于其泰勒多项式的前任意有限项和。}

\medskip
\textbf{\textcolor{CSummary}{使用条件}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{条件1（自变量非负）：} $x \geq 0$，这是不等式成立的前提。
    \item \textbf{条件2（多项式阶数正整数）：} $n$ 为正整数，表示泰勒多项式的项数。
\end{itemize}

\medskip
\textbf{\textcolor{CSummary}{关键提醒}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{易错点（负数域无效）：} 切记 $x < 0$ 时不成立，避免误用。
    \item \textbf{检查项（取等唯一）：} 等号仅在 $x=0$ 时成立，用于最值问题需验证取等。
\end{itemize}
\end{summarybox}
