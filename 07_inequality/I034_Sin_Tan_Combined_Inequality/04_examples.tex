\begin{examplebox}
\textbf{\textcolor{CExample}{例 1（基础）}}

\textbf{\textcolor{CExample}{题目：}} 已知 $x = 0.3$，判断 $e^{0.3}$ 与 $1.345$ 的大小关系。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（验证条件）：}] 由于 $0.3 \geq 0$，满足 $x \geq 0$ 的条件。
    \par\textbf{理由：}
    不等式 $e^x \geq 1 + x + \frac{x^2}{2}$ 要求 $x \geq 0$。

    \item[\textbf{第二步（应用不等式）：}] 根据结论，有 $e^{0.3} \geq 1 + 0.3 + \frac{0.3^2}{2}$。
    \[
    e^{0.3} \geq 1 + 0.3 + \frac{0.09}{2}
    \]
    \par\textbf{理由：}
    直接代入 $x=0.3$ 到不等式中。

    \item[\textbf{第三步（计算比较）：}] 计算右边值： $1 + 0.3 + 0.045 = 1.345$。
    \[
    e^{0.3} \geq 1.345
    \]
    \par\textbf{理由：}
    多项式计算得出下界。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{因此 $e^{0.3} \geq 1.345$，且等号不成立因为 $x \neq 0$。}

\textbf{\textcolor{CExample}{例 2（稍变形）}}

\textbf{\textcolor{CExample}{题目：}} 设 $x \geq 0$，求函数 $h(x) = e^x - x$ 的最小值，并说明取到最小值的条件。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（利用下界）：}] 由 $e^x \geq 1 + x + \frac{x^2}{2}$，可得 $e^x - x \geq 1 + \frac{x^2}{2}$。
    \[
    h(x) = e^x - x \geq 1 + \frac{x^2}{2}
    \]
    \par\textbf{理由：}
    将不等式 $e^x \geq 1 + x + \frac{x^2}{2}$ 两边同时减去 $x$。

    \item[\textbf{第二步（分析下界函数）：}] 由于 $x \geq 0$，有 $\frac{x^2}{2} \geq 0$，所以 $1 + \frac{x^2}{2} \geq 1$。
    \[
    h(x) \geq 1 + \frac{x^2}{2} \geq 1
    \]
    \par\textbf{理由：}
    平方项非负，因此下界函数的最小值为1。

    \item[\textbf{第三步（取等条件）：}] 原不等式 $e^x \geq 1 + x + \frac{x^2}{2}$ 等号当且仅当 $x=0$。此时 $h(0) = e^0 - 0 = 1$，且 $1 + \frac{0^2}{2} = 1$，所以等号成立。
    \[
    h(0) = 1
    \]
    \par\textbf{理由：}
    检查 $x=0$ 时，所有不等式等号同时成立。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{函数 $h(x) = e^x - x$ 在 $x \geq 0$ 上的最小值为 $1$，当且仅当 $x=0$ 时取到。}
\end{examplebox}
