\textbf{\textcolor{CStatement}{条件}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{条件1（非负实数）：}}] $x \geq 0$。
\end{description}

\textbf{\textcolor{CStatement}{结论}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{\textcolor{CStatement}{结论一（二次下界）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 指数函数在非负实数上不低于其二次泰勒多项式。
    \[
    e^x \geq 1 + x + \frac{x^2}{2}
    \]

    \item[\textbf{\textcolor{CStatement}{推广结论（泰勒下界）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 指数函数在非负实数上不低于其任意阶泰勒多项式。
    \[
    e^x \geq \sum_{i=0}^{n} \frac{x^i}{i!}
    \]
\end{description}

\textbf{\textcolor{CStatement}{等号/取等条件：}} 当且仅当 $x=0$ 时等号成立。
