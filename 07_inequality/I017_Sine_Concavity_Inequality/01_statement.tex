\textbf{\textcolor{CStatement}{条件}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{条件 1（区间限制）：}}] $x, y \in (0, \pi)$。
    \item[\textbf{\textcolor{CStatement}{条件 2（内角条件）：}}] $A, B, C$ 为三角形内角。
\end{description}

\textbf{\textcolor{CStatement}{结论}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{\textcolor{CStatement}{结论一（凹性不等式）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 正弦函数凹性导致算术平均小于中点正弦值。
    \[
    \frac{\sin x + \sin y}{2} \leq \sin\!\left(\frac{x+y}{2}\right).
    \]

    \item[\textbf{\textcolor{CStatement}{推广结论（三角形正弦和）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 三角形内角正弦和的最大值为 $\frac{3\sqrt{3}}{2}$。
    \[
    \frac{\sin A + \sin B + \sin C}{3} \leq \sin\frac{\pi}{3} = \frac{\sqrt{3}}{2}.
    \]
\end{description}

\textbf{\textcolor{CStatement}{等号/取等条件：}} 对于结论一，等号成立当且仅当 $x = y$；对于推广结论，等号成立当且仅当 $\triangle ABC$ 为正三角形。

\end{statementbox}
