\begin{summarybox}
\textbf{\textcolor{CSummary}{一句话核心}}

\textbf{正弦函数在 $(0, \pi)$ 上为凹函数，因此其函数值的算术平均不大于自变量中点处的函数值。}

\medskip
\textbf{\textcolor{CSummary}{使用条件}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{条件 1（区间内点）：} 自变量 $x, y$ 必须属于开区间 $(0, \pi)$。
    \item \textbf{条件 2（三角形角）：} 在三角形中应用时，$A, B, C$ 需为内角且均在 $(0, \pi)$ 内。
\end{itemize}

\medskip
\textbf{\textcolor{CSummary}{关键提醒}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{易错点（方向混淆）：} 凹性不等式不等号方向为 $\leq$，勿与凸性混淆。
    \item \textbf{检查项（取等验证）：} 应用后务必检查等号成立条件是否满足题目要求。
\end{itemize}
\end{summarybox}
