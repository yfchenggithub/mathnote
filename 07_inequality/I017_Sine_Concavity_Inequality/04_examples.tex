\begin{examplebox}
\textbf{\textcolor{CExample}{例 1（基础）}}

\textbf{\textcolor{CExample}{题目：}} 设 $x, y \in (0, \pi)$，且 $x+y = \frac{\pi}{2}$，证明 $\sin x + \sin y \leq \sqrt{2}$。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（应用凹性）：}] 由正弦函数凹性不等式，有
    \[
    \frac{\sin x + \sin y}{2} \leq \sin\!\left(\frac{x+y}{2}\right).
    \]
    \par\textbf{理由：} 因为 $x, y \in (0, \pi)$，满足凹性条件。

    \item[\textbf{第二步（代入条件）：}] 代入 $x+y = \frac{\pi}{2}$，得 $\frac{x+y}{2} = \frac{\pi}{4}$。
    \[
    \frac{\sin x + \sin y}{2} \leq \sin\frac{\pi}{4} = \frac{\sqrt{2}}{2}.
    \]
    \par\textbf{理由：} 利用已知条件简化中点正弦值。

    \item[\textbf{第三步（导出结论）：}] 两边乘以 $2$，得到
    \[
    \sin x + \sin y \leq 2 \cdot \frac{\sqrt{2}}{2} = \sqrt{2}.
    \]
    \par\textbf{理由：} 不等式两边同乘正数 $2$，不等号方向不变。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{当 $x+y=\frac{\pi}{2}$ 时，$\sin x + \sin y$ 的最大值为 $\sqrt{2}$，取等当 $x=y=\frac{\pi}{4}$。}

\medskip
\textbf{\textcolor{CExample}{例 2（稍变形）}}

\textbf{\textcolor{CExample}{题目：}} 在 $\triangle ABC$ 中，证明 $\sin A + \sin B \leq 2 \cos\frac{C}{2}$。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（应用凹性）：}] 由正弦函数凹性不等式，有
    \[
    \frac{\sin A + \sin B}{2} \leq \sin\!\left(\frac{A+B}{2}\right).
    \]
    \par\textbf{理由：} 因为 $A, B \in (0, \pi)$ 且为三角形内角，满足凹性条件。

    \item[\textbf{第二步（利用内角和）：}] 在三角形中，$A+B = \pi - C$，所以 $\frac{A+B}{2} = \frac{\pi}{2} - \frac{C}{2}$。
    \[
    \sin\!\left(\frac{A+B}{2}\right) = \sin\!\left(\frac{\pi}{2} - \frac{C}{2}\right) = \cos\frac{C}{2}.
    \]
    \par\textbf{理由：} 三角函数的诱导公式。

    \item[\textbf{第三步（结合得证）：}] 代入第一步的不等式，得到
    \[
    \frac{\sin A + \sin B}{2} \leq \cos\frac{C}{2} \quad \Rightarrow \quad \sin A + \sin B \leq 2 \cos\frac{C}{2}.
    \]
    \par\textbf{理由：} 将第二步结果代入并乘以 $2$。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{在三角形中，$\sin A + \sin B \leq 2 \cos\frac{C}{2}$，取等当 $A=B$。}
\end{examplebox}
