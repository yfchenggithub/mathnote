\begin{proofbox}
\textbf{\textcolor{CProof}{思路提示}}

利用指数函数的凸性，直接应用Jensen不等式。

\medskip
\textbf{\textcolor{CProof}{正式推导}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.55em]
    \item[\textbf{步骤一（凸性验证）：}] 证明 $f(x)=e^x$ 是凸函数。
    \[
    f''(x) = e^x > 0, \quad \forall x \in \mathbb{R}.
    \]
    \par\textbf{理由：}
    二阶导数恒正说明函数是严格凸的。

    \item[\textbf{步骤二（Jensen不等式）：}] 对凸函数 $f$，有Jensen不等式：
    \[
    \sum_{i=1}^n \lambda_i f(x_i) \geq f\left( \sum_{i=1}^n \lambda_i x_i \right),
    \]
    其中 $\lambda_i \geq 0$ 且 $\sum_{i=1}^n \lambda_i = 1$。
    \par\textbf{理由：}
    凸函数的定义推广到多点加权平均。

    \item[\textbf{步骤三（取等权重）：}] 取 $\lambda_i = \frac{1}{n}$，则
    \[
    \frac{1}{n} \sum_{i=1}^n f(x_i) \geq f\left( \frac{1}{n} \sum_{i=1}^n x_i \right).
    \]
    \par\textbf{理由：}
    权重相等时是Jensen不等式的特例。

    \item[\textbf{步骤四（代入函数）：}] 代入 $f(x)=e^x$，得到
    \[
    \frac{1}{n} \sum_{i=1}^n e^{x_i} \geq e^{\frac{1}{n} \sum_{i=1}^n x_i}.
    \]
    \par\textbf{理由：}
    函数的具体形式。

    \item[\textbf{步骤五（取等条件）：}] 等号成立当且仅当 $x_1 = x_2 = \cdots = x_n$。
    \par\textbf{理由：}
    由于 $f$ 是严格凸函数，等号成立当且仅当所有 $x_i$ 相等。
\end{description}

\medskip
\textbf{\textcolor{CProof}{结论回扣}}

\textbf{因此原不等式成立，且取等条件为所有变量相等。}
\end{proofbox}
