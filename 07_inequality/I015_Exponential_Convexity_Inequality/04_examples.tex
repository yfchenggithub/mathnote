\begin{examplebox}
\textbf{\textcolor{CExample}{例 1（基础）}}

\textbf{\textcolor{CExample}{题目：}} 已知实数 $a, b, c$ 满足 $a+b+c=0$，求证：$e^a + e^b + e^c \geq 3$。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（应用不等式）：}] 由指数凸性不等式，有
    \[
    \frac{e^a + e^b + e^c}{3} \geq e^{\frac{a+b+c}{3}}.
    \]
    \par\textbf{理由：}
    直接代入 $n=3$。

    \item[\textbf{第二步（利用条件）：}] 因为 $a+b+c=0$，所以
    \[
    e^{\frac{a+b+c}{3}} = e^0 = 1.
    \]
    \par\textbf{理由：}
    代入已知条件。

    \item[\textbf{第三步（得出结论）：}] 于是
    \[
    \frac{e^a + e^b + e^c}{3} \geq 1 \quad \Rightarrow \quad e^a + e^b + e^c \geq 3.
    \]
    \par\textbf{理由：}
    整理不等式。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{当 $a+b+c=0$ 时，$e^a + e^b + e^c \geq 3$，等号当 $a=b=c=0$ 时成立。}

\medskip
\textbf{\textcolor{CExample}{例 2（稍变形）}}

\textbf{\textcolor{CExample}{题目：}} 已知实数 $x, y, z$ 满足 $x+y+z=3$，求证：$e^{x-1}+e^{y-1}+e^{z-1} \geq 3$。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（换元转化）：}] 令 $a = x-1, b = y-1, c = z-1$，则 $a+b+c = (x+y+z)-3 = 0$。
    \par\textbf{理由：}
    将条件转化为 $a+b+c=0$，从而与例1类似。

    \item[\textbf{第二步（应用不等式）：}] 由指数凸性不等式，有
    \[
    \frac{e^a+e^b+e^c}{3} \geq e^{\frac{a+b+c}{3}} = e^0 = 1.
    \]
    \par\textbf{理由：}
    直接代入 $n=3$ 和条件 $a+b+c=0$。

    \item[\textbf{第三步（回代得证）：}] 所以 $e^a+e^b+e^c \geq 3$，即 $e^{x-1}+e^{y-1}+e^{z-1} \geq 3$。
    \par\textbf{理由：}
    将 $a, b, c$ 换回 $x-1, y-1, z-1$。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{当 $x+y+z=3$ 时，$e^{x-1}+e^{y-1}+e^{z-1} \geq 3$，等号当 $x=y=z=1$ 时成立。}
\end{examplebox}
