\begin{statementbox}

\textbf{\textcolor{CStatement}{条件}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{条件 1（任意实数）：}}] 设 $x_1, x_2, \dots, x_n$ 是任意实数。
\end{description}

\textbf{\textcolor{CStatement}{结论}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{\textcolor{CStatement}{结论一（指数平均大）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 指数平均不小于指数底数的算术平均的指数。
    \[
    \frac{e^{x_1}+e^{x_2}+\cdots+e^{x_n}}{n} \geq e^{\frac{x_1+x_2+\cdots+x_n}{n}}
    \]

    \item[\textbf{\textcolor{CStatement}{推广结论（加权形式）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 权重非负时，加权指数平均不小于加权算术平均的指数。
    \[
    \sum_{i=1}^n \lambda_i e^{x_i} \geq e^{\sum_{i=1}^n \lambda_i x_i}, \quad \text{其中 } \lambda_i \geq 0, \sum_{i=1}^n \lambda_i = 1.
    \]
\end{description}

\textbf{\textcolor{CStatement}{等号/取等条件：}} 当且仅当 $x_1 = x_2 = \cdots = x_n$ 时等号成立。

\end{statementbox}
