\begin{proofbox}
\textbf{\textcolor{CProof}{思路提示}}
通过分析幂函数 $f(t)=t^n$ 的凸凹性（利用二阶导数），应用凸函数不等式（Jensen 不等式特例）直接导出结论。

\medskip
\textbf{\textcolor{CProof}{正式推导}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.55em]
    \item[\textbf{步骤一（函数定义）：}] 定义函数 $f(t)=t^n$，其中 $t>0$，$n \in \mathbb{R}$。
    \[ f(t) = t^{n}. \]
    \par\textbf{理由：} 将原不等式转化为函数凸性问题，便于分析。

    \item[\textbf{步骤二（导数计算）：}] 求 $f(t)$ 的一阶和二阶导数。
    \[
    f'(t) = n t^{n-1}, \quad f''(t) = n(n-1) t^{n-2}.
    \]
    \par\textbf{理由：} 二阶导数的符号决定函数的凸凹性。

    \item[\textbf{步骤三（凸性判断）：}] 分析 $f''(t)$ 的符号。
    \[
    \text{当 } n>1 \text{ 或 } n<0 \text{ 时}, \, n(n-1)>0 \Rightarrow f''(t)>0, \, f \text{ 为凸函数}.
    \]
    \[
    \text{当 } 0<n<1 \text{ 时}, \, n(n-1)<0 \Rightarrow f''(t)<0, \, f \text{ 为凹函数}.
    \]
    \par\textbf{理由：} 二阶导正则凸，负则凹，这是凸凹性的充要条件。

    \item[\textbf{步骤四（凸性不等式）：}] 应用凸函数性质。对于凸函数 $f$，有
    \[
    f\!\left(\frac{x+y}{2}\right) \le \frac{f(x)+f(y)}{2}.
    \]
    对于凹函数 $f$，不等号反向。等号成立当且仅当 $x=y$。
    \par\textbf{理由：} 这是凸函数 Jensen 不等式的二元等权重特例，可直接引用。

    \item[\textbf{步骤五（代入得证）：}] 将 $f(t)=t^n$ 代入上式。
    \[
    \text{当 } n>1 \text{ 或 } n<0 \text{ 时：} \quad \left(\frac{x+y}{2}\right)^{n} \le \frac{x^{n}+y^{n}}{2}.
    \]
    \[
    \text{当 } 0<n<1 \text{ 时：} \quad \frac{x^{n}+y^{n}}{2} \le \left(\frac{x+y}{2}\right)^{n}.
    \]
    \par\textbf{理由：} 直接替换 $f(x)=x^n$, $f(y)=y^n$，即得原不等式。
\end{description}

\medskip
\textbf{\textcolor{CProof}{结论回扣}}
\textbf{因此，原命题成立：不等式方向由指数 $n$ 的范围决定，等号当且仅当 $x=y$ 时取得。证明完毕。}

\end{proofbox}
