\begin{examplebox}
\textbf{\textcolor{CExample}{例 1（基础）}}

\textbf{\textcolor{CExample}{题目：}} 设 $x=2$, $y=8$, $n=2$，比较 $\frac{x^n+y^n}{2}$ 与 $\left(\frac{x+y}{2}\right)^n$ 的大小。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（判断指数）：}] 由 $n=2$，满足 $n>1$。
    \par\textbf{理由：} 根据结论，当 $n>1$ 时，不等式方向为 $\frac{x^n+y^n}{2} \ge \left(\frac{x+y}{2}\right)^n$。

    \item[\textbf{第二步（计算左边）：}] 计算 $\frac{x^n+y^n}{2} = \frac{2^2 + 8^2}{2}$。
    \[
    \frac{2^2 + 8^2}{2} = \frac{4 + 64}{2} = \frac{68}{2} = 34.
    \]
    \par\textbf{理由：} 直接代入数值进行算术运算。

    \item[\textbf{第三步（计算右边）：}] 计算 $\left(\frac{x+y}{2}\right)^n = \left(\frac{2+8}{2}\right)^2$。
    \[
    \left(\frac{2+8}{2}\right)^2 = \left(\frac{10}{2}\right)^2 = 5^2 = 25.
    \]
    \par\textbf{理由：} 先求算术平均，再取幂次。

    \item[\textbf{第四步（比较大小）：}] 比较 $34$ 与 $25$。
    \[
    34 > 25 \quad \Rightarrow \quad \frac{x^n+y^n}{2} > \left(\frac{x+y}{2}\right)^n.
    \]
    \par\textbf{理由：} 数值验证符合不等式方向。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{当 $n>1$ 时，确有 $\frac{x^n+y^n}{2} \ge \left(\frac{x+y}{2}\right)^n$，等号当 $x=y$ 时成立。本例中 $x \ne y$，故为严格大于。}

\textbf{\textcolor{CExample}{例 2（稍变形）}}

\textbf{\textcolor{CExample}{题目：}} 设 $a,b>0$，且 $a+b=1$，求 $a^2+b^2$ 的最小值。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（识别指数）：}] 表达式 $a^2+b^2$ 对应 $n=2$，满足 $n>1$。
    \par\textbf{理由：} 可应用结论，形式为 $\frac{a^2+b^2}{2} \ge \left(\frac{a+b}{2}\right)^2$。

    \item[\textbf{第二步（应用不等式）：}] 由结论，直接写出不等式。
    \[
    \frac{a^2+b^2}{2} \ge \left(\frac{a+b}{2}\right)^2.
    \]
    \par\textbf{理由：} 这是 $n=2>1$ 时的标准形式。

    \item[\textbf{第三步（代入条件）：}] 代入 $a+b=1$。
    \[
    \frac{a^2+b^2}{2} \ge \left(\frac{1}{2}\right)^2 = \frac{1}{4}.
    \]
    \par\textbf{理由：} 将已知条件代入右边，简化计算。

    \item[\textbf{第四步（解出目标）：}] 两边乘以 $2$，得到 $a^2+b^2$ 的下界。
    \[
    a^2+b^2 \ge 2 \times \frac{1}{4} = \frac{1}{2}.
    \]
    \par\textbf{理由：} 代数变形求出最小值。

    \item[\textbf{第五步（取等条件）：}] 等号成立当且仅当 $a=b$。结合 $a+b=1$，得 $a=b=\frac{1}{2}$。
    \par\textbf{理由：} 原不等式等号当 $a=b$ 时成立，代入条件解出具体值。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{$a^2+b^2$ 的最小值为 $\frac{1}{2}$，当且仅当 $a=b=\frac{1}{2}$ 时取得。此题展示了不等式在求最值中的应用。}

\end{examplebox}
