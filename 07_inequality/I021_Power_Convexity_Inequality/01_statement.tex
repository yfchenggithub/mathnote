\begin{statementbox}

\textbf{\textcolor{CStatement}{条件}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{条件 1（正实数变量）：}}] 设 $x, y > 0$，$n \in \mathbb{R}$。
\end{description}

\textbf{\textcolor{CStatement}{结论}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{\textcolor{CStatement}{结论一（指数外凸）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 当指数 $n>1$ 或 $n<0$ 时，幂函数上凸，算术平均不小于幂平均。
    \[
    \frac{x^{n} + y^{n}}{2} \geq \left(\frac{x+y}{2}\right)^{n}.
    \]

    \item[\textbf{\textcolor{CStatement}{结论二（指数内凹）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 当指数 $0 < n < 1$ 时，幂函数下凸，算术平均不大于幂平均。
    \[
    \frac{x^{n} + y^{n}}{2} \leq \left(\frac{x+y}{2}\right)^{n}.
    \]

    \item[\textbf{\textcolor{CStatement}{推广结论（Jensen特例）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 此结论是凸函数 Jensen 不等式的二元特例。
    \[
    \text{对于凸函数 } f, \quad f\!\left(\frac{x+y}{2}\right) \le \frac{f(x)+f(y)}{2}.
    \]
\end{description}

\textbf{\textcolor{CStatement}{等号/取等条件：}} 等号成立当且仅当 $x = y$。

\end{statementbox}
