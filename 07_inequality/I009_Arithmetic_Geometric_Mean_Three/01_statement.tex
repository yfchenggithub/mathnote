\textbf{\textcolor{CStatement}{条件}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{条件 1（非负实数）：}}] 变量 $a, b, c$ 都是非负实数，即 $a \geq 0, b \geq 0, c \geq 0$。
\end{description}

\textbf{\textcolor{CStatement}{结论}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{\textcolor{CStatement}{结论一（三元均值不等式）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 三个数的算术平均值不小于它们的几何平均值。
    \[
    \frac{a+b+c}{3} \geq \sqrt[3]{abc}
    \]

    \item[\textbf{\textcolor{CStatement}{推广结论（等价形式）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 该不等式有几种常用且等价的表述形式。
    \[
    a+b+c \geq 3\sqrt[3]{abc} \quad \text{或} \quad abc \leq \left(\frac{a+b+c}{3}\right)^3
    \]
\end{description}

\textbf{\textcolor{CStatement}{等号/取等条件：}} 当且仅当三个数相等，即 $a = b = c$ 时，不等式中的等号成立。

\end{statementbox}
