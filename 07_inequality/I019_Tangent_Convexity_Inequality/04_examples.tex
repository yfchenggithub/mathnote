\begin{examplebox}
\textbf{\textcolor{CExample}{例 1（基础）}}

\textbf{\textcolor{CExample}{题目：}} 已知 $x, y \in (0, \pi/2)$，且 $x+y = \frac{\pi}{3}$，证明 $\tan x + \tan y \geq \frac{2}{\sqrt{3}}$。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（应用主定理）：}] 根据主定理，有 $\frac{\tan x + \tan y}{2} \geq \tan\left(\frac{x+y}{2}\right)$。
    \[
    \frac{\tan x + \tan y}{2} \geq \tan\left(\frac{x+y}{2}\right)
    \]
    \par\textbf{理由：}
    直接使用正切凸性不等式。

    \item[\textbf{第二步（代入已知）：}] 已知 $x+y = \frac{\pi}{3}$，所以 $\frac{x+y}{2} = \frac{\pi}{6}$，且 $\tan(\pi/6) = \frac{1}{\sqrt{3}}$。
    \[
    \frac{\tan x + \tan y}{2} \geq \tan\left(\frac{\pi}{6}\right) = \frac{1}{\sqrt{3}}
    \]
    \par\textbf{理由：}
    将已知条件代入不等式。

    \item[\textbf{第三步（整理结论）：}] 两边乘以2，得到 $\tan x + \tan y \geq \frac{2}{\sqrt{3}}$。
    \[
    \tan x + \tan y \geq 2 \cdot \frac{1}{\sqrt{3}} = \frac{2}{\sqrt{3}}
    \]
    \par\textbf{理由：}
    简单代数变形，得到所求不等式。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{对于满足 $x+y = \pi/3$ 的锐角 $x,y$，有 $\tan x + \tan y \geq 2/\sqrt{3}$，等号当 $x=y=\pi/6$ 时成立。}

\medskip
\textbf{\textcolor{CExample}{例 2（稍变形）}}

\textbf{\textcolor{CExample}{题目：}} 设 $\triangle ABC$ 为锐角三角形，求 $\tan A + \tan B + \tan C$ 的最小值，并说明取到最小值时三角形的形状。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（应用推论）：}] 根据推论，对于锐角三角形，有 $\tan A + \tan B + \tan C \geq 3\sqrt{3}$。
    \[
    \tan A + \tan B + \tan C \geq 3\sqrt{3}
    \]
    \par\textbf{理由：}
    直接使用三角形正切和不等式。

    \item[\textbf{第二步（验证等号）：}] 等号成立当且仅当 $A = B = C = \pi/3$，即三角形为等边三角形。
    \[
    \tan A + \tan B + \tan C = 3\sqrt{3} \iff A=B=C=\frac{\pi}{3}
    \]
    \par\textbf{理由：}
    推论中明确了取等条件。

    \item[\textbf{第三步（结论）：}] 因此，$\tan A + \tan B + \tan C$ 的最小值为 $3\sqrt{3}$，当且仅当三角形为等边三角形时取到。
    \[
    \min(\tan A + \tan B + \tan C) = 3\sqrt{3}
    \]
    \par\textbf{理由：}
    由不等式知下界为 $3\sqrt{3}$，且等号可达，故为最小值。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{在锐角三角形中，$\tan A + \tan B + \tan C$ 的最小值为 $3\sqrt{3}$，取到最小值时三角形为等边三角形。}
\end{examplebox}
