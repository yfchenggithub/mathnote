\begin{statementbox}

\textbf{\textcolor{CStatement}{条件}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{条件 1（角范围）：}}] 两个角 $x, y$ 都在开区间 $(0, \pi/2)$ 内。
\end{description}

\textbf{\textcolor{CStatement}{结论}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{\textcolor{CStatement}{结论一（正切凸性不等式）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 正切值的算术平均不小于这两角算术平均的正切值。
    \[
    \frac{\tan x + \tan y}{2} \geq \tan\left(\frac{x+y}{2}\right),
    \]
    等号成立当且仅当 $x = y$。

    \item[\textbf{\textcolor{CStatement}{推广结论（三角形正切和下界）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 锐角三角形三个内角正切之和不小于 $3\sqrt{3}$。
    \[
    \tan A + \tan B + \tan C \geq 3\sqrt{3},
    \]
    等号成立当且仅当 $A = B = C = \pi/3$（即三角形为等边三角形）。
\end{description}

\textbf{\textcolor{CStatement}{等号/取等条件：}} 对于结论一，等号成立当且仅当 $x = y$；对于推广结论，等号成立当且仅当三角形为等边三角形。

\end{statementbox}
