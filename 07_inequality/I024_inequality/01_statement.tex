\textbf{\textcolor{CStatement}{条件}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{条件 1（实数范围）：}}] $a, b, c$ 是任意实数。
\end{description}

\textbf{\textcolor{CStatement}{结论}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{\textcolor{CStatement}{结论一（不等式链）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 两两乘积和、平方平均、平方和间的大小关系。
    \[
    ab + bc + ca \leq \frac{(a+b+c)^2}{3} \leq a^2 + b^2 + c^2
    \]

    \item[\textbf{\textcolor{CStatement}{推广结论（柯西特例）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 中间项展开后，即平方和与乘积和的不等式。
    \[
    a^2+b^2+c^2 \ge ab+bc+ca
    \]
\end{description}

\textbf{\textcolor{CStatement}{等号/取等条件：}} 所有等号同时成立当且仅当 $a = b = c$。

\end{statementbox}
