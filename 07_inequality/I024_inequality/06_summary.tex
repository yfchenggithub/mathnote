\begin{summarybox}
\textbf{\textcolor{CSummary}{一句话核心}}
\textbf{三元对称式基本强度链：两两乘积和 $\le$ 平均尺度 $\le$ 平方和，等号仅在三数相等时成立。}

\medskip
\textbf{\textcolor{CSummary}{使用条件}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{条件 1（实数变量）：} $a, b, c$ 必须是实数，不限定正负。
    \item \textbf{条件 2（对称结构）：} 问题中的表达式具有关于 $a, b, c$ 的轮换对称性。
\end{itemize}

\medskip
\textbf{\textcolor{CSummary}{关键提醒}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{易错点（取等验证）：} 用其求最值后，务必回代 $a=b=c$ 到所有约束中检验。
    \item \textbf{检查项（方向特例）：} 不确定方向时，代一组特值（如 1,0,0）立即判断。
\end{itemize}
\end{summarybox}
