\begin{examplebox}
\textbf{\textcolor{CExample}{例 1（基础）}}

\textbf{\textcolor{CExample}{题目：}} 设 $a, b, c$ 为正实数，证明 $(a+b+c)\left(\frac{1}{a}+\frac{1}{b}+\frac{1}{c}\right) \geq 9$。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（确认条件）：}] 首先确认 $a, b, c$ 均为正实数。
    \par\textbf{理由：}
    不等式成立的前提是所有数均为正实数。

    \item[\textbf{第二步（应用不等式）：}] 直接应用倒数和乘积不等式（三元特例）：
    \[
    (a+b+c)\left(\frac{1}{a}+\frac{1}{b}+\frac{1}{c}\right) \geq 3^2 = 9.
    \]
    \par\textbf{理由：}
    对于 $n=3$ 个正实数，和与倒数和之积不小于 $3^2=9$。

    \item[\textbf{第三步（取等条件）：}] 等号成立当且仅当 $a=b=c$。
    \par\textbf{理由：}
    原不等式等号成立条件是所有数相等。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{因此不等式成立，最小值为 $9$，在 $a=b=c$ 时取得。}

\textbf{\textcolor{CExample}{例 2（稍变形）}}

\textbf{\textcolor{CExample}{题目：}} 设 $x, y$ 为正实数，求 $(x+y)\left(\frac{1}{x}+\frac{1}{y}\right)$ 的最小值。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（确认条件）：}] 确认 $x, y$ 为正实数。
    \par\textbf{理由：}
    确保不等式适用。

    \item[\textbf{第二步（应用不等式）：}] 应用倒数和乘积不等式（$n=2$ 情形）：
    \[
    (x+y)\left(\frac{1}{x}+\frac{1}{y}\right) \geq 2^2 = 4.
    \]
    \par\textbf{理由：}
    对于两个正实数，和与倒数和之积不小于 $2^2=4$。

    \item[\textbf{第三步（取等条件）：}] 等号成立当且仅当 $x=y$。
    \par\textbf{理由：}
    当且仅当两数相等时不等式等号成立。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{最小值为 $4$，在 $x=y$ 时取得。}
\end{examplebox}
