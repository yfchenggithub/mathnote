\begin{summarybox}
\textbf{\textcolor{CSummary}{一句话核心}}
\textbf{正实数的和与倒数和之积有下界 $n^2$，等号仅当所有数相等时取得。}

\medskip
\textbf{\textcolor{CSummary}{使用条件}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{条件 1（正实数）：} 所有涉及的数必须为正实数。
    \item \textbf{条件 2（倒数形式）：} 第二个和必须是第一个和中每个数的倒数之和。
\end{itemize}

\medskip
\textbf{\textcolor{CSummary}{关键提醒}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{易错点（忽略正负）：} 忘记检查正实数条件，导致不等式方向错误。
    \item \textbf{检查项（全等取等）：} 等号成立要求所有数相等，应用后务必验证取等可能性。
\end{itemize}
\end{summarybox}
