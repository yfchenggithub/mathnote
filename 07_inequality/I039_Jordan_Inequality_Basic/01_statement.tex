\textbf{\textcolor{CStatement}{条件}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{条件 1（正实数）：}}] 所有 $a_1, a_2, \ldots, a_n$ 均为正实数。
\end{description}

\textbf{\textcolor{CStatement}{结论}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{\textcolor{CStatement}{结论一（倒数和积不等式）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 和与倒数和之积不小于 $n$ 的平方。
    \[
    \left( \sum_{i=1}^n a_i \right) \left( \sum_{i=1}^n \frac{1}{a_i} \right) \geq n^2.
    \]

    \item[\textbf{\textcolor{CStatement}{推广结论（三元特例）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 三元时乘积不小于 $9$。
    \[
    (a+b+c)\left(\frac{1}{a}+\frac{1}{b}+\frac{1}{c}\right) \geq 9.
    \]
\end{description}

\textbf{\textcolor{CStatement}{等号/取等条件：}} 当且仅当所有 $a_1 = a_2 = \cdots = a_n$ 时等号成立；对于三元特例，当且仅当 $a=b=c$。

\end{statementbox}
