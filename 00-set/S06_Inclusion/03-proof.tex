\begin{proofbox}
\textbf{证明} $A\cap B = A \Leftrightarrow A\subseteq B$：
\begin{enumerate}
    \item[($\Rightarrow$)] 若 $A\cap B = A$，则对任意 $x\in A$，有 $x\in A\cap B$，故 $x\in B$，因此 $A\subseteq B$。
    \item[($\Leftarrow$)] 若 $A\subseteq B$，则对任意 $x\in A$，$x\in B$，从而 $x\in A\cap B$，即 $A\subseteq A\cap B$；又 $A\cap B\subseteq A$ 恒成立，故 $A\cap B = A$。
\end{enumerate}
\end{proofbox}

\begin{proofbox}
\textbf{证明} $A\subseteq B \Leftrightarrow A\cup B = B$：
\begin{enumerate}
    \item[($\Rightarrow$)] 若 $A\subseteq B$，则 $A\cup B \subseteq B\cup B = B$；又 $B\subseteq A\cup B$ 恒成立，故 $A\cup B = B$。
    \item[($\Leftarrow$)] 若 $A\cup B = B$，则对任意 $x\in A$，有 $x\in A\cup B = B$，所以 $x\in B$，即 $A\subseteq B$。
\end{enumerate}
\end{proofbox}

\begin{proofbox}
\textbf{证明} $A\subseteq B \Leftrightarrow C_U B \subseteq C_U A$：
\begin{enumerate}
    \item[($\Rightarrow$)] 若 $A\subseteq B$，则对任意 $x\in C_U B$，有 $x\notin B$，故 $x\notin A$（否则若 $x\in A\subseteq B$，矛盾），所以 $x\in C_U A$，即 $C_U B \subseteq C_U A$。
    \item[($\Leftarrow$)] 若 $C_U B \subseteq C_U A$，则对任意 $x\in A$，有 $x\notin C_U A$，从而 $x\notin C_U B$（因为 $C_U B\subseteq C_U A$），故 $x\in B$，即 $A\subseteq B$。
\end{enumerate}
\end{proofbox}
