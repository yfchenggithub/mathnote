\begin{examplebox}
\textbf{例题1：} 设全集 $U=\{1,2,3,4,5\}$，$A=\{1,2\}$，$B=\{1,2,3\}$。验证四个等价命题均成立。

\textit{解析：} 
\begin{itemize}
    \item $A\cap B = \{1,2\} = A$；
    \item $A\cup B = \{1,2,3\} = B$；
    \item 显然 $A\subseteq B$；
    \item $C_U A = \{3,4,5\}$，$C_U B = \{4,5\}$，有 $C_U B \subseteq C_U A$。
\end{itemize}
因此四个命题同时成立。
\end{examplebox}

\begin{examplebox}
\textbf{例题2：} 已知 $A\cap B = A$，判断下列命题的真假：
\begin{enumerate}
    \item $A\subseteq B$；
    \item $B\subseteq A$；
    \item $A\cup B = A$；
    \item $C_U A \subseteq C_U B$。
\end{enumerate}

\textit{解析：} 由 $A\cap B = A$ 知 $A\subseteq B$（根据等价链条）。
\begin{enumerate}
    \item 真。
    \item 假（反例：$A=\{1\},B=\{1,2\}$，$A\cap B=A$ 但 $B\not\subseteq A$）。
    \item 假（$A\cup B = B \neq A$，除非 $A=B$）。
    \item 由 $A\subseteq B$ 得 $C_U B \subseteq C_U A$，故原命题"$C_U A \subseteq C_U B$"应为假。
\end{enumerate}
\end{examplebox}
