\begin{proofbox}
\textbf{条件判定的标准化流程（定义法）：}
\begin{enumerate}
    \item \textbf{第一步：尝试由 $p$ 推 $q$}。若 $p \Rightarrow q$ 为真，则 $p$ 是 $q$ 的充分条件。
    \item \textbf{第二步：尝试由 $q$ 推 $p$}。若 $q \Rightarrow p$ 为真，则 $p$ 是 $q$ 的必要条件。
    \item \textbf{第三步：综合结论}：
    \begin{itemize}
        \item \textit{充分不必要}：$p \Rightarrow q$ 且 $q \nRightarrow p$；
        \item \textit{必要不充分}：$p \nRightarrow q$ 且 $q \Rightarrow p$；
        \item \textit{充要条件}：$p \Leftrightarrow q$；
        \item \textit{既不充分也不必要}：$p \nRightarrow q$ 且 $q \nRightarrow p$。
    \end{itemize}
\end{enumerate}
\end{proofbox}
