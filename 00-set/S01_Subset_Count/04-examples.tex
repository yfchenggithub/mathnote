\begin{examplebox}
\textbf{例题：} 已知集合 $A = \{1,2,3,4,5\}$，求：
\begin{enumerate}
    \item 集合 $A$ 的所有子集个数；
    \item 集合 $A$ 的真子集个数；
    \item 集合 $A$ 的非空子集个数；
    \item 集合 $A$ 的非空真子集个数。
\end{enumerate}

\textit{解析：} 集合 $A$ 含有 $5$ 个元素，因此：
\begin{enumerate}
    \item 子集个数为 $2^5 = 32$；
    \item 真子集个数为 $2^5 - 1 = 31$；
    \item 非空子集个数为 $2^5 - 1 = 31$；
    \item 非空真子集个数为 $2^5 - 2 = 30$。
\end{enumerate}
\end{examplebox}
