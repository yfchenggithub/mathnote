\begin{trapbox}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
  \item[\textbf{\textcolor{CTrap}{易错点一（忽略范围）}}]
  忘记 $1\le m<n$，直接把 $m=n$ 或 $m>n$ 代入公式.
  \item[\textbf{\textcolor{CTrap}{正确理解（条件限制）：}}]
  本公式中分母是 $n-m$，所以必须 $m<n$；同时 $A_{n-1}^m$ 也要求 $m\le n-1$.
  \item[\textbf{\textcolor{CTrap}{错因分析（只看 $A_n^m$）：}}]
  当 $m=n$ 时，$A_n^n$ 本身有定义，但 $A_{n-1}^n$ 没有定义，且 $\frac{n}{n-m}$ 分母为 $0$，所以不能套用本递推.
\end{description}

\medskip
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
  \item[\textbf{\textcolor{CTrap}{易错点二（方向记反）}}]
  把公式错写成
  \[
    A_{n-1}^m=\frac{n}{n-m}A_n^m.
  \]
  \item[\textbf{\textcolor{CTrap}{正确理解（公式形式）：}}]
  正确公式是
  \[
    A_n^m=\frac{n}{n-m}A_{n-1}^m,
  \]
  如果反过来表示，应为
  \[
    A_{n-1}^m=\frac{n-m}{n}A_n^m.
  \]
  \item[\textbf{\textcolor{CTrap}{错因分析（分子分母混淆）：}}]
  从 $n-1$ 推到 $n$ 时排列数变大，应乘以大于 $1$ 的 $\frac{n}{n-m}$；反向时才乘以小于 $1$ 的 $\frac{n-m}{n}$.
\end{description}

\medskip
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
  \item[\textbf{\textcolor{CTrap}{易错点三（混淆两种递推）}}]
  把本公式与另一个排列数递推
  \[
    A_n^m=(n-m+1)A_n^{m-1}
  \]
  混淆.
  \item[\textbf{\textcolor{CTrap}{正确理解（看谁在变）：}}]
  本讲公式是“上标不变、下标变化”：
  \[
    A_n^m=\frac{n}{n-m}A_{n-1}^m.
  \]
  另一个公式是“下标不变、上标变化”：
  \[
    A_n^m=(n-m+1)A_n^{m-1}.
  \]
  \item[\textbf{\textcolor{CTrap}{错因分析（只背名字）：}}]
  两个都叫“排列数递推公式”，必须根据题目中变化的是 $n$ 还是 $m$ 来选择.
\end{description}

\medskip
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
  \item[\textbf{\textcolor{CTrap}{易错点四（乱约分）}}]
  把 $\frac{n}{n-m}$ 错误约分，甚至误认为它等于 $\frac{1}{1-m}$.
  \item[\textbf{\textcolor{CTrap}{正确理解（整体分母）：}}]
  $n-m$ 是一个整体，不能把分子中的 $n$ 与分母中的 $n$ 单独消去.
  \item[\textbf{\textcolor{CTrap}{错因分析（代数习惯错误）：}}]
  只有公因式才能约去，$n$ 不是 $n-m$ 的公因式.
\end{description}
\end{trapbox}
