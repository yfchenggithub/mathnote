\begin{summarybox}
\textbf{\textcolor{CSummary}{一句话核心}}\par
排列数按 $n$ 递推公式
\[
  A_n^m=\frac{n}{n-m}A_{n-1}^m
\]
表示：固定取出个数 $m$，把元素总数从 $n-1$ 增加到 $n$ 时，排列数按比例 $\frac{n}{n-m}$ 放大.

\medskip
\textbf{\textcolor{CSummary}{使用条件}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
  \item \textbf{条件1（整数要求）：} $n,m$ 为正整数，且 $1\le m<n$.
  \item \textbf{条件2（递推方向）：} 用 $A_{n-1}^m$ 推 $A_n^m$，上标 $m$ 保持不变.
\end{itemize}

\medskip
\textbf{\textcolor{CSummary}{关键提醒}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
  \item \textbf{先看谁变：} 如果变的是下标 $n$，用 $A_n^m=\frac{n}{n-m}A_{n-1}^m$.
  \item \textbf{注意分母：} 必须 $m<n$，否则分母 $n-m$ 为 $0$ 或相关排列数不存在.
  \item \textbf{不要倒置：} 反向表达时应写成 $A_{n-1}^m=\frac{n-m}{n}A_n^m$.
\end{itemize}
\end{summarybox}
