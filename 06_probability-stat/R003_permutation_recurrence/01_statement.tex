\begin{statementbox}
\textbf{\textcolor{CStatement}{条件}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{条件 1（正整数范围）：}}] $n,m\in\mathbb{N}^*$，且 $1\leq m\leq n$。
    \item[\textbf{\textcolor{CStatement}{条件 2（边界约定）：}}] 为覆盖 $m=1$ 的情形，约定 $A_k^0=1$。若教材暂未引入该约定，则本公式主要用于 $2\leq m\leq n$，而 $m=1$ 时单独使用 $A_n^1=n$。
\end{description}

\textbf{\textcolor{CStatement}{结论}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{\textcolor{CStatement}{结论一（首位分步递推）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 先确定第一个位置，有 $n$ 种选法；剩余 $n-1$ 个元素中再取 $m-1$ 个排列。
    \[
    A_n^m = n\cdot A_{n-1}^{m-1}.
    \]

    \item[\textbf{\textcolor{CStatement}{相关公式（按取出个数递推）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 固定总元素数 $n$，也可由 $A_n^{m-1}$ 推到 $A_n^m$。
    \[
    A_n^m = (n-m+1)\cdot A_n^{m-1}.
    \]
    \textbf{注意：} 这是另一条常用递推式，应与本结论的“$n,m$ 同时减 $1$”区分记忆。
\end{description}
\end{statementbox}
