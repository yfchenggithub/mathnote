\begin{summarybox}
\textbf{\textcolor{CSummary}{一句话核心}}

\textbf{排列数分类递推：从 $n+1$ 个元素中取 $m$ 个排列，按是否含新增元素分类，得到 $A_{n+1}^m=A_n^m+mA_n^{m-1}$。}

\medskip
\textbf{\textcolor{CSummary}{使用条件}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{条件 1（原式范围）：} $n,m$ 为正整数，且 $1\leq m\leq n$。
    \item \textbf{条件 2（边界约定）：} 使用 $m=1$ 时需采用 $A_n^0=1$ 的约定。
    \item \textbf{条件 3（平移式范围）：} $A_n^m=A_{n-1}^m+mA_{n-1}^{m-1}$ 通常要求 $n\geq2$，$1\leq m\leq n-1$。
\end{itemize}

\medskip
\textbf{\textcolor{CSummary}{关键提醒}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{易错点（漏乘 $m$）：} 含新增元素时要先选位置，所以第二项是 $mA_n^{m-1}$。
    \item \textbf{检查项（上下标对齐）：} 原式右端两个排列数下标都是 $n$，上标分别是 $m$ 与 $m-1$。
    \item \textbf{区分点（与组合数递推）：} 组合数递推没有系数 $m$，是因为排列数递推整体除以了 $m!$。
\end{itemize}
\end{summarybox}
