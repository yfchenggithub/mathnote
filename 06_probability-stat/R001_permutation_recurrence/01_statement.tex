\textbf{\textcolor{CStatement}{条件}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{条件1（正整数范围）：}}] 给定正整数 $n, m$，满足 $2 \le m \le n$。
\end{description}

\textbf{\textcolor{CStatement}{结论}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{\textcolor{CStatement}{结论一（递推公式）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 排列数 $A_n^m$ 可从前一个排列数 $A_n^{m-1}$ 乘以一个因子得到。
    \[
    A_n^m = (n - m + 1) A_n^{m-1}.
    \]

    \item[\textbf{\textcolor{CStatement}{常见变形（另一递推）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 也可从 $A_{n-1}^{m-1}$ 乘以 $n$ 得到。
    \[
    A_n^m = n \cdot A_{n-1}^{m-1}.
    \]
\end{description}
\end{statementbox}
