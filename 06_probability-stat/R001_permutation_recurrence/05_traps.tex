\begin{trapbox}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CTrap}{易错点一（忘记范围）}}] 当 $m=1$ 时，认为 $A_n^1 = (n-1+1)A_n^0$，但 $A_n^0$ 未定义，导致错误。
    \item[\textbf{\textcolor{CTrap}{正确理解（范围限制）：}}] 递推公式仅适用于 $2\le m\le n$；$m=1$ 时 $A_n^1 = n$ 为已知。
    \item[\textbf{\textcolor{CTrap}{错因分析（定义缺失）：}}] 未注意到 $A_n^{m-1}$ 当 $m-1=0$ 时没有排列意义。
\end{description}

\medskip
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CTrap}{易错点二（因子错误）}}] 误写为 $A_n^m = (n-m)A_n^{m-1}$，漏掉 $+1$。
    \item[\textbf{\textcolor{CTrap}{正确理解（因子正确）：}}] 正确因子是 $n-m+1$，可通过定义验证：$A_n^m = (n-m+1)A_n^{m-1}$。
    \item[\textbf{\textcolor{CTrap}{错因分析（记忆混淆）：}}] 与组合数递推 $C_n^m = \frac{n-m+1}{m}C_n^{m-1}$ 混淆，或误以为最后一位有 $n-m$ 种选择。
\end{description}

\medskip
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CTrap}{易错点三（参数对应错误）}}] 在使用递推时，将 $n$ 与 $m$ 位置搞反，例如写成 $A_n^m = (n-m+1)A_{n-1}^{m}$。
    \item[\textbf{\textcolor{CTrap}{正确理解（参数对应）：}}] 递推公式中下标 $n$ 不变，上标减小 $1$，系数为 $n-m+1$，即 $A_n^m = (n-m+1)A_n^{m-1}$。
    \item[\textbf{\textcolor{CTrap}{错因分析（结构误解）：}}] 误以为递推是降低 $n$ 而非降低 $m$，或不理解系数与剩余元素个数的关系。
\end{description}
\end{trapbox}
