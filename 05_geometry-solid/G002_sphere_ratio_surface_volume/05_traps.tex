\begin{trapbox}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CTrap}{易错点一（混淆平方与立方）}}] 直接把半径之比 $m:n$ 当作表面积比 $m:n$ 或体积比 $m:n$。
    \item[\textbf{\textcolor{CTrap}{正确理解（维度决定次数）：}}] 表面积是二维量，所以表面积比是 $m^2:n^2$；体积是三维量，所以体积比是 $m^3:n^3$。
    \item[\textbf{\textcolor{CTrap}{错因分析（线性惯性）：}}] 误以为“半径变几倍，所有量都变几倍”，忽视了面积和体积的维度差异。
\end{description}

\medskip
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CTrap}{易错点二（单位维度混乱）}}] 计算具体数值时，把半径平方或立方后的结果仍当作长度单位。
    \item[\textbf{\textcolor{CTrap}{正确理解（单位对应）：}}] 若半径单位是 $\text{cm}$，则表面积单位是 $\text{cm}^2$，体积单位是 $\text{cm}^3$。半径平方对应面积单位，半径立方对应体积单位。
    \item[\textbf{\textcolor{CTrap}{错因分析（只算数值不看量纲）：}}] 只关注数字运算，忘了长度、面积、体积属于不同维度。
\end{description}

\medskip
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CTrap}{易错点三（比例反向使用）}}] 已知面积比 $a:b$，错误地认为半径比为 $a:b$ 或 $a^2:b^2$；已知体积比时也犯类似错误。
    \item[\textbf{\textcolor{CTrap}{正确理解（逆运算开方或开立方）：}}] 当 $a,b>0$ 时，若表面积比为 $a:b$，则半径比为
    \[
    R_1:R_2=\sqrt{a}:\sqrt{b}.
    \]
    若体积比为 $a:b$，则半径比为
    \[
    R_1:R_2=\sqrt[3]{a}:\sqrt[3]{b}.
    \]
    \item[\textbf{\textcolor{CTrap}{错因分析（逆运算不熟）：}}] 正向由半径到面积、体积要平方或立方；反向由面积、体积回到半径，就要开平方或开立方。
\end{description}

\medskip
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CTrap}{易错点四（“是几倍”和“大几倍”混淆）}}] 把“大球体积是小球体积的多少倍”和“大球体积比小球体积大多少倍”混为一谈。
    \item[\textbf{\textcolor{CTrap}{正确理解（倍数口径）：}}] “A 是 B 的多少倍”用 $\dfrac{A}{B}$；“A 比 B 大多少倍”通常用 $\dfrac{A-B}{B}$。
    \item[\textbf{\textcolor{CTrap}{错因分析（题眼没看清）：}}] 没有区分“是”和“比……大”，导致最后答案差 $1$ 倍。
\end{description}

\medskip
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CTrap}{易错点五（相似体推广乱用）}}] 看到“面积比平方、体积比立方”就直接套到任意两个立体上。
    \item[\textbf{\textcolor{CTrap}{正确理解（必须相似）：}}] 该推广只适用于相似图形或相似立体，并且必须比较对应长度、对应面积、对应体积；若两个立体不相似，不能只凭某一条长度之比推出面积比或体积比。
    \item[\textbf{\textcolor{CTrap}{错因分析（忽略前提）：}}] 把“球一定相似”的结论误推广到所有空间几何体，漏掉了“相似”这个关键条件。
\end{description}
\end{trapbox}
