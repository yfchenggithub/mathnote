\begin{proofbox}

\textbf{\textcolor{CProof}{思路提示}}

由球表面积公式 $S=4\pi R^2$ 和体积公式 $V=\frac{4}{3}\pi R^3$，将两个球的半径之比代入，比值化简即可得到结论。

\medskip
\textbf{\textcolor{CProof}{正式推导}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.55em]
    \item[\textbf{步骤一（列出公式）：}] 设球 1 半径为 $R_1$，球 2 半径为 $R_2$。
    \[
    S_1 = 4\pi R_1^2, \quad S_2 = 4\pi R_2^2; \quad V_1 = \frac{4}{3}\pi R_1^3, \quad V_2 = \frac{4}{3}\pi R_2^3.
    \]
    \par\textbf{理由：} 直接使用标准表面积和体积公式。

    \item[\textbf{步骤二（代入半径比）：}] 已知 $R_1:R_2 = m:n$，即 $\frac{R_1}{R_2} = \frac{m}{n}$。
    \[
    \frac{S_1}{S_2} = \frac{4\pi R_1^2}{4\pi R_2^2} = \left(\frac{R_1}{R_2}\right)^2 = \left(\frac{m}{n}\right)^2 = \frac{m^2}{n^2}.
    \]
    \par\textbf{理由：} 约去公因数 $4\pi$，利用比例平方。

    \item[\textbf{步骤三（同理得体积比）：}]
    \[
    \frac{V_1}{V_2} = \frac{\frac{4}{3}\pi R_1^3}{\frac{4}{3}\pi R_2^3} = \left(\frac{R_1}{R_2}\right)^3 = \left(\frac{m}{n}\right)^3 = \frac{m^3}{n^3}.
    \]
    \par\textbf{理由：} 约去公因数 $\frac{4}{3}\pi$，利用比例立方。
\end{description}

\medskip
\textbf{\textcolor{CProof}{结论回扣}}

\textbf{因此，当两球半径之比为 $m:n$ 时，表面积之比为 $m^2:n^2$，体积之比为 $m^3:n^3$。} 证明过程仅依赖于球的表面积和体积公式，以及比例的基本性质。

\end{proofbox}
