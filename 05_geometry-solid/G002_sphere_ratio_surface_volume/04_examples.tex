\textbf{\textcolor{CExample}{例 1（基础）}}

\textbf{\textcolor{CExample}{题目：}} 已知两个球的半径之比为 $2:3$，求它们的表面积之比和体积之比。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（识别比例）：}] 半径比 $m:n = 2:3$。
    \par\textbf{理由：} 直接提取输入条件。
    \item[\textbf{第二步（套用面积比公式）：}] 表面积之比 $S_1:S_2 = m^2:n^2 = 2^2:3^2 = 4:9$。
    \par\textbf{理由：} 由结论，面积比等于半径比的平方。
    \item[\textbf{第三步（套用体积比公式）：}] 体积之比 $V_1:V_2 = m^3:n^3 = 2^3:3^3 = 8:27$。
    \par\textbf{理由：} 由结论，体积比等于半径比的立方。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{表面积比 $4:9$，体积比 $8:27$。}

\medskip
\textbf{\textcolor{CExample}{例 2（稍变形）}}

\textbf{\textcolor{CExample}{题目：}} 两个球半径分别为 $3\,\text{cm}$ 和 $5\,\text{cm}$，求大球体积比小球体积大多少倍？

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（先求半径比）：}] 半径比 $R_1:R_2 = 3:5$，所以 $m:n=3:5$。
    \par\textbf{理由：} 先统一成比例形式。
    \item[\textbf{第二步（套用体积比公式）：}] 体积比 $V_1:V_2 = 3^3:5^3 = 27:125$。
    \par\textbf{理由：} 直接使用结论。
    \item[\textbf{第三步（计算倍数）：}] 大球体积是小球的 $\frac{125}{27} \approx 4.63$ 倍，即大体积比小体积大 $\frac{125}{27} - 1 = \frac{98}{27}$ 倍。
    \par\textbf{理由：} “大多少倍”指差值倍数。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{大球体积比小球体积大 $\frac{98}{27}$ 倍。}

\textbf{\textcolor{CExample}{例 3（综合应用）}}

\textbf{\textcolor{CExample}{题目：}} 两个球的半径之比为 $1:2$，它们的体积之和为 $36\pi$，求较大球的表面积。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（设半径与体积比）：}] 设小球半径 $r$，大球半径 $2r$。体积比 $V_1:V_2 = 1^3:2^3 = 1:8$。
    \par\textbf{理由：} 由结论得体积比。
    \item[\textbf{第二步（求具体体积）：}] 设小球体积 $V_1 = V$，则大球体积 $V_2 = 8V$，$
    V + 8V = 9V = 36\pi$，解得 $V = 4\pi$。
    \par\textbf{理由：} 体积和已知，列方程。
    \item[\textbf{第三步（求大球半径）：}] 由 $V_2 = \frac{4}{3}\pi (2r)^3 = 8V = 8\times 4\pi = 32\pi$，所以 $\frac{4}{3}\pi (8r^3) = 32\pi$，解得 $r^3 = 3$，$r = \sqrt[3]{3}$。
    \par\textbf{理由：} 用体积公式反求半径。
    \item[\textbf{第四步（求大球表面积）：}] 大球半径 $2r = 2\sqrt[3]{3}$，表面积 $S = 4\pi (2r)^2 = 4\pi \times 4r^2 = 16\pi r^2 = 16\pi (\sqrt[3]{3})^2 = 16\pi \cdot 3^{2/3}$。
    \par\textbf{理由：} 代入表面积公式。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{较大球的表面积为 $16\pi \cdot 3^{2/3}$。}
