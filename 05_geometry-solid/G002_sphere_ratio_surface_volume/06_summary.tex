\begin{summarybox}

\textbf{\textcolor{CSummary}{一句话核心}}

\textbf{同一顺序下，半径比平方得表面积比，半径比立方得体积比。}

\medskip
\textbf{\textcolor{CSummary}{公式总览}}
\[
R_1:R_2=m:n
\quad\Longrightarrow\quad
S_1:S_2=m^2:n^2,
\qquad
V_1:V_2=m^3:n^3.
\]

\medskip
\textbf{\textcolor{CSummary}{使用条件}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{条件 1（已知半径比）：} 直接把半径比平方得到表面积比，立方得到体积比。
    \item \textbf{条件 2（已知具体半径）：} 先化为半径比，能约分就约分，再套用平方、立方关系。
    \item \textbf{条件 3（已知面积比或体积比）：} 反求半径比时，面积比开平方，体积比开立方。
    \item \textbf{条件 4（推广到相似体）：} 必须是相似图形或相似立体，并且比较的是对应长度、对应面积、对应体积。
\end{itemize}

\medskip
\textbf{\textcolor{CSummary}{关键提醒}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{不要混用：} 表面积比用平方，体积比用立方。
    \item \textbf{注意顺序：} 小球:大球 与 大球:小球 是相反的比例。
    \item \textbf{看清问法：} “是几倍”用商，“比……大几倍”用差值再除以基准量。
    \item \textbf{别漏前提：} 球之间天然相似；一般立体要先确认相似，才能使用面积比平方、体积比立方。
\end{itemize}
\end{summarybox}
