\begin{summarybox}

\textbf{\textcolor{CSummary}{一句话核心}}

\textbf{半径比平方得面积比，半径比立方得体积比。}

\medskip
\textbf{\textcolor{CSummary}{使用条件}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{条件 1（已知半径比）：} 直接得到 $m:n$ 或 $R_1:R_2$。
    \item \textbf{条件 2（已知具体半径）：} 先化为最简整数比再套用公式。
\end{itemize}

\medskip
\textbf{\textcolor{CSummary}{关键提醒}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{易错点（混淆平方立方）：} 面积比用平方，体积比用立方，不能混用。
    \item \textbf{检查项（逆运算开方）：} 已知面积比反求半径比需开平方，已知体积比反求半径比需开立方。
\end{itemize}
\end{summarybox}
