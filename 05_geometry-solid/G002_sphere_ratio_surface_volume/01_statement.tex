\begin{statementbox}

\textbf{\textcolor{CStatement}{条件}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{条件 1（半径比例）：}}] 两个球的半径分别为 $R_1,R_2$，且按“球 1 : 球 2”的顺序有
    \[
    R_1:R_2=m:n,
    \]
    其中 $R_1,R_2>0$，$m,n>0$。
\end{description}

\textbf{\textcolor{CStatement}{结论}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{\textcolor{CStatement}{结论一（表面积比例）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 表面积之比等于半径比的平方，且比例顺序保持一致。
    \[
    S_1:S_2 = m^2:n^2,
    \qquad \text{即}\qquad
    \frac{S_1}{S_2}=\left(\frac{R_1}{R_2}\right)^2.
    \]

    \item[\textbf{\textcolor{CStatement}{结论二（体积比例）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 体积之比等于半径比的立方，且比例顺序保持一致。
    \[
    V_1:V_2 = m^3:n^3,
    \qquad \text{即}\qquad
    \frac{V_1}{V_2}=\left(\frac{R_1}{R_2}\right)^3.
    \]

    \item[\textbf{\textcolor{CStatement}{常见推广（相似体）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 对任意两个相似立体，若对应线段之比为 $m:n$，则对应面积之比为 $m^2:n^2$，对应体积之比为 $m^3:n^3$。
    \[
    S_1^{(\text{对应})}:S_2^{(\text{对应})}=m^2:n^2,
    \qquad
    V_1^{(\text{对应})}:V_2^{(\text{对应})}=m^3:n^3.
    \]
    特别地，若用放大倍数
    \[
    k=\frac{R_1}{R_2}
    \quad\text{（球）}\qquad\text{或}\qquad
    k=\frac{\text{对应线段}_1}{\text{对应线段}_2}
    \quad\text{（一般相似体）},
    \]
    表示，则
    \[
    \frac{S_1}{S_2}=k^2,
    \qquad
    \frac{V_1}{V_2}=k^3.
    \]
    注意：若只是平面相似图形，只谈对应面积比；只有空间相似体才谈体积比。
\end{description}

\textbf{\textcolor{CStatement}{参数限制：}} 半径必须为正数。若 $m:n$ 能化为最简整数比，写成最简整数比最便于计算；若不能化为整数比，保持根式或分式形式即可。

\end{statementbox}
