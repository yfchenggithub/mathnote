\begin{statementbox}

\textbf{\textcolor{CStatement}{条件}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{条件 1（半径比例）：}}] 两个球的半径之比为 $R_1:R_2=m:n$，其中 $m,n>0$。
\end{description}

\textbf{\textcolor{CStatement}{结论}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{\textcolor{CStatement}{结论一（面积比例）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 表面积之比等于半径比的平方。
    \[
    S_1:S_2 = m^2:n^2
    \]

    \item[\textbf{\textcolor{CStatement}{结论二（体积比例）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 体积之比等于半径比的立方。
    \[
    V_1:V_2 = m^3:n^3
    \]

    \item[\textbf{\textcolor{CStatement}{常见变形（相似体）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 任意相似体（如正方体、正四面体）中，对应面积比等于相似比的平方，体积比等于相似比的立方。
    \[
    S_{\text{对应}}:S_{\text{对应}} = k^2, \quad V_{\text{对应}}:V_{\text{对应}} = k^3
    \]
    （其中 $k$ 为对应线段之比）
\end{description}

\textbf{\textcolor{CStatement}{参数限制：}} $m,n$ 均为正实数，且 $m:n$ 为最简整数比时结论形式最简洁。

\end{statementbox}
