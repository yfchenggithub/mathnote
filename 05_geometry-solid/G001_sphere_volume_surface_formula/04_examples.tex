\begin{examplebox}
\textbf{\textcolor{CExample}{例1（基础）}}

\textbf{\textcolor{CExample}{题目：}} 已知一个球的半径为 $R = 3\,\mathrm{cm}$，求它的体积 $V$ 和表面积 $S$。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
\item[\textbf{第一步（代入体积公式）：}] 将 $R=3$ 代入 $V = \frac{4}{3} \pi R^{3}$，得
\[
V=\frac{4}{3}\pi\times 3^{3}=\frac{4}{3}\pi\times 27=36\pi\,\mathrm{cm}^{3}.
\]
\par\textbf{理由：} 直接套用球的体积公式。

\item[\textbf{第二步（代入表面积公式）：}] 将 $R=3$ 代入 $S = 4\pi R^{2}$，得
\[
S=4\pi\times 3^{2}=4\pi\times 9=36\pi\,\mathrm{cm}^{2}.
\]
\par\textbf{理由：} 直接套用球的表面积公式。

\item[\textbf{第三步（给出答案）：}] 体积为 $36\pi\,\mathrm{cm}^{3}$，表面积为 $36\pi\,\mathrm{cm}^{2}$。
\par\textbf{理由：} 题目要求求体积和表面积，注意二者单位不同。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{代入半径直接计算即可。体积和表面积可能数值相等，但单位不同，不能说“体积等于表面积”。}

\textbf{\textcolor{CExample}{例2（稍变形）}}

\textbf{\textcolor{CExample}{题目：}} 已知一个球的体积为 $V = 288\pi\,\mathrm{cm}^{3}$，求它的表面积 $S$。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
\item[\textbf{第一步（逆向求半径）：}] 由 $V = \frac{4}{3} \pi R^{3}$ 得
\[
R^{3}=\frac{3V}{4\pi}
=\frac{3\times 288\pi}{4\pi}
=216,
\]
所以
\[
R=\sqrt[3]{216}=6\,\mathrm{cm}.
\]
\par\textbf{理由：} 先由体积求出半径，再求表面积。

\item[\textbf{第二步（代入表面积公式）：}]
\[
S=4\pi R^{2}=4\pi\times 6^{2}=144\pi\,\mathrm{cm}^{2}.
\]
\par\textbf{理由：} 得到半径后直接套用表面积公式。

\item[\textbf{第三步（给出答案）：}] 表面积为 $144\pi\,\mathrm{cm}^{2}$。
\par\textbf{理由：} 题目要求表面积。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{已知体积求表面积时，先通过开立方解出半径，再代入表面积公式。}

\textbf{\textcolor{CExample}{例3（综合应用）}}

\textbf{\textcolor{CExample}{题目：}} 两个球半径之比为 $R_1 : R_2 = 2 : 3$，求它们的体积之比 $V_1 : V_2$ 和表面积之比 $S_1 : S_2$。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
\item[\textbf{第一步（设比例参数）：}] 设 $R_1 = 2k$，$R_2 = 3k$（$k>0$）。
\par\textbf{理由：} 用同一参数表示半径，方便计算比值。

\item[\textbf{第二步（求体积之比）：}] 因为球的体积与半径的三次方成正比，所以
\[
\frac{V_1}{V_2}=\frac{(2k)^{3}}{(3k)^{3}}=\frac{8}{27}.
\]
\par\textbf{理由：} 体积公式中含有 $R^{3}$，比值为半径比的三次方。

\item[\textbf{第三步（求表面积之比）：}] 因为球的表面积与半径的平方成正比，所以
\[
\frac{S_1}{S_2}=\frac{(2k)^{2}}{(3k)^{2}}=\frac{4}{9}.
\]
\par\textbf{理由：} 表面积公式中含有 $R^{2}$，比值为半径比的平方。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{两个球的体积之比等于它们半径之比的三次方；
两个球的表面积之比等于它们半径之比的平方。}
\end{examplebox}
