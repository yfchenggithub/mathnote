\begin{examplebox}
\textbf{\textcolor{CExample}{例1（基础）}}

\textbf{\textcolor{CExample}{题目：}} 已知一个球的半径为 $R = 3$ cm，求它的体积 $V$ 和表面积 $S$。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
\item[\textbf{第一步（代入体积公式）：}] 将 $R=3$ 代入 $V = \frac43 \pi R^3$，得 $V = \frac43 \pi \times 27 = 36\pi$（cm$^3$）。
\par\textbf{理由：} 直接套用球体积公式。

\item[\textbf{第二步（代入表面积公式）：}] 将 $R=3$ 代入 $S = 4\pi R^2$，得 $S = 4\pi \times 9 = 36\pi$（cm$^2$）。
\par\textbf{理由：} 直接套用球表面积公式。

\item[\textbf{第三步（给出答案）：}] 体积为 $36\pi$ cm$^3$，表面积为 $36\pi$ cm$^2$。
\par\textbf{理由：} 题目要求求体积和表面积。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{代入半径直接计算即可，注意体积和表面积可能数值相等但单位不同。}

\textbf{\textcolor{CExample}{例2（稍变形）}}

\textbf{\textcolor{CExample}{题目：}} 已知一个球的体积为 $V = 288\pi$，求它的表面积 $S$。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
\item[\textbf{第一步（逆向求半径）：}] 由 $V = \frac43 \pi R^3$ 得 $R^3 = \frac{3V}{4\pi} = \frac{3 \times 288\pi}{4\pi} = \frac{864\pi}{4\pi} = 216$，所以 $R = \sqrt[3]{216} = 6$。
\par\textbf{理由：} 解出半径是求表面积的前提。

\item[\textbf{第二步（代入表面积公式）：}] $S = 4\pi R^2 = 4\pi \times 36 = 144\pi$。
\par\textbf{理由：} 得到半径后直接套用表面积公式。

\item[\textbf{第三步（给出答案）：}] 表面积为 $144\pi$。
\par\textbf{理由：} 题目要求表面积。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{已知体积求表面积时，先通过开立方解出半径，再代入表面积公式。}

\textbf{\textcolor{CExample}{例3（综合应用）}}

\textbf{\textcolor{CExample}{题目：}} 两个球半径之比为 $R_1 : R_2 = 2 : 3$，求它们的体积之比 $V_1 : V_2$ 和表面积之比 $S_1 : S_2$。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
\item[\textbf{第一步（设比例参数）：}] 设 $R_1 = 2k$，$R_2 = 3k$（$k>0$）。
\par\textbf{理由：} 用同一参数表示半径，方便计算比。

\item[\textbf{第二步（求体积之比）：}] $V_1 = \frac43 \pi (2k)^3 = \frac43 \pi \cdot 8k^3$，$V_2 = \frac43 \pi (3k)^3 = \frac43 \pi \cdot 27k^3$，所以 $\frac{V_1}{V_2} = \frac{8}{27}$。
\[
\frac{V_1}{V_2} = \frac{8}{27}.
\]
\par\textbf{理由：} 体积中含有 $R^3$，比值为半径比的三次方。

\item[\textbf{第三步（求表面积之比）：}] $S_1 = 4\pi (2k)^2 = 4\pi \cdot 4k^2$，$S_2 = 4\pi (3k)^2 = 4\pi \cdot 9k^2$，所以 $\frac{S_1}{S_2} = \frac{4}{9}$。
\[
\frac{S_1}{S_2} = \frac{4}{9}.
\]
\par\textbf{理由：} 表面积中含有 $R^2$，比值为半径比的平方。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{球的体积比等于半径比的立方，表面积比等于半径比的平方。}
\end{examplebox}
