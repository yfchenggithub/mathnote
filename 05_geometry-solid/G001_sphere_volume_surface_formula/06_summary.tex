\begin{summarybox}
\textbf{\textcolor{CSummary}{一句话核心}}

\textbf{球的体积 $V = \frac43\pi R^3$，表面积 $S = 4\pi R^2$，仅与半径 $R$ 有关。}

\medskip
\textbf{\textcolor{CSummary}{使用条件}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
\item \textbf{条件1（已知半径）：} 必须知道球的半径 $R$，或能通过已知量求出 $R$。
\item \textbf{条件2（单位一致）：} 代入计算时注意长度单位统一，结果单位对应（体积为立方单位，表面积为平方单位）。
\end{itemize}

\medskip
\textbf{\textcolor{CSummary}{关键提醒}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
\item \textbf{易错点（公式混淆）：} 牢记体积系数 $\frac43\pi$，表面积系数 $4\pi$，不要交换。
\item \textbf{检查项（半径确认）：} 若题目给直径，务必将直径除以 2 得到半径后再套公式。
\end{itemize}
\end{summarybox}
