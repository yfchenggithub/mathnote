\begin{statementbox}
\textbf{\textcolor{CStatement}{条件}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
\item[\textbf{\textcolor{CStatement}{条件1（球与半径）：}}] 设球的半径为 $R$，其中 $R>0$。
\end{description}

\textbf{\textcolor{CStatement}{结论}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
\item[\textbf{\textcolor{CStatement}{结论一（体积公式）：}}]
\textbf{\textcolor{CStatement}{直观描述：}} 球的体积与半径的三次方成正比。
\[
V = \frac{4}{3}\pi R^{3} .
\]

\item[\textbf{\textcolor{CStatement}{结论二（表面积公式）：}}]
\textbf{\textcolor{CStatement}{直观描述：}} 球的表面积（即球面面积）与半径的平方成正比。
\[
S = 4\pi R^{2} .
\]
\end{description}

\medskip
\textbf{\textcolor{CStatement}{推广结论（逆用公式）：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
\item[\textbf{（已知体积求半径）：}]
若球的体积为 $V>0$，则
\[
R = \sqrt[3]{\frac{3V}{4\pi}} .
\]
\item[\textbf{（已知表面积求半径）：}]
若球的表面积为 $S>0$，则
\[
R = \sqrt{\frac{S}{4\pi}} .
\]
\item[\textbf{（已知表面积求体积）：}]
若球的表面积为 $S>0$，则先由
\[
R=\sqrt{\frac{S}{4\pi}}
\]
求出半径，再代入体积公式
\[
V=\frac{4}{3}\pi R^{3}.
\]
等价地，
\[
V=\frac{S}{3}\sqrt{\frac{S}{4\pi}}.
\]
\end{description}
\end{statementbox}
