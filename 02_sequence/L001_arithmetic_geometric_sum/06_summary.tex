\begin{summarybox}

\textbf{\textcolor{CSummary}{一句话核心}}

\textbf{差比数列求和可用公式 \(S_n = \frac{c_{n+1} - q^2 c_n + q c_1}{(q-1)^2}\) 或显式形式快速计算，本质是错位相减的固化。}

\medskip
\textbf{\textcolor{CSummary}{使用条件}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{条件 1（等差数列）：} 清楚公差 \(d\) 和首项 \(a_1\)。
    \item \textbf{条件 2（等比数列）：} 公比 \(q \neq 1\)，并明确首项 \(b_1\)。
    \item \textbf{条件 3（乘积项）：} 正确定义 \(c_n = a_n b_n\)，不可错位。
\end{itemize}

\medskip
\textbf{\textcolor{CSummary}{关键提醒}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{易错点（q=1 禁忌）：} 必须先验证 \(q \neq 1\)，否则公式不适用。
    \item \textbf{检查项（符号核对）：} 代入后计算 \(c_{n+1}, c_n, c_1\)，按“新减平方旧加首”核对分子符号。
\end{itemize}
\end{summarybox}
