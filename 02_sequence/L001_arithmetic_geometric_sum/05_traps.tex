\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CTrap}{易错点一（忽略q≠1）：}}] 当公比 \(q=1\) 时，分母为零，公式失效，此时需回归等差数列求和或直接用定义。
    \item[\textbf{\textcolor{CTrap}{正确理解（条件限制）：}}] 使用该公式前必须确认 \(q \neq 1\)。若 \(q=1\)，\(S_n\) 应作为等差与常数乘积的和处理。
    \item[\textbf{\textcolor{CTrap}{错因分析（惯性套用）：}}] 学生习惯直接代公式而忽视基本条件，导致分母无意义或计算错误。
\end{description}

\medskip
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CTrap}{易错点二（混淆cn定义）：}}] 误将 \(c_n\) 记作 \(a_n b_{n+1}\) 或 \(a_{n+1} b_n\) 代入公式，导致结果错误。
    \item[\textbf{\textcolor{CTrap}{正确理解（严格相乘）：}}] 牢记 \(c_n = a_n b_n\)，且 \(a_n\) 与 \(b_n\) 的下标必须一致，否则破坏了乘积结构。
    \item[\textbf{\textcolor{CTrap}{错因分析（符号混淆）：}}] 在错位相减过程中，容易出现下标错位的衍生记号，需核对每一项的原始定义。
\end{description}

\medskip
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CTrap}{易错点三（分子符号反记）：}}] 公式分子为 \(c_{n+1} - q^2 c_n + q c_1\)，易写成 \(q^2 c_n - c_{n+1} + q c_1\) 或漏写项。
    \item[\textbf{\textcolor{CTrap}{正确理解（记忆口诀）：}}] 可以用“新减平方旧加首”辅助记忆：\(c_{n+1}\) 减 \(q^2 c_n\) 加 \(q c_1\)。
    \item[\textbf{\textcolor{CTrap}{错因分析（推导跳跃）：}}] 推导时未注意移项与通分的符号变化，凭感觉记错顺序。
\end{description}
