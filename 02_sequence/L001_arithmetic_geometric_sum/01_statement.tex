\textbf{\textcolor{CStatement}{条件}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{条件 1（等差定义）：}}] 数列 \(\{a_n\}\) 为等差数列，设公差为 \(d\)。
    \item[\textbf{\textcolor{CStatement}{条件 2（等比定义）：}}] 数列 \(\{b_n\}\) 为等比数列，公比 \(q \neq 1\)。
    \item[\textbf{\textcolor{CStatement}{条件 3（乘积定义）：}}] 令 \(c_n = a_n b_n\)，其前 \(n\) 项和为 \(S_n = \sum_{k=1}^{n} c_k\)。
\end{description}

\textbf{\textcolor{CStatement}{结论}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{\textcolor{CStatement}{结论一（差比乘积和）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 前 \(n\) 项和可用项与首项交叉表达。
    \[
    S_n = \frac{c_{n+1} - q^{2} c_n + q c_1}{(q-1)^{2}}.
    \]

    \item[\textbf{\textcolor{CStatement}{推广结论（显式求和）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 直接由首项、公差和公比给出显式。
    \[
    S_n = \frac{a_1 b_1 - a_{n+1} b_{n+1}}{1-q} + \frac{d b_1 q (1 - q^{n})}{(1-q)^{2}}.
    \]
\end{description}

\textbf{\textcolor{CStatement}{等号/取等条件：}} 无（公式为恒等式，仅需 \(q \neq 1\)）。

\end{statementbox}
