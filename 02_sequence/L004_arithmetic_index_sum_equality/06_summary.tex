\begin{summarybox}
\textbf{\textcolor{CSummary}{一句话核心}}

\textbf{在等差数列中，若$m+n=p+q$，则$a_m+a_n=a_p+a_q$。} 这是等差中项性质的推广。

\medskip
\textbf{\textcolor{CSummary}{使用条件}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{条件1（数列等差）：} 必须确认数列是等差数列，公差为常数。
    \item \textbf{条件2（下标和相等）：} 四个下标$m,n,p,q$满足$m+n=p+q$（均为正整数）。
\end{itemize}

\medskip
\textbf{\textcolor{CSummary}{关键提醒}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{易错点（前提遗忘）：} 不要在其他数列（如等比数列）中滥用此结论。
    \item \textbf{检查项（下标配对）：} 计算下标和时注意是否确实相等，避免计算失误。
\end{itemize}
\end{summarybox}
