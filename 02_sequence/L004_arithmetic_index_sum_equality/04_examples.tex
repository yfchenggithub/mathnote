\begin{examplebox}
\textbf{\textcolor{CExample}{例1（基础直接应用）}}

\textbf{\textcolor{CExample}{题目：}} 已知等差数列$\{a_n\}$中，$a_2=5$，$a_6=13$，求$a_3+a_5$的值。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（找下标关系）：}] 观察到$2+6=8$，$3+5=8$，满足$2+6=3+5$。
    \par\textbf{理由：} 需要找到满足下标和相等的配对。
    \item[\textbf{第二步（套用性质）：}] 由等差数列性质，$a_2+a_6 = a_3+a_5$。
    \par\textbf{理由：} 直接应用下标和性质。
    \item[\textbf{第三步（代值计算）：}] $a_2+a_6 = 5+13 = 18$，故$a_3+a_5 = 18$。
    \par\textbf{理由：} 代入已知数值。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{$a_3+a_5=18$}。

\medskip
\textbf{\textcolor{CExample}{例2（稍变形：已知和求某项）}}

\textbf{\textcolor{CExample}{题目：}} 在等差数列$\{a_n\}$中，$a_1+a_5=10$，求$a_3$的值。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（构造中项）：}] 注意到$1+5=6$，$3+3=6$，由推广结论（中项特例）：$a_1+a_5 = 2a_3$。
    \par\textbf{理由：} 当下标和相等且为偶数时，两项和等于中间项的两倍。
    \item[\textbf{第二步（解方程）：}] 由$2a_3 = 10$，得$a_3=5$。
    \par\textbf{理由：} 直接解一元一次方程。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{$a_3=5$}。

\medskip
\textbf{\textcolor{CExample}{例3（综合应用：结合公差）}}

\textbf{\textcolor{CExample}{题目：}} 等差数列$\{a_n\}$中，$a_2+a_8=12$，且$a_4=3$，求公差$d$。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（配对关系）：}] 观察$2+8=10$，$4+6=10$，则$a_2+a_8 = a_4+a_6$。
    \par\textbf{理由：} 下标和相等性质。
    \item[\textbf{第二步（代入已知）：}] 已知$a_2+a_8=12$，$a_4=3$，故$3 + a_6 = 12$，得$a_6=9$。
    \par\textbf{理由：} 等式转移。
    \item[\textbf{第三步（求公差）：}] 由$a_6$与$a_4$，有$a_6 = a_4 + (6-4)d = 3 + 2d = 9$，解得$d=3$。
    \par\textbf{理由：} 等差数列相邻项差为公差$d$。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{公差$d=3$}。
\end{examplebox}
