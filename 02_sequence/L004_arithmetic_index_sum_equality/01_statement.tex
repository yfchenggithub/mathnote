\textbf{\textcolor{CStatement}{条件}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{条件1（等差数列定义）：}}] 数列$\{a_n\}$是等差数列，即存在常数$d$使得$a_{n+1}-a_n=d$对任意正整数$n$成立。
\end{description}

\textbf{\textcolor{CStatement}{结论}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{\textcolor{CStatement}{结论一（下标和性质）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 下标和相等时，对应两项之和相等。
    \[
    \text{若}m+n=p+q\text{，则}a_m+a_n=a_p+a_q\text{。}
    \]

    \item[\textbf{\textcolor{CStatement}{推广结论（中项特例）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 当下标和为偶数时，两项和等于中间项的两倍。
    \[
    \text{若}m+n=2k\text{，则}a_m+a_n=2a_k\text{。}
    \]
\end{description}

\textbf{\textcolor{CStatement}{等号/取等条件：}} 等号恒成立，无需额外条件。

\end{statementbox}
