\begin{examplebox}
\textbf{\textcolor{CExample}{例 1（基础）}}

\textbf{\textcolor{CExample}{题目：}} 在等差数列 $\{a_n\}$ 中，已知 $a_1=5$，$a_{10}=41$，求前 $10$ 项的和 $S_{10}$。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（提取已知量）：}] 
    由题知 $a_1=5$，$a_{10}=41$，项数 $n=10$。
    \par\textbf{理由：}
    明确已知量与求解目标，选定含 $a_1,a_n,n$ 的求和公式。

    \item[\textbf{第二步（选用公式）：}] 
    使用 $S_n = \dfrac{n(a_1+a_n)}{2}$。
    \par\textbf{理由：}
    已知首项和末项时，该式无需先求公差，计算最简。

    \item[\textbf{第三步（代入计算）：}] 
    \[
    S_{10} = \frac{10 \times (5 + 41)}{2} = \frac{10 \times 46}{2} = 230.
    \]
    \par\textbf{理由：}
    直接代入数值，注意先算括号内，再乘除避免错误。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{$S_{10}=230$，使用首末项公式可快速求解。}

\textbf{\textcolor{CExample}{例 2（稍变形）}}

\textbf{\textcolor{CExample}{题目：}} 等差数列的前 $5$ 项和为 $35$，前 $10$ 项和为 $120$，求首项 $a_1$ 和公差 $d$。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（设出求和公式）：}] 
    由 $S_n = na_1 + \dfrac{n(n-1)}{2}d$ 得
    \[
    S_5 = 5a_1 + 10d,\quad S_{10} = 10a_1 + 45d.
    \]
    \par\textbf{理由：}
    选用含 $a_1,d,n$ 的公式，便于直接列方程。

    \item[\textbf{第二步（建立方程组）：}] 
    依题意得
    \[
    \begin{cases}
    5a_1 + 10d = 35,\\
    10a_1 + 45d = 120.
    \end{cases}
    \]
    \par\textbf{理由：}
    将已知和转化为关于 $a_1,d$ 的线性方程组。

    \item[\textbf{第三步（解方程组）：}] 
    化简第一个方程得 $a_1 + 2d = 7$，第二个方程两边除以 $5$ 得 $2a_1 + 9d = 24$。消去 $a_1$：将第一式乘 $2$ 得 $2a_1+4d=14$，与第二式相减得 $5d=10$，所以 $d=2$，代回得 $a_1 = 7-4=3$。
    \par\textbf{理由：}
    采用加减消元法，逐步求出 $d$ 再回代求 $a_1$。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{$a_1=3$，$d=2$，利用求和公式建立方程组是核心方法。}
\end{examplebox}
