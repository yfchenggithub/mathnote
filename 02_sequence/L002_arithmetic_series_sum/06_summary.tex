\begin{summarybox}
\textbf{\textcolor{CSummary}{一句话核心}}
\textbf{等差数列前 $n$ 项和等于“首尾平均”乘“项数”，本质是倒序相加构造配对常数。}

\medskip
\textbf{\textcolor{CSummary}{使用条件}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{条件 1（数列等差）：} $\{a_n\}$ 满足 $a_{n+1}-a_n=d$（常数），即等差定义。
    \item \textbf{条件 2（项数限制）：} $n$ 为正整数，公式对 $n\geq 1$ 恒成立。
\end{itemize}

\medskip
\textbf{\textcolor{CSummary}{关键提醒}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{易错点（公式选用）：} 已知首末项时直接用 $S_n = \frac{n(a_1+a_n)}{2}$；已知首项与公差时用含 $d$ 的公式，避免走弯路。
    \item \textbf{检查项（下标统一）：} 确认题目给出的项数与公式中的 $n$ 一致，尤其片段和问题中不要混淆 $S_n$ 与 $a_n$。
\end{itemize}
\end{summarybox}
