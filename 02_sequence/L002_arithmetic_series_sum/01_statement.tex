\textbf{\textcolor{CStatement}{条件}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{条件 1（等差数列定义）：}}] 数列 $\{a_n\}$ 为等差数列，首项 $a_1$，公差 $d$，即对任意 $n\in\mathbb{N}^*$ 有 $a_{n+1}-a_n=d$。
\end{description}

\textbf{\textcolor{CStatement}{结论}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{\textcolor{CStatement}{结论一（前n项和公式）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 数列和等于首尾平均乘项数，像梯形面积。
    \[
    S_n = \frac{n(a_1+a_n)}{2} = na_1 + \frac{n(n-1)}{2}d .
    \]

    \item[\textbf{\textcolor{CStatement}{推广结论（倒序相加形式）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 采用首尾配对相加，乘以项数一半。
    \[
    S_n = \frac{n}{2}(a_1 + a_n) .
    \]
\end{description}

\end{statementbox}
