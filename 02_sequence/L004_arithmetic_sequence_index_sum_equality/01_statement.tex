\begin{statementbox}

\textbf{\textcolor{CStatement}{条件}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{条件 1（数列等差）：}}] 数列 $\{a_n\}$ 是等差数列。
    \item[\textbf{\textcolor{CStatement}{条件 2（下标和等）：}}] $m,n,p,q\in\mathbb{N}^*$ 且 $m+n=p+q$。
\end{description}

\textbf{\textcolor{CStatement}{结论}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{\textcolor{CStatement}{结论一（下标和定和）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 两对下标之和相等时，对应两项之和也相等。
    \[
    a_m + a_n = a_p + a_q.
    \]

    \item[\textbf{\textcolor{CStatement}{推广结论（等差中项）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 当下标满足 $m+n=2k$ 时，两项和等于中项的两倍。
    \[
    a_m + a_n = 2a_k \quad (m+n=2k,\ k\in\mathbb{N}^*).
    \]
\end{description}

\textbf{\textcolor{CStatement}{等号/取等条件：}} 本结论为恒等式，在条件满足时自动成立，无额外取等条件。

\end{statementbox}
