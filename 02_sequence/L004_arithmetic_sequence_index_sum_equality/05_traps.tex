\begin{trapbox}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CTrap}{易错点一（忽略等差前提）}}] 以为该结论对所有数列都成立，错误代入等比或其他递推数列。
    \item[\textbf{\textcolor{CTrap}{正确理解（仅限等差数列）：}}] 结论严格依赖于等差数列通项为一次函数，仅在等差数列中整体成立。
    \item[\textbf{\textcolor{CTrap}{错因分析（概念混淆）：}}] 未区分数列类型，将通项公式为线性的特性误当作数列通用性质。
\end{description}

\medskip
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CTrap}{易错点二（符号理解错误）}}] 将 $a_m + a_n = a_{m+n}$ 或类似错误公式当成正确结论使用。
    \item[\textbf{\textcolor{CTrap}{正确理解（区分下标与项）：}}] 结论只涉及两项直接相加，下标和相等指的是序号相加，不是把序号和当作新序号。
    \item[\textbf{\textcolor{CTrap}{错因分析（符号辨识弱）：}}] 初次学习时容易混淆 $a_{m+n}$ 与 $a_m+a_n$，需反复辨析符号含义。
\end{description}

\medskip
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CTrap}{易错点三（错误推广到差）}}] 认为若 $m+n=p+q$，便有 $a_m-a_n = a_p-a_q$，并据此计算。
    \item[\textbf{\textcolor{CTrap}{正确理解（差无此性质）：}}] 等差数列中 $a_m-a_n = (m-n)d$，结果依赖于差值 $m-n$ 而不是和 $m+n$，不能由下标和相等导出。
    \item[\textbf{\textcolor{CTrap}{错因分析（类比不当）：}}] 仅看到下标关系，未检查运算类型，将加法性质错误迁移到减法。
\end{description}
\end{trapbox}
