\begin{summarybox}
\textbf{\textcolor{CSummary}{一句话核心}}

\textbf{等差数列中，下标和相等则项和相等。}

\medskip
\textbf{\textcolor{CSummary}{使用条件}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{条件 1（数列等差）：} 数列必须是等差数列，公差任意。
    \item \textbf{条件 2（下标和等）：} $m,n,p,q$ 为正整数，且 $m+n = p+q$。
\end{itemize}

\medskip
\textbf{\textcolor{CSummary}{关键提醒}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{易错点（非等差勿用）：} 等比或其他数列不可使用此结论。
    \item \textbf{检查项（符号区分）：} 注意 $a_m + a_n$ 不等于 $a_{m+n}$，别混淆。
\end{itemize}
\end{summarybox}
