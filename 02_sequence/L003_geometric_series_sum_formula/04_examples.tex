\begin{examplebox}
\textbf{\textcolor{CExample}{例 1（基础）}}

\textbf{\textcolor{CExample}{题目：}} 等比数列 $\{a_n\}$ 的首项 $a_1=2$，公比 $q=3$，求前 $5$ 项和 $S_5$。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（识别参数）：}] $a_1=2$, $q=3$, $n=5$。
    \par\textbf{理由：} 从题目直接提取条件，确认公比不为 $1$。
    \item[\textbf{第二步（套公式）：}] 因为 $q\neq 1$，使用 $S_n = \frac{a_1(1-q^n)}{1-q}$。
    \par\textbf{理由：} 选择对应情形的求和公式。
    \item[\textbf{第三步（代入求值）：}] 
    \[
    \begin{aligned}
    S_5 &= \frac{2(1-3^5)}{1-3} \\
    &= \frac{2(1-243)}{-2} \\
    &= 242.
    \end{aligned}
    \]
    \par\textbf{理由：} 计算 $3^5=243$，分子分母同除 $-2$，得 $242$。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{该等比数列前 $5$ 项和为 $242$。}

\medskip
\textbf{\textcolor{CExample}{例 2（稍变形）}}

\textbf{\textcolor{CExample}{题目：}} 已知等比数列首项 $a_1=5$，公比 $q=\dfrac12$。求前 $5$ 项和 $S_5$，并求该数列所有项的和 $S$。

\textbf{\textcolor{CExample}{解题步骤：}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{第一步（判断公比）：}] $q=\frac12\neq 1$，且 $|q|<1$。
    \par\textbf{理由：} 确保可使用求和公式及无穷项和公式。
    \item[\textbf{第二步（求前5项和）：}] 
    \[
    \begin{aligned}
    S_5 &= \frac{5\bigl(1-(\frac12)^5\bigr)}{1-\frac12} \\
    &= \frac{5(1-\frac{1}{32})}{\frac12} \\
    &= \frac{5\cdot\frac{31}{32}}{\frac12} \\
    &= 5\cdot\frac{31}{32}\cdot2 = \frac{155}{16}.
    \end{aligned}
    \]
    \par\textbf{理由：} 依次化简指数、通分、乘倒数，得到最简分数。
    \item[\textbf{第三步（求无穷和）：}] 
    \[
    S = \frac{a_1}{1-q} = \frac{5}{1-\frac12}=10.
    \]
    \par\textbf{理由：} 由 $|q|<1$ 直接套无穷项和公式。
\end{description}

\textbf{\textcolor{CExample}{关键结论：}} \textbf{前 $5$ 项和 $S_5 = \frac{155}{16}$，所有项的和 $S = 10$。}
\end{examplebox}
