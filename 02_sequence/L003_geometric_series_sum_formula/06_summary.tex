\begin{summarybox}
\textbf{\textcolor{CSummary}{一句话核心}}

\textbf{等比数列求和，就是利用错位相减将指数增长转化为一次方程，得到分段公式。}

\medskip
\textbf{\textcolor{CSummary}{使用条件}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{条件 1（等比定义）：} 数列从第二项起每一项与前一项的比恒为常数 $q$（$q\neq0$）。
    \item \textbf{条件 2（参数充分）：} 必须已知首项 $a_1$ 和公比 $q$，或能通过条件求出它们。
\end{itemize}

\medskip
\textbf{\textcolor{CSummary}{关键提醒}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{易错点（分类讨论）：} 第一步务必检查 $q$ 是否等于 $1$，避免除零。
    \item \textbf{检查项（指数验证）：} 代入小值（如 $n=1,2$）检验公式结果是否等于 $a_1$ 和 $a_1+a_2$。
\end{itemize}
\end{summarybox}
