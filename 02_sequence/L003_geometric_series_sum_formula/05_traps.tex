\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CTrap}{易错点一（忽略q=1）：}}] 直接使用分母 $1-q$ 导致除以零，尤其在字母题中忽略 $q=1$ 的可能性。
    \item[\textbf{\textcolor{CTrap}{正确理解（分类讨论）：}}] 遇到等比求和，首要一步是判断 $q$ 是否等于 $1$，再选用对应公式。
    \item[\textbf{\textcolor{CTrap}{错因分析（定义域遗忘）：}}] 学生习惯套用分式而忘记分母不可为零，未养成分类讨论的习惯。
\end{description}

\medskip
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CTrap}{易错点二（项数指数混淆）：}}] 误以为通项是 $a_1 q^n$，导致求和公式写成 $\frac{a_1(1-q^{n+1})}{1-q}$ 等错误。
    \item[\textbf{\textcolor{CTrap}{正确理解（指数对应）：}}] 第 $n$ 项是 $a_n = a_1 q^{n-1}$，前 $n$ 项和公式中的指数是 $q^n$，记住“首项指数 $0$”的约定。
    \item[\textbf{\textcolor{CTrap}{错因分析（首项指数误）：}}] 把首项当成 $q^1$，漏掉 $n-1$ 的减一导致指数整体偏移。
\end{description}

\medskip
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CTrap}{易错点三（末项公式符号）：}}] 使用 $\frac{a_n q - a_1}{1-q}$ 或漏乘 $q$ 而写成 $\frac{a_1 - a_n}{1-q}$。
    \item[\textbf{\textcolor{CTrap}{正确理解（符号检查）：}}] 末项表示形式应为 $\frac{a_1 - a_n q}{1-q}$，可通过代入 $n=1$ 检验是否回退为 $a_1$ 来验证。
    \item[\textbf{\textcolor{CTrap}{错因分析（推导变形错）：}}] 在变形过程中符号处理随意，不做回代检验，导致公式记忆错误。
\end{description}
\end{trapbox}
