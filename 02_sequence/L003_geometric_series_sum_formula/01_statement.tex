\textbf{\textcolor{CStatement}{条件}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{条件 1（等比定义）：}}] 数列 $\{a_n\}$ 满足 $\frac{a_{n}}{a_{n-1}} = q$（$n\ge 2$），即各项均为前一项的 $q$ 倍，首项 $a_1\neq 0$，$q\neq 0$。
\end{description}

\textbf{\textcolor{CStatement}{结论}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{\textcolor{CStatement}{结论一（分段求和）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 等比数列前 $n$ 项和按公比是否等于 $1$ 分成两种情形计算。
    \[
    S_n=
    \begin{cases}
    n a_1, & q=1,\\[6pt]
    \dfrac{a_1(1-q^n)}{1-q}
    =\dfrac{a_1-a_n q}{1-q}, & q\neq 1.
    \end{cases}
    \]

    \item[\textbf{\textcolor{CStatement}{推广结论(无穷项和):}}]
    \textbf{\textcolor{CStatement}{直观描述:}} 当公比的绝对值小于 $1$ 时,无限项求和收敛于常数.
    \[
    S = \frac{a_1}{1-q},\qquad |q|<1.
    \]
\end{description}

\end{statementbox}
