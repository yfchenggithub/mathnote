\begin{trapbox}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CTrap}{易错点一（漏掉减一）}}] 错误认为
    \[
    1!+2\cdot2!+\cdots+n\cdot n!=(n+1)!.
    \]
    \item[\textbf{\textcolor{CTrap}{正确理解（完整公式）：}}] 正确结果是
    \[
    1!+2\cdot2!+\cdots+n\cdot n!=(n+1)!-1.
    \]
    \item[\textbf{\textcolor{CTrap}{错因分析（漏看首项）：}}] 裂项后首尾为 $(n+1)!-1!$，而 $1!=1$，所以必须减 $1$。
\end{description}

\medskip
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CTrap}{易错点二（裂项混淆）}}] 将 $k\cdot k!$ 误写成 $(k+1)!-k$、$k!-(k-1)!$ 或 $(k+1)!-1$。
    \item[\textbf{\textcolor{CTrap}{正确理解（标准裂项）：}}] 标准裂项是
    \[
    k\cdot k!=(k+1)!-k!.
    \]
    \item[\textbf{\textcolor{CTrap}{错因分析（阶乘展开错误）：}}] 没有先把 $(k+1)!$ 写成 $(k+1)k!$，导致相邻阶乘差写错。
\end{description}

\medskip
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CTrap}{易错点三（下限变化仍照搬）}}] 看到 $2\cdot2!+3\cdot3!+\cdots+n\cdot n!$，仍写成 $(n+1)!-1$。
    \item[\textbf{\textcolor{CTrap}{正确理解（首项随下限变化）：}}] 若从 $r$ 开始，则
    \[
    \sum_{k=r}^{n}k\cdot k!=(n+1)!-r!.
    \]
    例如从 $2$ 开始，应为 $(n+1)!-2!$。
    \item[\textbf{\textcolor{CTrap}{错因分析（只记结论不看下限）：}}] $(n+1)!-1$ 只对应从 $k=1$ 开始的求和。
\end{description}

\medskip
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CTrap}{易错点四（定义域说法过死）}}] 认为阶乘“只定义在正整数上”，从而把 $0!$ 一概排除。
    \item[\textbf{\textcolor{CTrap}{正确理解（教材约定）：}}] 高中常用约定包含 $0!=1$；本公式主讲 $n\in\mathbb{N}^+$，若采用空和约定，则 $n=0$ 时也有 $0=(0+1)!-1$。
    \item[\textbf{\textcolor{CTrap}{错因分析（范围与约定混淆）：}}] “本题范围限制”与“阶乘定义约定”不是一回事，讲义中应避免绝对化表述。
\end{description}
\end{trapbox}
