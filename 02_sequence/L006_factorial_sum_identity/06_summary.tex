\begin{summarybox}
\textbf{\textcolor{CSummary}{一句话核心}}

\textbf{看到 $k\cdot k!$，先裂成 $(k+1)!-k!$；从 $1$ 加到 $n$，答案就是 $(n+1)!-1$。}

\medskip
\textbf{\textcolor{CSummary}{使用条件}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{条件 1（正整数限定）：} 主公式用于 $n\in\mathbb{N}^+$。
    \item \textbf{条件 2（通项匹配）：} 通项必须是 $k\cdot k!$，不能把 $k!$、$k\cdot(k+1)!$ 等误套入公式。
    \item \textbf{条件 3（下限检查）：} 若不是从 $k=1$ 开始，应使用 $\sum_{k=r}^{n}k\cdot k!=(n+1)!-r!$。
\end{itemize}

\medskip
\textbf{\textcolor{CSummary}{关键提醒}}
\begin{itemize}[leftmargin=*, itemsep=0.5em, topsep=0.4em]
    \item \textbf{易错点（须减一）：} 从 $1$ 开始时，结果是 $(n+1)!-1$，不是 $(n+1)!$。
    \item \textbf{检查项（先验算裂项）：} 用 $k=1$ 检验：左边 $1\cdot1!=1$，右边 $2!-1!=1$。
    \item \textbf{记忆法：} “阶乘求和看差分，下一阶乘减首阶乘。”
\end{itemize}
\end{summarybox}
