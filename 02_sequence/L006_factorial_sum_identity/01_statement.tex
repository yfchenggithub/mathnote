\begin{statementbox}
\textbf{\textcolor{CStatement}{条件}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{条件 1（定义域限制）：}}] $n\in\mathbb{N}^+$，求和下标 $k=1,2,\dots,n$。
    \item[\textbf{\textcolor{CStatement}{条件 2（通项匹配）：}}] 通项必须是 $k\cdot k!$，即“系数”和“阶乘下标”相同。
\end{description}

\textbf{\textcolor{CStatement}{结论}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{\textcolor{CStatement}{结论一（阶乘裂项求和公式）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 从 $1\cdot 1!$ 到 $n\cdot n!$ 求和，结果等于“下一阶乘减 $1$”。
    \[
    1\cdot 1!+2\cdot 2!+3\cdot 3!+\cdots+n\cdot n!=(n+1)!-1.
    \]
    由于 $1\cdot 1!=1!$，也常写成
    \[
    1!+2\cdot 2!+3\cdot 3!+\cdots+n\cdot n!=(n+1)!-1.
    \]

    \item[\textbf{\textcolor{CStatement}{核心裂项：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 每一项 $k\cdot k!$ 都可以拆成相邻两个阶乘之差。
    \[
    k\cdot k!=(k+1)!-k!.
    \]

    \item[\textbf{\textcolor{CStatement}{常用变形（非从 $1$ 开始）：}}]
    若 $1\le r\le n$，则
    \[
    \sum_{k=r}^{n}k\cdot k!=(n+1)!-r!.
    \]
    \textbf{\textcolor{CStatement}{说明：}} 下限不是 $1$ 时，不能直接写成 $(n+1)!-1$。
\end{description}
\end{statementbox}
