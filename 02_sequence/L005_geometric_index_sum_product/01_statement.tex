\begin{statementbox}
\textbf{\textcolor{CStatement}{条件}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.35em]
    \item[\textbf{\textcolor{CStatement}{条件1（下标和相等）：}}] 等比数列 $\{a_n\}$ 中，下标 $m,n,p,q\in\mathbb{N}^*$ 满足 $m+n=p+q$。
\end{description}

\textbf{\textcolor{CStatement}{结论}}
\begin{description}[style=nextline, leftmargin=2.8em, labelsep=0.5em, itemsep=0.45em]
    \item[\textbf{\textcolor{CStatement}{结论一（两项乘积相等）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 下标和相等时，对应两项的乘积相等。
    \[
    a_m\cdot a_n = a_p\cdot a_q.
    \]

    \item[\textbf{\textcolor{CStatement}{推广结论（等比中项性质）：}}]
    \textbf{\textcolor{CStatement}{直观描述：}} 当下标和是偶数时，乘积等于中间项的平方。
    \[
    m+n=2k \quad\Rightarrow\quad a_m\cdot a_n = a_k^2.
    \]
\end{description}

\textbf{\textcolor{CStatement}{等号/取等条件：}} 无额外取等条件，等式恒成立。
\end{statementbox}
