\begin{proofbox}
\textbf{向量数量积边长公式的证明}：
\begin{enumerate}
    \item \textbf{向量表示对边}：
    在 $\triangle ABC$ 中，设 $\vec{AB} = \mathbf{c}, \vec{AC} = \mathbf{b}$，则对边 $\vec{BC} = \vec{AC} - \vec{AB} = \mathbf{b} - \mathbf{c}$。
    
    \item \textbf{模长平方运算}：
    对向量 $\vec{BC}$ 进行模长平方运算：
    $a^2 = |\vec{BC}|^2 = |\vec{AC} - \vec{AB}|^2$
    
    \item \textbf{展开向量数量积}：
    根据完全平方公式：
    $a^2 = |\vec{AC}|^2 + |\vec{AB}|^2 - 2\vec{AB} \cdot \vec{AC}$
    
    \item \textbf{代入边长符号}：
    $a^2 = b^2 + c^2 - 2\vec{AB} \cdot \vec{AC}$
    
    \item \textbf{移项整理}：
    $2\vec{AB} \cdot \vec{AC} = b^2 + c^2 - a^2$
    由此得出：$\vec{AB} \cdot \vec{AC} = \frac{b^2 + c^2 - a^2}{2}$。
\end{enumerate}
\end{proofbox}
